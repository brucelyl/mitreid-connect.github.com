017package org.mitre.oauth2.introspectingfilter;
018
019import java.util.Date;
020import java.util.HashMap;
021import java.util.Map;
022
023import org.slf4j.LoggerFactory;
024import org.springframework.security.core.Authentication;
025import org.springframework.security.core.AuthenticationException;
026import org.springframework.security.core.authority.AuthorityUtils;
027import org.springframework.security.oauth2.common.OAuth2AccessToken;
028import org.springframework.security.oauth2.provider.AuthorizationRequest;
029import org.springframework.security.oauth2.provider.OAuth2Authentication;
030import org.springframework.security.oauth2.provider.token.ResourceServerTokenServices;
031import org.springframework.security.web.authentication.preauth.PreAuthenticatedAuthenticationToken;
032import org.springframework.util.LinkedMultiValueMap;
033import org.springframework.util.MultiValueMap;
034import org.springframework.web.client.RestClientException;
035import org.springframework.web.client.RestTemplate;
036
037import com.google.gson.JsonElement;
038import com.google.gson.JsonObject;
039import com.google.gson.JsonParser;
040
041public class IntrospectingTokenService implements ResourceServerTokenServices {
042
043
044        private String clientId;
045        private String clientSecret;
046        private String introspectionUrl;
047
048        // Inner class to store in the hash map
049        private class TokenCacheObject { OAuth2AccessToken token; OAuth2Authentication auth;
050        private TokenCacheObject(OAuth2AccessToken token, OAuth2Authentication auth) {
051                this.token = token;
052                this.auth = auth;
053        }
054        }
055        private Map<String, TokenCacheObject> authCache = new HashMap<String, TokenCacheObject>();
056
057        public String getIntrospectionUrl() {
058                return introspectionUrl;
059        }
060
061        public void setIntrospectionUrl(String introspectionUrl) {
062                this.introspectionUrl = introspectionUrl;
063        }
064
065        public String getClientId() {
066                return clientId;
067        }
068
069        public void setClientId(String clientId) {
070                this.clientId = clientId;
071        }
072
073        public String getClientSecret() {
074                return clientSecret;
075        }
076
077        public void setClientSecret(String clientSecret) {
078                this.clientSecret = clientSecret;
079        }
080
081        // Check if there is a token and authentication in the cache
082        //   and check if it is not expired.
083        private TokenCacheObject checkCache(String key) {
084                if(authCache.containsKey(key)) {
085                        TokenCacheObject tco = authCache.get(key);
086                        if (tco.token.getExpiration().after(new Date())) {
087                                return tco;
088                        } else {
089                                // if the token is expired, don't keep things around.
090                                authCache.remove(key);
091                        }
092                }
093                return null;
094        }
095
096        private AuthorizationRequest createAuthRequest(final JsonObject token) {
097                AuthorizationRequest authReq = new AuthorizationRequestImpl(token);
098                return authReq;
099        }
100
101        // create a default authentication object with authority ROLE_API
102        private Authentication createAuthentication(JsonObject token){
103                // TODO: make role/authority configurable somehow
104                return new PreAuthenticatedAuthenticationToken(token.get("sub").getAsString(), null, AuthorityUtils.createAuthorityList("ROLE_API"));
105        }
106
107        private OAuth2AccessToken createAccessToken(final JsonObject token, final String tokenString){
108                OAuth2AccessToken accessToken = new OAuth2AccessTokenImpl(token, tokenString);
109                return accessToken;
110        }
111
112        // Validate a token string against the introspection endpoint,
113        //   then parse it and store it in the local cache. Return true on
114        //   sucess, false otherwise.
115        private boolean parseToken(String accessToken) {
116                String validatedToken = null;
117                // Use the SpringFramework RestTemplate to send the request to the endpoint
118
119                RestTemplate restTemplate = new RestTemplate();
120                MultiValueMap<String, String> form = new LinkedMultiValueMap<String, String>();
121                form.add("token",accessToken);
122                form.add("client_id", this.clientId);
123                form.add("client_secret", this.clientSecret);
124
125                try {
126                        validatedToken = restTemplate.postForObject(introspectionUrl, form, String.class);
127                } catch (RestClientException rce) {
128                        // TODO: LOG THIS!?
129                        LoggerFactory.getLogger(IntrospectingTokenService.class).error("validateToken", rce);
130                }
131                if (validatedToken != null) {
132                        // parse the json
133                        JsonElement jsonRoot = new JsonParser().parse(validatedToken);
134                        if (!jsonRoot.isJsonObject()) {
135                                return false; // didn't get a proper JSON object
136                        }
137
138                        JsonObject tokenResponse = jsonRoot.getAsJsonObject();
139
140                        if (tokenResponse.get("error") != null) {
141                                // report an error?
142                                return false;
143                        }
144
145                        if (!tokenResponse.get("active").getAsBoolean()){
146                                // non-valid token
147                                return false;
148                        }
149                        // create an OAuth2Authentication
150                        OAuth2Authentication auth = new OAuth2Authentication(createAuthRequest(tokenResponse), createAuthentication(tokenResponse));
151                        // create an OAuth2AccessToken
152                        OAuth2AccessToken token = createAccessToken(tokenResponse, accessToken);
153
154                        if (token.getExpiration().after(new Date())){
155                                // Store them in the cache
156                                authCache.put(accessToken, new TokenCacheObject(token,auth));
157
158                                return true;
159                        }
160                }
161
162                // If we never put a token and an authentication in the cache...
163                return false;
164        }
165
166        @Override
167        public OAuth2Authentication loadAuthentication(String accessToken) throws AuthenticationException {
168                // First check if the in memory cache has an Authentication object, and that it is still valid
169                // If Valid, return it
170                TokenCacheObject cacheAuth = checkCache(accessToken);
171                if (cacheAuth != null) {
172                        return cacheAuth.auth;
173                } else {
174                        if (parseToken(accessToken)) {
175                                cacheAuth = authCache.get(accessToken);
176                                if (cacheAuth != null && (cacheAuth.token.getExpiration().after(new Date()))) {
177                                        return cacheAuth.auth;
178                                } else {
179                                        return null;
180                                }
181                        } else {
182                                return null;
183                        }
184                }
185        }
186
187        @Override
188        public OAuth2AccessToken readAccessToken(String accessToken) {
189                // First check if the in memory cache has a Token object, and that it is still valid
190                // If Valid, return it
191                TokenCacheObject cacheAuth = checkCache(accessToken);
192                if (cacheAuth != null) {
193                        return cacheAuth.token;
194                } else {
195                        if (parseToken(accessToken)) {
196                                cacheAuth = authCache.get(accessToken);
197                                if (cacheAuth != null && (cacheAuth.token.getExpiration().after(new Date()))) {
198                                        return cacheAuth.token;
199                                } else {
200                                        return null;
201                                }
202                        } else {
203                                return null;
204                        }
205                }
206        }
