004package org.mitre.openid.connect.client.service.impl;
005
006import java.util.concurrent.ExecutionException;
007
008import org.apache.http.client.HttpClient;
009import org.apache.http.impl.client.DefaultHttpClient;
010import org.mitre.openid.connect.client.service.ServerConfigurationService;
011import org.mitre.openid.connect.config.ServerConfiguration;
012import org.slf4j.Logger;
013import org.slf4j.LoggerFactory;
014import org.springframework.http.client.HttpComponentsClientHttpRequestFactory;
015import org.springframework.web.client.RestTemplate;
016
017import com.google.common.cache.CacheBuilder;
018import com.google.common.cache.CacheLoader;
019import com.google.common.cache.LoadingCache;
020import com.google.gson.JsonElement;
021import com.google.gson.JsonObject;
022import com.google.gson.JsonParser;
023
024/**
025 * 
026 * Dynamically fetches OpenID Connect server configurations based on the issuer. Caches the server configurations.
027 * 
028 * @author jricher
029 *
030 */
031public class DynamicServerConfigurationService implements ServerConfigurationService {
032
033        private static Logger logger = LoggerFactory.getLogger(DynamicServerConfigurationService.class);
034        
035        // map of issuer -> server configuration, loaded dynamically from service discovery
036        private LoadingCache<String, ServerConfiguration> servers;
037        
038        public DynamicServerConfigurationService() {
039                // initialize the cache
040                servers = CacheBuilder.newBuilder().build(new OpenIDConnectServiceConfigurationFetcher());
041        }
042        
043        @Override
044        public ServerConfiguration getServerConfiguration(String issuer) {
045                try {
046                return servers.get(issuer);
047        } catch (ExecutionException e) {
048                logger.warn("Couldn't load configuration for " + issuer, e);
049                return null; 
050        }
051                
052        }
053
054        /**
055     * @author jricher
056     *
057     */
058    private class OpenIDConnectServiceConfigurationFetcher extends CacheLoader<String, ServerConfiguration> {
059        private HttpClient httpClient = new DefaultHttpClient();
060        private HttpComponentsClientHttpRequestFactory httpFactory = new HttpComponentsClientHttpRequestFactory(httpClient);
061                private JsonParser parser = new JsonParser();
062
063        @Override
064        public ServerConfiguration load(String issuer) throws Exception {
065                RestTemplate restTemplate = new RestTemplate(httpFactory);
066
067                // data holder
068                ServerConfiguration conf = new ServerConfiguration();
069                
070                // construct the well-known URI
071                String url = issuer + "/.well-known/openid-configuration";
072                
073                // fetch the value
074                String jsonString = restTemplate.getForObject(url, String.class);
075
076                JsonElement parsed = parser.parse(jsonString);
077                if (parsed.isJsonObject()) {
078                        
079                        JsonObject o = parsed.getAsJsonObject();
080                        
081                        // sanity checks
082                        if (!issuer.equals(o.get("issuer").getAsString())) {
083                                throw new IllegalStateException("Discovered issuers didn't match, expected " + issuer + " got " + o.get("issuer").getAsString());
084                        }
085                        
086                        conf.setIssuer(o.get("issuer").getAsString());
087                        conf.setAuthorizationEndpointUri(o.get("authorization_endpoint").getAsString());
088                        conf.setTokenEndpointUri(o.get("token_endpoint").getAsString());
089                        conf.setJwksUri(o.get("jwks_uri").getAsString());
090                        conf.setUserInfoUri(o.get("userinfo_endpoint").getAsString());
091                        conf.setRegistrationEndpointUri(o.get("registration_endpoint").getAsString());
092
093                        return conf;
094                } else {
095                        throw new IllegalStateException("Couldn't parse server discovery results for " + url); 
096                }
097                
098        }
099
100    }
