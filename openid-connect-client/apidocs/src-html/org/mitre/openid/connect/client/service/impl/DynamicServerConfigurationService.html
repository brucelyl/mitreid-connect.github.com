020package org.mitre.openid.connect.client.service.impl;
021
022import java.util.HashSet;
023import java.util.Set;
024import java.util.concurrent.ExecutionException;
025
026import org.apache.http.client.HttpClient;
027import org.apache.http.impl.client.HttpClientBuilder;
028import org.mitre.openid.connect.client.service.ServerConfigurationService;
029import org.mitre.openid.connect.config.ServerConfiguration;
030import org.slf4j.Logger;
031import org.slf4j.LoggerFactory;
032import org.springframework.http.client.HttpComponentsClientHttpRequestFactory;
033import org.springframework.security.authentication.AuthenticationServiceException;
034import org.springframework.web.client.RestTemplate;
035
036import com.google.common.cache.CacheBuilder;
037import com.google.common.cache.CacheLoader;
038import com.google.common.cache.LoadingCache;
039import com.google.common.util.concurrent.UncheckedExecutionException;
040import com.google.gson.JsonElement;
041import com.google.gson.JsonObject;
042import com.google.gson.JsonParser;
043
044import static org.mitre.util.JsonUtils.getAsBoolean;
045import static org.mitre.util.JsonUtils.getAsEncryptionMethodList;
046import static org.mitre.util.JsonUtils.getAsJweAlgorithmList;
047import static org.mitre.util.JsonUtils.getAsJwsAlgorithmList;
048import static org.mitre.util.JsonUtils.getAsString;
049import static org.mitre.util.JsonUtils.getAsStringList;
050
051/**
052 * 
053 * Dynamically fetches OpenID Connect server configurations based on the issuer. Caches the server configurations.
054 * 
055 * @author jricher
056 *
057 */
058public class DynamicServerConfigurationService implements ServerConfigurationService {
059
060        /**
061         * Logger for this class
062         */
063        private static final Logger logger = LoggerFactory.getLogger(DynamicServerConfigurationService.class);
064
065        // map of issuer -> server configuration, loaded dynamically from service discovery
066        private LoadingCache<String, ServerConfiguration> servers;
067
068        private Set<String> whitelist = new HashSet<>();
069        private Set<String> blacklist = new HashSet<>();
070
071        public DynamicServerConfigurationService() {
072                // initialize the cache
073                servers = CacheBuilder.newBuilder().build(new OpenIDConnectServiceConfigurationFetcher());
074        }
075
076        /**
077         * @return the whitelist
078         */
079        public Set<String> getWhitelist() {
080                return whitelist;
081        }
082
083        /**
084         * @param whitelist the whitelist to set
085         */
086        public void setWhitelist(Set<String> whitelist) {
087                this.whitelist = whitelist;
088        }
089
090        /**
091         * @return the blacklist
092         */
093        public Set<String> getBlacklist() {
094                return blacklist;
095        }
096
097        /**
098         * @param blacklist the blacklist to set
099         */
100        public void setBlacklist(Set<String> blacklist) {
101                this.blacklist = blacklist;
102        }
103
104        @Override
105        public ServerConfiguration getServerConfiguration(String issuer) {
106                try {
107
108                        if (!whitelist.isEmpty() && !whitelist.contains(issuer)) {
109                                throw new AuthenticationServiceException("Whitelist was nonempty, issuer was not in whitelist: " + issuer);
110                        }
111
112                        if (blacklist.contains(issuer)) {
113                                throw new AuthenticationServiceException("Issuer was in blacklist: " + issuer);
114                        }
115
116                        return servers.get(issuer);
117                } catch (UncheckedExecutionException | ExecutionException e) {
118                        logger.warn("Couldn't load configuration for " + issuer + ": " + e);
119                        return null;
120                }
121
122        }
123
124        /**
125         * @author jricher
126         *
127         */
128        private class OpenIDConnectServiceConfigurationFetcher extends CacheLoader<String, ServerConfiguration> {
129                private HttpClient httpClient = HttpClientBuilder.create()
130                                .useSystemProperties()
131                                .build();
132                private HttpComponentsClientHttpRequestFactory httpFactory = new HttpComponentsClientHttpRequestFactory(httpClient);
133                private JsonParser parser = new JsonParser();
134
135                @Override
136                public ServerConfiguration load(String issuer) throws Exception {
137                        RestTemplate restTemplate = new RestTemplate(httpFactory);
138
139                        // data holder
140                        ServerConfiguration conf = new ServerConfiguration();
141
142                        // construct the well-known URI
143                        String url = issuer + "/.well-known/openid-configuration";
144
145                        // fetch the value
146                        String jsonString = restTemplate.getForObject(url, String.class);
147
148                        JsonElement parsed = parser.parse(jsonString);
149                        if (parsed.isJsonObject()) {
150
151                                JsonObject o = parsed.getAsJsonObject();
152
153                                // sanity checks
154                                if (!o.has("issuer")) {
155                                        throw new IllegalStateException("Returned object did not have an 'issuer' field");
156                                }
157
158                                if (!issuer.equals(o.get("issuer").getAsString())) {
159                                        logger.info("Issuer used for discover was " + issuer + " but final issuer is " + o.get("issuer").getAsString());
160                                }
161
162                                conf.setIssuer(o.get("issuer").getAsString());
163
164
165                                conf.setAuthorizationEndpointUri(getAsString(o, "authorization_endpoint"));
166                                conf.setTokenEndpointUri(getAsString(o, "token_endpoint"));
167                                conf.setJwksUri(getAsString(o, "jwks_uri"));
168                                conf.setUserInfoUri(getAsString(o, "userinfo_endpoint"));
169                                conf.setRegistrationEndpointUri(getAsString(o, "registration_endpoint"));
170                                conf.setIntrospectionEndpointUri(getAsString(o, "introspection_endpoint"));
171                                conf.setAcrValuesSupported(getAsStringList(o, "acr_values_supported"));
172                                conf.setCheckSessionIframe(getAsString(o, "check_session_iframe"));
173                                conf.setClaimsLocalesSupported(getAsStringList(o, "claims_locales_supported"));
174                                conf.setClaimsParameterSupported(getAsBoolean(o, "claims_parameter_supported"));
175                                conf.setClaimsSupported(getAsStringList(o, "claims_supported"));
176                                conf.setDisplayValuesSupported(getAsStringList(o, "display_values_supported"));
177                                conf.setEndSessionEndpoint(getAsString(o, "end_session_endpoint"));
178                                conf.setGrantTypesSupported(getAsStringList(o, "grant_types_supported"));
179                                conf.setIdTokenSigningAlgValuesSupported(getAsJwsAlgorithmList(o, "id_token_signing_alg_values_supported"));
180                                conf.setIdTokenEncryptionAlgValuesSupported(getAsJweAlgorithmList(o, "id_token_encryption_alg_values_supported"));
181                                conf.setIdTokenEncryptionEncValuesSupported(getAsEncryptionMethodList(o, "id_token_encryption_enc_values_supported"));
182                                conf.setOpPolicyUri(getAsString(o, "op_policy_uri"));
183                                conf.setOpTosUri(getAsString(o, "op_tos_uri"));
184                                conf.setRequestObjectEncryptionAlgValuesSupported(getAsJweAlgorithmList(o, "request_object_encryption_alg_values_supported"));
185                                conf.setRequestObjectEncryptionEncValuesSupported(getAsEncryptionMethodList(o, "request_object_encryption_enc_values_supported"));
186                                conf.setRequestObjectSigningAlgValuesSupported(getAsJwsAlgorithmList(o, "request_object_signing_alg_values_supported"));
187                                conf.setRequestParameterSupported(getAsBoolean(o, "request_parameter_supported"));
188                                conf.setRequestUriParameterSupported(getAsBoolean(o, "request_uri_parameter_supported"));
189                                conf.setResponseTypesSupported(getAsStringList(o, "response_types_supported"));
190                                conf.setScopesSupported(getAsStringList(o, "scopes_supported"));
191                                conf.setSubjectTypesSupported(getAsStringList(o, "subject_types_supported"));
192                                conf.setServiceDocumentation(getAsString(o, "service_documentation"));
193                                conf.setTokenEndpointAuthMethodsSupported(getAsStringList(o, "token_endpoint_auth_methods"));
194                                conf.setTokenEndpointAuthSigningAlgValuesSupported(getAsJwsAlgorithmList(o, "token_endpoint_auth_signing_alg_values_supported"));
195                                conf.setUiLocalesSupported(getAsStringList(o, "ui_locales_supported"));
196                                conf.setUserinfoEncryptionAlgValuesSupported(getAsJweAlgorithmList(o, "userinfo_encryption_alg_values_supported"));
197                                conf.setUserinfoEncryptionEncValuesSupported(getAsEncryptionMethodList(o, "userinfo_encryption_enc_values_supported"));
198                                conf.setUserinfoSigningAlgValuesSupported(getAsJwsAlgorithmList(o, "userinfo_signing_alg_values_supported"));
199
200                                return conf;
201                        } else {
202                                throw new IllegalStateException("Couldn't parse server discovery results for " + url);
203                        }
204
205                }
206
207        }
