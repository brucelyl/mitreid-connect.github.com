004package org.mitre.openid.connect.client;
005
006import org.springframework.security.core.GrantedAuthority;
007
008import com.google.common.base.Strings;
009
010/**
011 * 
012 * Simple authority representing a user at an issuer.
013 * 
014 * @author jricher
015 *
016 */
017public class SubjectIssuerGrantedAuthority implements GrantedAuthority {
018
019    private static final long serialVersionUID = 5584978219226664794L;
020    
021        private final String subject;
022        private final String issuer;
023        
024        /**
025         * @param subject
026         * @param issuer
027         */
028    public SubjectIssuerGrantedAuthority(String subject, String issuer) {
029        if (Strings.isNullOrEmpty(subject) || Strings.isNullOrEmpty(issuer)) {
030                throw new IllegalArgumentException("Neither subject nor issuer may be null or empty");
031        }
032        this.subject = subject;
033        this.issuer = issuer;
034    }
035
036        /**
037         * Returns a string formed by concatenating the subject with the issuer, separated by _ and prepended with OIDC_
038         * 
039         * For example, the user "bob" from issuer "http://id.example.com/" would return the authority string of:
040         * 
041         * OIDC_bob_http://id.example.com/
042         */
043        @Override
044        public String getAuthority() {
045                return "OIDC_" + subject + "_" + issuer;
046        }
047        
048        /**
049         * @return the subject
050         */
051    public String getSubject() {
052            return subject;
053    }
054
055    /**
056         * @return the issuer
057         */
058    public String getIssuer() {
059            return issuer;
060    }
061
062        /* (non-Javadoc)
063         * @see java.lang.Object#hashCode()
064         */
065    @Override
066    public int hashCode() {
067            final int prime = 31;
068            int result = 1;
069            result = prime * result + ((issuer == null) ? 0 : issuer.hashCode());
070            result = prime * result + ((subject == null) ? 0 : subject.hashCode());
071            return result;
072    }
073
074        /* (non-Javadoc)
075         * @see java.lang.Object#equals(java.lang.Object)
076         */
077    @Override
078    public boolean equals(Object obj) {
079            if (this == obj) {
080                    return true;
081            }
082            if (obj == null) {
083                    return false;
084            }
085            if (!(obj instanceof SubjectIssuerGrantedAuthority)) {
086                    return false;
087            }
088            SubjectIssuerGrantedAuthority other = (SubjectIssuerGrantedAuthority) obj;
089            if (issuer == null) {
090                    if (other.issuer != null) {
091                            return false;
092                    }
093            } else if (!issuer.equals(other.issuer)) {
094                    return false;
095            }
096            if (subject == null) {
097                    if (other.subject != null) {
098                            return false;
099                    }
100            } else if (!subject.equals(other.subject)) {
101                    return false;
102            }
103            return true;
104    }
105    
106    @Override
107    public String toString() {
108        return getAuthority();
109    }
110}
