144        /*
145         * This is the main entry point for the filter.
146         * 
147         * (non-Javadoc)
148         * 
149         * @see org.springframework.security.web.authentication.
150         * AbstractAuthenticationProcessingFilter
151         * #attemptAuthentication(javax.servlet.http.HttpServletRequest,
152         * javax.servlet.http.HttpServletResponse)
153         */
154        @Override
155        public Authentication attemptAuthentication(HttpServletRequest request, HttpServletResponse response) throws AuthenticationException, IOException, ServletException {
156
157                if (!Strings.isNullOrEmpty(request.getParameter("error"))) {
158
159                        // there's an error coming back from the server, need to handle this
160                        handleError(request, response);
161                        return null; // no auth, response is sent to display page or something
162
163                } else if (!Strings.isNullOrEmpty(request.getParameter("code"))) {
164
165                        // we got back the code, need to process this to get our tokens
166                        Authentication auth = handleAuthorizationCodeResponse(request, response);
167                        return auth;
168
169                } else {
170
171                        // not an error, not a code, must be an initial login of some type
172                        handleAuthorizationRequest(request, response);
173
174                        return null; // no auth, response redirected to the server's Auth Endpoint (or possibly to the account chooser)
175                }
176
177        }

179        /**
180         * Initiate an Authorization request
181         * 
182         * @param request
183         *            The request from which to extract parameters and perform the
184         *            authentication
185         * @param response
186         * @throws IOException
187         *             If an input or output exception occurs
188         */
189        protected void handleAuthorizationRequest(HttpServletRequest request, HttpServletResponse response) throws IOException {
190
191                HttpSession session = request.getSession();
192
193                IssuerServiceResponse issResp = issuerService.getIssuer(request);
194
195                if (issResp == null) {
196                        logger.error("Null issuer response returned from service.");
197                        throw new AuthenticationServiceException("No issuer found.");
198                }
199
200                if (issResp.shouldRedirect()) {
201                        response.sendRedirect(issResp.getRedirectUrl());
202                } else {
203                        String issuer = issResp.getIssuer();
204
205                        if (!Strings.isNullOrEmpty(issResp.getTargetLinkUri())) {
206                                // there's a target URL in the response, we should save this so we can forward to it later
207                                session.setAttribute(TARGET_SESSION_VARIABLE, issResp.getTargetLinkUri());
208                        }
209
210                        if (Strings.isNullOrEmpty(issuer)) {
211                                logger.error("No issuer found: " + issuer);
212                                throw new AuthenticationServiceException("No issuer found: " + issuer);
213                        }
214
215                        session.setAttribute(ISSUER_SESSION_VARIABLE, issuer);
216
217                        ServerConfiguration serverConfig = servers.getServerConfiguration(issuer);
218                        if (serverConfig == null) {
219                                logger.error("No server configuration found for issuer: " + issuer);
220                                throw new AuthenticationServiceException("No server configuration found for issuer: " + issuer);
221                        }
222
223
224                        RegisteredClient clientConfig = clients.getClientConfiguration(serverConfig);
225                        if (clientConfig == null) {
226                                logger.error("No client configuration found for issuer: " + issuer);
227                                throw new AuthenticationServiceException("No client configuration found for issuer: " + issuer);
228                        }
229
230                        String redirectUri = null;
231                        if (clientConfig.getRegisteredRedirectUri() != null && clientConfig.getRegisteredRedirectUri().size() == 1) {
232                                // if there's a redirect uri configured (and only one), use that
233                                redirectUri = clientConfig.getRegisteredRedirectUri().toArray(new String[] {})[0];
234                        } else {
235                                // otherwise our redirect URI is this current URL, with no query parameters
236                                redirectUri = request.getRequestURL().toString();
237                        }
238                        session.setAttribute(REDIRECT_URI_SESION_VARIABLE, redirectUri);
239
240                        // this value comes back in the id token and is checked there
241                        String nonce = createNonce(session);
242
243                        // this value comes back in the auth code response
244                        String state = createState(session);
245
246                        Map<String, String> options = authOptions.getOptions(serverConfig, clientConfig, request);
247
248                        String authRequest = authRequestBuilder.buildAuthRequestUrl(serverConfig, clientConfig, redirectUri, nonce, state, options);
249
250                        logger.debug("Auth Request:  " + authRequest);
251
252                        response.sendRedirect(authRequest);
253                }
254        }

256        /**
257         * @param request
258         *            The request from which to extract parameters and perform the
259         *            authentication
260         * @return The authenticated user token, or null if authentication is
261         *         incomplete.
262         */
263        protected Authentication handleAuthorizationCodeResponse(HttpServletRequest request, HttpServletResponse response) {
264
265                String authorizationCode = request.getParameter("code");
266
267                HttpSession session = request.getSession();
268
269                // check for state, if it doesn't match we bail early
270                String storedState = getStoredState(session);
271                if (!Strings.isNullOrEmpty(storedState)) {
272                        String state = request.getParameter("state");
273                        if (!storedState.equals(state)) {
274                                throw new AuthenticationServiceException("State parameter mismatch on return. Expected " + storedState + " got " + state);
275                        }
276                }
277
278                // look up the issuer that we set out to talk to
279                String issuer = getStoredSessionString(session, ISSUER_SESSION_VARIABLE);
280
281                // pull the configurations based on that issuer
282                ServerConfiguration serverConfig = servers.getServerConfiguration(issuer);
283                final RegisteredClient clientConfig = clients.getClientConfiguration(serverConfig);
284
285                MultiValueMap<String, String> form = new LinkedMultiValueMap<String, String>();
286                form.add("grant_type", "authorization_code");
287                form.add("code", authorizationCode);
288
289                String redirectUri = getStoredSessionString(session, REDIRECT_URI_SESION_VARIABLE);
290                if (redirectUri != null) {
291                        form.add("redirect_uri", redirectUri);
292                }
293
294                // Handle Token Endpoint interaction
295                HttpClient httpClient = new SystemDefaultHttpClient();
296
297                httpClient.getParams().setParameter("http.socket.timeout", new Integer(httpSocketTimeout));
298
299                HttpComponentsClientHttpRequestFactory factory = new HttpComponentsClientHttpRequestFactory(httpClient);
300
301                RestTemplate restTemplate;
302
303                if (SECRET_BASIC.equals(clientConfig.getTokenEndpointAuthMethod())){
304                        // use BASIC auth if configured to do so
305                        restTemplate = new RestTemplate(factory) {
306
307                                @Override
308                                protected ClientHttpRequest createRequest(URI url, HttpMethod method) throws IOException {
309                                        ClientHttpRequest httpRequest = super.createRequest(url, method);
310                                        httpRequest.getHeaders().add("Authorization",
311                                                        String.format("Basic %s", Base64.encode(String.format("%s:%s", clientConfig.getClientId(), clientConfig.getClientSecret())) ));
312
313
314
315                                        return httpRequest;
316                                }
317                        };
318                } else {
319                        // we're not doing basic auth, figure out what other flavor we have
320                        restTemplate = new RestTemplate(factory);
321
322                        if (SECRET_JWT.equals(clientConfig.getTokenEndpointAuthMethod()) || PRIVATE_KEY.equals(clientConfig.getTokenEndpointAuthMethod())) {
323                                // do a symmetric secret signed JWT for auth
324
325
326                                JwtSigningAndValidationService signer = null;
327                                JWSAlgorithm alg = clientConfig.getTokenEndpointAuthSigningAlg();
328
329                                if (SECRET_JWT.equals(clientConfig.getTokenEndpointAuthMethod()) &&
330                                                (alg.equals(JWSAlgorithm.HS256)
331                                                                || alg.equals(JWSAlgorithm.HS384)
332                                                                || alg.equals(JWSAlgorithm.HS512))) {
333
334                                        // generate one based on client secret
335                                        signer = symmetricCacheService.getSymmetricValidtor(clientConfig.getClient());
336
337                                } else if (PRIVATE_KEY.equals(clientConfig.getTokenEndpointAuthMethod())) {
338
339                                        // needs to be wired in to the bean
340                                        signer = authenticationSignerService;
341                                }
342
343                                if (signer == null) {
344                                        throw new AuthenticationServiceException("Couldn't find required signer service for use with private key auth.");
345                                }
346
347                                JWTClaimsSet claimsSet = new JWTClaimsSet();
348
349                                claimsSet.setIssuer(clientConfig.getClientId());
350                                claimsSet.setSubject(clientConfig.getClientId());
351                                claimsSet.setAudience(Lists.newArrayList(serverConfig.getTokenEndpointUri()));
352
353                                // TODO: make this configurable
354                                Date exp = new Date(System.currentTimeMillis() + (60 * 1000)); // auth good for 60 seconds
355                                claimsSet.setExpirationTime(exp);
356
357                                Date now = new Date(System.currentTimeMillis());
358                                claimsSet.setIssueTime(now);
359                                claimsSet.setNotBeforeTime(now);
360
361                                SignedJWT jwt = new SignedJWT(new JWSHeader(alg), claimsSet);
362
363                                signer.signJwt(jwt, alg);
364
365                                form.add("client_assertion_type", "urn:ietf:params:oauth:client-assertion-type:jwt-bearer");
366                                form.add("client_assertion", jwt.serialize());
367                        } else {
368                                //Alternatively use form based auth
369                                form.add("client_id", clientConfig.getClientId());
370                                form.add("client_secret", clientConfig.getClientSecret());
371                        }
372
373                }
374
375                logger.debug("tokenEndpointURI = " + serverConfig.getTokenEndpointUri());
376                logger.debug("form = " + form);
377
378                String jsonString = null;
379
380                try {
381                        jsonString = restTemplate.postForObject(serverConfig.getTokenEndpointUri(), form, String.class);
382                } catch (HttpClientErrorException httpClientErrorException) {
383
384                        // Handle error
385
386                        logger.error("Token Endpoint error response:  "
387                                        + httpClientErrorException.getStatusText() + " : "
388                                        + httpClientErrorException.getMessage());
389
390                        throw new AuthenticationServiceException("Unable to obtain Access Token: " + httpClientErrorException.getMessage());
391                }
392
393                logger.debug("from TokenEndpoint jsonString = " + jsonString);
394
395                JsonElement jsonRoot = new JsonParser().parse(jsonString);
396                if (!jsonRoot.isJsonObject()) {
397                        throw new AuthenticationServiceException("Token Endpoint did not return a JSON object: " + jsonRoot);
398                }
399
400                JsonObject tokenResponse = jsonRoot.getAsJsonObject();
401
402                if (tokenResponse.get("error") != null) {
403
404                        // Handle error
405
406                        String error = tokenResponse.get("error").getAsString();
407
408                        logger.error("Token Endpoint returned: " + error);
409
410                        throw new AuthenticationServiceException("Unable to obtain Access Token.  Token Endpoint returned: " + error);
411
412                } else {
413
414                        // Extract the id_token to insert into the
415                        // OIDCAuthenticationToken
416
417                        // get out all the token strings
418                        String accessTokenValue = null;
419                        String idTokenValue = null;
420                        String refreshTokenValue = null;
421
422                        if (tokenResponse.has("access_token")) {
423                                accessTokenValue = tokenResponse.get("access_token").getAsString();
424                        } else {
425                                throw new AuthenticationServiceException("Token Endpoint did not return an access_token: " + jsonString);
426                        }
427
428                        if (tokenResponse.has("id_token")) {
429                                idTokenValue = tokenResponse.get("id_token").getAsString();
430                        } else {
431                                logger.error("Token Endpoint did not return an id_token");
432                                throw new AuthenticationServiceException("Token Endpoint did not return an id_token");
433                        }
434
435                        if (tokenResponse.has("refresh_token")) {
436                                refreshTokenValue = tokenResponse.get("refresh_token").getAsString();
437                        }
438
439                        try {
440                                SignedJWT idToken = SignedJWT.parse(idTokenValue);
441
442                                // validate our ID Token over a number of tests
443                                ReadOnlyJWTClaimsSet idClaims = idToken.getJWTClaimsSet();
444
445                                // check the signature
446                                JwtSigningAndValidationService jwtValidator = null;
447
448                                JWSAlgorithm alg = idToken.getHeader().getAlgorithm();
449                                if (alg.equals(JWSAlgorithm.HS256)
450                                        || alg.equals(JWSAlgorithm.HS384)
451                                        || alg.equals(JWSAlgorithm.HS512)) {
452                                        
453                                        // generate one based on client secret
454                                        jwtValidator = symmetricCacheService.getSymmetricValidtor(clientConfig.getClient());
455                                } else {
456                                        // otherwise load from the server's public key
457                                        jwtValidator = validationServices.getValidator(serverConfig.getJwksUri());
458                                }
459                                
460                                if (jwtValidator != null) {
461                                        if(!jwtValidator.validateSignature(idToken)) {
462                                                throw new AuthenticationServiceException("Signature validation failed");
463                                        }
464                                } else {
465                                        logger.error("No validation service found. Skipping signature validation");
466                                        throw new AuthenticationServiceException("Unable to find an appropriate signature validator for ID Token.");
467                                }
468
469                                // check the issuer
470                                if (idClaims.getIssuer() == null) {
471                                        throw new AuthenticationServiceException("Id Token Issuer is null");
472                                } else if (!idClaims.getIssuer().equals(serverConfig.getIssuer())){
473                                        throw new AuthenticationServiceException("Issuers do not match, expected " + serverConfig.getIssuer() + " got " + idClaims.getIssuer());
474                                }
475
476                                // check expiration
477                                if (idClaims.getExpirationTime() == null) {
478                                        throw new AuthenticationServiceException("Id Token does not have required expiration claim");
479                                } else {
480                                        // it's not null, see if it's expired
481                                        Date now = new Date(System.currentTimeMillis() - (timeSkewAllowance * 1000));
482                                        if (now.after(idClaims.getExpirationTime())) {
483                                                throw new AuthenticationServiceException("Id Token is expired: " + idClaims.getExpirationTime());
484                                        }
485                                }
486
487                                // check not before
488                                if (idClaims.getNotBeforeTime() != null) {
489                                        Date now = new Date(System.currentTimeMillis() + (timeSkewAllowance * 1000));
490                                        if (now.before(idClaims.getNotBeforeTime())){
491                                                throw new AuthenticationServiceException("Id Token not valid untill: " + idClaims.getNotBeforeTime());
492                                        }
493                                }
494
495                                // check issued at
496                                if (idClaims.getIssueTime() == null) {
497                                        throw new AuthenticationServiceException("Id Token does not have required issued-at claim");
498                                } else {
499                                        // since it's not null, see if it was issued in the future
500                                        Date now = new Date(System.currentTimeMillis() + (timeSkewAllowance * 1000));
501                                        if (now.before(idClaims.getIssueTime())) {
502                                                throw new AuthenticationServiceException("Id Token was issued in the future: " + idClaims.getIssueTime());
503                                        }
504                                }
505
506                                // check audience
507                                if (idClaims.getAudience() == null) {
508                                        throw new AuthenticationServiceException("Id token audience is null");
509                                } else if (!idClaims.getAudience().contains(clientConfig.getClientId())) {
510                                        throw new AuthenticationServiceException("Audience does not match, expected " + clientConfig.getClientId() + " got " + idClaims.getAudience());
511                                }
512
513                                // compare the nonce to our stored claim
514                                String nonce = idClaims.getStringClaim("nonce");
515                                if (Strings.isNullOrEmpty(nonce)) {
516
517                                        logger.error("ID token did not contain a nonce claim.");
518
519                                        throw new AuthenticationServiceException("ID token did not contain a nonce claim.");
520                                }
521
522                                String storedNonce = getStoredNonce(session);
523                                if (!nonce.equals(storedNonce)) {
524                                        logger.error("Possible replay attack detected! The comparison of the nonce in the returned "
525                                                        + "ID Token to the session " + NONCE_SESSION_VARIABLE + " failed. Expected " + storedNonce + " got " + nonce + ".");
526
527                                        throw new AuthenticationServiceException(
528                                                        "Possible replay attack detected! The comparison of the nonce in the returned "
529                                                                        + "ID Token to the session " + NONCE_SESSION_VARIABLE + " failed. Expected " + storedNonce + " got " + nonce + ".");
530                                }
531
532                                // pull the subject (user id) out as a claim on the id_token
533
534                                String userId = idClaims.getSubject();
535
536                                // construct an OIDCAuthenticationToken and return a Authentication object w/the userId and the idToken
537
538                                OIDCAuthenticationToken token = new OIDCAuthenticationToken(userId, idClaims.getIssuer(), serverConfig, idTokenValue, accessTokenValue, refreshTokenValue);
539
540                                Authentication authentication = this.getAuthenticationManager().authenticate(token);
541
542                                return authentication;
543                        } catch (ParseException e) {
544                                throw new AuthenticationServiceException("Couldn't parse idToken: ", e);
545                        }
546
547
548
549                }
550        }

572        /**
573         * Get the named stored session variable as a string. Return null if not found or not a string.
574         * @param session
575         * @param key
576         * @return
577         */
578        private static String getStoredSessionString(HttpSession session, String key) {
579                Object o = session.getAttribute(key);
580                if (o != null && o instanceof String) {
581                        return o.toString();
582                } else {
583                        return null;
584                }

639        /**
640         * Handle a successful authentication event. If the issuer service sets
641         * a target URL, we'll go to that. Otherwise we'll let the superclass handle
642         * it for us with the configured behavior.
643         */
644        protected class TargetLinkURIAuthenticationSuccessHandler implements AuthenticationSuccessHandler {
645
646                private AuthenticationSuccessHandler passthrough;
647
648                @Override
649                public void onAuthenticationSuccess(HttpServletRequest request,
650                                HttpServletResponse response, Authentication authentication)
651                                                throws IOException, ServletException {
652
653                        HttpSession session = request.getSession();
654
655                        // check to see if we've got a target
656                        String target = getStoredSessionString(session, TARGET_SESSION_VARIABLE);
657
658                        if (!Strings.isNullOrEmpty(target)) {
659                                session.removeAttribute(TARGET_SESSION_VARIABLE);
660
661                                target = deepLinkFilter.filter(target);
662
663                                response.sendRedirect(target);
664                        } else {
665                                // if the target was blank, use the default behavior here
666                                passthrough.onAuthenticationSuccess(request, response, authentication);
667                        }
668
669                }
670
671        }
