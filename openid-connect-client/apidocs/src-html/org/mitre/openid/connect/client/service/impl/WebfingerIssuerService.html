050/**
051 * Use Webfinger to discover the appropriate issuer for a user-given input string.
052 * @author jricher
053 *
054 */
055public class WebfingerIssuerService implements IssuerService {
056
057        private static Logger logger = LoggerFactory.getLogger(WebfingerIssuerService.class);
058
059        // map of user input -> issuer, loaded dynamically from webfinger discover
060        private LoadingCache<UriComponents, String> issuers;
061
062        private Set<String> whitelist = new HashSet<String>();
063        private Set<String> blacklist = new HashSet<String>();
064
065        /**
066         * Name of the incoming parameter to check for discovery purposes.
067         */
068        private String parameterName = "identifier";
069
070        /**
071         * URL of the page to forward to if no identifier is given.
072         */
073        private String loginPageUrl;
074
075        public WebfingerIssuerService() {
076                issuers = CacheBuilder.newBuilder().build(new WebfingerIssuerFetcher());
077        }
078
079        /* (non-Javadoc)
080         * @see org.mitre.openid.connect.client.service.IssuerService#getIssuer(javax.servlet.http.HttpServletRequest)
081         */
082        @Override
083        public IssuerServiceResponse getIssuer(HttpServletRequest request) {
084
085                String identifier = request.getParameter(parameterName);
086                if (!Strings.isNullOrEmpty(identifier)) {
087                        try {
088                                String issuer = issuers.get(WebfingerURLNormalizer.normalizeResource(identifier));
089                                if (!whitelist.isEmpty() && !whitelist.contains(issuer)) {
090                                        throw new AuthenticationServiceException("Whitelist was nonempty, issuer was not in whitelist: " + issuer);
091                                }
092
093                                if (blacklist.contains(issuer)) {
094                                        throw new AuthenticationServiceException("Issuer was in blacklist: " + issuer);
095                                }
096
097                                return new IssuerServiceResponse(issuer, null, null);
098                        } catch (ExecutionException e) {
099                                logger.warn("Issue fetching issuer for user input: " + identifier, e);
100                                return null;
101                        }
102
103                } else {
104                        logger.warn("No user input given, directing to login page: " + loginPageUrl);
105                        return new IssuerServiceResponse(loginPageUrl);
106                }
107        }
108

110         * @return the parameterName

113                return parameterName;
114        }

135                this.loginPageUrl = loginPageUrl;

163                this.blacklist = blacklist;

165
166        /**
167         * @author jricher
168         *
169         */
170        private class WebfingerIssuerFetcher extends CacheLoader<UriComponents, String> {
171                private HttpClient httpClient = new DefaultHttpClient();
172                private HttpComponentsClientHttpRequestFactory httpFactory = new HttpComponentsClientHttpRequestFactory(httpClient);
173                private JsonParser parser = new JsonParser();
174
175                @Override
176                public String load(UriComponents key) throws Exception {
177
178                        RestTemplate restTemplate = new RestTemplate(httpFactory);
179                        // construct the URL to go to
180
181                        // preserving http scheme is strictly for demo system use only.
182                        String scheme = key.getScheme();
183                        if (!Strings.isNullOrEmpty(scheme) && scheme.equals("http")) {
184                                scheme = "http://"; // add on colon and slashes.
185                                logger.warn("Webfinger endpoint MUST use the https URI scheme.");
186                        } else {
187                                scheme = "https://";
188                        }
189
190                        // do a webfinger lookup
191                        URIBuilder builder = new URIBuilder(scheme
192                                        + key.getHost()
193                                        + (key.getPort() >= 0 ? ":" + key.getPort() : "")
194                                        + Strings.nullToEmpty(key.getPath())
195                                        + "/.well-known/webfinger"
196                                        + (Strings.isNullOrEmpty(key.getQuery()) ? "" : "?" + key.getQuery())
197                                        );
198                        builder.addParameter("resource", key.toString());
199                        builder.addParameter("rel", "http://openid.net/specs/connect/1.0/issuer");
200
201                        // do the fetch
202                        logger.info("Loading: " + builder.toString());
203                        String webfingerResponse = restTemplate.getForObject(builder.build(), String.class);
204
205                        // TODO: catch and handle HTTP errors
206
207                        JsonElement json = parser.parse(webfingerResponse);
208
209                        // TODO: catch and handle JSON errors
210
211                        if (json != null && json.isJsonObject()) {
212                                // find the issuer
213                                JsonArray links = json.getAsJsonObject().get("links").getAsJsonArray();
214                                for (JsonElement link : links) {
215                                        if (link.isJsonObject()) {
216                                                JsonObject linkObj = link.getAsJsonObject();
217                                                if (linkObj.has("href")
218                                                                && linkObj.has("rel")
219                                                                && linkObj.get("rel").getAsString().equals("http://openid.net/specs/connect/1.0/issuer")) {
220
221                                                        // we found the issuer, return it
222                                                        return linkObj.get("href").getAsString();
223                                                }
224                                        }
225                                }
226                        }
227
228                        // we couldn't find it
229
230                        if (key.getScheme().equals("http") || key.getScheme().equals("https")) {
231                                // if it looks like HTTP then punt and return the input
232                                logger.warn("Returning normalized input string as issuer, hoping for the best: " + key.toString());
233                                return key.toString();
234                        } else {
235                                // if it's not HTTP, give up
236                                logger.warn("Couldn't find issuer: " + key.toString());
237                                return null;
238                        }
239
240                }
241
242        }
