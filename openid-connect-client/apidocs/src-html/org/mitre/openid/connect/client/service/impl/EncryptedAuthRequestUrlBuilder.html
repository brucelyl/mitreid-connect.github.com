020package org.mitre.openid.connect.client.service.impl;
021
022import java.net.URISyntaxException;
023import java.util.Map;
024import java.util.Map.Entry;
025
026import org.apache.http.client.utils.URIBuilder;
027import org.mitre.jwt.encryption.service.JwtEncryptionAndDecryptionService;
028import org.mitre.jwt.signer.service.impl.JWKSetCacheService;
029import org.mitre.oauth2.model.RegisteredClient;
030import org.mitre.openid.connect.client.service.AuthRequestUrlBuilder;
031import org.mitre.openid.connect.config.ServerConfiguration;
032import org.springframework.security.authentication.AuthenticationServiceException;
033
034import com.google.common.base.Joiner;
035import com.nimbusds.jose.EncryptionMethod;
036import com.nimbusds.jose.JWEAlgorithm;
037import com.nimbusds.jose.JWEHeader;
038import com.nimbusds.jwt.EncryptedJWT;
039import com.nimbusds.jwt.JWTClaimsSet;
040
041/**
042 * @author jricher
043 *
044 */
045public class EncryptedAuthRequestUrlBuilder implements AuthRequestUrlBuilder {
046
047        private JWKSetCacheService encrypterService;
048
049        private JWEAlgorithm alg;
050        private EncryptionMethod enc;
051
052
053        /* (non-Javadoc)
054         * @see org.mitre.openid.connect.client.service.AuthRequestUrlBuilder#buildAuthRequestUrl(org.mitre.openid.connect.config.ServerConfiguration, org.mitre.oauth2.model.RegisteredClient, java.lang.String, java.lang.String, java.lang.String, java.util.Map)
055         */
056        @Override
057        public String buildAuthRequestUrl(ServerConfiguration serverConfig, RegisteredClient clientConfig, String redirectUri, String nonce, String state, Map<String, String> options) {
058
059                // create our signed JWT for the request object
060                JWTClaimsSet claims = new JWTClaimsSet();
061
062                //set parameters to JwtClaims
063                claims.setClaim("response_type", "code");
064                claims.setClaim("client_id", clientConfig.getClientId());
065                claims.setClaim("scope", Joiner.on(" ").join(clientConfig.getScope()));
066
067                // build our redirect URI
068                claims.setClaim("redirect_uri", redirectUri);
069
070                // this comes back in the id token
071                claims.setClaim("nonce", nonce);
072
073                // this comes back in the auth request return
074                claims.setClaim("state", state);
075
076                // Optional parameters
077                for (Entry<String, String> option : options.entrySet()) {
078                        claims.setClaim(option.getKey(), option.getValue());
079                }
080
081                EncryptedJWT jwt = new EncryptedJWT(new JWEHeader(alg, enc), claims);
082
083                JwtEncryptionAndDecryptionService encryptor = encrypterService.getEncrypter(serverConfig.getJwksUri());
084
085                encryptor.encryptJwt(jwt);
086
087                try {
088                        URIBuilder uriBuilder = new URIBuilder(serverConfig.getAuthorizationEndpointUri());
089                        uriBuilder.addParameter("request", jwt.serialize());
090
091                        // build out the URI
092                        return uriBuilder.build().toString();
093                } catch (URISyntaxException e) {
094                        throw new AuthenticationServiceException("Malformed Authorization Endpoint Uri", e);
095                }
096        }
097
098        /**
099         * @return the encrypterService
100         */
101        public JWKSetCacheService getEncrypterService() {
102                return encrypterService;
103        }
104
105        /**
106         * @param encrypterService the encrypterService to set
107         */
108        public void setEncrypterService(JWKSetCacheService encrypterService) {
109                this.encrypterService = encrypterService;
110        }
111
112        /**
113         * @return the alg
114         */
115        public JWEAlgorithm getAlg() {
116                return alg;
117        }
118
119        /**
120         * @param alg the alg to set
121         */
122        public void setAlg(JWEAlgorithm alg) {
123                this.alg = alg;
124        }
125
126        /**
127         * @return the enc
128         */
129        public EncryptionMethod getEnc() {
130                return enc;
131        }
132
133        /**
134         * @param enc the enc to set
135         */
136        public void setEnc(EncryptionMethod enc) {
137                this.enc = enc;
138        }
139
140}
