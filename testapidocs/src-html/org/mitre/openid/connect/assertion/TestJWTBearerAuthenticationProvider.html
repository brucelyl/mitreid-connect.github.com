132        @Test
133        public void should_throw_AuthenticationServiceException_for_SignedJWT_when_signing_algorithms_do_not_match() {
134                when(client.getTokenEndpointAuthSigningAlg()).thenReturn(JWSAlgorithm.RS256);
135                SignedJWT signedJWT = createSignedJWT(JWSAlgorithm.ES384);
136                when(token.getJwt()).thenReturn(signedJWT);
137
138                Throwable thrown = authenticateAndReturnThrownException();
139
140                assertThat(thrown, instanceOf(AuthenticationServiceException.class));
141                assertThat(thrown.getMessage(), is("Client's registered token endpoint signing algorithm (RS256) does not match token's actual algorithm (ES384)"));
142        }

163                List<JWSAlgorithm> invalidAlgorithms = Arrays.asList(JWSAlgorithm.HS256, JWSAlgorithm.HS384, JWSAlgorithm.HS512);
164
165                for (JWSAlgorithm algorithm : invalidAlgorithms) {
166                        SignedJWT signedJWT = createSignedJWT(algorithm);
167                        when(token.getJwt()).thenReturn(signedJWT);

357        private void mockPlainJWTAuthAttempt() {
358                PlainJWT plainJWT = new PlainJWT(createJwtClaimsSet());
359                when(token.getJwt()).thenReturn(plainJWT);
360        }
361
362        private void mockEncryptedJWTAuthAttempt() {
363                JWEHeader jweHeader = new JWEHeader.Builder(JWEAlgorithm.A128GCMKW, EncryptionMethod.A256GCM).build();
364                EncryptedJWT encryptedJWT = new EncryptedJWT(jweHeader, createJwtClaimsSet());
365                when(token.getJwt()).thenReturn(encryptedJWT);
366        }
367
368        private SignedJWT mockSignedJWTAuthAttempt() {
369                return mockSignedJWTAuthAttempt(createJwtClaimsSet());
370        }
371
372        private SignedJWT mockSignedJWTAuthAttempt(JWTClaimsSet jwtClaimsSet) {
373                SignedJWT signedJWT = createSignedJWT(JWSAlgorithm.RS256, jwtClaimsSet);
374                when(token.getJwt()).thenReturn(signedJWT);
375                when(client.getTokenEndpointAuthMethod()).thenReturn(AuthMethod.PRIVATE_KEY);
376                when(client.getTokenEndpointAuthSigningAlg()).thenReturn(JWSAlgorithm.RS256);
377                return signedJWT;
378        }
379
380        private Throwable authenticateAndReturnThrownException() {
381                try {
382                        jwtBearerAuthenticationProvider.authenticate(token);
383                } catch (Throwable throwable) {
384                        return throwable;
385                }
386                throw new AssertionError("No exception thrown when expected");
387        }
388
389        private SignedJWT createSignedJWT() {
390                return createSignedJWT(JWSAlgorithm.RS256);
391        }
392
393        private SignedJWT createSignedJWT(JWSAlgorithm jwsAlgorithm) {
394                JWSHeader jwsHeader = new JWSHeader.Builder(jwsAlgorithm).build();
395                JWTClaimsSet claims = createJwtClaimsSet();
396
397                return new SignedJWT(jwsHeader, claims);
398        }
399
400        private SignedJWT createSignedJWT(JWSAlgorithm jwsAlgorithm, JWTClaimsSet jwtClaimsSet) {
401                JWSHeader jwsHeader = new JWSHeader.Builder(jwsAlgorithm).build();
402
403                return new SignedJWT(jwsHeader, jwtClaimsSet);
404        }
405
406        private JWTClaimsSet createJwtClaimsSet() {
407                return new JWTClaimsSet.Builder()
408                        .issuer(CLIENT_ID)
409                        .expirationTime(new Date())
410                        .audience("http://issuer.com/")
411                        .build();
412        }
