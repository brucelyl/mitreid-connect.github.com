017package org.mitre.openid.connect.service.impl;
018
019import java.util.Date;
020
021import org.joda.time.DateTime;
022import org.joda.time.Period;
023import org.junit.Before;
024import org.junit.Test;
025import org.junit.runner.RunWith;
026import org.mitre.openid.connect.model.Nonce;
027import org.mitre.openid.connect.repository.NonceRepository;
028import org.mockito.InjectMocks;
029import org.mockito.Mock;
030import org.mockito.Mockito;
031import org.mockito.runners.MockitoJUnitRunner;
032
033import com.google.common.collect.Sets;
034
035import static org.junit.Assert.assertEquals;
036import static org.junit.Assert.assertFalse;
037import static org.junit.Assert.assertTrue;
038
039/**
040 * @author wkim
041 *
042 */
043@RunWith(MockitoJUnitRunner.class)
044public class TestDefaultNonceService {
045
046        // Test value for nonce storage duration time.
047        private Period nonceStorageDuration = new Period().withSeconds(1);
048
049        private String clientId = "123abc";
050        private String value1 = "n1";
051        private String value2 = "n2";
052        private String value3 = "n3";
053
054        @Mock
055        private NonceRepository repository;
056
057        @InjectMocks
058        private DefaultNonceService service = new DefaultNonceService();
059
060        @Before
061        public void prepare() {
062
063                Mockito.reset(repository);
064
065                service.setNonceStorageDuration(nonceStorageDuration);
066
067
068        }
069
070        /**
071         * Tests the correctness of created nonces.
072         */
073        @Test
074        public void create() {
075
076                Date start = new Date(System.currentTimeMillis() - 100); // time skew of 100ms on either side
077
078                Nonce nonce = service.create(clientId, value1);
079
080                Date end = new Date(System.currentTimeMillis() + 100); // time skew of 100ms on either side
081
082                assertEquals(clientId, nonce.getClientId());
083                assertEquals(value1, nonce.getValue());
084                assertTrue(nonce.getUseDate().after(start) && nonce.getUseDate().before(end)); // make sure the date is within the right range (within 100ms on either side)
085
086                // Check expiration date.
087                assertEquals(new DateTime(nonce.getUseDate()).plus(nonceStorageDuration), new DateTime(nonce.getExpireDate()));
088        }
089
090        /**
091         * Verifies that if any nonce returned by the repository already has the value being checked for,
092         * then the service method returns true.
093         * 
094         * Also checks a nonce value not returned by the repository to verify the method returns false in this case.
095         */
096        @Test
097        public void alreadyUsed() {
098
099                Nonce nonce1 = service.create(clientId, value1);
100                Nonce nonce2 = service.create(clientId, value2);
101
102                Mockito.when(repository.getByClientId(clientId)).thenReturn(Sets.newHashSet(nonce1, nonce2));
103
104                assertTrue(service.alreadyUsed(clientId, value1));
105                assertTrue(service.alreadyUsed(clientId, value2));
106
107                assertFalse(service.alreadyUsed(clientId, value3));
108        }
109}
