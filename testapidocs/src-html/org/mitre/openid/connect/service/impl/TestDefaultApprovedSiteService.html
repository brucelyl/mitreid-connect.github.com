017package org.mitre.openid.connect.service.impl;
018
019import static org.mockito.Matchers.any;
020
021import java.util.HashSet;
022import java.util.List;
023import java.util.Set;
024
025import org.junit.Before;
026import org.junit.Test;
027import org.junit.runner.RunWith;
028import org.mitre.oauth2.model.ClientDetailsEntity;
029import org.mitre.oauth2.model.OAuth2AccessTokenEntity;
030import org.mitre.oauth2.repository.OAuth2TokenRepository;
031import org.mitre.openid.connect.model.ApprovedSite;
032import org.mitre.openid.connect.repository.ApprovedSiteRepository;
033import org.mitre.openid.connect.service.ApprovedSiteService;
034import org.mitre.openid.connect.service.StatsService;
035import org.mockito.InjectMocks;
036import org.mockito.Mock;
037import org.mockito.Mockito;
038import org.mockito.runners.MockitoJUnitRunner;
039import org.springframework.test.annotation.Rollback;
040
041import com.google.common.collect.ImmutableList;
042import com.google.common.collect.Sets;
043
044import static org.mockito.Mockito.never;
045import static org.mockito.Mockito.times;
046
047@RunWith(MockitoJUnitRunner.class)
048public class TestDefaultApprovedSiteService {
049
050        private ApprovedSite site1;
051        private ApprovedSite site2;
052        private ApprovedSite site3;
053
054        private ClientDetailsEntity client;
055        private final String clientId = "client";
056
057        @Mock
058        private ApprovedSiteRepository repository;
059
060        @Mock
061        private OAuth2TokenRepository tokenRepository;
062
063        @Mock
064        private StatsService statsService;
065
066        @InjectMocks
067        private ApprovedSiteService service = new DefaultApprovedSiteService();
068
069
070        /**
071         * Initialize the service and repository mock. Initialize a client and
072         * several ApprovedSite objects for use in unit tests.
073         */
074        @Before
075        public void prepare() {
076
077                client = new ClientDetailsEntity();
078                client.setClientId(clientId);
079
080                site1 = new ApprovedSite();
081                site1.setId(1L);
082                site1.setUserId("user1");
083                site1.setClientId("other");
084
085                site2 = new ApprovedSite();
086                site2.setId(2L);
087                site2.setUserId("user1");
088                site2.setClientId(clientId);
089
090                site3 = new ApprovedSite();
091                site3.setId(3L);
092                site3.setUserId("user2");
093                site3.setClientId(clientId);
094
095                Mockito.reset(repository, statsService);
096
097        }
098
099        /**
100         * Test clearing approved sites for a client that has 2 stored approved sites.
101         * Ensure that the repository's remove() method is called twice.
102         */
103        @Test
104        public void clearApprovedSitesForClient_success() {
105                Set<ApprovedSite> setToReturn = Sets.newHashSet(site2, site3);
106                Mockito.when(repository.getByClientId(client.getClientId())).thenReturn(setToReturn);
107                List<OAuth2AccessTokenEntity> tokens = ImmutableList.of();
108                Mockito.when(tokenRepository.getAccessTokensForApprovedSite(any(ApprovedSite.class))).thenReturn(tokens);
109
110                service.clearApprovedSitesForClient(client);
111
112                Mockito.verify(repository, times(2)).remove(any(ApprovedSite.class));
113        }
114
115        /**
116         * Test clearing approved sites for a client that doesn't have any stored approved
117         * sites. Ensure that the repository's remove() method is never called in this case.
118         */
119        @Test
120        @Rollback
121        public void clearApprovedSitesForClient_null() {
122                String otherId = "a different id";
123                client.setClientId(otherId);
124                service.clearApprovedSitesForClient(client);
125                Mockito.when(repository.getByClientId(otherId)).thenReturn(new HashSet<ApprovedSite>());
126                Mockito.verify(repository, never()).remove(any(ApprovedSite.class));
127        }
128
129
130}
