017package org.mitre.openid.connect.client.service.impl;
018
019
020import org.junit.Before;
021import org.junit.Test;
022import org.junit.runner.RunWith;
023import org.mitre.openid.connect.config.ServerConfiguration;
024import org.mockito.InjectMocks;
025import org.mockito.Matchers;
026import org.mockito.Mock;
027import org.mockito.Mockito;
028import org.mockito.runners.MockitoJUnitRunner;
029
030import static org.hamcrest.CoreMatchers.is;
031import static org.hamcrest.CoreMatchers.nullValue;
032import static org.junit.Assert.assertEquals;
033import static org.junit.Assert.assertThat;
034
035/**
036 * @author wkim
037 *
038 */
039@RunWith(MockitoJUnitRunner.class)
040public class TestHybridServerConfigurationService {
041
042        @Mock
043        private StaticServerConfigurationService mockStaticService;
044
045        @Mock
046        private DynamicServerConfigurationService mockDynamicService;
047
048        @InjectMocks
049        private HybridServerConfigurationService hybridService;
050
051        @Mock
052        private ServerConfiguration mockServerConfig;
053
054        private String issuer = "https://www.example.com/";
055
056        @Before
057        public void prepare() {
058
059                Mockito.reset(mockDynamicService, mockStaticService);
060
061        }
062
063
064        @Test
065        public void getServerConfiguration_useStatic() {
066
067                Mockito.when(mockStaticService.getServerConfiguration(issuer)).thenReturn(mockServerConfig);
068
069                ServerConfiguration result = hybridService.getServerConfiguration(issuer);
070
071                Mockito.verify(mockStaticService).getServerConfiguration(issuer);
072                Mockito.verify(mockDynamicService, Mockito.never()).getServerConfiguration(Matchers.anyString());
073                assertEquals(mockServerConfig, result);
074        }
075
076        @Test
077        public void getServerConfiguration_useDynamic() {
078
079                Mockito.when(mockStaticService.getServerConfiguration(issuer)).thenReturn(null);
080                Mockito.when(mockDynamicService.getServerConfiguration(issuer)).thenReturn(mockServerConfig);
081
082                ServerConfiguration result = hybridService.getServerConfiguration(issuer);
083
084                Mockito.verify(mockStaticService).getServerConfiguration(issuer);
085                Mockito.verify(mockDynamicService).getServerConfiguration(issuer);
086                assertEquals(mockServerConfig, result);
087        }
088
089        /**
090         * Checks the behavior when the issuer is not known.
091         */
092        @Test
093        public void getServerConfiguration_noIssuer() {
094
095                Mockito.when(mockStaticService.getServerConfiguration(issuer)).thenReturn(mockServerConfig);
096                Mockito.when(mockDynamicService.getServerConfiguration(issuer)).thenReturn(mockServerConfig);
097
098                String badIssuer = "www.badexample.com";
099
100                ServerConfiguration result = hybridService.getServerConfiguration(badIssuer);
101
102                Mockito.verify(mockStaticService).getServerConfiguration(badIssuer);
103                Mockito.verify(mockDynamicService).getServerConfiguration(badIssuer);
104                assertThat(result, is(nullValue()));
105        }
106}
