111        /**
112         * This test takes the URI from the result of building a signed request
113         * and checks that the JWS object parsed from the request URI matches up
114         * with the expected claim values.
115         */
116        @Test
117        public void buildAuthRequestUrl() {
118
119                String requestUri = urlBuilder.buildAuthRequestUrl(serverConfig, clientConfig, redirectUri, nonce, state, options);
120
121                // parsing the result
122                UriComponentsBuilder builder = null;
123
124                try {
125                        builder = UriComponentsBuilder.fromUri(new URI(requestUri));
126                } catch (URISyntaxException e1) {
127                        fail("URISyntaxException was thrown.");
128                }
129
130                UriComponents components = builder.build();
131                String jwtString = components.getQueryParams().get("request").get(0);
132                ReadOnlyJWTClaimsSet claims = null;
133
134                try {
135                        SignedJWT jwt = SignedJWT.parse(jwtString);
136                        claims = jwt.getJWTClaimsSet();
137                } catch (ParseException e) {
138                        fail("ParseException was thrown.");
139                }
140
141                assertEquals(responseType, claims.getClaim("response_type"));
142                assertEquals(clientConfig.getClientId(), claims.getClaim("client_id"));
143
144                List<String> scopeList = Arrays.asList(((String) claims.getClaim("scope")).split(" "));
145                assertTrue(scopeList.containsAll(clientConfig.getScope()));
146
147                assertEquals(redirectUri, claims.getClaim("redirect_uri"));
148                assertEquals(nonce, claims.getClaim("nonce"));
149                assertEquals(state, claims.getClaim("state"));
150                for (String claim : options.keySet()) {
151                        assertEquals(options.get(claim), claims.getClaim(claim));
152                }
153        }
