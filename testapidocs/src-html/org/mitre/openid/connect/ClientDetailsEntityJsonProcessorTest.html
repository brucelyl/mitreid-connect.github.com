028        /**
029         * Test method for {@link org.mitre.openid.connect.ClientDetailsEntityJsonProcessor#parse(java.lang.String)}.
030         */
031        @Test
032        public void testParse() {
033                String json = "  {\n" + 
034                                "   \"application_type\": \"web\",\n" + 
035                                "   \"redirect_uris\":\n" + 
036                                "     [\"https://client.example.org/callback\",\n" + 
037                                "      \"https://client.example.org/callback2\"],\n" + 
038                                "   \"client_name\": \"My Example\",\n" + 
039                                "   \"client_name#ja-Jpan-JP\":\n" + 
040                                "     \"クライアント名\",\n" + 
041                                "   \"logo_uri\": \"https://client.example.org/logo.png\",\n" + 
042                                "   \"subject_type\": \"pairwise\",\n" + 
043                                "   \"sector_identifier_uri\":\n" + 
044                                "     \"https://other.example.net/file_of_redirect_uris.json\",\n" + 
045                                "   \"token_endpoint_auth_method\": \"client_secret_basic\",\n" + 
046                                "   \"jwks_uri\": \"https://client.example.org/my_public_keys.jwks\",\n" + 
047                                "   \"userinfo_encrypted_response_alg\": \"RSA1_5\",\n" + 
048                                "   \"userinfo_encrypted_response_enc\": \"A128CBC-HS256\",\n" + 
049                                "   \"contacts\": [\"ve7jtb@example.org\", \"mary@example.org\"],\n" + 
050                                "   \"request_uris\":\n" + 
051                                "     [\"https://client.example.org/rf.txt#qpXaRLh_n93TTR9F252ValdatUQvQiJi5BDub2BeznA\"]\n" + 
052                                "  }";
053                ClientDetailsEntity c = ClientDetailsEntityJsonProcessor.parse(json);
054                
055                assertEquals(ClientDetailsEntity.AppType.WEB, c.getApplicationType());
056                assertEquals(ImmutableSet.of("https://client.example.org/callback", "https://client.example.org/callback2"), c.getRedirectUris());
057                assertEquals("My Example", c.getClientName());
058                assertEquals("https://client.example.org/logo.png", c.getLogoUri());
059                assertEquals(ClientDetailsEntity.SubjectType.PAIRWISE, c.getSubjectType());
060                assertEquals("https://other.example.net/file_of_redirect_uris.json", c.getSectorIdentifierUri());
061                assertEquals(ClientDetailsEntity.AuthMethod.SECRET_BASIC, c.getTokenEndpointAuthMethod());
062                assertEquals("https://client.example.org/my_public_keys.jwks", c.getJwksUri());
063                assertEquals(JWEAlgorithm.RSA1_5, c.getUserInfoEncryptedResponseAlg().getAlgorithm());
064                assertEquals(EncryptionMethod.A128CBC_HS256, c.getUserInfoEncryptedResponseEnc().getAlgorithm());
065                assertEquals(ImmutableSet.of("ve7jtb@example.org", "mary@example.org"), c.getContacts());
066                assertEquals(ImmutableSet.of("https://client.example.org/rf.txt#qpXaRLh_n93TTR9F252ValdatUQvQiJi5BDub2BeznA"), c.getRequestUris());
067                
068        }

128        /**
129         * Test method for {@link org.mitre.openid.connect.ClientDetailsEntityJsonProcessor#serialize(org.mitre.oauth2.model.RegisteredClient)}.
130         */
131        @Test
132        public void testSerialize() {
133                RegisteredClient c = new RegisteredClient();
134                
135                c.setClientId("s6BhdRkqt3");
136                c.setClientSecret("ZJYCqe3GGRvdrudKyZS0XhGv_Z45DuKhCUk0gBR1vZk");
137                c.setClientSecretExpiresAt(new Date(1577858400L * 1000L));
138                c.setRegistrationAccessToken("this.is.an.access.token.value.ffx83");
139                c.setRegistrationClientUri("https://server.example.com/connect/register?client_id=s6BhdRkqt3");
140                c.setApplicationType(ClientDetailsEntity.AppType.WEB);
141                c.setRedirectUris(ImmutableSet.of("https://client.example.org/callback", "https://client.example.org/callback2"));
142                c.setClientName("My Example");
143                c.setLogoUri("https://client.example.org/logo.png");
144                c.setSubjectType(ClientDetailsEntity.SubjectType.PAIRWISE);
145                c.setSectorIdentifierUri("https://other.example.net/file_of_redirect_uris.json");
146                c.setTokenEndpointAuthMethod(ClientDetailsEntity.AuthMethod.SECRET_BASIC);
147                c.setJwksUri("https://client.example.org/my_public_keys.jwks");
148                c.setUserInfoEncryptedResponseAlg(new JWEAlgorithmEmbed(JWEAlgorithm.RSA1_5));
149                c.setUserInfoEncryptedResponseEnc(new JWEEncryptionMethodEmbed(EncryptionMethod.A128CBC_HS256));
150                c.setContacts(ImmutableSet.of("ve7jtb@example.org", "mary@example.org"));
151                c.setRequestUris(ImmutableSet.of("https://client.example.org/rf.txt#qpXaRLh_n93TTR9F252ValdatUQvQiJi5BDub2BeznA"));
152                
153                JsonObject j = ClientDetailsEntityJsonProcessor.serialize(c);
154
155                assertEquals("s6BhdRkqt3", j.get("client_id").getAsString());
156                assertEquals("ZJYCqe3GGRvdrudKyZS0XhGv_Z45DuKhCUk0gBR1vZk", j.get("client_secret").getAsString());
157                assertEquals(1577858400L, j.get("client_secret_expires_at").getAsNumber());
158                assertEquals("this.is.an.access.token.value.ffx83", j.get("registration_access_token").getAsString());
159                assertEquals("https://server.example.com/connect/register?client_id=s6BhdRkqt3", j.get("registration_client_uri").getAsString());
160                assertEquals(ClientDetailsEntity.AppType.WEB.getValue(), j.get("application_type").getAsString());
161                for (JsonElement e : j.get("redirect_uris").getAsJsonArray()) { 
162                        assertTrue(ImmutableSet.of("https://client.example.org/callback", "https://client.example.org/callback2").contains(e.getAsString()));
163        }
164                assertEquals("My Example", j.get("client_name").getAsString());
165                assertEquals("https://client.example.org/logo.png", j.get("logo_uri").getAsString());
166                assertEquals(ClientDetailsEntity.SubjectType.PAIRWISE.getValue(), j.get("subject_type").getAsString());
167                assertEquals("https://other.example.net/file_of_redirect_uris.json", j.get("sector_identifier_uri").getAsString());
168                assertEquals(ClientDetailsEntity.AuthMethod.SECRET_BASIC.getValue(), j.get("token_endpoint_auth_method").getAsString());
169                assertEquals("https://client.example.org/my_public_keys.jwks", j.get("jwks_uri").getAsString());
170                assertEquals(JWEAlgorithm.RSA1_5.getName(), j.get("userinfo_encrypted_response_alg").getAsString());
171                assertEquals(EncryptionMethod.A128CBC_HS256.getName(), j.get("userinfo_encrypted_response_enc").getAsString());
172                for (JsonElement e : j.get("contacts").getAsJsonArray()) {
173                        assertTrue(ImmutableSet.of("ve7jtb@example.org", "mary@example.org").contains(e.getAsString()));
174                }
175                for (JsonElement e : j.get("request_uris").getAsJsonArray()) {
176                        assertTrue(ImmutableSet.of("https://client.example.org/rf.txt#qpXaRLh_n93TTR9F252ValdatUQvQiJi5BDub2BeznA").contains(e.getAsString()));
177                }
178                
179        }
