018package org.mitre.uma.service.impl;
019
020import java.util.Set;
021import java.util.UUID;
022
023import org.junit.Before;
024import org.junit.Test;
025import org.junit.runner.RunWith;
026import org.mitre.oauth2.service.SystemScopeService;
027import org.mitre.uma.model.PermissionTicket;
028import org.mitre.uma.model.ResourceSet;
029import org.mitre.uma.repository.PermissionRepository;
030import org.mockito.AdditionalAnswers;
031import org.mockito.InjectMocks;
032import org.mockito.Matchers;
033import org.mockito.Mock;
034import org.mockito.invocation.InvocationOnMock;
035import org.mockito.runners.MockitoJUnitRunner;
036import org.mockito.stubbing.Answer;
037import org.springframework.security.oauth2.common.exceptions.InsufficientScopeException;
038
039import com.google.common.collect.ImmutableSet;
040
041import static org.hamcrest.CoreMatchers.equalTo;
042import static org.hamcrest.CoreMatchers.not;
043
044import static org.mockito.Matchers.anySetOf;
045
046import static org.mockito.Mockito.when;
047
048import static org.junit.Assert.assertNotNull;
049import static org.junit.Assert.assertThat;
050
051/**
052 * @author jricher
053 *
054 */
055@RunWith(MockitoJUnitRunner.class)
056public class TestDefaultPermissionService {
057
058        @Mock
059        private PermissionRepository permissionRepository;
060
061        @Mock
062        private SystemScopeService scopeService;
063
064        @InjectMocks
065        private DefaultPermissionService permissionService;
066
067        private Set<String> scopes1 = ImmutableSet.of("foo", "bar", "baz");
068        private Set<String> scopes2 = ImmutableSet.of("alpha", "beta", "betest");
069
070        private ResourceSet rs1;
071        private ResourceSet rs2;
072
073        private String rs1Name = "resource set 1";
074        private String rs1Owner = "resource set owner 1";
075        private Long rs1Id = 1L;
076
077        private String rs2Name = "resource set 2";
078        private String rs2Owner = "resource set owner 2";
079        private Long rs2Id = 2L;
080
081
082        @Before
083        public void prepare() {
084                rs1 = new ResourceSet();
085                rs1.setName(rs1Name);
086                rs1.setOwner(rs1Owner);
087                rs1.setId(rs1Id );
088                rs1.setScopes(scopes1);
089
090                rs2 = new ResourceSet();
091                rs2.setName(rs2Name);
092                rs2.setOwner(rs2Owner);
093                rs2.setId(rs2Id);
094                rs2.setScopes(scopes2);
095
096                // have the repository just pass the argument through
097                when(permissionRepository.save(Matchers.any(PermissionTicket.class))).then(AdditionalAnswers.returnsFirstArg());
098
099                when(scopeService.scopesMatch(anySetOf(String.class), anySetOf(String.class))).then(new Answer<Boolean>() {
100
101                        @Override
102                        public Boolean answer(InvocationOnMock invocation) throws Throwable {
103                                Object[] arguments = invocation.getArguments();
104                                @SuppressWarnings("unchecked")
105                                Set<String> expected = (Set<String>) arguments[0];
106                                @SuppressWarnings("unchecked")
107                                Set<String> actual = (Set<String>) arguments[1];
108
109                                return expected.containsAll(actual);
110                        }
111                });
112
113        }
114
115
116        /**
117         * Test method for {@link org.mitre.uma.service.impl.DefaultPermissionService#createTicket(org.mitre.uma.model.ResourceSet, java.util.Set)}.
118         */
119        @Test
120        public void testCreate_ticket() {
121
122                PermissionTicket perm = permissionService.createTicket(rs1, scopes1);
123
124                // we want there to be a non-null ticket
125                assertNotNull(perm.getTicket());
126        }
127
128        @Test
129        public void testCreate_uuid() {
130                PermissionTicket perm = permissionService.createTicket(rs1, scopes1);
131
132                // we expect this to be a UUID
133                UUID uuid = UUID.fromString(perm.getTicket());
134
135                assertNotNull(uuid);
136
137        }
138
139        @Test
140        public void testCreate_differentTicketsSameClient() {
141
142                PermissionTicket perm1 = permissionService.createTicket(rs1, scopes1);
143                PermissionTicket perm2 = permissionService.createTicket(rs1, scopes1);
144
145                assertNotNull(perm1.getTicket());
146                assertNotNull(perm2.getTicket());
147
148                // make sure these are different from each other
149                assertThat(perm1.getTicket(), not(equalTo(perm2.getTicket())));
150
151        }
152
153        @Test
154        public void testCreate_differentTicketsDifferentClient() {
155
156                PermissionTicket perm1 = permissionService.createTicket(rs1, scopes1);
157                PermissionTicket perm2 = permissionService.createTicket(rs2, scopes2);
158
159                assertNotNull(perm1.getTicket());
160                assertNotNull(perm2.getTicket());
161
162                // make sure these are different from each other
163                assertThat(perm1.getTicket(), not(equalTo(perm2.getTicket())));
164
165        }
166
167        @Test(expected = InsufficientScopeException.class)
168        public void testCreate_scopeMismatch() {
169                @SuppressWarnings("unused")
170                // try to get scopes outside of what we're allowed to do, this should throw an exception
171                PermissionTicket perm = permissionService.createTicket(rs1, scopes2);
172        }
