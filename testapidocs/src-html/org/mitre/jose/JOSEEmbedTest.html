004package org.mitre.jose;
005
006import org.junit.Test;
007
008import com.nimbusds.jose.EncryptionMethod;
009import com.nimbusds.jose.JWEAlgorithm;
010import com.nimbusds.jose.JWSAlgorithm;
011
012import static org.junit.Assert.*;
013
014/**
015 * 
016 * These tests make sure that the algorithm name processing 
017 * is functional on the three embedded JOSE classes.
018 * 
019 * @author jricher
020 *
021 */
022public class JOSEEmbedTest {
023
024        @Test
025        public void testJWSAlgorithmEmbed() {
026                JWSAlgorithmEmbed a = new JWSAlgorithmEmbed(JWSAlgorithm.HS256);
027                
028                assertEquals(JWSAlgorithm.HS256, a.getAlgorithm());
029                assertEquals("HS256", a.getAlgorithmName());
030        }
031        
032        @Test
033        public void testJWSAlgorithmEmbedGetForAlgoirthmName() {
034                JWSAlgorithmEmbed a = JWSAlgorithmEmbed.getForAlgorithmName("RS256");
035                
036                assertEquals(JWSAlgorithm.RS256, a.getAlgorithm());
037                assertEquals("RS256", a.getAlgorithmName());
038        }
039
040        @Test
041        public void testJWEAlgorithmEmbed() {
042                JWEAlgorithmEmbed a = new JWEAlgorithmEmbed(JWEAlgorithm.A128KW);
043                
044                assertEquals(JWEAlgorithm.A128KW, a.getAlgorithm());
045                assertEquals("A128KW", a.getAlgorithmName());
046        }
047        
048        @Test
049        public void testJWEAlgorithmEmbedGetForAlgoirthmName() {
050                JWEAlgorithmEmbed a = JWEAlgorithmEmbed.getForAlgorithmName("RSA1_5");
051                
052                assertEquals(JWEAlgorithm.RSA1_5, a.getAlgorithm());
053                assertEquals("RSA1_5", a.getAlgorithmName());
054        }
055
056        @Test
057        public void testJWEEncryptionMethodEmbed() {
058                JWEEncryptionMethodEmbed a = new JWEEncryptionMethodEmbed(EncryptionMethod.A128CBC_HS256);
059                
060                assertEquals(EncryptionMethod.A128CBC_HS256, a.getAlgorithm());
061                assertEquals("A128CBC-HS256", a.getAlgorithmName());
062        }
063        
064        @Test
065        public void testJWEEncryptionMethodEmbedGetForAlgoirthmName() {
066                JWEEncryptionMethodEmbed a = JWEEncryptionMethodEmbed.getForAlgorithmName("A256GCM");
067                
068                assertEquals(EncryptionMethod.A256GCM, a.getAlgorithm());
069                assertEquals("A256GCM", a.getAlgorithmName());
070        }
071
072}
