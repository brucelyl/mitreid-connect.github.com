018package org.mitre.oauth2.service.impl;
019
020import java.util.Set;
021
022import org.junit.Before;
023import org.junit.Test;
024import org.junit.runner.RunWith;
025import org.mitre.oauth2.model.SystemScope;
026import org.mitre.oauth2.repository.SystemScopeRepository;
027import org.mockito.InjectMocks;
028import org.mockito.Mock;
029import org.mockito.Mockito;
030import org.mockito.runners.MockitoJUnitRunner;
031
032import com.google.common.collect.Sets;
033
034import static org.hamcrest.CoreMatchers.equalTo;
035import static org.hamcrest.CoreMatchers.is;
036import static org.hamcrest.CoreMatchers.nullValue;
037
038import static org.junit.Assert.assertThat;
039
040/**
041 * @author wkim
042 *
043 */
044@RunWith(MockitoJUnitRunner.class)
045public class TestDefaultSystemScopeService {
046
047        // test fixture
048        private SystemScope defaultDynScope1;
049        private SystemScope defaultDynScope2;
050        private SystemScope defaultScope1;
051        private SystemScope defaultScope2;
052        private SystemScope dynScope1;
053        private SystemScope restrictedScope1;
054
055        private String defaultDynScope1String = "defaultDynScope1";
056        private String defaultDynScope2String = "defaultDynScope2";
057        private String defaultScope1String = "defaultScope1";
058        private String defaultScope2String = "defaultScope2";
059        private String dynScope1String = "dynScope1";
060        private String restrictedScope1String = "restrictedScope1";
061
062        private Set<SystemScope> allScopes;
063        private Set<String> allScopeStrings;
064        private Set<SystemScope> allScopesWithValue;
065        private Set<String> allScopeStringsWithValue;
066
067        @Mock
068        private SystemScopeRepository repository;
069
070        @InjectMocks
071        private DefaultSystemScopeService service;
072
073        /**
074         * Assumes these SystemScope defaults: isDefaultScope=false and isAllowDynReg=false.
075         */
076        @Before
077        public void prepare() {
078
079                Mockito.reset(repository);
080
081                // two default and dynamically registerable scopes (unrestricted)
082                defaultDynScope1 = new SystemScope(defaultDynScope1String);
083                defaultDynScope2 = new SystemScope(defaultDynScope2String);
084                defaultDynScope1.setDefaultScope(true);
085                defaultDynScope2.setDefaultScope(true);
086
087                // two strictly default scopes (restricted)
088                defaultScope1 = new SystemScope(defaultScope1String);
089                defaultScope2 = new SystemScope(defaultScope2String);
090                defaultScope1.setRestricted(true);
091                defaultScope2.setRestricted(true);
092                defaultScope1.setDefaultScope(true);
093                defaultScope2.setDefaultScope(true);
094
095                // one strictly dynamically registerable scope (isDefault false)
096                dynScope1 = new SystemScope(dynScope1String);
097
098                // extraScope1 : extra scope that is neither restricted nor default (defaults to false/false)
099                restrictedScope1 = new SystemScope(restrictedScope1String);
100                restrictedScope1.setRestricted(true);
101
102
103                allScopes = Sets.newHashSet(defaultDynScope1, defaultDynScope2, defaultScope1, defaultScope2, dynScope1, restrictedScope1);
104                allScopeStrings = Sets.newHashSet(defaultDynScope1String, defaultDynScope2String, defaultScope1String, defaultScope2String, dynScope1String, restrictedScope1String);
105
106                allScopesWithValue = Sets.newHashSet(defaultDynScope1, defaultDynScope2, defaultScope1, defaultScope2, dynScope1, restrictedScope1);
107                allScopeStringsWithValue = Sets.newHashSet(defaultDynScope1String, defaultDynScope2String, defaultScope1String, defaultScope2String, dynScope1String, restrictedScope1String);
108
109                Mockito.when(repository.getByValue(defaultDynScope1String)).thenReturn(defaultDynScope1);
110                Mockito.when(repository.getByValue(defaultDynScope2String)).thenReturn(defaultDynScope2);
111                Mockito.when(repository.getByValue(defaultScope1String)).thenReturn(defaultScope1);
112                Mockito.when(repository.getByValue(defaultScope2String)).thenReturn(defaultScope2);
113                Mockito.when(repository.getByValue(dynScope1String)).thenReturn(dynScope1);
114                Mockito.when(repository.getByValue(restrictedScope1String)).thenReturn(restrictedScope1);
115
116                Mockito.when(repository.getAll()).thenReturn(allScopes);
117        }
118
119        @Test
120        public void getAll() {
121
122                assertThat(service.getAll(), equalTo(allScopes));
123        }
124
125        @Test
126        public void getDefaults() {
127
128                Set<SystemScope> defaults = Sets.newHashSet(defaultDynScope1, defaultDynScope2, defaultScope1, defaultScope2);
129
130                assertThat(service.getDefaults(), equalTo(defaults));
131        }
132
133        @Test
134        public void getUnrestricted() {
135
136                Set<SystemScope> unrestricted = Sets.newHashSet(defaultDynScope1, defaultDynScope2, dynScope1);
137
138                assertThat(service.getUnrestricted(), equalTo(unrestricted));
139        }
140
141        @Test
142        public void getRestricted() {
143                Set<SystemScope> restricted = Sets.newHashSet(defaultScope1, defaultScope2, restrictedScope1);
144
145                assertThat(service.getRestricted(), equalTo(restricted));
146
147        }
148
149        @Test
150        public void fromStrings() {
151
152                // check null condition
153                assertThat(service.fromStrings(null), is(nullValue()));
154
155                assertThat(service.fromStrings(allScopeStrings), equalTo(allScopes));
156
157                assertThat(service.fromStrings(allScopeStringsWithValue), equalTo(allScopesWithValue));
158        }
159
160        @Test
161        public void toStrings() {
162
163                // check null condition
164                assertThat(service.toStrings(null), is(nullValue()));
165
166                assertThat(service.toStrings(allScopes), equalTo(allScopeStrings));
167
168                assertThat(service.toStrings(allScopesWithValue), equalTo(allScopeStringsWithValue));
169        }
170
171        @Test
172        public void scopesMatch() {
173
174                Set<String> expected = Sets.newHashSet("foo", "bar", "baz");
175                Set<String> actualGood = Sets.newHashSet("foo", "baz", "bar");
176                Set<String> actualGood2 = Sets.newHashSet("foo", "bar");
177                Set<String> actualBad = Sets.newHashSet("foo", "bob", "bar");
178
179                // same scopes, different order
180                assertThat(service.scopesMatch(expected, actualGood), is(true));
181
182                // subset
183                assertThat(service.scopesMatch(expected, actualGood2), is(true));
184
185                // extra scope (fail)
186                assertThat(service.scopesMatch(expected, actualBad), is(false));
187        }
188
189}
