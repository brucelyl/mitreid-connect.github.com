031import org.mockito.stubbing.Answer;
032
033import com.google.common.collect.Sets;
034
035import static org.hamcrest.CoreMatchers.equalTo;
036import static org.hamcrest.CoreMatchers.is;
037import static org.hamcrest.CoreMatchers.nullValue;
038
039import static org.junit.Assert.assertThat;
040
041/**
042 * @author wkim
043 *
044 */
045@RunWith(MockitoJUnitRunner.class)
046public class TestDefaultSystemScopeService {
047
048        // test fixture
049        private SystemScope defaultDynScope1;
050        private SystemScope defaultDynScope2;
051        private SystemScope defaultScope1;
052        private SystemScope defaultScope2;
053        private SystemScope dynScope1;
054        private SystemScope restrictedScope1;
055        private SystemScope structuredScope1;
056        private SystemScope structuredScope1Value;
057
058        private String defaultDynScope1String = "defaultDynScope1";
059        private String defaultDynScope2String = "defaultDynScope2";
060        private String defaultScope1String = "defaultScope1";
061        private String defaultScope2String = "defaultScope2";
062        private String dynScope1String = "dynScope1";
063        private String restrictedScope1String = "restrictedScope1";
064        private String structuredScope1String = "structuredScope1";
065        private String structuredValue = "structuredValue";
066
067        private Set<SystemScope> allScopes;
068        private Set<String> allScopeStrings;
069        private Set<SystemScope> allScopesWithValue;
070        private Set<String> allScopeStringsWithValue;
071
072        @Mock
073        private SystemScopeRepository repository;
074
075        @InjectMocks
076        private DefaultSystemScopeService service;
077
078        /**
079         * Assumes these SystemScope defaults: isDefaultScope=false and isAllowDynReg=false.
080         */
081        @Before
082        public void prepare() {
083
084                Mockito.reset(repository);
085
086                // two default and dynamically registerable scopes (unrestricted)
087                defaultDynScope1 = new SystemScope(defaultDynScope1String);
088                defaultDynScope2 = new SystemScope(defaultDynScope2String);
089                defaultDynScope1.setDefaultScope(true);
090                defaultDynScope2.setDefaultScope(true);
091
092                // two strictly default scopes (restricted)
093                defaultScope1 = new SystemScope(defaultScope1String);
094                defaultScope2 = new SystemScope(defaultScope2String);
095                defaultScope1.setRestricted(true);
096                defaultScope2.setRestricted(true);
097                defaultScope1.setDefaultScope(true);
098                defaultScope2.setDefaultScope(true);
099
100                // one strictly dynamically registerable scope (isDefault false)
101                dynScope1 = new SystemScope(dynScope1String);
102
103                // extraScope1 : extra scope that is neither restricted nor default (defaults to false/false)
104                restrictedScope1 = new SystemScope(restrictedScope1String);
105                restrictedScope1.setRestricted(true);
106
107
108                // structuredScope1 : structured scope
109                structuredScope1 = new SystemScope(structuredScope1String);
110                structuredScope1.setStructured(true);
111
112                // structuredScope1Value : structured scope with value
113                structuredScope1Value = new SystemScope(structuredScope1String);
114                structuredScope1Value.setStructured(true);
115                structuredScope1Value.setStructuredValue(structuredValue);
116
117                allScopes = Sets.newHashSet(defaultDynScope1, defaultDynScope2, defaultScope1, defaultScope2, dynScope1, restrictedScope1, structuredScope1);
118                allScopeStrings = Sets.newHashSet(defaultDynScope1String, defaultDynScope2String, defaultScope1String, defaultScope2String, dynScope1String, restrictedScope1String, structuredScope1String);
119
120                allScopesWithValue = Sets.newHashSet(defaultDynScope1, defaultDynScope2, defaultScope1, defaultScope2, dynScope1, restrictedScope1, structuredScope1, structuredScope1Value);
121                allScopeStringsWithValue = Sets.newHashSet(defaultDynScope1String, defaultDynScope2String, defaultScope1String, defaultScope2String, dynScope1String, restrictedScope1String, structuredScope1String, structuredScope1String + ":" + structuredValue);
122
123                Mockito.when(repository.getByValue(defaultDynScope1String)).thenReturn(defaultDynScope1);
124                Mockito.when(repository.getByValue(defaultDynScope2String)).thenReturn(defaultDynScope2);
125                Mockito.when(repository.getByValue(defaultScope1String)).thenReturn(defaultScope1);
126                Mockito.when(repository.getByValue(defaultScope2String)).thenReturn(defaultScope2);
127                Mockito.when(repository.getByValue(dynScope1String)).thenReturn(dynScope1);
128                Mockito.when(repository.getByValue(restrictedScope1String)).thenReturn(restrictedScope1);
129                // we re-use this value so we've got to use thenAnswer instead
130                Mockito.when(repository.getByValue(structuredScope1String)).thenAnswer(new Answer<SystemScope>() {
131                        @Override
132                        public SystemScope answer(InvocationOnMock invocation) throws Throwable {
133                                SystemScope s = new SystemScope(structuredScope1String);
134                                s.setStructured(true);
135                                return s;
136                        }
137
138                });
139
140                Mockito.when(repository.getAll()).thenReturn(allScopes);
141        }
142
143        @Test
144        public void getAll() {
145
146                assertThat(service.getAll(), equalTo(allScopes));
147        }
148
149        @Test
150        public void getDefaults() {
151
152                Set<SystemScope> defaults = Sets.newHashSet(defaultDynScope1, defaultDynScope2, defaultScope1, defaultScope2);
153
154                assertThat(service.getDefaults(), equalTo(defaults));
155        }
156
157        @Test
158        public void getUnrestricted() {
159
160                Set<SystemScope> unrestricted = Sets.newHashSet(defaultDynScope1, defaultDynScope2, dynScope1, structuredScope1);
161
162                assertThat(service.getUnrestricted(), equalTo(unrestricted));
163        }
164
165        @Test
166        public void getRestricted() {
167                Set<SystemScope> restricted = Sets.newHashSet(defaultScope1, defaultScope2, restrictedScope1);
168
169                assertThat(service.getRestricted(), equalTo(restricted));
170
171        }
172
173        @Test
174        public void fromStrings() {
175
176                // check null condition
177                assertThat(service.fromStrings(null), is(nullValue()));
178
179                assertThat(service.fromStrings(allScopeStrings), equalTo(allScopes));
180
181                assertThat(service.fromStrings(allScopeStringsWithValue), equalTo(allScopesWithValue));
182        }
183
184        @Test
185        public void toStrings() {
186
187                // check null condition
188                assertThat(service.toStrings(null), is(nullValue()));
189
190                assertThat(service.toStrings(allScopes), equalTo(allScopeStrings));
191
192                assertThat(service.toStrings(allScopesWithValue), equalTo(allScopeStringsWithValue));
193        }
194
195        @Test
196        public void scopesMatch() {
197
198                Set<String> expected = Sets.newHashSet("foo", "bar", "baz");
199                Set<String> actualGood = Sets.newHashSet("foo", "baz", "bar");
200                Set<String> actualGood2 = Sets.newHashSet("foo", "bar");
201                Set<String> actualBad = Sets.newHashSet("foo", "bob", "bar");
202
203                // same scopes, different order
204                assertThat(service.scopesMatch(expected, actualGood), is(true));
205
206                // subset
207                assertThat(service.scopesMatch(expected, actualGood2), is(true));
208
209                // extra scope (fail)
210                assertThat(service.scopesMatch(expected, actualBad), is(false));
211        }
212
213        @Test
214        public void scopesMatch_structured() {
215                Set<String> expected = Sets.newHashSet("foo", "bar", "baz");
216                Set<String> actualGood = Sets.newHashSet("foo:value", "baz", "bar");
217                Set<String> actualBad = Sets.newHashSet("foo:value", "bar:value");
218
219                // note: we have to use "thenAnswer" here to mimic the repository not serializing the structuredValue field
220                Mockito.when(repository.getByValue("foo")).thenAnswer(new Answer<SystemScope>() {
221                        @Override
222                        public SystemScope answer(InvocationOnMock invocation) throws Throwable {
223                                SystemScope foo = new SystemScope("foo");
224                                foo.setStructured(true);
225                                return foo;
226                        }
227
228                });
229
230                assertThat(service.scopesMatch(expected, actualGood), is(true));
231
232                assertThat(service.scopesMatch(expected, actualBad), is(false));
233        }
234}
