021package org.mitre.oauth2.model;
022
023import java.util.Date;
024
025import org.junit.Test;
026
027import com.google.common.collect.ImmutableSet;
028import com.nimbusds.jose.EncryptionMethod;
029import com.nimbusds.jose.JWEAlgorithm;
030
031import static org.junit.Assert.assertEquals;
032
033/**
034 * @author jricher
035 *
036 */
037public class ClientDetailsEntityTest {
038
039        /**
040         * Test method for {@link org.mitre.oauth2.model.ClientDetailsEntity#ClientDetailsEntity()}.
041         */
042        @Test
043        public void testClientDetailsEntity() {
044                Date now = new Date();
045
046                ClientDetailsEntity c = new ClientDetailsEntity();
047
048                c.setClientId("s6BhdRkqt3");
049                c.setClientSecret("ZJYCqe3GGRvdrudKyZS0XhGv_Z45DuKhCUk0gBR1vZk");
050                c.setApplicationType(ClientDetailsEntity.AppType.WEB);
051                c.setRedirectUris(ImmutableSet.of("https://client.example.org/callback", "https://client.example.org/callback2"));
052                c.setClientName("My Example");
053                c.setLogoUri("https://client.example.org/logo.png");
054                c.setSubjectType(ClientDetailsEntity.SubjectType.PAIRWISE);
055                c.setSectorIdentifierUri("https://other.example.net/file_of_redirect_uris.json");
056                c.setTokenEndpointAuthMethod(ClientDetailsEntity.AuthMethod.SECRET_BASIC);
057                c.setJwksUri("https://client.example.org/my_public_keys.jwks");
058                c.setUserInfoEncryptedResponseAlg(JWEAlgorithm.RSA1_5);
059                c.setUserInfoEncryptedResponseEnc(EncryptionMethod.A128CBC_HS256);
060                c.setContacts(ImmutableSet.of("ve7jtb@example.org", "mary@example.org"));
061                c.setRequestUris(ImmutableSet.of("https://client.example.org/rf.txt#qpXaRLh_n93TTR9F252ValdatUQvQiJi5BDub2BeznA"));
062                c.setCreatedAt(now);
063                c.setAccessTokenValiditySeconds(600);
064
065                assertEquals("s6BhdRkqt3", c.getClientId());
066                assertEquals("ZJYCqe3GGRvdrudKyZS0XhGv_Z45DuKhCUk0gBR1vZk", c.getClientSecret());
067                assertEquals(ClientDetailsEntity.AppType.WEB, c.getApplicationType());
068                assertEquals(ImmutableSet.of("https://client.example.org/callback", "https://client.example.org/callback2"), c.getRedirectUris());
069                assertEquals("My Example", c.getClientName());
070                assertEquals("https://client.example.org/logo.png", c.getLogoUri());
071                assertEquals(ClientDetailsEntity.SubjectType.PAIRWISE, c.getSubjectType());
072                assertEquals("https://other.example.net/file_of_redirect_uris.json", c.getSectorIdentifierUri());
073                assertEquals(ClientDetailsEntity.AuthMethod.SECRET_BASIC, c.getTokenEndpointAuthMethod());
074                assertEquals("https://client.example.org/my_public_keys.jwks", c.getJwksUri());
075                assertEquals(JWEAlgorithm.RSA1_5, c.getUserInfoEncryptedResponseAlg());
076                assertEquals(EncryptionMethod.A128CBC_HS256, c.getUserInfoEncryptedResponseEnc());
077                assertEquals(ImmutableSet.of("ve7jtb@example.org", "mary@example.org"), c.getContacts());
078                assertEquals(ImmutableSet.of("https://client.example.org/rf.txt#qpXaRLh_n93TTR9F252ValdatUQvQiJi5BDub2BeznA"), c.getRequestUris());
079                assertEquals(now, c.getCreatedAt());
080                assertEquals(600, c.getAccessTokenValiditySeconds().intValue());
081
082        }
083
084}
