039        /**
040         * Test method for {@link org.mitre.oauth2.model.RegisteredClient#RegisteredClient()}.
041         */
042        @Test
043        public void testRegisteredClient() {
044
045                // make sure all the pass-through getters and setters work
046
047                RegisteredClient c = new RegisteredClient();
048
049                c.setClientId("s6BhdRkqt3");
050                c.setClientSecret("ZJYCqe3GGRvdrudKyZS0XhGv_Z45DuKhCUk0gBR1vZk");
051                c.setClientSecretExpiresAt(new Date(1577858400L * 1000L));
052                c.setRegistrationAccessToken("this.is.an.access.token.value.ffx83");
053                c.setRegistrationClientUri("https://server.example.com/connect/register?client_id=s6BhdRkqt3");
054                c.setApplicationType(ClientDetailsEntity.AppType.WEB);
055                c.setRedirectUris(ImmutableSet.of("https://client.example.org/callback", "https://client.example.org/callback2"));
056                c.setClientName("My Example");
057                c.setLogoUri("https://client.example.org/logo.png");
058                c.setSubjectType(ClientDetailsEntity.SubjectType.PAIRWISE);
059                c.setSectorIdentifierUri("https://other.example.net/file_of_redirect_uris.json");
060                c.setTokenEndpointAuthMethod(ClientDetailsEntity.AuthMethod.SECRET_BASIC);
061                c.setJwksUri("https://client.example.org/my_public_keys.jwks");
062                c.setUserInfoEncryptedResponseAlg(JWEAlgorithm.RSA1_5);
063                c.setUserInfoEncryptedResponseEnc(EncryptionMethod.A128CBC_HS256);
064                c.setContacts(ImmutableSet.of("ve7jtb@example.org", "mary@example.org"));
065                c.setRequestUris(ImmutableSet.of("https://client.example.org/rf.txt#qpXaRLh_n93TTR9F252ValdatUQvQiJi5BDub2BeznA"));
066
067                assertEquals("s6BhdRkqt3", c.getClientId());
068                assertEquals("ZJYCqe3GGRvdrudKyZS0XhGv_Z45DuKhCUk0gBR1vZk", c.getClientSecret());
069                assertEquals(new Date(1577858400L * 1000L), c.getClientSecretExpiresAt());
070                assertEquals("this.is.an.access.token.value.ffx83", c.getRegistrationAccessToken());
071                assertEquals("https://server.example.com/connect/register?client_id=s6BhdRkqt3", c.getRegistrationClientUri());
072                assertEquals(ClientDetailsEntity.AppType.WEB, c.getApplicationType());
073                assertEquals(ImmutableSet.of("https://client.example.org/callback", "https://client.example.org/callback2"), c.getRedirectUris());
074                assertEquals("My Example", c.getClientName());
075                assertEquals("https://client.example.org/logo.png", c.getLogoUri());
076                assertEquals(ClientDetailsEntity.SubjectType.PAIRWISE, c.getSubjectType());
077                assertEquals("https://other.example.net/file_of_redirect_uris.json", c.getSectorIdentifierUri());
078                assertEquals(ClientDetailsEntity.AuthMethod.SECRET_BASIC, c.getTokenEndpointAuthMethod());
079                assertEquals("https://client.example.org/my_public_keys.jwks", c.getJwksUri());
080                assertEquals(JWEAlgorithm.RSA1_5, c.getUserInfoEncryptedResponseAlg());
081                assertEquals(EncryptionMethod.A128CBC_HS256, c.getUserInfoEncryptedResponseEnc());
082                assertEquals(ImmutableSet.of("ve7jtb@example.org", "mary@example.org"), c.getContacts());
083                assertEquals(ImmutableSet.of("https://client.example.org/rf.txt#qpXaRLh_n93TTR9F252ValdatUQvQiJi5BDub2BeznA"), c.getRequestUris());
084        }
