475        /**
476         * @param writer
477         */
478        private void writeAuthenticationHolders(JsonWriter writer) throws IOException {
479                for (AuthenticationHolderEntity holder : authHolderRepository.getAll()) {
480                        writer.beginObject();
481                        writer.name(ID).value(holder.getId());
482
483                        writer.name(REQUEST_PARAMETERS);
484                        writer.beginObject();
485                        for (Entry<String, String> entry : holder.getRequestParameters().entrySet()) {
486                                writer.name(entry.getKey()).value(entry.getValue());
487                        }
488                        writer.endObject();
489                        writer.name(CLIENT_ID).value(holder.getClientId());
490                        Set<String> scope = holder.getScope();
491                        writer.name(SCOPE);
492                        writer.beginArray();
493                        for (String s : scope) {
494                                writer.value(s);
495                        }
496                        writer.endArray();
497                        writer.name(RESOURCE_IDS);
498                        writer.beginArray();
499                        if (holder.getResourceIds() != null) {
500                                for (String s : holder.getResourceIds()) {
501                                        writer.value(s);
502                                }
503                        }
504                        writer.endArray();
505                        writer.name(AUTHORITIES);
506                        writer.beginArray();
507                        for (GrantedAuthority authority : holder.getAuthorities()) {
508                                writer.value(authority.getAuthority());
509                        }
510                        writer.endArray();
511                        writer.name(APPROVED).value(holder.isApproved());
512                        writer.name(REDIRECT_URI).value(holder.getRedirectUri());
513                        writer.name(RESPONSE_TYPES);
514                        writer.beginArray();
515                        for (String s : holder.getResponseTypes()) {
516                                writer.value(s);
517                        }
518                        writer.endArray();
519                        writer.name(EXTENSIONS);
520                        writer.beginObject();
521                        for (Entry<String, Serializable> entry : holder.getExtensions().entrySet()) {
522                                // while the extension map itself is Serializable, we enforce storage of Strings
523                                if (entry.getValue() instanceof String) {
524                                        writer.name(entry.getKey()).value((String) entry.getValue());
525                                } else {
526                                        logger.warn("Skipping non-string extension: " + entry);
527                                }
528                        }
529                        writer.endObject();
530
531                        writer.name(SAVED_USER_AUTHENTICATION);
532                        if (holder.getUserAuth() != null) {
533                                writer.beginObject();
534                                writer.name(NAME).value(holder.getUserAuth().getName());
535                                writer.name(SOURCE_CLASS).value(holder.getUserAuth().getSourceClass());
536                                writer.name(AUTHENTICATED).value(holder.getUserAuth().isAuthenticated());
537                                writer.name(AUTHORITIES);
538                                writer.beginArray();
539                                for (GrantedAuthority authority : holder.getUserAuth().getAuthorities()) {
540                                        writer.value(authority.getAuthority());
541                                }
542                                writer.endArray();
543
544                                writer.endObject();
545                        } else {
546                                writer.nullValue();
547                        }
548
549
550                        writer.endObject();
551                        logger.debug("Wrote authentication holder {}", holder.getId());
552                }
553                logger.info("Done writing authentication holders");
554        }
