017package org.mitre.openid.connect.service.impl;
018
019import static org.hamcrest.core.IsNot.not;
020import static org.junit.Assert.assertEquals;
021import static org.junit.Assert.assertThat;
022import static org.junit.matchers.JUnitMatchers.hasItem;
023
024import java.util.ArrayList;
025
026import org.junit.Before;
027import org.junit.Test;
028import org.junit.runner.RunWith;
029import org.mitre.openid.connect.model.DefaultUserInfo;
030import org.mitre.openid.connect.model.UserInfo;
031import org.mitre.openid.connect.repository.UserInfoRepository;
032import org.mockito.InjectMocks;
033import org.mockito.Mock;
034import org.mockito.Mockito;
035import org.mockito.runners.MockitoJUnitRunner;
036import org.springframework.security.core.GrantedAuthority;
037import org.springframework.security.core.userdetails.UserDetails;
038import org.springframework.security.core.userdetails.UsernameNotFoundException;
039
040import com.google.common.collect.Lists;
041
042@RunWith(MockitoJUnitRunner.class)
043public class TestDefaultUserInfoUserDetailsService {
044
045        @InjectMocks
046        private DefaultUserInfoUserDetailsService service = new DefaultUserInfoUserDetailsService();
047
048        @Mock
049        private UserInfoRepository userInfoRepository;
050
051        private UserInfo userInfoAdmin;
052        private UserInfo userInfoRegular;
053
054        private String adminUsername = "username";
055        private String regularUsername = "regular";
056        private String adminSub = "adminSub12d3a1f34a2";
057        private String regularSub = "regularSub652ha23b";
058
059        /**
060         * Initialize the service and the mocked repository.
061         * Initialize 2 users, one of them an admin, for use in unit tests.
062         */
063        @Before
064        public void prepare() {
065
066
067                service.setAdmins(Lists.newArrayList(adminUsername));
068
069                userInfoAdmin = new DefaultUserInfo();
070                userInfoAdmin.setPreferredUsername(adminUsername);
071                userInfoAdmin.setSub(adminSub);
072
073                userInfoRegular = new DefaultUserInfo();
074                userInfoRegular.setPreferredUsername(regularUsername);
075                userInfoRegular.setSub(regularSub);
076
077        }
078
079        /**
080         * Test loading an admin user, ensuring that the UserDetails object returned
081         * has both the ROLE_USER and ROLE_ADMIN authorities.
082         */
083        @Test
084        public void loadByUsername_admin_success() {
085
086                Mockito.when(userInfoRepository.getByUsername(adminUsername)).thenReturn(userInfoAdmin);
087                UserDetails user = service.loadUserByUsername(adminUsername);
088                ArrayList<GrantedAuthority> userAuthorities = Lists.newArrayList(user.getAuthorities());
089                assertThat(userAuthorities, hasItem(DefaultUserInfoUserDetailsService.ROLE_ADMIN));
090                assertThat(userAuthorities, hasItem(DefaultUserInfoUserDetailsService.ROLE_USER));
091                assertEquals(user.getUsername(), adminSub);
092
093        }
094
095        /**
096         * Test loading a regular, non-admin user, ensuring that the returned UserDetails
097         * object has ROLE_USER but *not* ROLE_ADMIN.
098         */
099        @Test
100        public void loadByUsername_regular_success() {
101
102                Mockito.when(userInfoRepository.getByUsername(regularUsername)).thenReturn(userInfoRegular);
103                UserDetails user = service.loadUserByUsername(regularUsername);
104                ArrayList<GrantedAuthority> userAuthorities = Lists.newArrayList(user.getAuthorities());
105                assertThat(userAuthorities, not(hasItem(DefaultUserInfoUserDetailsService.ROLE_ADMIN)));
106                assertThat(userAuthorities, hasItem(DefaultUserInfoUserDetailsService.ROLE_USER));
107                assertEquals(user.getUsername(), regularSub);
108
109        }
110
111        /**
112         * If a user is not found, the loadByUsername method should throw an exception.
113         */
114        @Test(expected = UsernameNotFoundException.class)
115        public void loadByUsername_nullUser() {
116
117                Mockito.when(userInfoRepository.getByUsername(adminUsername)).thenReturn(null);
118                service.loadUserByUsername(adminUsername);
119
120        }
121}
