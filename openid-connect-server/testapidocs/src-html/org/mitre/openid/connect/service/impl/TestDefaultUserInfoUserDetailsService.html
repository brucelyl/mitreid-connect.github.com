001package org.mitre.openid.connect.service.impl;
002
003import static org.hamcrest.core.IsNot.not;
004import static org.junit.Assert.assertEquals;
005import static org.junit.Assert.assertThat;
006import static org.junit.matchers.JUnitMatchers.hasItem;
007
008import java.util.ArrayList;
009
010import org.junit.Before;
011import org.junit.Test;
012import org.junit.runner.RunWith;
013import org.mitre.openid.connect.model.DefaultUserInfo;
014import org.mitre.openid.connect.model.UserInfo;
015import org.mitre.openid.connect.repository.UserInfoRepository;
016import org.mockito.InjectMocks;
017import org.mockito.Mock;
018import org.mockito.Mockito;
019import org.mockito.runners.MockitoJUnitRunner;
020import org.springframework.security.core.GrantedAuthority;
021import org.springframework.security.core.userdetails.UserDetails;
022import org.springframework.security.core.userdetails.UsernameNotFoundException;
023
024import com.google.common.collect.Lists;
025
026@RunWith(MockitoJUnitRunner.class)
027public class TestDefaultUserInfoUserDetailsService {
028
029        @InjectMocks
030        private DefaultUserInfoUserDetailsService service = new DefaultUserInfoUserDetailsService();
031        
032        @Mock
033        private UserInfoRepository userInfoRepository;
034        
035        private UserInfo userInfoAdmin;
036        private UserInfo userInfoRegular;
037        private String adminUsername = "username";
038        private String regularUsername = "regular";
039        private String adminSub = "adminSub12d3a1f34a2";
040        private String regularSub = "regularSub652ha23b";
041        
042        /**
043         * Initialize the service and the mocked repository.
044         * Initialize 2 users, one of them an admin, for use in unit tests.
045         */
046        @Before
047        public void prepare() {
048                
049                
050                service.setAdmins(Lists.newArrayList(adminUsername));
051                
052                userInfoAdmin = new DefaultUserInfo();
053                userInfoAdmin.setPreferredUsername(adminUsername);
054                userInfoAdmin.setSub(adminSub);
055                
056                userInfoRegular = new DefaultUserInfo();
057                userInfoRegular.setPreferredUsername(regularUsername);
058                userInfoRegular.setSub(regularSub);     
059        }
060        
061        /**
062         * Test loading an admin user, ensuring that the UserDetails object returned
063         * has both the ROLE_USER and ROLE_ADMIN authorities.
064         */
065        @Test
066        public void loadByUsername_admin_success() {
067                
068                Mockito.when(userInfoRepository.getByUsername(adminUsername)).thenReturn(userInfoAdmin);
069                UserDetails user = service.loadUserByUsername(adminUsername);
070                ArrayList<GrantedAuthority> userAuthorities = Lists.newArrayList(user.getAuthorities());
071                assertThat(userAuthorities, hasItem(DefaultUserInfoUserDetailsService.ROLE_ADMIN));
072                assertThat(userAuthorities, hasItem(DefaultUserInfoUserDetailsService.ROLE_USER));
073                assertEquals(user.getUsername(), adminSub);
074                
075        }
076        
077        /**
078         * Test loading a regular, non-admin user, ensuring that the returned UserDetails
079         * object has ROLE_USER but *not* ROLE_ADMIN.
080         */
081        @Test
082        public void loadByUsername_regular_success() {
083                
084                Mockito.when(userInfoRepository.getByUsername(regularUsername)).thenReturn(userInfoRegular);
085                UserDetails user = service.loadUserByUsername(regularUsername);
086                ArrayList<GrantedAuthority> userAuthorities = Lists.newArrayList(user.getAuthorities());
087                assertThat(userAuthorities, not(hasItem(DefaultUserInfoUserDetailsService.ROLE_ADMIN)));
088                assertThat(userAuthorities, hasItem(DefaultUserInfoUserDetailsService.ROLE_USER));
089                assertEquals(user.getUsername(), regularSub);
090                
091        }       
092        
093        /**
094         * If a user is not found, the loadByUsername method should throw an exception.
095         */
096        @Test(expected = UsernameNotFoundException.class)
097        public void loadByUsername_nullUser() {
098                
099                Mockito.when(userInfoRepository.getByUsername(adminUsername)).thenReturn(null);
100                service.loadUserByUsername(adminUsername);
101                
102        }
103        
104}
