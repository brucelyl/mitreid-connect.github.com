001package org.mitre.openid.connect.service.impl;
002
003import static org.mockito.Matchers.any;
004import static org.mockito.Mockito.never;
005import static org.mockito.Mockito.times;
006
007import java.util.HashSet;
008import java.util.Set;
009
010import org.junit.Before;
011import org.junit.Test;
012import org.mitre.oauth2.model.ClientDetailsEntity;
013import org.mitre.openid.connect.model.ApprovedSite;
014import org.mitre.openid.connect.repository.ApprovedSiteRepository;
015import org.mitre.openid.connect.service.ApprovedSiteService;
016import org.mockito.Mockito;
017import org.springframework.test.annotation.Rollback;
018
019import com.google.common.collect.Sets;
020
021public class TestDefaultApprovedSiteService {
022
023        private ApprovedSite site1;
024        private ApprovedSite site2;
025        private ApprovedSite site3;
026        
027        private ClientDetailsEntity client;
028        private final String clientId = "client";
029        
030        private ApprovedSiteService service;
031        private ApprovedSiteRepository repository;
032        
033        /**
034         * Initialize the service and repository mock. Initialize a client and
035         * several ApprovedSite objects for use in unit tests.
036         */
037        @Before
038        public void prepare() {
039                
040                client = new ClientDetailsEntity();
041                client.setClientId(clientId);
042                
043                site1 = new ApprovedSite();
044                site1.setId(1L);
045                site1.setUserId("user1");
046                site1.setClientId("other");
047                
048                site2 = new ApprovedSite();
049                site2.setId(2L);
050                site2.setUserId("user1");
051                site2.setClientId(clientId);
052                
053                site3 = new ApprovedSite();
054                site3.setId(3L);
055                site3.setUserId("user2");
056                site3.setClientId(clientId);
057                
058                repository = Mockito.mock(ApprovedSiteRepository.class);
059                
060                service = new DefaultApprovedSiteService(repository);
061        }
062        
063        /**
064         * Test clearing approved sites for a client that has 2 stored approved sites.
065         * Ensure that the repository's remove() method is called twice.
066         */
067        @Test
068        public void clearApprovedSitesForClient_success() {
069                Set<ApprovedSite> setToReturn = Sets.newHashSet(site2, site3);
070                Mockito.when(repository.getByClientId(client.getClientId())).thenReturn(setToReturn);
071                
072                service.clearApprovedSitesForClient(client);
073                
074                Mockito.verify(repository, times(2)).remove(any(ApprovedSite.class));
075        }
076        
077        /**
078         * Test clearing approved sites for a client that doesn't have any stored approved
079         * sites. Ensure that the repository's remove() method is never called in this case.
080         */
081        @Test
082        @Rollback
083        public void clearApprovedSitesForClient_null() {
084                String otherId = "a different id";
085                client.setClientId(otherId);
086                service.clearApprovedSitesForClient(client);
087                Mockito.when(repository.getByClientId(otherId)).thenReturn(new HashSet<ApprovedSite>());
088                Mockito.verify(repository, never()).remove(any(ApprovedSite.class));
089        }
090        
091        
092}
