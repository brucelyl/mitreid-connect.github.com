118        /**
119         * Test method for {@link org.mitre.openid.connect.service.impl.UUIDPairwiseIdentiferService#getIdentifier(org.mitre.openid.connect.model.UserInfo, org.mitre.oauth2.model.ClientDetailsEntity)}.
120         */
121        @Test
122        public void testGetIdentifier_existingEqual() {
123
124                Mockito.when(pairwiseIdentifierRepository.getBySectorIdentifier(regularSub, sectorHost12)).thenReturn(savedPairwiseIdentifier);
125                
126                String pairwise1 = service.getIdentifier(userInfoRegular, pairwiseClient1);
127                String pairwise2 = service.getIdentifier(userInfoRegular, pairwiseClient2);
128
129                assertEquals(pairwiseSub, pairwise1);
130                assertEquals(pairwiseSub, pairwise2);
131                
132        }
133        
134        @Test
135        public void testGetIdentifier_newEqual() {
136
137                String pairwise1 = service.getIdentifier(userInfoRegular, pairwiseClient1);
138                Mockito.verify(pairwiseIdentifierRepository, Mockito.atLeast(1)).save(Mockito.any(PairwiseIdentifier.class));
139                
140                PairwiseIdentifier pairwiseId = new PairwiseIdentifier();
141                pairwiseId.setUserSub(regularSub);
142                pairwiseId.setIdentifier(pairwise1);
143                pairwiseId.setSectorIdentifier(sectorHost12);
144                
145                Mockito.when(pairwiseIdentifierRepository.getBySectorIdentifier(regularSub, sectorHost12)).thenReturn(pairwiseId);
146                
147                String pairwise2 = service.getIdentifier(userInfoRegular, pairwiseClient2);
148
149                assertNotSame(pairwiseSub, pairwise1);
150                assertNotSame(pairwiseSub, pairwise2);
151
152                assertEquals(pairwise1, pairwise2);
153                
154                // see if the pairwise id's are actual UUIDs
155                UUID uudi1 = UUID.fromString(pairwise1);
156                UUID uuid2 = UUID.fromString(pairwise2);
157                
158        }
159        
160        @Test
161        public void testGetIdentifer_unique() {
162                String pairwise1 = service.getIdentifier(userInfoRegular, pairwiseClient1);
163                String pairwise3 = service.getIdentifier(userInfoRegular, pairwiseClient3);
164                String pairwise4 = service.getIdentifier(userInfoRegular, pairwiseClient4);
165                
166                // make sure nothing's equal
167                assertNotSame(pairwise1, pairwise3);
168                assertNotSame(pairwise1, pairwise4);
169                assertNotSame(pairwise3, pairwise4);
170                
171                // see if the pairwise id's are actual UUIDs
172                UUID uudi1 = UUID.fromString(pairwise1);
173                UUID uudi3 = UUID.fromString(pairwise3);
174                UUID uudi4 = UUID.fromString(pairwise4);
175
176        }
