017package org.mitre.openid.connect.service.impl;
018
019import java.util.Set;
020
021import org.junit.Before;
022import org.junit.Test;
023import org.junit.runner.RunWith;
024import org.mitre.openid.connect.model.BlacklistedSite;
025import org.mitre.openid.connect.repository.BlacklistedSiteRepository;
026import org.mockito.InjectMocks;
027import org.mockito.Mock;
028import org.mockito.Mockito;
029import org.mockito.runners.MockitoJUnitRunner;
030
031import com.google.common.collect.Sets;
032
033import static org.junit.Assert.assertFalse;
034import static org.junit.Assert.assertTrue;
035import static org.mockito.Mockito.times;
036
037/**
038 * @author wkim
039 *
040 */
041@RunWith(MockitoJUnitRunner.class)
042public class TestDefaultBlacklistedSiteService {
043
044        private BlacklistedSite site1;
045        private BlacklistedSite site2;
046
047        private String uri1 = "black1";
048        private String uri2 = "black2";
049        private String uri3 = "not-black";
050
051        private Set<BlacklistedSite> blackListedSitesSet;
052
053        @Mock
054        private BlacklistedSiteRepository mockRepository;
055
056        @InjectMocks
057        private DefaultBlacklistedSiteService service = new DefaultBlacklistedSiteService();
058
059        /**
060         * @throws java.lang.Exception
061         */
062        @Before
063        public void prepare() throws Exception {
064
065                site1 = new BlacklistedSite();
066                site2 = new BlacklistedSite();
067
068                site1.setUri(uri1);
069                site2.setUri(uri2);
070
071                blackListedSitesSet = Sets.newHashSet(site1, site2);
072        }
073
074        /**
075         * Test finding blacklisted sites from the repository.
076         */
077        @Test
078        public void isBlacklisted_yes() {
079
080                Mockito.when(mockRepository.getAll()).thenReturn(blackListedSitesSet);
081
082                assertTrue(service.isBlacklisted(uri1));
083                assertTrue(service.isBlacklisted(uri2));
084
085                Mockito.verify(mockRepository, times(2)).getAll();
086        }
087
088        /**
089         * Tests for finding a site that is not blacklisted in the repository.
090         */
091        @Test
092        public void isBlacklisted_no() {
093
094                Mockito.when(mockRepository.getAll()).thenReturn(blackListedSitesSet);
095
096                assertFalse(service.isBlacklisted(uri3));
097
098                Mockito.verify(mockRepository).getAll();
099        }
100
101}
