213        /**
214         * Checks to see that the expiration date of new tokens is being set accurately to within some delta for time skew.
215         */
216        @Test
217        public void createAccessToken_expiration() {
218
219                Integer accessTokenValiditySeconds = 3600;
220                Integer refreshTokenValiditySeconds = 600;
221
222                Mockito.when(client.getAccessTokenValiditySeconds()).thenReturn(accessTokenValiditySeconds);
223                Mockito.when(client.getRefreshTokenValiditySeconds()).thenReturn(refreshTokenValiditySeconds);
224
225                long start = System.currentTimeMillis();
226                OAuth2AccessTokenEntity token = service.createAccessToken(authentication);
227                long end = System.currentTimeMillis();
228
229                // Accounting for some delta for time skew on either side.
230                Date lowerBoundAccessTokens = new Date(start + (accessTokenValiditySeconds * 1000L) - DELTA);
231                Date upperBoundAccessTokens = new Date(end + (accessTokenValiditySeconds * 1000L) + DELTA);
232                Date lowerBoundRefreshTokens = new Date(start + (refreshTokenValiditySeconds * 1000L) - DELTA);
233                Date upperBoundRefreshTokens = new Date(end + (refreshTokenValiditySeconds * 1000L) + DELTA);
234
235                assertTrue(token.getExpiration().after(lowerBoundAccessTokens) && token.getExpiration().before(upperBoundAccessTokens));
236                assertTrue(token.getRefreshToken().getExpiration().after(lowerBoundRefreshTokens) && token.getRefreshToken().getExpiration().before(upperBoundRefreshTokens));
237        }
