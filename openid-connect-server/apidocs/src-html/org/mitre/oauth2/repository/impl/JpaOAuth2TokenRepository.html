017package org.mitre.oauth2.repository.impl;
018
019import java.util.Date;
020import java.util.LinkedHashSet;
021import java.util.List;
022import java.util.Set;
023
024import javax.persistence.EntityManager;
025import javax.persistence.PersistenceContext;
026import javax.persistence.TypedQuery;
027
028import org.mitre.oauth2.model.ClientDetailsEntity;
029import org.mitre.oauth2.model.OAuth2AccessTokenEntity;
030import org.mitre.oauth2.model.OAuth2RefreshTokenEntity;
031import org.mitre.oauth2.repository.OAuth2TokenRepository;
032import org.mitre.util.jpa.JpaUtil;
033import org.springframework.security.oauth2.provider.OAuth2Authentication;
034import org.springframework.stereotype.Repository;
035import org.springframework.transaction.annotation.Transactional;
036
037@Repository
038public class JpaOAuth2TokenRepository implements OAuth2TokenRepository {
039
040        private static final int MAXEXPIREDRESULTS = 1000;
041
042        @PersistenceContext
043        private EntityManager manager;
044
045        @Override
046        public Set<OAuth2AccessTokenEntity> getAllAccessTokens() {
047                TypedQuery<OAuth2AccessTokenEntity> query = manager.createNamedQuery("OAuth2AccessTokenEntity.getAll", OAuth2AccessTokenEntity.class);
048                return new LinkedHashSet<OAuth2AccessTokenEntity>(query.getResultList());
049        }
050
051        @Override
052        public Set<OAuth2RefreshTokenEntity> getAllRefreshTokens() {
053                TypedQuery<OAuth2RefreshTokenEntity> query = manager.createNamedQuery("OAuth2RefreshTokenEntity.getAll", OAuth2RefreshTokenEntity.class);
054                return new LinkedHashSet<OAuth2RefreshTokenEntity>(query.getResultList());
055        }
056
057
058        @Override
059        public OAuth2AccessTokenEntity getAccessTokenByValue(String accessTokenValue) {
060                TypedQuery<OAuth2AccessTokenEntity> query = manager.createNamedQuery("OAuth2AccessTokenEntity.getByTokenValue", OAuth2AccessTokenEntity.class);
061                query.setParameter("tokenValue", accessTokenValue);
062                return JpaUtil.getSingleResult(query.getResultList());
063        }
064
065        @Override
066        public OAuth2AccessTokenEntity getAccessTokenById(Long id) {
067                return manager.find(OAuth2AccessTokenEntity.class, id);
068        }
069
070        @Override
071        @Transactional
072        public OAuth2AccessTokenEntity saveAccessToken(OAuth2AccessTokenEntity token) {
073                return JpaUtil.saveOrUpdate(token.getId(), manager, token);
074        }
075
076        @Override
077        @Transactional
078        public void removeAccessToken(OAuth2AccessTokenEntity accessToken) {
079                OAuth2AccessTokenEntity found = getAccessTokenByValue(accessToken.getValue());
080                if (found != null) {
081                        manager.remove(found);
082                } else {
083                        throw new IllegalArgumentException("Access token not found: " + accessToken);
084                }
085        }
086
087        @Override
088        @Transactional
089        public void clearAccessTokensForRefreshToken(OAuth2RefreshTokenEntity refreshToken) {
090                TypedQuery<OAuth2AccessTokenEntity> query = manager.createNamedQuery("OAuth2AccessTokenEntity.getByRefreshToken", OAuth2AccessTokenEntity.class);
091                query.setParameter("refreshToken", refreshToken);
092                List<OAuth2AccessTokenEntity> accessTokens = query.getResultList();
093                for (OAuth2AccessTokenEntity accessToken : accessTokens) {
094                        removeAccessToken(accessToken);
095                }
096        }
097
098        @Override
099        public OAuth2RefreshTokenEntity getRefreshTokenByValue(String refreshTokenValue) {
100                TypedQuery<OAuth2RefreshTokenEntity> query = manager.createNamedQuery("OAuth2RefreshTokenEntity.getByTokenValue", OAuth2RefreshTokenEntity.class);
101                query.setParameter("tokenValue", refreshTokenValue);
102                return JpaUtil.getSingleResult(query.getResultList());
103        }
104
105        @Override
106        public OAuth2RefreshTokenEntity getRefreshTokenById(Long id) {
107                return manager.find(OAuth2RefreshTokenEntity.class, id);
108        }
109
110        @Override
111        @Transactional
112        public OAuth2RefreshTokenEntity saveRefreshToken(OAuth2RefreshTokenEntity refreshToken) {
113                return JpaUtil.saveOrUpdate(refreshToken.getId(), manager, refreshToken);
114        }
115
116        @Override
117        @Transactional
118        public void removeRefreshToken(OAuth2RefreshTokenEntity refreshToken) {
119                OAuth2RefreshTokenEntity found = getRefreshTokenByValue(refreshToken.getValue());
120                if (found != null) {
121                        manager.remove(found);
122                } else {
123                        throw new IllegalArgumentException("Refresh token not found: " + refreshToken);
124                }
125        }
126
127        @Override
128        @Transactional
129        public void clearTokensForClient(ClientDetailsEntity client) {
130                TypedQuery<OAuth2AccessTokenEntity> queryA = manager.createNamedQuery("OAuth2AccessTokenEntity.getByClient", OAuth2AccessTokenEntity.class);
131                queryA.setParameter("client", client);
132                List<OAuth2AccessTokenEntity> accessTokens = queryA.getResultList();
133                for (OAuth2AccessTokenEntity accessToken : accessTokens) {
134                        removeAccessToken(accessToken);
135                }
136                TypedQuery<OAuth2RefreshTokenEntity> queryR = manager.createNamedQuery("OAuth2RefreshTokenEntity.getByClient", OAuth2RefreshTokenEntity.class);
137                queryR.setParameter("client", client);
138                List<OAuth2RefreshTokenEntity> refreshTokens = queryR.getResultList();
139                for (OAuth2RefreshTokenEntity refreshToken : refreshTokens) {
140                        removeRefreshToken(refreshToken);
141                }
142        }
143
144        /* (non-Javadoc)
145         * @see org.mitre.oauth2.repository.OAuth2TokenRepository#getAccessTokensForClient(org.mitre.oauth2.model.ClientDetailsEntity)
146         */
147        @Override
148        public List<OAuth2AccessTokenEntity> getAccessTokensForClient(ClientDetailsEntity client) {
149                TypedQuery<OAuth2AccessTokenEntity> queryA = manager.createNamedQuery("OAuth2AccessTokenEntity.getByClient", OAuth2AccessTokenEntity.class);
150                queryA.setParameter("client", client);
151                List<OAuth2AccessTokenEntity> accessTokens = queryA.getResultList();
152                return accessTokens;
153        }
154
155        /* (non-Javadoc)
156         * @see org.mitre.oauth2.repository.OAuth2TokenRepository#getRefreshTokensForClient(org.mitre.oauth2.model.ClientDetailsEntity)
157         */
158        @Override
159        public List<OAuth2RefreshTokenEntity> getRefreshTokensForClient(ClientDetailsEntity client) {
160                TypedQuery<OAuth2RefreshTokenEntity> queryR = manager.createNamedQuery("OAuth2RefreshTokenEntity.getByClient", OAuth2RefreshTokenEntity.class);
161                queryR.setParameter("client", client);
162                List<OAuth2RefreshTokenEntity> refreshTokens = queryR.getResultList();
163                return refreshTokens;
164        }
165
166        @Override
167        public OAuth2AccessTokenEntity getByAuthentication(OAuth2Authentication auth) {
168                TypedQuery<OAuth2AccessTokenEntity> queryA = manager.createNamedQuery("OAuth2AccessTokenEntity.getByAuthentication", OAuth2AccessTokenEntity.class);
169                queryA.setParameter("authentication", auth);
170                List<OAuth2AccessTokenEntity> accessTokens = queryA.getResultList();
171                return JpaUtil.getSingleResult(accessTokens);
172        }
173
174        /* (non-Javadoc)
175         * @see org.mitre.oauth2.repository.OAuth2TokenRepository#getAccessTokenForIdToken(org.mitre.oauth2.model.OAuth2AccessTokenEntity)
176         */
177        @Override
178        public OAuth2AccessTokenEntity getAccessTokenForIdToken(OAuth2AccessTokenEntity idToken) {
179                TypedQuery<OAuth2AccessTokenEntity> queryA = manager.createNamedQuery("OAuth2AccessTokenEntity.getByIdToken", OAuth2AccessTokenEntity.class);
180                queryA.setParameter("idToken", idToken);
181                List<OAuth2AccessTokenEntity> accessTokens = queryA.getResultList();
182                return JpaUtil.getSingleResult(accessTokens);
183        }
184
185        @Override
186        public Set<OAuth2AccessTokenEntity> getAllExpiredAccessTokens() {
187                TypedQuery<OAuth2AccessTokenEntity> query = manager.createNamedQuery("OAuth2AccessTokenEntity.getAllExpiredByDate", OAuth2AccessTokenEntity.class);
188                query.setParameter("date", new Date());
189                query.setMaxResults(MAXEXPIREDRESULTS);
190                return new LinkedHashSet<OAuth2AccessTokenEntity>(query.getResultList());
191        }
192
193        @Override
194        public Set<OAuth2RefreshTokenEntity> getAllExpiredRefreshTokens() {
195                TypedQuery<OAuth2RefreshTokenEntity> query = manager.createNamedQuery("OAuth2RefreshTokenEntity.getAllExpiredByDate", OAuth2RefreshTokenEntity.class);
196                query.setParameter("date", new Date());
197                query.setMaxResults(MAXEXPIREDRESULTS);
198                return new LinkedHashSet<OAuth2RefreshTokenEntity>(query.getResultList());
199        }
