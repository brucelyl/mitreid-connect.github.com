051@Repository

058        @PersistenceContext(unitName="defaultPersistenceUnit")
059        private EntityManager manager;

061        @Override
062        public Set<OAuth2AccessTokenEntity> getAllAccessTokens() {
063                TypedQuery<OAuth2AccessTokenEntity> query = manager.createNamedQuery(OAuth2AccessTokenEntity.QUERY_ALL, OAuth2AccessTokenEntity.class);
064                return new LinkedHashSet<>(query.getResultList());
065        }
066
067        @Override
068        public Set<OAuth2RefreshTokenEntity> getAllRefreshTokens() {
069                TypedQuery<OAuth2RefreshTokenEntity> query = manager.createNamedQuery(OAuth2RefreshTokenEntity.QUERY_ALL, OAuth2RefreshTokenEntity.class);
070                return new LinkedHashSet<>(query.getResultList());
071        }

073
074        @Override
075        public OAuth2AccessTokenEntity getAccessTokenByValue(String accessTokenValue) {
076                try {
077                        JWT jwt = JWTParser.parse(accessTokenValue);
078                        TypedQuery<OAuth2AccessTokenEntity> query = manager.createNamedQuery(OAuth2AccessTokenEntity.QUERY_BY_TOKEN_VALUE, OAuth2AccessTokenEntity.class);
079                        query.setParameter(OAuth2AccessTokenEntity.PARAM_TOKEN_VALUE, jwt);
080                        return JpaUtil.getSingleResult(query.getResultList());
081                } catch (ParseException e) {
082                        return null;
083                }
084        }
085
086        @Override
087        public OAuth2AccessTokenEntity getAccessTokenById(Long id) {
088                return manager.find(OAuth2AccessTokenEntity.class, id);
089        }
090
091        @Override
092        @Transactional(value="defaultTransactionManager")
093        public OAuth2AccessTokenEntity saveAccessToken(OAuth2AccessTokenEntity token) {
094                return JpaUtil.saveOrUpdate(token.getId(), manager, token);
095        }
096
097        @Override
098        @Transactional(value="defaultTransactionManager")
099        public void removeAccessToken(OAuth2AccessTokenEntity accessToken) {
100                OAuth2AccessTokenEntity found = getAccessTokenById(accessToken.getId());
101                if (found != null) {
102                        manager.remove(found);
103                } else {
104                        throw new IllegalArgumentException("Access token not found: " + accessToken);
105                }
106        }
107
108        @Override
109        @Transactional(value="defaultTransactionManager")
110        public void clearAccessTokensForRefreshToken(OAuth2RefreshTokenEntity refreshToken) {
111                TypedQuery<OAuth2AccessTokenEntity> query = manager.createNamedQuery(OAuth2AccessTokenEntity.QUERY_BY_REFRESH_TOKEN, OAuth2AccessTokenEntity.class);
112                query.setParameter(OAuth2AccessTokenEntity.PARAM_REFERSH_TOKEN, refreshToken);
113                List<OAuth2AccessTokenEntity> accessTokens = query.getResultList();
114                for (OAuth2AccessTokenEntity accessToken : accessTokens) {
115                        removeAccessToken(accessToken);
116                }
117        }

153        @Override
154        @Transactional(value="defaultTransactionManager")
155        public void clearTokensForClient(ClientDetailsEntity client) {
156                TypedQuery<OAuth2AccessTokenEntity> queryA = manager.createNamedQuery(OAuth2AccessTokenEntity.QUERY_BY_CLIENT, OAuth2AccessTokenEntity.class);
157                queryA.setParameter(OAuth2AccessTokenEntity.PARAM_CLIENT, client);
158                List<OAuth2AccessTokenEntity> accessTokens = queryA.getResultList();
159                for (OAuth2AccessTokenEntity accessToken : accessTokens) {
160                        removeAccessToken(accessToken);
161                }
162                TypedQuery<OAuth2RefreshTokenEntity> queryR = manager.createNamedQuery(OAuth2RefreshTokenEntity.QUERY_BY_CLIENT, OAuth2RefreshTokenEntity.class);
163                queryR.setParameter(OAuth2RefreshTokenEntity.PARAM_CLIENT, client);
164                List<OAuth2RefreshTokenEntity> refreshTokens = queryR.getResultList();
165                for (OAuth2RefreshTokenEntity refreshToken : refreshTokens) {
166                        removeRefreshToken(refreshToken);
167                }
168        }

230        /* (non-Javadoc)
231         * @see org.mitre.oauth2.repository.OAuth2TokenRepository#clearDuplicateAccessTokens()
232         */
233        @Override
234        @Transactional(value="defaultTransactionManager")
235        public void clearDuplicateAccessTokens() {
236
237                Query query = manager.createQuery("select a.jwt, count(1) as c from OAuth2AccessTokenEntity a GROUP BY a.jwt HAVING count(1) > 1");
238                @SuppressWarnings("unchecked")
239                List<Object[]> resultList = query.getResultList();
240                List<JWT> values = new ArrayList<>();
241                for (Object[] r : resultList) {
242                        logger.warn("Found duplicate access tokens: {}, {}", ((JWT)r[0]).serialize(), r[1]);
243                        values.add((JWT) r[0]);
244                }
245                if (values.size() > 0) {
246                        CriteriaBuilder cb = manager.getCriteriaBuilder();
247                        CriteriaDelete<OAuth2AccessTokenEntity> criteriaDelete = cb.createCriteriaDelete(OAuth2AccessTokenEntity.class);
248                        Root<OAuth2AccessTokenEntity> root = criteriaDelete.from(OAuth2AccessTokenEntity.class);
249                        criteriaDelete.where(root.get("jwt").in(values));
250                        int result = manager.createQuery(criteriaDelete).executeUpdate();
251                        logger.warn("Deleted {} duplicate access tokens", result);
252                }
253        }
254
255        /* (non-Javadoc)
256         * @see org.mitre.oauth2.repository.OAuth2TokenRepository#clearDuplicateRefreshTokens()
257         */
258        @Override
259        @Transactional(value="defaultTransactionManager")
260        public void clearDuplicateRefreshTokens() {
261                Query query = manager.createQuery("select a.jwt, count(1) as c from OAuth2RefreshTokenEntity a GROUP BY a.jwt HAVING count(1) > 1");
262                @SuppressWarnings("unchecked")
263                List<Object[]> resultList = query.getResultList();
264                List<JWT> values = new ArrayList<>();
265                for (Object[] r : resultList) {
266                        logger.warn("Found duplicate refresh tokens: {}, {}", ((JWT)r[0]).serialize(), r[1]);
267                        values.add((JWT) r[0]);
268                }
269                if (values.size() > 0) {
270                        CriteriaBuilder cb = manager.getCriteriaBuilder();
271                        CriteriaDelete<OAuth2RefreshTokenEntity> criteriaDelete = cb.createCriteriaDelete(OAuth2RefreshTokenEntity.class);
272                        Root<OAuth2RefreshTokenEntity> root = criteriaDelete.from(OAuth2RefreshTokenEntity.class);
273                        criteriaDelete.where(root.get("jwt").in(values));
274                        int result = manager.createQuery(criteriaDelete).executeUpdate();
275                        logger.warn("Deleted {} duplicate refresh tokens", result);
276                }
