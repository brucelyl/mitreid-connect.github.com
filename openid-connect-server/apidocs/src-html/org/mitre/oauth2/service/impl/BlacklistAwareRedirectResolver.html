004package org.mitre.oauth2.service.impl;
005
006import org.mitre.openid.connect.service.BlacklistedSiteService;
007import org.springframework.beans.factory.annotation.Autowired;
008import org.springframework.security.oauth2.common.exceptions.InvalidRequestException;
009import org.springframework.security.oauth2.common.exceptions.OAuth2Exception;
010import org.springframework.security.oauth2.provider.ClientDetails;
011import org.springframework.security.oauth2.provider.endpoint.DefaultRedirectResolver;
012import org.springframework.stereotype.Component;
013
014import com.google.common.base.Strings;
015
016/**
017 * 
018 * A redirect resolver that knows how to check against the blacklisted URIs
019 * for forbidden values. Can be configured to do strict string matching also.
020 * 
021 * @author jricher
022 *
023 */
024@Component("blacklistAwareRedirectResolver")
025public class BlacklistAwareRedirectResolver extends DefaultRedirectResolver {
026
027        @Autowired
028        private BlacklistedSiteService blacklistService;
029        
030        private boolean strictMatch = false;
031        
032        /* (non-Javadoc)
033         * @see org.springframework.security.oauth2.provider.endpoint.RedirectResolver#resolveRedirect(java.lang.String, org.springframework.security.oauth2.provider.ClientDetails)
034         */
035        @Override
036        public String resolveRedirect(String requestedRedirect, ClientDetails client) throws OAuth2Exception {
037                String redirect = super.resolveRedirect(requestedRedirect, client);
038                if (blacklistService.isBlacklisted(redirect)) {
039                        // don't let it go through
040                        throw new InvalidRequestException("The supplied redirect_uri is not allowed on this server.");
041                } else {
042                        // not blacklisted, passed the parent test, we're fine
043                        return redirect;
044                }
045        }
046
047        /* (non-Javadoc)
048         * @see org.springframework.security.oauth2.provider.endpoint.DefaultRedirectResolver#redirectMatches(java.lang.String, java.lang.String)
049         */
050        @Override
051        protected boolean redirectMatches(String requestedRedirect, String redirectUri) {
052                
053                if (isStrictMatch()) {
054                        // we're doing a strict string match for all clients
055                        return Strings.nullToEmpty(requestedRedirect).equals(redirectUri);
056                } else {
057                        // otherwise do the prefix-match from the library
058                        return super.redirectMatches(requestedRedirect, redirectUri);
059                }
060                
061        }
062
063        /**
064         * @return the strictMatch
065         */
066        public boolean isStrictMatch() {
067                return strictMatch;
068        }
069
070        /**
071         * Set this to true to require exact string matches for all redirect URIs. (Default is false)
072         * 
073         * @param strictMatch the strictMatch to set
074         */
075        public void setStrictMatch(boolean strictMatch) {
076                this.strictMatch = strictMatch;
077        }
078
079        
080        
081}
