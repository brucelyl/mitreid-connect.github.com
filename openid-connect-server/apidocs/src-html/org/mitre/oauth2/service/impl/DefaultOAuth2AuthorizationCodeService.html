020package org.mitre.oauth2.service.impl;
021
022import java.util.Collection;
023import java.util.Date;
024
025import org.mitre.data.AbstractPageOperationTemplate;
026import org.mitre.oauth2.model.AuthenticationHolderEntity;
027import org.mitre.oauth2.model.AuthorizationCodeEntity;
028import org.mitre.oauth2.repository.AuthenticationHolderRepository;
029import org.mitre.oauth2.repository.AuthorizationCodeRepository;
030import org.slf4j.Logger;
031import org.slf4j.LoggerFactory;
032import org.springframework.beans.factory.annotation.Autowired;
033import org.springframework.security.oauth2.common.exceptions.InvalidGrantException;
034import org.springframework.security.oauth2.common.util.RandomValueStringGenerator;
035import org.springframework.security.oauth2.provider.OAuth2Authentication;
036import org.springframework.security.oauth2.provider.code.AuthorizationCodeServices;
037import org.springframework.stereotype.Service;
038import org.springframework.transaction.annotation.Transactional;
039
040/**
041 * Database-backed, random-value authorization code service implementation.
042 *
043 * @author aanganes
044 *
045 */
046@Service("defaultOAuth2AuthorizationCodeService")
047public class DefaultOAuth2AuthorizationCodeService implements AuthorizationCodeServices {
048        // Logger for this class
049        private static final Logger logger = LoggerFactory.getLogger(DefaultOAuth2AuthorizationCodeService.class);
050
051        @Autowired
052        private AuthorizationCodeRepository repository;
053
054        @Autowired
055        private AuthenticationHolderRepository authenticationHolderRepository;
056
057        private int authCodeExpirationSeconds = 60 * 5; // expire in 5 minutes by default
058
059        private RandomValueStringGenerator generator = new RandomValueStringGenerator();
060
061        /**
062         * Generate a random authorization code and create an AuthorizationCodeEntity,
063         * which will be stored in the repository.
064         *
065         * @param authentication        the authentication of the current user, to be retrieved when the
066         *                                                      code is consumed
067         * @return                                      the authorization code
068         */
069        @Override
070        @Transactional(value="defaultTransactionManager")
071        public String createAuthorizationCode(OAuth2Authentication authentication) {
072                String code = generator.generate();
073
074                // attach the authorization so that we can look it up later
075                AuthenticationHolderEntity authHolder = new AuthenticationHolderEntity();
076                authHolder.setAuthentication(authentication);
077                authHolder = authenticationHolderRepository.save(authHolder);
078
079                // set the auth code to expire
080                Date expiration = new Date(System.currentTimeMillis() + (getAuthCodeExpirationSeconds() * 1000L));
081
082                AuthorizationCodeEntity entity = new AuthorizationCodeEntity(code, authHolder, expiration);
083                repository.save(entity);
084
085                return code;
086        }
087
088        /**
089         * Consume a given authorization code.
090         * Match the provided string to an AuthorizationCodeEntity. If one is found, return
091         * the authentication associated with the code. If one is not found, throw an
092         * InvalidGrantException.
093         *
094         * @param code          the authorization code
095         * @return                      the authentication that made the original request
096         * @throws                      InvalidGrantException, if an AuthorizationCodeEntity is not found with the given value
097         */
098        @Override
099        public OAuth2Authentication consumeAuthorizationCode(String code) throws InvalidGrantException {
100
101                AuthorizationCodeEntity result = repository.getByCode(code);
102
103                if (result == null) {
104                        throw new InvalidGrantException("JpaAuthorizationCodeRepository: no authorization code found for value " + code);
105                }
106
107                OAuth2Authentication auth = result.getAuthenticationHolder().getAuthentication();
108
109                repository.remove(result);
110
111                return auth;
112        }
113
114        /**
115         * Find and remove all expired auth codes.
116         */
117        @Transactional(value="defaultTransactionManager")
118        public void clearExpiredAuthorizationCodes() {
119
120                new AbstractPageOperationTemplate<AuthorizationCodeEntity>("clearExpiredAuthorizationCodes"){
121                        @Override
122                        public Collection<AuthorizationCodeEntity> fetchPage() {
123                                return repository.getExpiredCodes();
124                        }
125
126                        @Override
127                        protected void doOperation(AuthorizationCodeEntity item) {
128                                repository.remove(item);
129                        }
130                }.execute();
131        }
132
133        /**
134         * @return the repository
135         */
136        public AuthorizationCodeRepository getRepository() {
137                return repository;
138        }
139
140        /**
141         * @param repository the repository to set
142         */
143        public void setRepository(AuthorizationCodeRepository repository) {
144                this.repository = repository;
145        }
146
147        /**
148         * @return the authCodeExpirationSeconds
149         */
150        public int getAuthCodeExpirationSeconds() {
151                return authCodeExpirationSeconds;
152        }
153
154        /**
155         * @param authCodeExpirationSeconds the authCodeExpirationSeconds to set
156         */
157        public void setAuthCodeExpirationSeconds(int authCodeExpirationSeconds) {
158                this.authCodeExpirationSeconds = authCodeExpirationSeconds;
159        }
160
161}
