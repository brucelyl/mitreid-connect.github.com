104        @Override
105        public Set<OAuth2AccessTokenEntity> getAllAccessTokensForUser(String id) {
106
107                Set<OAuth2AccessTokenEntity> all = tokenRepository.getAllAccessTokens();
108                Set<OAuth2AccessTokenEntity> results = Sets.newLinkedHashSet();
109
110                for (OAuth2AccessTokenEntity token : all) {
111                        if (clearExpiredAccessToken(token) != null && token.getAuthenticationHolder().getAuthentication().getName().equals(id)) {
112                                results.add(token);
113                        }
114                }
115
116                return results;
117        }

144        /**
145         * Utility function to delete an access token that's expired before returning it.
146         * @param token the token to check
147         * @return null if the token is null or expired, the input token (unchanged) if it hasn't
148         */
149        private OAuth2AccessTokenEntity clearExpiredAccessToken(OAuth2AccessTokenEntity token) {
150                if (token == null) {
151                        return null;
152                } else if (token.isExpired()) {
153                        // immediately revoke expired token
154                        logger.debug("Clearing expired access token: " + token.getValue());
155                        revokeAccessToken(token);
156                        return null;
157                } else {
158                        return token;
159                }
160        }

162        /**
163         * Utility function to delete a refresh token that's expired before returning it.
164         * @param token the token to check
165         * @return null if the token is null or expired, the input token (unchanged) if it hasn't
166         */
167        private OAuth2RefreshTokenEntity clearExpiredRefreshToken(OAuth2RefreshTokenEntity token) {
168                if (token == null) {
169                        return null;
170                } else if (token.isExpired()) {
171                        // immediately revoke expired token
172                        logger.debug("Clearing expired refresh token: " + token.getValue());
173                        revokeRefreshToken(token);
174                        return null;
175                } else {
176                        return token;
177                }
178        }

180        @Override
181        @Transactional(value="defaultTransactionManager")
182        public OAuth2AccessTokenEntity createAccessToken(OAuth2Authentication authentication) throws AuthenticationException, InvalidClientException {
183                if (authentication != null && authentication.getOAuth2Request() != null) {
184                        // look up our client
185                        OAuth2Request request = authentication.getOAuth2Request();
186
187                        ClientDetailsEntity client = clientDetailsService.loadClientByClientId(request.getClientId());
188
189                        if (client == null) {
190                                throw new InvalidClientException("Client not found: " + request.getClientId());
191                        }
192
193
194                        // handle the PKCE code challenge if present
195                        if (request.getExtensions().containsKey(CODE_CHALLENGE)) {
196                                String challenge = (String) request.getExtensions().get(CODE_CHALLENGE);
197                                PKCEAlgorithm alg = PKCEAlgorithm.parse((String) request.getExtensions().get(CODE_CHALLENGE_METHOD));
198
199                                String verifier = request.getRequestParameters().get(CODE_VERIFIER);
200
201                                if (alg.equals(PKCEAlgorithm.plain)) {
202                                        // do a direct string comparison
203                                        if (!challenge.equals(verifier)) {
204                                                throw new InvalidRequestException("Code challenge and verifier do not match");
205                                        }
206                                } else if (alg.equals(PKCEAlgorithm.S256)) {
207                                        // hash the verifier
208                                        try {
209                                                MessageDigest digest = MessageDigest.getInstance("SHA-256");
210                                                String hash = Base64URL.encode(digest.digest(verifier.getBytes(StandardCharsets.US_ASCII))).toString();
211                                                if (!challenge.equals(hash)) {
212                                                        throw new InvalidRequestException("Code challenge and verifier do not match");
213                                                }
214                                        } catch (NoSuchAlgorithmException e) {
215                                                logger.error("Unknown algorithm for PKCE digest", e);
216                                        }
217                                }
218
219                        }
220
221
222                        OAuth2AccessTokenEntity token = new OAuth2AccessTokenEntity();//accessTokenFactory.createNewAccessToken();
223
224                        // attach the client
225                        token.setClient(client);
226
227                        // inherit the scope from the auth, but make a new set so it is
228                        //not unmodifiable. Unmodifiables don't play nicely with Eclipselink, which
229                        //wants to use the clone operation.
230                        Set<SystemScope> scopes = scopeService.fromStrings(request.getScope());
231
232                        // remove any of the special system scopes
233                        scopes = scopeService.removeReservedScopes(scopes);
234
235                        token.setScope(scopeService.toStrings(scopes));
236
237                        // make it expire if necessary
238                        if (client.getAccessTokenValiditySeconds() != null && client.getAccessTokenValiditySeconds() > 0) {
239                                Date expiration = new Date(System.currentTimeMillis() + (client.getAccessTokenValiditySeconds() * 1000L));
240                                token.setExpiration(expiration);
241                        }
242
243                        // attach the authorization so that we can look it up later
244                        AuthenticationHolderEntity authHolder = new AuthenticationHolderEntity();
245                        authHolder.setAuthentication(authentication);
246                        authHolder = authenticationHolderRepository.save(authHolder);
247
248                        token.setAuthenticationHolder(authHolder);
249
250                        // attach a refresh token, if this client is allowed to request them and the user gets the offline scope
251                        if (client.isAllowRefresh() && token.getScope().contains(SystemScopeService.OFFLINE_ACCESS)) {
252                                OAuth2RefreshTokenEntity savedRefreshToken = createRefreshToken(client, authHolder);
253
254                                token.setRefreshToken(savedRefreshToken);
255                        }
256
257                        //Add approved site reference, if any
258                        OAuth2Request originalAuthRequest = authHolder.getAuthentication().getOAuth2Request();
259
260                        if (originalAuthRequest.getExtensions() != null && originalAuthRequest.getExtensions().containsKey("approved_site")) {
261
262                                Long apId = Long.parseLong((String) originalAuthRequest.getExtensions().get("approved_site"));
263                                ApprovedSite ap = approvedSiteService.getById(apId);
264
265                                token.setApprovedSite(ap);
266                        }
267
268                        OAuth2AccessTokenEntity enhancedToken = (OAuth2AccessTokenEntity) tokenEnhancer.enhance(token, authentication);
269
270                        OAuth2AccessTokenEntity savedToken = saveAccessToken(enhancedToken);
271
272                        if (savedToken.getRefreshToken() != null) {
273                                tokenRepository.saveRefreshToken(savedToken.getRefreshToken()); // make sure we save any changes that might have been enhanced
274                        }
275
276                        return savedToken;
277                }
278
279                throw new AuthenticationCredentialsNotFoundException("No authentication credentials found");
280        }

283        private OAuth2RefreshTokenEntity createRefreshToken(ClientDetailsEntity client, AuthenticationHolderEntity authHolder) {
284                OAuth2RefreshTokenEntity refreshToken = new OAuth2RefreshTokenEntity(); //refreshTokenFactory.createNewRefreshToken();
285                JWTClaimsSet.Builder refreshClaims = new JWTClaimsSet.Builder();
286
287
288                // make it expire if necessary
289                if (client.getRefreshTokenValiditySeconds() != null) {
290                        Date expiration = new Date(System.currentTimeMillis() + (client.getRefreshTokenValiditySeconds() * 1000L));
291                        refreshToken.setExpiration(expiration);
292                        refreshClaims.expirationTime(expiration);
293                }
294
295                // set a random identifier
296                refreshClaims.jwtID(UUID.randomUUID().toString());
297
298                // TODO: add issuer fields, signature to JWT
299
300                PlainJWT refreshJwt = new PlainJWT(refreshClaims.build());
301                refreshToken.setJwt(refreshJwt);
302
303                //Add the authentication
304                refreshToken.setAuthenticationHolder(authHolder);
305                refreshToken.setClient(client);
306
307
308
309                // save the token first so that we can set it to a member of the access token (NOTE: is this step necessary?)
310                OAuth2RefreshTokenEntity savedRefreshToken = tokenRepository.saveRefreshToken(refreshToken);
311                return savedRefreshToken;
312        }

314        @Override
315        @Transactional(value="defaultTransactionManager")
316        public OAuth2AccessTokenEntity refreshAccessToken(String refreshTokenValue, TokenRequest authRequest) throws AuthenticationException {
317
318                OAuth2RefreshTokenEntity refreshToken = clearExpiredRefreshToken(tokenRepository.getRefreshTokenByValue(refreshTokenValue));
319
320                if (refreshToken == null) {
321                        throw new InvalidTokenException("Invalid refresh token: " + refreshTokenValue);
322                }
323
324                ClientDetailsEntity client = refreshToken.getClient();
325
326                AuthenticationHolderEntity authHolder = refreshToken.getAuthenticationHolder();
327
328                // make sure that the client requesting the token is the one who owns the refresh token
329                ClientDetailsEntity requestingClient = clientDetailsService.loadClientByClientId(authRequest.getClientId());
330                if (!client.getClientId().equals(requestingClient.getClientId())) {
331                        tokenRepository.removeRefreshToken(refreshToken);
332                        throw new InvalidClientException("Client does not own the presented refresh token");
333                }
334
335                //Make sure this client allows access token refreshing
336                if (!client.isAllowRefresh()) {
337                        throw new InvalidClientException("Client does not allow refreshing access token!");
338                }
339
340                // clear out any access tokens
341                if (client.isClearAccessTokensOnRefresh()) {
342                        tokenRepository.clearAccessTokensForRefreshToken(refreshToken);
343                }
344
345                if (refreshToken.isExpired()) {
346                        tokenRepository.removeRefreshToken(refreshToken);
347                        throw new InvalidTokenException("Expired refresh token: " + refreshTokenValue);
348                }
349
350                OAuth2AccessTokenEntity token = new OAuth2AccessTokenEntity();
351
352                // get the stored scopes from the authentication holder's authorization request; these are the scopes associated with the refresh token
353                Set<String> refreshScopesRequested = new HashSet<>(refreshToken.getAuthenticationHolder().getAuthentication().getOAuth2Request().getScope());
354                Set<SystemScope> refreshScopes = scopeService.fromStrings(refreshScopesRequested);
355                // remove any of the special system scopes
356                refreshScopes = scopeService.removeReservedScopes(refreshScopes);
357
358                Set<String> scopeRequested = authRequest.getScope() == null ? new HashSet<String>() : new HashSet<>(authRequest.getScope());
359                Set<SystemScope> scope = scopeService.fromStrings(scopeRequested);
360
361                // remove any of the special system scopes
362                scope = scopeService.removeReservedScopes(scope);
363
364                if (scope != null && !scope.isEmpty()) {
365                        // ensure a proper subset of scopes
366                        if (refreshScopes != null && refreshScopes.containsAll(scope)) {
367                                // set the scope of the new access token if requested
368                                token.setScope(scopeService.toStrings(scope));
369                        } else {
370                                String errorMsg = "Up-scoping is not allowed.";
371                                logger.error(errorMsg);
372                                throw new InvalidScopeException(errorMsg);
373                        }
374                } else {
375                        // otherwise inherit the scope of the refresh token (if it's there -- this can return a null scope set)
376                        token.setScope(scopeService.toStrings(refreshScopes));
377                }
378
379                token.setClient(client);
380
381                if (client.getAccessTokenValiditySeconds() != null) {
382                        Date expiration = new Date(System.currentTimeMillis() + (client.getAccessTokenValiditySeconds() * 1000L));
383                        token.setExpiration(expiration);
384                }
385
386                if (client.isReuseRefreshToken()) {
387                        // if the client re-uses refresh tokens, do that
388                        token.setRefreshToken(refreshToken);
389                } else {
390                        // otherwise, make a new refresh token
391                        OAuth2RefreshTokenEntity newRefresh = createRefreshToken(client, authHolder);
392                        token.setRefreshToken(newRefresh);
393
394                        // clean up the old refresh token
395                        tokenRepository.removeRefreshToken(refreshToken);
396                }
397
398                token.setAuthenticationHolder(authHolder);
399
400                tokenEnhancer.enhance(token, authHolder.getAuthentication());
401
402                tokenRepository.saveAccessToken(token);
403
404                return token;
405
406        }

408        @Override
409        public OAuth2Authentication loadAuthentication(String accessTokenValue) throws AuthenticationException {
410
411                OAuth2AccessTokenEntity accessToken = clearExpiredAccessToken(tokenRepository.getAccessTokenByValue(accessTokenValue));
412
413                if (accessToken == null) {
414                        throw new InvalidTokenException("Invalid access token: " + accessTokenValue);
415                } else {
416                        return accessToken.getAuthenticationHolder().getAuthentication();
417                }
418        }

493        /**
494         * Clears out expired tokens and any abandoned authentication objects
495         */
496        @Override
497        public void clearExpiredTokens() {
498                logger.debug("Cleaning out all expired tokens");
499
500                new AbstractPageOperationTemplate<OAuth2AccessTokenEntity>("clearExpiredAccessTokens") {
501                        @Override
502                        public Collection<OAuth2AccessTokenEntity> fetchPage() {
503                                return tokenRepository.getAllExpiredAccessTokens(new DefaultPageCriteria());
504                        }
505
506                        @Override
507                        public void doOperation(OAuth2AccessTokenEntity item) {
508                                revokeAccessToken(item);
509                        }
510                }.execute();
511
512                new AbstractPageOperationTemplate<OAuth2RefreshTokenEntity>("clearExpiredRefreshTokens") {
513                        @Override
514                        public Collection<OAuth2RefreshTokenEntity> fetchPage() {
515                                return tokenRepository.getAllExpiredRefreshTokens(new DefaultPageCriteria());
516                        }
517
518                        @Override
519                        public void doOperation(OAuth2RefreshTokenEntity item) {
520                                revokeRefreshToken(item);
521                        }
522                }.execute();
523
524                new AbstractPageOperationTemplate<AuthenticationHolderEntity>("clearExpiredAuthenticationHolders") {
525                        @Override
526                        public Collection<AuthenticationHolderEntity> fetchPage() {
527                                return authenticationHolderRepository.getOrphanedAuthenticationHolders(new DefaultPageCriteria());
528                        }
529
530                        @Override
531                        public void doOperation(AuthenticationHolderEntity item) {
532                                authenticationHolderRepository.remove(item);
533                        }
534                }.execute();
535        }

538         * @see org.mitre.oauth2.service.OAuth2TokenEntityService#saveAccessToken(org.mitre.oauth2.model.OAuth2AccessTokenEntity)
539         */
540        @Override
541        @Transactional(value="defaultTransactionManager")
542        public OAuth2AccessTokenEntity saveAccessToken(OAuth2AccessTokenEntity accessToken) {
543                OAuth2AccessTokenEntity newToken = tokenRepository.saveAccessToken(accessToken);
544
545                // if the old token has any additional information for the return from the token endpoint, carry it through here after save
546                if (accessToken.getAdditionalInformation() != null && !accessToken.getAdditionalInformation().isEmpty()) {
547                        newToken.getAdditionalInformation().putAll(accessToken.getAdditionalInformation());
548                }
549
550                return newToken;
551        }

576        @Override
577        public OAuth2AccessTokenEntity getRegistrationAccessTokenForClient(ClientDetailsEntity client) {
578                List<OAuth2AccessTokenEntity> allTokens = getAccessTokensForClient(client);
579
580                for (OAuth2AccessTokenEntity token : allTokens) {
581                        if ((token.getScope().contains(SystemScopeService.REGISTRATION_TOKEN_SCOPE) || token.getScope().contains(SystemScopeService.RESOURCE_TOKEN_SCOPE))
582                                        && token.getScope().size() == 1) {
583                                // if it only has the registration scope, then it's a registration token
584                                return token;
585                        }
586                }
587
588                return null;
589        }
