004package org.mitre.oauth2.web;
005
006import java.io.IOException;
007
008import javax.servlet.FilterChain;
009import javax.servlet.ServletException;
010import javax.servlet.http.HttpServletRequest;
011import javax.servlet.http.HttpServletResponse;
012
013import org.springframework.stereotype.Component;
014import org.springframework.web.filter.OncePerRequestFilter;
015
016/**
017 * 
018 * Implements Cross-Origin Resource Sharing (CORS) headers. This filter adds the CORS
019 * headers to all requests that pass through it, and as such it should be used only
020 * on endpoints that require CORS support.
021 * 
022 * @author jricher
023 *
024 */
025@Component("corsFilter")
026public class CorsFilter extends OncePerRequestFilter {
027
028        /* (non-Javadoc)
029         * @see javax.servlet.Filter#doFilter(javax.servlet.ServletRequest, javax.servlet.ServletResponse, javax.servlet.FilterChain)
030         */
031        @Override
032        public void doFilterInternal(HttpServletRequest request, HttpServletResponse response, FilterChain filterChain) throws IOException, ServletException {
033                
034                response.addHeader("Access-Control-Allow-Origin", "*");
035                if (request.getHeader("Access-Control-Request-Method") != null && "OPTIONS".equals(request.getMethod())) {
036                        // CORS "pre-flight" request
037                        response.addHeader("Access-Control-Allow-Methods", "GET, POST, PUT, DELETE");
038                        response.addHeader("Access-Control-Allow-Headers", "X-Requested-With,Origin,Content-Type, Accept, Authorization");
039                }
040                filterChain.doFilter(request, response);
041        }
042
043}
