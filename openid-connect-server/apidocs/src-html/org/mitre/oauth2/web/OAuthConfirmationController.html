060@Controller
061@SessionAttributes("authorizationRequest")
062public class OAuthConfirmationController {
063
064        @Autowired
065        private ClientDetailsEntityService clientService;
066
067        @Autowired
068        private SystemScopeService scopeService;
069
070        @Autowired
071        private ScopeClaimTranslationService scopeClaimTranslationService;
072        
073        @Autowired
074        private UserInfoService userInfoService;
075        
076        @Autowired
077        private StatsService statsService;
078        
079        private static Logger logger = LoggerFactory.getLogger(OAuthConfirmationController.class);
080
081        public OAuthConfirmationController() {
082
083        }
084
085        public OAuthConfirmationController(ClientDetailsEntityService clientService) {
086                this.clientService = clientService;
087        }
088
089        @PreAuthorize("hasRole('ROLE_USER')")
090        @RequestMapping("/oauth/confirm_access")
091        public String confimAccess(Map<String, Object> model, @ModelAttribute("authorizationRequest") AuthorizationRequest clientAuth,
092                        Principal p) {
093
094                // Check the "prompt" parameter to see if we need to do special processing
095
096                String prompt = (String)clientAuth.getExtensions().get("prompt");
097                List<String> prompts = Splitter.on(" ").splitToList(Strings.nullToEmpty(prompt));
098                if (prompts.contains("none")) {
099                        // we're not supposed to prompt, so "return an error"
100                        logger.info("Client requested no prompt, returning 403 from confirmation endpoint");
101                        model.put("code", HttpStatus.FORBIDDEN);
102                        return "httpCodeView";
103                }
104
105                //AuthorizationRequest clientAuth = (AuthorizationRequest) model.remove("authorizationRequest");
106
107                ClientDetailsEntity client = null;
108
109                try {
110                        client = clientService.loadClientByClientId(clientAuth.getClientId());
111                } catch (OAuth2Exception e) {
112                        logger.error("confirmAccess: OAuth2Exception was thrown when attempting to load client", e);
113                        model.put("code", HttpStatus.BAD_REQUEST);
114                        return "httpCodeView";
115                } catch (IllegalArgumentException e) {
116                        logger.error("confirmAccess: IllegalArgumentException was thrown when attempting to load client", e);
117                        model.put("code", HttpStatus.BAD_REQUEST);
118                        return "httpCodeView";
119                }
120
121                if (client == null) {
122                        logger.error("confirmAccess: could not find client " + clientAuth.getClientId());
123                        model.put("code", HttpStatus.NOT_FOUND);
124                        return "httpCodeView";
125                }
126
127                model.put("auth_request", clientAuth);
128                model.put("client", client);
129
130                String redirect_uri = clientAuth.getRedirectUri();
131
132                model.put("redirect_uri", redirect_uri);
133
134                
135                // pre-process the scopes
136                Set<SystemScope> scopes = scopeService.fromStrings(clientAuth.getScope());
137
138                Set<SystemScope> sortedScopes = new LinkedHashSet<SystemScope>(scopes.size());
139                Set<SystemScope> systemScopes = scopeService.getAll();
140
141                // sort scopes for display based on the inherent order of system scopes
142                for (SystemScope s : systemScopes) {
143                        if (scopes.contains(s)) {
144                                sortedScopes.add(s);
145                        }
146                }
147
148                // add in any scopes that aren't system scopes to the end of the list
149                sortedScopes.addAll(Sets.difference(scopes, systemScopes));
150
151                model.put("scopes", sortedScopes);
152
153                // get the userinfo claims for each scope
154                UserInfo user = userInfoService.getByUsername(p.getName());
155                JsonObject userJson = user.toJson();
156                Map<String, Map<String, String>> claimsForScopes = new HashMap<String, Map<String, String>>();
157
158                for (SystemScope systemScope : sortedScopes) {
159                        Map<String, String> claimValues = new HashMap<String, String>();
160                        
161                        Set<String> claims = scopeClaimTranslationService.getClaimsForScope(systemScope.getValue());
162                        for (String claim : claims) {
163                                if (userJson.has(claim) && userJson.get(claim).isJsonPrimitive()) {
164                                        // TODO: this skips the address claim
165                                        claimValues.put(claim, userJson.get(claim).getAsString());
166                                }
167                        }
168                        
169                        claimsForScopes.put(systemScope.getValue(), claimValues);
170                }
171                
172                model.put("claims", claimsForScopes);
173
174                // client stats
175                Integer count = statsService.getCountForClientId(client.getId());
176                model.put("count", count);
177                
178                
179                // contacts
180                if (client.getContacts() != null) {
181                        String contacts = Joiner.on(", ").join(client.getContacts());
182                        model.put("contacts", contacts);
183                }
184                
185                // if the client is over a week old and has more than one registration, don't give such a big warning
186                // instead, tag as "Generally Recognized As Safe (gras)
187                Date lastWeek = new Date(System.currentTimeMillis() + (60 * 60 * 24 * 7 * 1000));
188                //Date lastWeek = new Date(System.currentTimeMillis() - (60 * 60 * 24 * 7 * 1000));
189                if (count > 1 && client.getCreatedAt() != null && client.getCreatedAt().before(lastWeek)) {
190                        model.put("gras", true);
191                } else {
192                        model.put("gras", false);
193                }
194                
195                
196                return "approve";
197        }
198
199        /**
200         * @return the clientService
201         */
202        public ClientDetailsEntityService getClientService() {
203                return clientService;
204        }
205

207         * @param clientService the clientService to set
208         */
209        public void setClientService(ClientDetailsEntityService clientService) {
210                this.clientService = clientService;
211        }
