001package org.mitre.openid.connect.repository.impl;
002
003import static org.mitre.util.jpa.JpaUtil.saveOrUpdate;
004
005import java.util.Collection;
006
007import javax.persistence.EntityManager;
008import javax.persistence.PersistenceContext;
009import javax.persistence.TypedQuery;
010
011import org.mitre.openid.connect.model.ApprovedSite;
012import org.mitre.openid.connect.model.Nonce;
013import org.mitre.openid.connect.repository.NonceRepository;
014import org.springframework.stereotype.Repository;
015import org.springframework.transaction.annotation.Transactional;
016
017@Repository
018public class JpaNonceRepository implements NonceRepository {
019
020        @PersistenceContext
021        private EntityManager manager;
022
023        
024        @Override
025        @Transactional
026        public Nonce getById(Long id) {
027                return manager.find(Nonce.class, id);
028        }
029
030        @Override
031        @Transactional
032        public void remove(Nonce nonce) {
033                Nonce found = manager.find(Nonce.class, nonce.getId());
034                
035                if (found != null) {
036                        manager.remove(found);
037                } else {
038                        throw new IllegalArgumentException("Nonce reporitory remove: Nonce with id " + nonce.getId() + " could not be found.");
039                }
040
041        }
042
043        @Override
044        @Transactional
045        public Nonce save(Nonce nonce) {
046                return saveOrUpdate(nonce.getId(), manager, nonce);
047        }
048
049        @Override
050        @Transactional
051        public Collection<Nonce> getAll() {
052                TypedQuery<Nonce> query = manager.createNamedQuery("Nonce.getAll", Nonce.class);
053                return query.getResultList();
054        }
055
056        @Override
057        @Transactional
058        public Collection<Nonce> getExpired() {
059                TypedQuery<Nonce> query = manager.createNamedQuery("Nonce.getExpired", Nonce.class);
060                return query.getResultList();
061        }
062
063        @Override
064        @Transactional
065        public Collection<Nonce> getByClientId(String clientId) {
066                TypedQuery<Nonce> query = manager.createNamedQuery("Nonce.getByClientId", Nonce.class);
067                query.setParameter("clientId", clientId);
068                
069                return query.getResultList();
070        }
071
072}
