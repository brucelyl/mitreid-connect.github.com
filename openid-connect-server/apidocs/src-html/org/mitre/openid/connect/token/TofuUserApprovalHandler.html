054/**
055 * Custom User Approval Handler implementation which uses a concept of a whitelist,
056 * blacklist, and greylist.
057 *
058 * Blacklisted sites will be caught and handled before this
059 * point.
060 *
061 * Whitelisted sites will be automatically approved, and an ApprovedSite entry will
062 * be created for the site the first time a given user access it.
063 *
064 * All other sites fall into the greylist - the user will be presented with the user
065 * approval page upon their first visit
066 * @author aanganes
067 *
068 */
069@Component("tofuUserApprovalHandler")
070public class TofuUserApprovalHandler implements UserApprovalHandler {
071
072        @Autowired
073        private ApprovedSiteService approvedSiteService;
074
075        @Autowired
076        private WhitelistedSiteService whitelistedSiteService;
077
078        @Autowired
079        private ClientDetailsService clientDetailsService;
080
081        @Autowired
082        private SystemScopeService systemScopes;
083
084        /**
085         * Check if the user has already stored a positive approval decision for this site; or if the
086         * site is whitelisted, approve it automatically.
087         *
088         * Otherwise, return false so that the user will see the approval page and can make their own decision.
089         *
090         * @param authorizationRequest  the incoming authorization request
091         * @param userAuthentication    the Principal representing the currently-logged-in user
092         *
093         * @return                                              true if the site is approved, false otherwise
094         */
095        @Override
096        public boolean isApproved(AuthorizationRequest authorizationRequest, Authentication userAuthentication) {
097
098                // if this request is already approved, pass that info through
099                // (this flag may be set by updateBeforeApproval, which can also do funny things with scopes, etc)
100                if (authorizationRequest.isApproved()) {
101                        return true;
102                } else {
103                        // if not, check to see if the user has approved it
104                        // TODO: make parameter name configurable?
105                        return Boolean.parseBoolean(authorizationRequest.getApprovalParameters().get("user_oauth_approval"));
106                }
107
108        }
109
110        /**
111         * Check if the user has already stored a positive approval decision for this site; or if the
112         * site is whitelisted, approve it automatically.
113         *
114         * Otherwise the user will be directed to the approval page and can make their own decision.
115         *
116         * @param authorizationRequest  the incoming authorization request
117         * @param userAuthentication    the Principal representing the currently-logged-in user
118         *
119         * @return                                              the updated AuthorizationRequest
120         */
121        @Override
122        public AuthorizationRequest checkForPreApproval(AuthorizationRequest authorizationRequest, Authentication userAuthentication) {
123
124                //First, check database to see if the user identified by the userAuthentication has stored an approval decision
125
126                String userId = userAuthentication.getName();
127                String clientId = authorizationRequest.getClientId();
128
129                //lookup ApprovedSites by userId and clientId
130                boolean alreadyApproved = false;
131
132                // find out if we're supposed to force a prompt on the user or not
133                String prompt = (String) authorizationRequest.getExtensions().get(PROMPT);
134                List<String> prompts = Splitter.on(PROMPT_SEPARATOR).splitToList(Strings.nullToEmpty(prompt));
135                if (!prompts.contains(PROMPT_CONSENT)) {
136                        // if the prompt parameter is set to "consent" then we can't use approved sites or whitelisted sites
137                        // otherwise, we need to check them below
138
139                        Collection<ApprovedSite> aps = approvedSiteService.getByClientIdAndUserId(clientId, userId);
140                        for (ApprovedSite ap : aps) {
141
142                                if (!ap.isExpired()) {
143
144                                        // if we find one that fits...
145                                        if (systemScopes.scopesMatch(ap.getAllowedScopes(), authorizationRequest.getScope())) {
146
147                                                //We have a match; update the access date on the AP entry and return true.
148                                                ap.setAccessDate(new Date());
149                                                approvedSiteService.save(ap);
150
151                                                String apId = ap.getId().toString();
152                                                authorizationRequest.getExtensions().put(APPROVED_SITE, apId);
153                                                authorizationRequest.setApproved(true);
154                                                alreadyApproved = true;
155
156                                                setAuthTime(authorizationRequest);
157                                        }
158                                }
159                        }
160
161                        if (!alreadyApproved) {
162                                WhitelistedSite ws = whitelistedSiteService.getByClientId(clientId);
163                                if (ws != null && systemScopes.scopesMatch(ws.getAllowedScopes(), authorizationRequest.getScope())) {
164                                        authorizationRequest.setApproved(true);
165
166                                        setAuthTime(authorizationRequest);
167                                }
168                        }
169                }
170
171                return authorizationRequest;
172
173        }
174
175
176        @Override
177        public AuthorizationRequest updateAfterApproval(AuthorizationRequest authorizationRequest, Authentication userAuthentication) {
178
179                String userId = userAuthentication.getName();
180                String clientId = authorizationRequest.getClientId();
181                ClientDetails client = clientDetailsService.loadClientByClientId(clientId);
182
183                // This must be re-parsed here because SECOAUTH forces us to call things in a strange order
184                if (Boolean.parseBoolean(authorizationRequest.getApprovalParameters().get("user_oauth_approval"))) {
185
186                        authorizationRequest.setApproved(true);
187
188                        // process scopes from user input
189                        Set<String> allowedScopes = Sets.newHashSet();
190                        Map<String,String> approvalParams = authorizationRequest.getApprovalParameters();
191
192                        Set<String> keys = approvalParams.keySet();
193
194                        for (String key : keys) {
195                                if (key.startsWith("scope_")) {
196                                        //This is a scope parameter from the approval page. The value sent back should
197                                        //be the scope string. Check to make sure it is contained in the client's
198                                        //registered allowed scopes.
199
200                                        String scope = approvalParams.get(key);
201                                        Set<String> approveSet = Sets.newHashSet(scope);
202
203                                        //Make sure this scope is allowed for the given client
204                                        if (systemScopes.scopesMatch(client.getScope(), approveSet)) {
205
206                                                allowedScopes.add(scope);
207                                        }
208
209                                }
210                        }
211
212                        // inject the user-allowed scopes into the auth request
213                        authorizationRequest.setScope(allowedScopes);
214
215                        //Only store an ApprovedSite if the user has checked "remember this decision":
216                        String remember = authorizationRequest.getApprovalParameters().get("remember");
217                        if (!Strings.isNullOrEmpty(remember) && !remember.equals("none")) {
218
219                                Date timeout = null;
220                                if (remember.equals("one-hour")) {
221                                        // set the timeout to one hour from now
222                                        Calendar cal = Calendar.getInstance();
223                                        cal.add(Calendar.HOUR, 1);
224                                        timeout = cal.getTime();
225                                }
226
227                                ApprovedSite newSite = approvedSiteService.createApprovedSite(clientId, userId, timeout, allowedScopes);
228                                String newSiteId = newSite.getId().toString();
229                                authorizationRequest.getExtensions().put(APPROVED_SITE, newSiteId);
230                        }
231
232                        setAuthTime(authorizationRequest);
233
234
235                }
236
237                return authorizationRequest;
238        }
239
240        /**
241         * Get the auth time out of the current session and add it to the
242         * auth request in the extensions map.
243         *
244         * @param authorizationRequest
245         */
246        private void setAuthTime(AuthorizationRequest authorizationRequest) {
247                // Get the session auth time, if we have it, and store it in the request
248                ServletRequestAttributes attr = (ServletRequestAttributes) RequestContextHolder.currentRequestAttributes();
249                if (attr != null) {
250                        HttpSession session = attr.getRequest().getSession();
251                        if (session != null) {
252                                Date authTime = (Date) session.getAttribute(AuthenticationTimeStamper.AUTH_TIMESTAMP);
253                                if (authTime != null) {
254                                        String authTimeString = Long.toString(authTime.getTime());
255                                        authorizationRequest.getExtensions().put(AuthenticationTimeStamper.AUTH_TIMESTAMP, authTimeString);
256                                }
257                        }
258                }
259        }
260
261        @Override
262        public Map<String, Object> getUserApprovalRequest(AuthorizationRequest authorizationRequest,
263                        Authentication userAuthentication) {
264                Map<String, Object> model = new HashMap<>();
265                // In case of a redirect we might want the request parameters to be included
266                model.putAll(authorizationRequest.getRequestParameters());
267                return model;
268        }
