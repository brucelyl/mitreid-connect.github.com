017package org.mitre.openid.connect.token;
018
019import java.security.InvalidKeyException;
020import java.security.NoSuchAlgorithmException;
021import java.util.Arrays;
022import java.util.Date;
023import java.util.Set;
024import java.util.UUID;
025import java.util.regex.Matcher;
026import java.util.regex.Pattern;
027
028import javax.crypto.Mac;
029import javax.crypto.spec.SecretKeySpec;
030import javax.servlet.http.HttpSession;
031
032import org.mitre.jwt.signer.service.JwtSigningAndValidationService;
033import org.mitre.oauth2.model.ClientDetailsEntity;
034import org.mitre.oauth2.model.OAuth2AccessTokenEntity;
035import org.mitre.oauth2.service.ClientDetailsEntityService;
036import org.mitre.openid.connect.config.ConfigurationPropertiesBean;
037import org.mitre.openid.connect.web.AuthenticationTimeStamper;
038import org.slf4j.Logger;
039import org.slf4j.LoggerFactory;
040import org.springframework.beans.factory.annotation.Autowired;
041import org.springframework.security.oauth2.common.OAuth2AccessToken;
042import org.springframework.security.oauth2.common.util.OAuth2Utils;
043import org.springframework.security.oauth2.provider.OAuth2Authentication;
044import org.springframework.security.oauth2.provider.token.TokenEnhancer;
045import org.springframework.stereotype.Service;
046import org.springframework.web.context.request.RequestContextHolder;
047import org.springframework.web.context.request.ServletRequestAttributes;
048
049import com.google.common.base.Strings;
050import com.google.common.collect.Lists;
051import com.google.common.collect.Sets;
052import com.nimbusds.jose.JWSAlgorithm;
053import com.nimbusds.jose.JWSHeader;
054import com.nimbusds.jose.util.Base64URL;
055import com.nimbusds.jwt.JWTClaimsSet;
056import com.nimbusds.jwt.SignedJWT;
057
058@Service
059public class ConnectTokenEnhancer implements TokenEnhancer {
060
061        Logger logger = LoggerFactory.getLogger(ConnectTokenEnhancer.class);
062
063        @Autowired
064        private ConfigurationPropertiesBean configBean;
065
066        @Autowired
067        private JwtSigningAndValidationService jwtService;
068
069        @Autowired
070        private ClientDetailsEntityService clientService;
071
072        @Override
073        public OAuth2AccessToken enhance(OAuth2AccessToken accessToken, OAuth2Authentication authentication) {
074
075                OAuth2AccessTokenEntity token = (OAuth2AccessTokenEntity) accessToken;
076
077                String clientId = authentication.getAuthorizationRequest().getClientId();
078                ClientDetailsEntity client = clientService.loadClientByClientId(clientId);
079
080                JWTClaimsSet claims = new JWTClaimsSet();
081
082                claims.setAudience(Lists.newArrayList(clientId));
083
084                claims.setIssuer(configBean.getIssuer());
085
086                claims.setIssueTime(new Date());
087
088                claims.setExpirationTime(token.getExpiration());
089
090                claims.setJWTID(UUID.randomUUID().toString()); // set a random NONCE in the middle of it
091
092                // TODO: use client's default signing algorithm
093
094                JWSAlgorithm signingAlg = jwtService.getDefaultSigningAlgorithm();
095                SignedJWT signed = new SignedJWT(new JWSHeader(signingAlg), claims);
096
097                jwtService.signJwt(signed);
098
099                token.setJwt(signed);
100
101                /**
102                 * Authorization request scope MUST include "openid" in OIDC, but access token request
103                 * may or may not include the scope parameter. As long as the AuthorizationRequest
104                 * has the proper scope, we can consider this a valid OpenID Connect request. Otherwise,
105                 * we consider it to be a vanilla OAuth2 request.
106                 */
107                if (authentication.getAuthorizationRequest().getScope().contains("openid")) {
108
109                        // TODO: maybe id tokens need a service layer
110
111                        String userId = authentication.getName();
112
113                        OAuth2AccessTokenEntity idTokenEntity = new OAuth2AccessTokenEntity();
114
115                        // FIXME: extend the "claims" section for id tokens
116                        JWTClaimsSet idClaims = new JWTClaimsSet();
117
118
119                        //
120                        // FIXME: storing the auth time in the session doesn't actually work, because we need access to it from the token endpoint when the user isn't present
121                        //
122
123                        // get the auth time from the session
124                        ServletRequestAttributes attr = (ServletRequestAttributes) RequestContextHolder.currentRequestAttributes();
125                        if (attr != null) {
126                                HttpSession session = attr.getRequest().getSession();
127                                if (session != null) {
128                                        Date authTime = (Date) session.getAttribute(AuthenticationTimeStamper.AUTH_TIMESTAMP);
129                                        if (authTime != null) {
130                                                idClaims.setClaim("auth_time", authTime.getTime() / 1000);
131                                        }
132                                }
133                        }
134
135                        idClaims.setIssueTime(claims.getIssueTime());
136
137                        if (client.getIdTokenValiditySeconds() != null) {
138                                Date expiration = new Date(System.currentTimeMillis() + (client.getIdTokenValiditySeconds() * 1000L));
139                                idClaims.setExpirationTime(expiration);
140                                idTokenEntity.setExpiration(expiration);
141                        }
142
143                        idClaims.setIssuer(configBean.getIssuer());
144                        idClaims.setSubject(userId);
145                        idClaims.setAudience(Lists.newArrayList(clientId));
146
147
148                        String nonce = authentication.getAuthorizationRequest().getAuthorizationParameters().get("nonce");
149                        if (!Strings.isNullOrEmpty(nonce)) {
150                                idClaims.setCustomClaim("nonce", nonce);
151                        }
152
153                        String responseType = authentication.getAuthorizationRequest().getAuthorizationParameters().get("response_type");
154                        Set<String> responseTypes = OAuth2Utils.parseParameterList(responseType);
155                        if (responseTypes.contains("token")) {
156                                // calculate the token hash
157
158                                // get the right algorithm size
159                                // TODO: all this string parsing feels like a bad hack
160                                String algName = signingAlg.getName();
161                                Pattern re = Pattern.compile("^[HRE]S(\\d+)$");
162                                Matcher match = re.matcher(algName);
163                                if (match.matches()) {
164                                        String bits = match.group(1);
165                                        String hmacAlg = "HMACSHA" + bits;
166                                        try {
167                                                Mac mac = Mac.getInstance(hmacAlg);
168                                                mac.init(new SecretKeySpec(token.getJwt().serialize().getBytes(), hmacAlg));
169
170                                                byte[] at_hash_bytes = mac.doFinal();
171                                                byte[] at_hash_bytes_left = Arrays.copyOf(at_hash_bytes, at_hash_bytes.length / 2);
172                                                Base64URL at_hash = Base64URL.encode(at_hash_bytes_left);
173
174                                                idClaims.setClaim("at_hash", at_hash);
175
176                                        } catch (NoSuchAlgorithmException e) {
177                                                // TODO Auto-generated catch block
178                                                e.printStackTrace();
179                                        } catch (InvalidKeyException e) {
180                                                // TODO Auto-generated catch block
181                                                e.printStackTrace();
182                                        }
183
184                                }
185
186                        }
187
188
189                        SignedJWT idToken = new SignedJWT(new JWSHeader(signingAlg), idClaims);
190
191                        //TODO: check for client's preferred signer alg and use that
192
193                        jwtService.signJwt(idToken);
194
195                        idTokenEntity.setJwt(idToken);
196
197                        // TODO: might want to create a specialty authentication object here instead of copying
198                        idTokenEntity.setAuthenticationHolder(token.getAuthenticationHolder());
199
200                        // create a scope set with just the special "id-token" scope
201                        //Set<String> idScopes = new HashSet<String>(token.getScope()); // this would copy the original token's scopes in, we don't really want that
202                        Set<String> idScopes = Sets.newHashSet(OAuth2AccessTokenEntity.ID_TOKEN_SCOPE);
203                        idTokenEntity.setScope(idScopes);
204
205                        idTokenEntity.setClient(token.getClient());
206
207                        // attach the id token to the parent access token
208                        // TODO: this relationship is one-to-one right now, this might change
209                        token.setIdToken(idTokenEntity);
210                }
211
212                return token;
213        }
214
215        public ConfigurationPropertiesBean getConfigBean() {
216                return configBean;
217        }
218
219        public void setConfigBean(ConfigurationPropertiesBean configBean) {
220                this.configBean = configBean;
221        }
222
223        public JwtSigningAndValidationService getJwtService() {
224                return jwtService;
225        }
226
227        public void setJwtService(JwtSigningAndValidationService jwtService) {
228                this.jwtService = jwtService;
229        }
230
231        public ClientDetailsEntityService getClientService() {
232                return clientService;
233        }
234
235        public void setClientService(ClientDetailsEntityService clientService) {
236                this.clientService = clientService;
237        }
