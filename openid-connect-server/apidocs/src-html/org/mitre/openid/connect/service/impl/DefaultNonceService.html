001package org.mitre.openid.connect.service.impl;
002
003import java.util.Collection;
004import java.util.Date;
005
006import org.joda.time.DateTime;
007import org.joda.time.Period;
008import org.mitre.openid.connect.model.Nonce;
009import org.mitre.openid.connect.repository.NonceRepository;
010import org.mitre.openid.connect.service.NonceService;
011import org.slf4j.Logger;
012import org.slf4j.LoggerFactory;
013import org.springframework.beans.factory.InitializingBean;
014import org.springframework.beans.factory.annotation.Autowired;
015import org.springframework.scheduling.annotation.Scheduled;
016import org.springframework.stereotype.Service;
017
018@Service("defaultNonceService")
019public class DefaultNonceService implements NonceService, InitializingBean {
020
021        private static Logger logger = LoggerFactory.getLogger(NonceService.class);     
022        
023        @Autowired
024        private NonceRepository repository;
025        
026        @Autowired
027        private Period nonceStorageDuration;
028        
029        /**
030         * Make sure that the nonce storage duration was set
031         */
032        public void afterPropertiesSet() throws Exception {
033                if (nonceStorageDuration == null) {
034                        logger.error("Nonce storage duration must be set!");
035                }
036        }
037
038        @Override
039        public Nonce create(String clientId, String value) {
040                //Store nonce
041                Nonce nonce = new Nonce();
042                nonce.setClientId(clientId);
043                nonce.setValue(value);
044                DateTime now = new DateTime(new Date());
045                nonce.setUseDate(now.toDate());
046                DateTime expDate = now.plus(nonceStorageDuration);
047                Date expirationJdkDate = expDate.toDate();
048                nonce.setExpireDate(expirationJdkDate);
049                return nonce;
050        }
051
052        @Override
053        public boolean alreadyUsed(String clientId, String value) {
054                
055                Collection<Nonce> clientNonces = getByClientId(clientId);
056                for (Nonce nonce : clientNonces) {
057                        String nonceVal = nonce.getValue();
058                        if (nonceVal.equals(value)) {
059                                return true;
060                        }
061                }
062                
063                return false;
064        }
065        
066        @Override
067        public Nonce getById(Long id) {
068                return repository.getById(id);
069        }
070
071        @Override
072        public void remove(Nonce nonce) {
073                repository.remove(nonce);
074        }
075
076        @Override
077        public Nonce save(Nonce nonce) {
078                return repository.save(nonce);
079        }
080
081        @Override
082        public Collection<Nonce> getAll() {
083                return repository.getAll();
084        }
085
086        @Override
087        public Collection<Nonce> getExpired() {
088                return repository.getExpired();
089        }
090
091        @Override
092        public Collection<Nonce> getByClientId(String clientId) {
093                return repository.getByClientId(clientId);
094        }
095        
096        @Override
097        @Scheduled(fixedRate = 5 * 60 * 1000) // schedule this task every five minutes
098        public void clearExpiredNonces() {
099                
100                logger.info("Clearing expired nonces");
101                
102                Collection<Nonce> expired = repository.getExpired();
103                logger.info("Found " + expired.size() + " expired nonces");
104                
105                for (Nonce nonce : expired) {
106                        remove(nonce);
107                }
108        }
109
110        public NonceRepository getRepository() {
111                return repository;
112        }
113
114        public void setRepository(NonceRepository repository) {
115                this.repository = repository;
116        }
117
118        public Period getNonceStorageDuration() {
119                return nonceStorageDuration;
120        }
121
122        public void setNonceStorageDuration(Period nonceStorageDuration) {
123                this.nonceStorageDuration = nonceStorageDuration;
124        }
125
126
127}
