125         * @see org.mitre.openid.connect.service.MITREidDataService#importData(com.google.gson.stream.JsonReader)
126         */
127        @Override
128        public void importData(JsonReader reader) throws IOException {
129
130                logger.info("Reading configuration for 1.0");
131
132                // this *HAS* to start as an object
133                reader.beginObject();
134
135                while (reader.hasNext()) {
136                        JsonToken tok = reader.peek();
137                        switch (tok) {
138                                case NAME:
139                                        String name = reader.nextName();
140                                        // find out which member it is
141                                        if (name.equals(CLIENTS)) {
142                                                readClients(reader);
143                                        } else if (name.equals(GRANTS)) {
144                                                readGrants(reader);
145                                        } else if (name.equals(WHITELISTEDSITES)) {
146                                                readWhitelistedSites(reader);
147                                        } else if (name.equals(BLACKLISTEDSITES)) {
148                                                readBlacklistedSites(reader);
149                                        } else if (name.equals(AUTHENTICATIONHOLDERS)) {
150                                                readAuthenticationHolders(reader);
151                                        } else if (name.equals(ACCESSTOKENS)) {
152                                                readAccessTokens(reader);
153                                        } else if (name.equals(REFRESHTOKENS)) {
154                                                readRefreshTokens(reader);
155                                        } else if (name.equals(SYSTEMSCOPES)) {
156                                                readSystemScopes(reader);
157                                        } else {
158                                                for (MITREidDataServiceExtension extension : extensions) {
159                                                        if (extension.supportsVersion(THIS_VERSION)) {
160                                                                if (extension.supportsVersion(THIS_VERSION)) {
161                                                                        extension.importExtensionData(name, reader);
162                                                                        break;
163                                                                }
164                                                        }
165                                                }
166                                                // unknown token, skip it
167                                                reader.skipValue();
168                                        }
169                                        break;
170                                case END_OBJECT:
171                                        // the object ended, we're done here
172                                        reader.endObject();
173                                        continue;
174                                default:
175                                        logger.debug("Found unexpected entry");
176                                        reader.skipValue();
177                                        continue;                       }
178                }
179                fixObjectReferences();
180                for (MITREidDataServiceExtension extension : extensions) {
181                        if (extension.supportsVersion(THIS_VERSION)) {
182                                extension.fixExtensionObjectReferences(maps);
183                                break;
184                        }
185                }
186                maps.clearAll();
187        }

196        private void readRefreshTokens(JsonReader reader) throws IOException {
197                reader.beginArray();
198                while (reader.hasNext()) {
199                        OAuth2RefreshTokenEntity token = new OAuth2RefreshTokenEntity();
200                        reader.beginObject();
201                        Long currentId = null;
202                        String clientId = null;
203                        Long authHolderId = null;
204                        while (reader.hasNext()) {
205                                switch (reader.peek()) {
206                                        case END_OBJECT:
207                                                continue;
208                                        case NAME:
209                                                String name = reader.nextName();
210                                                if (reader.peek() == JsonToken.NULL) {
211                                                        reader.skipValue();
212                                                } else if (name.equals("id")) {
213                                                        currentId = reader.nextLong();
214                                                } else if (name.equals("expiration")) {
215                                                        Date date = utcToDate(reader.nextString());
216                                                        token.setExpiration(date);
217                                                } else if (name.equals("value")) {
218                                                        String value = reader.nextString();
219                                                        try {
220                                                                token.setJwt(JWTParser.parse(value));
221                                                        } catch (ParseException ex) {
222                                                                logger.error("Unable to set refresh token value to {}", value, ex);
223                                                        }
224                                                } else if (name.equals("clientId")) {
225                                                        clientId = reader.nextString();
226                                                } else if (name.equals("authenticationHolderId")) {
227                                                        authHolderId = reader.nextLong();
228                                                } else {
229                                                        logger.debug("Found unexpected entry");
230                                                        reader.skipValue();
231                                                }
232                                                break;
233                                        default:
234                                                logger.debug("Found unexpected entry");
235                                                reader.skipValue();
236                                                continue;
237                                }
238                        }
239                        reader.endObject();
240                        Long newId = tokenRepository.saveRefreshToken(token).getId();
241                        maps.getRefreshTokenToClientRefs().put(currentId, clientId);
242                        maps.getRefreshTokenToAuthHolderRefs().put(currentId, authHolderId);
243                        maps.getRefreshTokenOldToNewIdMap().put(currentId, newId);
244                        logger.debug("Read refresh token {}", currentId);
245                }
246                reader.endArray();
247                logger.info("Done reading refresh tokens");
248        }

257        private void readAccessTokens(JsonReader reader) throws IOException {
258                reader.beginArray();
259                while (reader.hasNext()) {
260                        OAuth2AccessTokenEntity token = new OAuth2AccessTokenEntity();
261                        reader.beginObject();
262                        Long currentId = null;
263                        String clientId = null;
264                        Long authHolderId = null;
265                        Long refreshTokenId = null;
266                        while (reader.hasNext()) {
267                                switch (reader.peek()) {
268                                        case END_OBJECT:
269                                                continue;
270                                        case NAME:
271                                                String name = reader.nextName();
272                                                if (reader.peek() == JsonToken.NULL) {
273                                                        reader.skipValue();
274                                                } else if (name.equals("id")) {
275                                                        currentId = reader.nextLong();
276                                                } else if (name.equals("expiration")) {
277                                                        Date date = utcToDate(reader.nextString());
278                                                        token.setExpiration(date);
279                                                } else if (name.equals("value")) {
280                                                        String value = reader.nextString();
281                                                        try {
282                                                                // all tokens are JWTs
283                                                                token.setJwt(JWTParser.parse(value));
284                                                        } catch (ParseException ex) {
285                                                                logger.error("Unable to set refresh token value to {}", value, ex);
286                                                        }
287                                                } else if (name.equals("clientId")) {
288                                                        clientId = reader.nextString();
289                                                } else if (name.equals("authenticationHolderId")) {
290                                                        authHolderId = reader.nextLong();
291                                                } else if (name.equals("refreshTokenId")) {
292                                                        refreshTokenId = reader.nextLong();
293                                                } else if (name.equals("scope")) {
294                                                        Set<String> scope = readSet(reader);
295                                                        token.setScope(scope);
296                                                } else if (name.equals("type")) {
297                                                        token.setTokenType(reader.nextString());
298                                                } else {
299                                                        logger.debug("Found unexpected entry");
300                                                        reader.skipValue();
301                                                }
302                                                break;
303                                        default:
304                                                logger.debug("Found unexpected entry");
305                                                reader.skipValue();
306                                                continue;
307                                }
308                        }
309                        reader.endObject();
310                        Long newId = tokenRepository.saveAccessToken(token).getId();
311                        maps.getAccessTokenToClientRefs().put(currentId, clientId);
312                        maps.getAccessTokenToAuthHolderRefs().put(currentId, authHolderId);
313                        if (refreshTokenId != null) {
314                                maps.getAccessTokenToRefreshTokenRefs().put(currentId, refreshTokenId);
315                        }
316                        maps.getAccessTokenOldToNewIdMap().put(currentId, newId);
317                        logger.debug("Read access token {}", currentId);
318                }
319                reader.endArray();
320                logger.info("Done reading access tokens");
321        }

323         * @param reader

326        private void readAuthenticationHolders(JsonReader reader) throws IOException {
327                reader.beginArray();
328                while (reader.hasNext()) {
329                        AuthenticationHolderEntity ahe = new AuthenticationHolderEntity();
330                        reader.beginObject();
331                        Long currentId = null;
332                        while (reader.hasNext()) {
333                                switch (reader.peek()) {
334                                        case END_OBJECT:
335                                                continue;
336                                        case NAME:
337                                                String name = reader.nextName();
338                                                if (reader.peek() == JsonToken.NULL) {
339                                                        reader.skipValue();
340                                                } else if (name.equals("id")) {
341                                                        currentId = reader.nextLong();
342                                                } else if (name.equals("ownerId")) {
343                                                        //not needed
344                                                        reader.skipValue();
345                                                } else if (name.equals("authentication")) {
346                                                        OAuth2Request clientAuthorization = null;
347                                                        Authentication userAuthentication = null;
348                                                        reader.beginObject();
349                                                        while (reader.hasNext()) {
350                                                                switch (reader.peek()) {
351                                                                        case END_OBJECT:
352                                                                                continue;
353                                                                        case NAME:
354                                                                                String subName = reader.nextName();
355                                                                                if (reader.peek() == JsonToken.NULL) {
356                                                                                        reader.skipValue();
357                                                                                } else if (subName.equals("clientAuthorization")) {
358                                                                                        clientAuthorization = readAuthorizationRequest(reader);
359                                                                                } else if (subName.equals("userAuthentication")) {
360                                                                                        // skip binary encoded version
361                                                                                        reader.skipValue();
362
363                                                                                } else if (subName.equals("savedUserAuthentication")) {
364                                                                                        userAuthentication = readSavedUserAuthentication(reader);
365
366                                                                                } else {
367                                                                                        logger.debug("Found unexpected entry");
368                                                                                        reader.skipValue();
369                                                                                }
370                                                                                break;
371                                                                        default:
372                                                                                logger.debug("Found unexpected entry");
373                                                                                reader.skipValue();
374                                                                                continue;
375                                                                }
376                                                        }
377                                                        reader.endObject();
378                                                        OAuth2Authentication auth = new OAuth2Authentication(clientAuthorization, userAuthentication);
379                                                        ahe.setAuthentication(auth);
380                                                } else {
381                                                        logger.debug("Found unexpected entry");
382                                                        reader.skipValue();
383                                                }
384                                                break;
385                                        default:
386                                                logger.debug("Found unexpected entry");
387                                                reader.skipValue();
388                                                continue;
389                                }
390                        }
391                        reader.endObject();
392                        Long newId = authHolderRepository.save(ahe).getId();
393                        maps.getAuthHolderOldToNewIdMap().put(currentId, newId);
394                        logger.debug("Read authentication holder {}", currentId);
395                }
396                reader.endArray();
397                logger.info("Done reading authentication holders");
398        }

400        //used by readAuthenticationHolders
401        private OAuth2Request readAuthorizationRequest(JsonReader reader) throws IOException {
402                Set<String> scope = new LinkedHashSet<>();
403                Set<String> resourceIds = new HashSet<>();
404                boolean approved = false;
405                Collection<GrantedAuthority> authorities = new HashSet<>();
406                Map<String, String> authorizationParameters = new HashMap<>();
407                Set<String> responseTypes = new HashSet<>();
408                String redirectUri = null;
409                String clientId = null;
410                reader.beginObject();
411                while (reader.hasNext()) {
412                        switch (reader.peek()) {
413                                case END_OBJECT:
414                                        continue;
415                                case NAME:
416                                        String name = reader.nextName();
417                                        if (reader.peek() == JsonToken.NULL) {
418                                                reader.skipValue();
419                                        } else if (name.equals("authorizationParameters")) {
420                                                authorizationParameters = readMap(reader);
421                                        } else if (name.equals("approvalParameters")) {
422                                                reader.skipValue();
423                                        } else if (name.equals("clientId")) {
424                                                clientId = reader.nextString();
425                                        } else if (name.equals("scope")) {
426                                                scope = readSet(reader);
427                                        } else if (name.equals("resourceIds")) {
428                                                resourceIds = readSet(reader);
429                                        } else if (name.equals("authorities")) {
430                                                Set<String> authorityStrs = readSet(reader);
431                                                authorities = new HashSet<>();
432                                                for (String s : authorityStrs) {
433                                                        GrantedAuthority ga = new SimpleGrantedAuthority(s);
434                                                        authorities.add(ga);
435                                                }
436                                        } else if (name.equals("approved")) {
437                                                approved = reader.nextBoolean();
438                                        } else if (name.equals("denied")) {
439                                                if (approved == false) {
440                                                        approved = !reader.nextBoolean();
441                                                }
442                                        } else if (name.equals("redirectUri")) {
443                                                redirectUri = reader.nextString();
444                                        } else if (name.equals("responseTypes")) {
445                                                responseTypes = readSet(reader);
446                                        } else {
447                                                reader.skipValue();
448                                        }
449                                        break;
450                                default:
451                                        logger.debug("Found unexpected entry");
452                                        reader.skipValue();
453                                        continue;
454                        }
455                }
456                reader.endObject();
457                return new OAuth2Request(authorizationParameters, clientId, authorities, approved, scope, resourceIds, redirectUri, responseTypes, null);
458        }

465        private SavedUserAuthentication readSavedUserAuthentication(JsonReader reader) throws IOException {
466                SavedUserAuthentication savedUserAuth = new SavedUserAuthentication();
467                reader.beginObject();
468
469                while (reader.hasNext()) {
470                        switch(reader.peek()) {
471                                case END_OBJECT:
472                                        continue;
473                                case NAME:
474                                        String name = reader.nextName();
475                                        if (reader.peek() == JsonToken.NULL) {
476                                                reader.skipValue();
477                                        } else if (name.equals("name")) {
478                                                savedUserAuth.setName(reader.nextString());
479                                        } else if (name.equals("sourceClass")) {
480                                                savedUserAuth.setSourceClass(reader.nextString());
481                                        } else if (name.equals("authenticated")) {
482                                                savedUserAuth.setAuthenticated(reader.nextBoolean());
483                                        } else if (name.equals("authorities")) {
484                                                Set<String> authorityStrs = readSet(reader);
485                                                Set<GrantedAuthority> authorities = new HashSet<GrantedAuthority>();
486                                                for (String s : authorityStrs) {
487                                                        GrantedAuthority ga = new SimpleGrantedAuthority(s);
488                                                        authorities.add(ga);
489                                                }
490                                                savedUserAuth.setAuthorities(authorities);
491                                        } else {
492                                                logger.debug("Found unexpected entry");
493                                                reader.skipValue();
494                                        }
495                                        break;
496                                default:
497                                        logger.debug("Found unexpected entry");
498                                        reader.skipValue();
499                                        continue;
500                        }
501                }
502
503                reader.endObject();
504                return savedUserAuth;
505        }

511        private void readGrants(JsonReader reader) throws IOException {
512                reader.beginArray();
513                while (reader.hasNext()) {
514                        ApprovedSite site = new ApprovedSite();
515                        Long currentId = null;
516                        Long whitelistedSiteId = null;
517                        Set<Long> tokenIds = null;
518                        reader.beginObject();
519                        while (reader.hasNext()) {
520                                switch (reader.peek()) {
521                                        case END_OBJECT:
522                                                continue;
523                                        case NAME:
524                                                String name = reader.nextName();
525                                                if (reader.peek() == JsonToken.NULL) {
526                                                        reader.skipValue();
527                                                } else if (name.equals("id")) {
528                                                        currentId = reader.nextLong();
529                                                } else if (name.equals("accessDate")) {
530                                                        Date date = utcToDate(reader.nextString());
531                                                        site.setAccessDate(date);
532                                                } else if (name.equals("clientId")) {
533                                                        site.setClientId(reader.nextString());
534                                                } else if (name.equals("creationDate")) {
535                                                        Date date = utcToDate(reader.nextString());
536                                                        site.setCreationDate(date);
537                                                } else if (name.equals("timeoutDate")) {
538                                                        Date date = utcToDate(reader.nextString());
539                                                        site.setTimeoutDate(date);
540                                                } else if (name.equals("userId")) {
541                                                        site.setUserId(reader.nextString());
542                                                } else if (name.equals("allowedScopes")) {
543                                                        Set<String> allowedScopes = readSet(reader);
544                                                        site.setAllowedScopes(allowedScopes);
545                                                } else if (name.equals("whitelistedSiteId")) {
546                                                        whitelistedSiteId = reader.nextLong();
547                                                } else if (name.equals("approvedAccessTokens")) {
548                                                        tokenIds = readSet(reader);
549                                                } else {
550                                                        logger.debug("Found unexpected entry");
551                                                        reader.skipValue();
552                                                }
553                                                break;
554                                        default:
555                                                logger.debug("Found unexpected entry");
556                                                reader.skipValue();
557                                                continue;
558                                }
559                        }
560                        reader.endObject();
561                        Long newId = approvedSiteRepository.save(site).getId();
562                        maps.getGrantOldToNewIdMap().put(currentId, newId);
563                        if (whitelistedSiteId != null) {
564                                logger.debug("Ignoring whitelisted site marker on approved site.");
565                        }
566                        if (tokenIds != null) {
567                                maps.getGrantToAccessTokensRefs().put(currentId, tokenIds);
568                        }
569                        logger.debug("Read grant {}", currentId);
570                }
571                reader.endArray();
572                logger.info("Done reading grants");
573        }

794        /**
795         * Read the list of system scopes from the reader and insert them into the
796         * scope repository.
797         *
798         * @param reader
799         * @throws IOException
800         */
801        private void readSystemScopes(JsonReader reader) throws IOException {
802                reader.beginArray();
803                while (reader.hasNext()) {
804                        SystemScope scope = new SystemScope();
805                        reader.beginObject();
806                        while (reader.hasNext()) {
807                                switch (reader.peek()) {
808                                        case END_OBJECT:
809                                                continue;
810                                        case NAME:
811                                                String name = reader.nextName();
812                                                if (reader.peek() == JsonToken.NULL) {
813                                                        reader.skipValue();
814                                                } else if (name.equals("value")) {
815                                                        scope.setValue(reader.nextString());
816                                                } else if (name.equals("description")) {
817                                                        scope.setDescription(reader.nextString());
818                                                } else if (name.equals("allowDynReg")) {
819                                                        // previously "allowDynReg" scopes are now tagged as "not restricted" and vice versa
820                                                        scope.setRestricted(!reader.nextBoolean());
821                                                } else if (name.equals("defaultScope")) {
822                                                        scope.setDefaultScope(reader.nextBoolean());
823                                                } else if (name.equals("icon")) {
824                                                        scope.setIcon(reader.nextString());
825                                                } else {
826                                                        logger.debug("found unexpected entry");
827                                                        reader.skipValue();
828                                                }
829                                                break;
830                                        default:
831                                                logger.debug("Found unexpected entry");
832                                                reader.skipValue();
833                                                continue;
834                                }
835                        }
836                        reader.endObject();
837                        sysScopeRepository.save(scope);
838                }
839                reader.endArray();
840                logger.info("Done reading system scopes");
841        }

843        private void fixObjectReferences() {
844                for (Long oldRefreshTokenId : maps.getRefreshTokenToClientRefs().keySet()) {
845                        String clientRef = maps.getRefreshTokenToClientRefs().get(oldRefreshTokenId);
846                        ClientDetailsEntity client = clientRepository.getClientByClientId(clientRef);
847                        Long newRefreshTokenId = maps.getRefreshTokenOldToNewIdMap().get(oldRefreshTokenId);
848                        OAuth2RefreshTokenEntity refreshToken = tokenRepository.getRefreshTokenById(newRefreshTokenId);
849                        refreshToken.setClient(client);
850                        tokenRepository.saveRefreshToken(refreshToken);
851                }
852                for (Long oldRefreshTokenId : maps.getRefreshTokenToAuthHolderRefs().keySet()) {
853                        Long oldAuthHolderId = maps.getRefreshTokenToAuthHolderRefs().get(oldRefreshTokenId);
854                        Long newAuthHolderId = maps.getAuthHolderOldToNewIdMap().get(oldAuthHolderId);
855                        AuthenticationHolderEntity authHolder = authHolderRepository.getById(newAuthHolderId);
856                        Long newRefreshTokenId = maps.getRefreshTokenOldToNewIdMap().get(oldRefreshTokenId);
857                        OAuth2RefreshTokenEntity refreshToken = tokenRepository.getRefreshTokenById(newRefreshTokenId);
858                        refreshToken.setAuthenticationHolder(authHolder);
859                        tokenRepository.saveRefreshToken(refreshToken);
860                }
861                for (Long oldAccessTokenId : maps.getAccessTokenToClientRefs().keySet()) {
862                        String clientRef = maps.getAccessTokenToClientRefs().get(oldAccessTokenId);
863                        ClientDetailsEntity client = clientRepository.getClientByClientId(clientRef);
864                        Long newAccessTokenId = maps.getAccessTokenOldToNewIdMap().get(oldAccessTokenId);
865                        OAuth2AccessTokenEntity accessToken = tokenRepository.getAccessTokenById(newAccessTokenId);
866                        accessToken.setClient(client);
867                        tokenRepository.saveAccessToken(accessToken);
868                }
869                for (Long oldAccessTokenId : maps.getAccessTokenToAuthHolderRefs().keySet()) {
870                        Long oldAuthHolderId = maps.getAccessTokenToAuthHolderRefs().get(oldAccessTokenId);
871                        Long newAuthHolderId = maps.getAuthHolderOldToNewIdMap().get(oldAuthHolderId);
872                        AuthenticationHolderEntity authHolder = authHolderRepository.getById(newAuthHolderId);
873                        Long newAccessTokenId = maps.getAccessTokenOldToNewIdMap().get(oldAccessTokenId);
874                        OAuth2AccessTokenEntity accessToken = tokenRepository.getAccessTokenById(newAccessTokenId);
875                        accessToken.setAuthenticationHolder(authHolder);
876                        tokenRepository.saveAccessToken(accessToken);
877                }
878                maps.getAccessTokenToAuthHolderRefs().clear();
879                for (Long oldAccessTokenId : maps.getAccessTokenToRefreshTokenRefs().keySet()) {
880                        Long oldRefreshTokenId = maps.getAccessTokenToRefreshTokenRefs().get(oldAccessTokenId);
881                        Long newRefreshTokenId = maps.getRefreshTokenOldToNewIdMap().get(oldRefreshTokenId);
882                        OAuth2RefreshTokenEntity refreshToken = tokenRepository.getRefreshTokenById(newRefreshTokenId);
883                        Long newAccessTokenId = maps.getAccessTokenOldToNewIdMap().get(oldAccessTokenId);
884                        OAuth2AccessTokenEntity accessToken = tokenRepository.getAccessTokenById(newAccessTokenId);
885                        accessToken.setRefreshToken(refreshToken);
886                        tokenRepository.saveAccessToken(accessToken);
887                }
888                for (Long oldGrantId : maps.getGrantToAccessTokensRefs().keySet()) {
889                        Set<Long> oldAccessTokenIds = maps.getGrantToAccessTokensRefs().get(oldGrantId);
890
891                        Long newGrantId = maps.getGrantOldToNewIdMap().get(oldGrantId);
892                        ApprovedSite site = approvedSiteRepository.getById(newGrantId);
893
894                        for(Long oldTokenId : oldAccessTokenIds) {
895                                Long newTokenId = maps.getAccessTokenOldToNewIdMap().get(oldTokenId);
896                                OAuth2AccessTokenEntity token = tokenRepository.getAccessTokenById(newTokenId);
897                                token.setApprovedSite(site);
898                                tokenRepository.saveAccessToken(token);
899                        }
900
901                        approvedSiteRepository.save(site);
902                }
903        }
