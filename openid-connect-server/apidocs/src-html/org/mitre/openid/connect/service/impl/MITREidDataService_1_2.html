185        /* (non-Javadoc)
186         * @see org.mitre.openid.connect.service.MITREidDataService#importData(com.google.gson.stream.JsonReader)
187         */
188        @Override
189        public void importData(JsonReader reader) throws IOException {
190
191                logger.info("Reading configuration for 1.2");
192
193                // this *HAS* to start as an object
194                reader.beginObject();
195
196                while (reader.hasNext()) {
197                        JsonToken tok = reader.peek();
198                        switch (tok) {
199                                case NAME:
200                                        String name = reader.nextName();
201                                        // find out which member it is
202                                        if (name.equals(CLIENTS)) {
203                                                readClients(reader);
204                                        } else if (name.equals(GRANTS)) {
205                                                readGrants(reader);
206                                        } else if (name.equals(WHITELISTEDSITES)) {
207                                                readWhitelistedSites(reader);
208                                        } else if (name.equals(BLACKLISTEDSITES)) {
209                                                readBlacklistedSites(reader);
210                                        } else if (name.equals(AUTHENTICATIONHOLDERS)) {
211                                                readAuthenticationHolders(reader);
212                                        } else if (name.equals(ACCESSTOKENS)) {
213                                                readAccessTokens(reader);
214                                        } else if (name.equals(REFRESHTOKENS)) {
215                                                readRefreshTokens(reader);
216                                        } else if (name.equals(SYSTEMSCOPES)) {
217                                                readSystemScopes(reader);
218                                        } else {
219                                                for (MITREidDataServiceExtension extension : extensions) {
220                                                        if (extension.supportsVersion(THIS_VERSION)) {
221                                                                extension.importExtensionData(name, reader);
222                                                                break;
223                                                        }
224                                                }
225                                                // unknown token, skip it
226                                                reader.skipValue();
227                                        }
228                                        break;
229                                case END_OBJECT:
230                                        // the object ended, we're done here
231                                        reader.endObject();
232                                        continue;
233                                default:
234                                        logger.debug("Found unexpected entry");
235                                        reader.skipValue();
236                                        continue;
237                        }
238                }
239                fixObjectReferences();
240                for (MITREidDataServiceExtension extension : extensions) {
241                        if (extension.supportsVersion(THIS_VERSION)) {
242                                extension.fixExtensionObjectReferences(maps);
243                                break;
244                        }
245                }
246                maps.clearAll();
247        }

253         * @param reader
254         * @throws IOException
255         */
256        private void readRefreshTokens(JsonReader reader) throws IOException {
257                reader.beginArray();
258                while (reader.hasNext()) {
259                        OAuth2RefreshTokenEntity token = new OAuth2RefreshTokenEntity();
260                        reader.beginObject();
261                        Long currentId = null;
262                        String clientId = null;
263                        Long authHolderId = null;
264                        while (reader.hasNext()) {
265                                switch (reader.peek()) {
266                                        case END_OBJECT:
267                                                continue;
268                                        case NAME:
269                                                String name = reader.nextName();
270                                                if (reader.peek() == JsonToken.NULL) {
271                                                        reader.skipValue();
272                                                } else if (name.equals(ID)) {
273                                                        currentId = reader.nextLong();
274                                                } else if (name.equals(EXPIRATION)) {
275                                                        Date date = utcToDate(reader.nextString());
276                                                        token.setExpiration(date);
277                                                } else if (name.equals(VALUE)) {
278                                                        String value = reader.nextString();
279                                                        try {
280                                                                token.setJwt(JWTParser.parse(value));
281                                                        } catch (ParseException ex) {
282                                                                logger.error("Unable to set refresh token value to {}", value, ex);
283                                                        }
284                                                } else if (name.equals(CLIENT_ID)) {
285                                                        clientId = reader.nextString();
286                                                } else if (name.equals(AUTHENTICATION_HOLDER_ID)) {
287                                                        authHolderId = reader.nextLong();
288                                                } else {
289                                                        logger.debug("Found unexpected entry");
290                                                        reader.skipValue();
291                                                }
292                                                break;
293                                        default:
294                                                logger.debug("Found unexpected entry");
295                                                reader.skipValue();
296                                                continue;
297                                }
298                        }
299                        reader.endObject();
300                        Long newId = tokenRepository.saveRefreshToken(token).getId();
301                        maps.getRefreshTokenToClientRefs().put(currentId, clientId);
302                        maps.getRefreshTokenToAuthHolderRefs().put(currentId, authHolderId);
303                        maps.getRefreshTokenOldToNewIdMap().put(currentId, newId);
304                        logger.debug("Read refresh token {}", currentId);
305                }
306                reader.endArray();
307                logger.info("Done reading refresh tokens");
308        }

317        private void readAccessTokens(JsonReader reader) throws IOException {
318                reader.beginArray();
319                while (reader.hasNext()) {
320                        OAuth2AccessTokenEntity token = new OAuth2AccessTokenEntity();
321                        reader.beginObject();
322                        Long currentId = null;
323                        String clientId = null;
324                        Long authHolderId = null;
325                        Long refreshTokenId = null;
326                        while (reader.hasNext()) {
327                                switch (reader.peek()) {
328                                        case END_OBJECT:
329                                                continue;
330                                        case NAME:
331                                                String name = reader.nextName();
332                                                if (reader.peek() == JsonToken.NULL) {
333                                                        reader.skipValue();
334                                                } else if (name.equals(ID)) {
335                                                        currentId = reader.nextLong();
336                                                } else if (name.equals(EXPIRATION)) {
337                                                        Date date = utcToDate(reader.nextString());
338                                                        token.setExpiration(date);
339                                                } else if (name.equals(VALUE)) {
340                                                        String value = reader.nextString();
341                                                        try {
342                                                                // all tokens are JWTs
343                                                                token.setJwt(JWTParser.parse(value));
344                                                        } catch (ParseException ex) {
345                                                                logger.error("Unable to set refresh token value to {}", value, ex);
346                                                        }
347                                                } else if (name.equals(CLIENT_ID)) {
348                                                        clientId = reader.nextString();
349                                                } else if (name.equals(AUTHENTICATION_HOLDER_ID)) {
350                                                        authHolderId = reader.nextLong();
351                                                } else if (name.equals(REFRESH_TOKEN_ID)) {
352                                                        refreshTokenId = reader.nextLong();
353                                                } else if (name.equals(SCOPE)) {
354                                                        Set<String> scope = readSet(reader);
355                                                        token.setScope(scope);
356                                                } else if (name.equals(TYPE)) {
357                                                        token.setTokenType(reader.nextString());
358                                                } else {
359                                                        logger.debug("Found unexpected entry");
360                                                        reader.skipValue();
361                                                }
362                                                break;
363                                        default:
364                                                logger.debug("Found unexpected entry");
365                                                reader.skipValue();
366                                                continue;
367                                }
368                        }
369                        reader.endObject();
370                        Long newId = tokenRepository.saveAccessToken(token).getId();
371                        maps.getAccessTokenToClientRefs().put(currentId, clientId);
372                        maps.getAccessTokenToAuthHolderRefs().put(currentId, authHolderId);
373                        if (refreshTokenId != null) {
374                                maps.getAccessTokenToRefreshTokenRefs().put(currentId, refreshTokenId);
375                        }
376                        maps.getAccessTokenOldToNewIdMap().put(currentId, newId);
377                        logger.debug("Read access token {}", currentId);
378                }
379                reader.endArray();
380                logger.info("Done reading access tokens");
381        }

386        private void readAuthenticationHolders(JsonReader reader) throws IOException {
387                reader.beginArray();
388                while (reader.hasNext()) {
389                        AuthenticationHolderEntity ahe = new AuthenticationHolderEntity();
390                        reader.beginObject();
391                        Long currentId = null;
392                        while (reader.hasNext()) {
393                                switch (reader.peek()) {
394                                        case END_OBJECT:
395                                                continue;
396                                        case NAME:
397                                                String name = reader.nextName();
398                                                if (reader.peek() == JsonToken.NULL) {
399                                                        reader.skipValue();
400                                                } else if (name.equals(ID)) {
401                                                        currentId = reader.nextLong();
402                                                } else if (name.equals(REQUEST_PARAMETERS)) {
403                                                        ahe.setRequestParameters(readMap(reader));
404                                                } else if (name.equals(CLIENT_ID)) {
405                                                        ahe.setClientId(reader.nextString());
406                                                } else if (name.equals(SCOPE)) {
407                                                        ahe.setScope(readSet(reader));
408                                                } else if (name.equals(RESOURCE_IDS)) {
409                                                        ahe.setResourceIds(readSet(reader));
410                                                } else if (name.equals(AUTHORITIES)) {
411                                                        Set<String> authorityStrs = readSet(reader);
412                                                        Set<GrantedAuthority> authorities = new HashSet<GrantedAuthority>();
413                                                        for (String s : authorityStrs) {
414                                                                GrantedAuthority ga = new SimpleGrantedAuthority(s);
415                                                                authorities.add(ga);
416                                                        }
417                                                        ahe.setAuthorities(authorities);
418                                                } else if (name.equals(APPROVED)) {
419                                                        ahe.setApproved(reader.nextBoolean());
420                                                } else if (name.equals(REDIRECT_URI)) {
421                                                        ahe.setRedirectUri(reader.nextString());
422                                                } else if (name.equals(RESPONSE_TYPES)) {
423                                                        ahe.setResponseTypes(readSet(reader));
424                                                } else if (name.equals(EXTENSIONS)) {
425                                                        ahe.setExtensions(readMap(reader));
426                                                } else if (name.equals(SAVED_USER_AUTHENTICATION)) {
427                                                        ahe.setUserAuth(readSavedUserAuthentication(reader));
428                                                } else {
429                                                        logger.debug("Found unexpected entry");
430                                                        reader.skipValue();
431                                                }
432                                                break;
433                                        default:
434                                                logger.debug("Found unexpected entry");
435                                                reader.skipValue();
436                                                continue;
437                                }
438                        }
439                        reader.endObject();
440                        Long newId = authHolderRepository.save(ahe).getId();
441                        maps.getAuthHolderOldToNewIdMap().put(currentId, newId);
442                        logger.debug("Read authentication holder {}", currentId);
443                }
444                reader.endArray();
445                logger.info("Done reading authentication holders");
446        }

499        private void readGrants(JsonReader reader) throws IOException {
500                reader.beginArray();
501                while (reader.hasNext()) {
502                        ApprovedSite site = new ApprovedSite();
503                        Long currentId = null;
504                        Set<Long> tokenIds = null;
505                        reader.beginObject();
506                        while (reader.hasNext()) {
507                                switch (reader.peek()) {
508                                        case END_OBJECT:
509                                                continue;
510                                        case NAME:
511                                                String name = reader.nextName();
512                                                if (reader.peek() == JsonToken.NULL) {
513                                                        reader.skipValue();
514                                                } else if (name.equals(ID)) {
515                                                        currentId = reader.nextLong();
516                                                } else if (name.equals(ACCESS_DATE)) {
517                                                        Date date = utcToDate(reader.nextString());
518                                                        site.setAccessDate(date);
519                                                } else if (name.equals(CLIENT_ID)) {
520                                                        site.setClientId(reader.nextString());
521                                                } else if (name.equals(CREATION_DATE)) {
522                                                        Date date = utcToDate(reader.nextString());
523                                                        site.setCreationDate(date);
524                                                } else if (name.equals(TIMEOUT_DATE)) {
525                                                        Date date = utcToDate(reader.nextString());
526                                                        site.setTimeoutDate(date);
527                                                } else if (name.equals(USER_ID)) {
528                                                        site.setUserId(reader.nextString());
529                                                } else if (name.equals(ALLOWED_SCOPES)) {
530                                                        Set<String> allowedScopes = readSet(reader);
531                                                        site.setAllowedScopes(allowedScopes);
532                                                } else if (name.equals(APPROVED_ACCESS_TOKENS)) {
533                                                        tokenIds = readSet(reader);
534                                                } else {
535                                                        logger.debug("Found unexpected entry");
536                                                        reader.skipValue();
537                                                }
538                                                break;
539                                        default:
540                                                logger.debug("Found unexpected entry");
541                                                reader.skipValue();
542                                                continue;
543                                }
544                        }
545                        reader.endObject();
546                        Long newId = approvedSiteRepository.save(site).getId();
547                        maps.getGrantOldToNewIdMap().put(currentId, newId);
548                        if (tokenIds != null) {
549                                maps.getGrantToAccessTokensRefs().put(currentId, tokenIds);
550                        }
551                        logger.debug("Read grant {}", currentId);
552                }
553                reader.endArray();
554                logger.info("Done reading grants");
555        }

608                        reader.beginObject();
609                        while (reader.hasNext()) {
610                                switch (reader.peek()) {
611                                        case END_OBJECT:
612                                                continue;
613                                        case NAME:
614                                                String name = reader.nextName();
615                                                if (name.equals(ID)) {
616                                                        reader.skipValue();
617                                                } else if (name.equals(URI)) {
618                                                        blSite.setUri(reader.nextString());
619                                                } else {
620                                                        logger.debug("Found unexpected entry");
621                                                        reader.skipValue();
622                                                }
623                                                break;
624                                        default:
625                                                logger.debug("Found unexpected entry");
626                                                reader.skipValue();
627                                                continue;
628                                }
629                        }
630                        reader.endObject();
631                        blSiteRepository.save(blSite);
632                }
633                reader.endArray();
634                logger.info("Done reading blacklisted sites");
635        }

666                                                        Set<GrantedAuthority> authorities = new HashSet<GrantedAuthority>();
667                                                        for (String s : authorityStrs) {

839        private void fixObjectReferences() {
840                logger.info("Fixing object references...");
841                for (Long oldRefreshTokenId : maps.getRefreshTokenToClientRefs().keySet()) {
842                        String clientRef = maps.getRefreshTokenToClientRefs().get(oldRefreshTokenId);
843                        ClientDetailsEntity client = clientRepository.getClientByClientId(clientRef);
844                        Long newRefreshTokenId = maps.getRefreshTokenOldToNewIdMap().get(oldRefreshTokenId);
845                        OAuth2RefreshTokenEntity refreshToken = tokenRepository.getRefreshTokenById(newRefreshTokenId);
846                        refreshToken.setClient(client);
847                        tokenRepository.saveRefreshToken(refreshToken);
848                }
849                for (Long oldRefreshTokenId : maps.getRefreshTokenToAuthHolderRefs().keySet()) {
850                        Long oldAuthHolderId = maps.getRefreshTokenToAuthHolderRefs().get(oldRefreshTokenId);
851                        Long newAuthHolderId = maps.getAuthHolderOldToNewIdMap().get(oldAuthHolderId);
852                        AuthenticationHolderEntity authHolder = authHolderRepository.getById(newAuthHolderId);
853                        Long newRefreshTokenId = maps.getRefreshTokenOldToNewIdMap().get(oldRefreshTokenId);
854                        OAuth2RefreshTokenEntity refreshToken = tokenRepository.getRefreshTokenById(newRefreshTokenId);
855                        refreshToken.setAuthenticationHolder(authHolder);
856                        tokenRepository.saveRefreshToken(refreshToken);
857                }
858                for (Long oldAccessTokenId : maps.getAccessTokenToClientRefs().keySet()) {
859                        String clientRef = maps.getAccessTokenToClientRefs().get(oldAccessTokenId);
860                        ClientDetailsEntity client = clientRepository.getClientByClientId(clientRef);
861                        Long newAccessTokenId = maps.getAccessTokenOldToNewIdMap().get(oldAccessTokenId);
862                        OAuth2AccessTokenEntity accessToken = tokenRepository.getAccessTokenById(newAccessTokenId);
863                        accessToken.setClient(client);
864                        tokenRepository.saveAccessToken(accessToken);
865                }
866                for (Long oldAccessTokenId : maps.getAccessTokenToAuthHolderRefs().keySet()) {
867                        Long oldAuthHolderId = maps.getAccessTokenToAuthHolderRefs().get(oldAccessTokenId);
868                        Long newAuthHolderId = maps.getAuthHolderOldToNewIdMap().get(oldAuthHolderId);
869                        AuthenticationHolderEntity authHolder = authHolderRepository.getById(newAuthHolderId);
870                        Long newAccessTokenId = maps.getAccessTokenOldToNewIdMap().get(oldAccessTokenId);
871                        OAuth2AccessTokenEntity accessToken = tokenRepository.getAccessTokenById(newAccessTokenId);
872                        accessToken.setAuthenticationHolder(authHolder);
873                        tokenRepository.saveAccessToken(accessToken);
874                }
875                for (Long oldAccessTokenId : maps.getAccessTokenToRefreshTokenRefs().keySet()) {
876                        Long oldRefreshTokenId = maps.getAccessTokenToRefreshTokenRefs().get(oldAccessTokenId);
877                        Long newRefreshTokenId = maps.getRefreshTokenOldToNewIdMap().get(oldRefreshTokenId);
878                        OAuth2RefreshTokenEntity refreshToken = tokenRepository.getRefreshTokenById(newRefreshTokenId);
879                        Long newAccessTokenId = maps.getAccessTokenOldToNewIdMap().get(oldAccessTokenId);
880                        OAuth2AccessTokenEntity accessToken = tokenRepository.getAccessTokenById(newAccessTokenId);
881                        accessToken.setRefreshToken(refreshToken);
882                        tokenRepository.saveAccessToken(accessToken);
883                }
884                for (Long oldGrantId : maps.getGrantToAccessTokensRefs().keySet()) {
885                        Set<Long> oldAccessTokenIds = maps.getGrantToAccessTokensRefs().get(oldGrantId);
886
887                        Long newGrantId = maps.getGrantOldToNewIdMap().get(oldGrantId);
888                        ApprovedSite site = approvedSiteRepository.getById(newGrantId);
889
890                        for(Long oldTokenId : oldAccessTokenIds) {
891                                Long newTokenId = maps.getAccessTokenOldToNewIdMap().get(oldTokenId);
892                                OAuth2AccessTokenEntity token = tokenRepository.getAccessTokenById(newTokenId);
893                                token.setApprovedSite(site);
894                                tokenRepository.saveAccessToken(token);
895                        }
896
897                        approvedSiteRepository.save(site);
898                }
899                logger.info("Done fixing object references.");
900        }
