004package org.mitre.openid.connect.service.impl;
005
006import java.util.Collection;
007import java.util.HashMap;
008import java.util.HashSet;
009import java.util.Map;
010import java.util.Set;
011
012import org.mitre.openid.connect.model.ApprovedSite;
013import org.mitre.openid.connect.service.ApprovedSiteService;
014import org.mitre.openid.connect.service.StatsService;
015import org.springframework.beans.factory.annotation.Autowired;
016import org.springframework.stereotype.Service;
017
018/**
019 * @author jricher
020 *
021 */
022@Service
023public class DefaultStatsService implements StatsService {
024
025        @Autowired
026        private ApprovedSiteService approvedSiteService;
027        
028        @Override
029    public Map<String, Integer> calculateSummaryStats() {
030        // get all approved sites
031        Collection<ApprovedSite> allSites = approvedSiteService.getAll();
032    
033        // process to find number of unique users and sites
034        Set<String> userIds = new HashSet<String>();
035        Set<String> clientIds = new HashSet<String>();
036        for (ApprovedSite approvedSite : allSites) {
037            userIds.add(approvedSite.getUserId());
038            clientIds.add(approvedSite.getClientId());
039        }
040        
041        Map<String, Integer> e = new HashMap<String, Integer>();
042        
043        e.put("approvalCount", allSites.size());
044        e.put("userCount", userIds.size());
045        e.put("clientCount", clientIds.size());
046        return e;
047    }
048
049}
