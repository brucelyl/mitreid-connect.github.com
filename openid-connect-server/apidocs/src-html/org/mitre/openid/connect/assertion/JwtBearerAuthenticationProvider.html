020package org.mitre.openid.connect.assertion;
021
022import java.text.ParseException;
023import java.util.Date;
024import java.util.HashSet;
025import java.util.Set;
026
027import org.mitre.jwt.signer.service.JwtSigningAndValidationService;
028import org.mitre.jwt.signer.service.impl.JWKSetCacheService;
029import org.mitre.jwt.signer.service.impl.SymmetricCacheService;
030import org.mitre.oauth2.model.ClientDetailsEntity;
031import org.mitre.oauth2.model.ClientDetailsEntity.AuthMethod;
032import org.mitre.oauth2.service.ClientDetailsEntityService;
033import org.mitre.openid.connect.config.ConfigurationPropertiesBean;
034import org.slf4j.Logger;
035import org.slf4j.LoggerFactory;
036import org.springframework.beans.factory.annotation.Autowired;
037import org.springframework.security.authentication.AuthenticationProvider;
038import org.springframework.security.authentication.AuthenticationServiceException;
039import org.springframework.security.core.Authentication;
040import org.springframework.security.core.AuthenticationException;
041import org.springframework.security.core.GrantedAuthority;
042import org.springframework.security.core.authority.SimpleGrantedAuthority;
043import org.springframework.security.core.userdetails.UsernameNotFoundException;
044import org.springframework.security.oauth2.common.exceptions.InvalidClientException;
045
046import com.nimbusds.jose.JWSAlgorithm;
047import com.nimbusds.jwt.JWT;
048import com.nimbusds.jwt.ReadOnlyJWTClaimsSet;
049import com.nimbusds.jwt.SignedJWT;
050
051/**
052 * @author jricher
053 *
054 */
055public class JwtBearerAuthenticationProvider implements AuthenticationProvider {
056
057        private static final Logger logger = LoggerFactory.getLogger(JwtBearerAuthenticationProvider.class);
058
059        private static final GrantedAuthority ROLE_CLIENT = new SimpleGrantedAuthority("ROLE_CLIENT");
060        
061        // map of verifiers, load keys for clients
062        @Autowired
063        private JWKSetCacheService validators;
064
065        // map of symmetric verifiers for client secrets
066        @Autowired
067        private SymmetricCacheService symmetricCacheService;
068
069        // Allow for time sync issues by having a window of X seconds.
070        private int timeSkewAllowance = 300;
071
072        // to load clients
073        @Autowired
074        private ClientDetailsEntityService clientService;
075
076        // to get our server's issuer url
077        @Autowired
078        private ConfigurationPropertiesBean config;
079
080        /**
081         * Try to validate the client credentials by parsing and validating the JWT.
082         */
083        @Override
084        public Authentication authenticate(Authentication authentication) throws AuthenticationException {
085
086                JwtBearerAssertionAuthenticationToken jwtAuth = (JwtBearerAssertionAuthenticationToken)authentication;
087
088
089                try {
090                        ClientDetailsEntity client = clientService.loadClientByClientId(jwtAuth.getClientId());
091
092                        JWT jwt = jwtAuth.getJwt();
093                        ReadOnlyJWTClaimsSet jwtClaims = jwt.getJWTClaimsSet();
094
095                        // check the signature with nimbus
096                        if (jwt instanceof SignedJWT) {
097                                SignedJWT jws = (SignedJWT)jwt;
098
099                                JWSAlgorithm alg = jws.getHeader().getAlgorithm();
100
101                                if (client.getTokenEndpointAuthSigningAlg() != null &&
102                                                !client.getTokenEndpointAuthSigningAlg().equals(alg)) {
103                                        throw new InvalidClientException("Client's registered request object signing algorithm (" + client.getRequestObjectSigningAlg() + ") does not match request object's actual algorithm (" + alg.getName() + ")");
104                                }
105
106                                if (client.getTokenEndpointAuthMethod() == null ||
107                                                client.getTokenEndpointAuthMethod().equals(AuthMethod.NONE) ||
108                                                client.getTokenEndpointAuthMethod().equals(AuthMethod.SECRET_BASIC) ||
109                                                client.getTokenEndpointAuthMethod().equals(AuthMethod.SECRET_POST)) {
110                                        
111                                        // this client doesn't support this type of authentication
112                                        throw new AuthenticationServiceException("Client does not support this authentication method.");
113                                        
114                                } else if (client.getTokenEndpointAuthMethod().equals(AuthMethod.PRIVATE_KEY) &&
115                                                (alg.equals(JWSAlgorithm.RS256)
116                                                                || alg.equals(JWSAlgorithm.RS384)
117                                                                || alg.equals(JWSAlgorithm.RS512))) {
118
119                                        JwtSigningAndValidationService validator = validators.getValidator(client.getJwksUri());
120
121                                        if (validator == null) {
122                                                throw new AuthenticationServiceException("Unable to create signature validator for client's JWKS URI: " + client.getJwksUri());
123                                        }
124
125                                        if (!validator.validateSignature(jws)) {
126                                                throw new AuthenticationServiceException("Signature did not validate for presented JWT authentication.");
127                                        }
128                                } else if (client.getTokenEndpointAuthMethod().equals(AuthMethod.SECRET_JWT) &&
129                                                (alg.equals(JWSAlgorithm.HS256)
130                                                                || alg.equals(JWSAlgorithm.HS384)
131                                                                || alg.equals(JWSAlgorithm.HS512))) {
132
133                                        // it's HMAC, we need to make a validator based on the client secret
134
135                                        JwtSigningAndValidationService validator = symmetricCacheService.getSymmetricValidtor(client);
136
137                                        if (validator == null) {
138                                                throw new AuthenticationServiceException("Unable to create signature validator for client's secret: " + client.getClientSecret());
139                                        }
140
141                                        if (!validator.validateSignature(jws)) {
142                                                throw new AuthenticationServiceException("Signature did not validate for presented JWT authentication.");
143                                        }
144
145                                }
146                        }
147
148                        // check the issuer
149                        if (jwtClaims.getIssuer() == null) {
150                                throw new AuthenticationServiceException("Assertion Token Issuer is null");
151                        } else if (!jwtClaims.getIssuer().equals(client.getClientId())){
152                                throw new AuthenticationServiceException("Issuers do not match, expected " + client.getClientId() + " got " + jwtClaims.getIssuer());
153                        }
154
155                        // check expiration
156                        if (jwtClaims.getExpirationTime() == null) {
157                                throw new AuthenticationServiceException("Assertion Token does not have required expiration claim");
158                        } else {
159                                // it's not null, see if it's expired
160                                Date now = new Date(System.currentTimeMillis() - (timeSkewAllowance * 1000));
161                                if (now.after(jwtClaims.getExpirationTime())) {
162                                        throw new AuthenticationServiceException("Assertion Token is expired: " + jwtClaims.getExpirationTime());
163                                }
164                        }
165
166                        // check not before
167                        if (jwtClaims.getNotBeforeTime() != null) {
168                                Date now = new Date(System.currentTimeMillis() + (timeSkewAllowance * 1000));
169                                if (now.before(jwtClaims.getNotBeforeTime())){
170                                        throw new AuthenticationServiceException("Assertion Token not valid untill: " + jwtClaims.getNotBeforeTime());
171                                }
172                        }
173
174                        // check issued at
175                        if (jwtClaims.getIssueTime() != null) {
176                                // since it's not null, see if it was issued in the future
177                                Date now = new Date(System.currentTimeMillis() + (timeSkewAllowance * 1000));
178                                if (now.before(jwtClaims.getIssueTime())) {
179                                        throw new AuthenticationServiceException("Assertion Token was issued in the future: " + jwtClaims.getIssueTime());
180                                }
181                        }
182
183                        // check audience
184                        if (jwtClaims.getAudience() == null) {
185                                throw new AuthenticationServiceException("Assertion token audience is null");
186                        } else if (!(jwtClaims.getAudience().contains(config.getIssuer()) || jwtClaims.getAudience().contains(config.getIssuer() + "token"))) {
187                                throw new AuthenticationServiceException("Audience does not match, expected " + config.getIssuer() + " or " + (config.getIssuer() + "token") + " got " + jwtClaims.getAudience());
188                        }
189
190                        // IFF we managed to get all the way down here, the token is valid
191                        
192                        // add in the ROLE_CLIENT authority
193                        Set<GrantedAuthority> authorities = new HashSet<GrantedAuthority>(client.getAuthorities());
194                        authorities.add(ROLE_CLIENT);
195                        
196                        return new JwtBearerAssertionAuthenticationToken(client.getClientId(), jwt, authorities);
197
198                } catch (InvalidClientException e) {
199                        throw new UsernameNotFoundException("Could not find client: " + jwtAuth.getClientId());
200                } catch (ParseException e) {
201
202                        logger.error("Failure during authentication, error was: ", e);
203
204                        throw new AuthenticationServiceException("Invalid JWT format");
205                }
206        }
207
208        /**
209         * We support {@link JwtBearerAssertionAuthenticationToken}s only.
210         */
211        @Override
212        public boolean supports(Class<?> authentication) {
213                return (JwtBearerAssertionAuthenticationToken.class.isAssignableFrom(authentication));
214        }
