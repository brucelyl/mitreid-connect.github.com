016package org.mitre.openid.connect.util;
017
018import java.text.ParseException;
019import java.text.SimpleDateFormat;
020import java.util.Date;
021import java.util.TimeZone;
022
023import org.slf4j.Logger;
024import org.slf4j.LoggerFactory;
025
026/**
027 *
028 * @author arielak
029 */
030public class DateUtil {
031    private static final Logger log = LoggerFactory.getLogger(DateUtil.class);
032    private static final String ISO_FORMAT = "yyyy-MM-dd'T'HH:mm:ss.SSSZ";
033    private static final SimpleDateFormat sdf = new SimpleDateFormat(ISO_FORMAT);
034    private static final TimeZone utc = TimeZone.getTimeZone("UTC");    
035    
036    public static String toUTCString(Date date) {
037        if (date == null) {
038            return null;
039        }
040        sdf.setTimeZone(utc);
041        return sdf.format(date);
042    }
043
044    public static Date utcToDate(String s) {
045        if (s == null) {
046            return null;
047        }
048        Date d = null;
049        try {
050            d = sdf.parse(s);
051        } catch(ParseException ex) {
052            log.error("Unable to parse date string {}", s, ex);
053        }
054        return d;
055    }
056}
