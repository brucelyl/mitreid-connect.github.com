020package org.mitre.openid.connect.view;
021
022import java.io.IOException;
023import java.io.StringWriter;
024import java.io.Writer;
025import java.text.ParseException;
026import java.util.Date;
027import java.util.Map;
028import java.util.UUID;
029
030import javax.servlet.http.HttpServletRequest;
031import javax.servlet.http.HttpServletResponse;
032
033import org.mitre.jwt.encryption.service.JwtEncryptionAndDecryptionService;
034import org.mitre.jwt.signer.service.JwtSigningAndValidationService;
035import org.mitre.jwt.signer.service.impl.JWKSetCacheService;
036import org.mitre.jwt.signer.service.impl.SymmetricCacheService;
037import org.mitre.oauth2.model.ClientDetailsEntity;
038import org.mitre.openid.connect.config.ConfigurationPropertiesBean;
039import org.slf4j.Logger;
040import org.slf4j.LoggerFactory;
041import org.springframework.beans.factory.annotation.Autowired;
042import org.springframework.stereotype.Component;
043
044import com.google.common.base.Strings;
045import com.google.common.collect.Lists;
046import com.google.gson.JsonObject;
047import com.nimbusds.jose.Algorithm;
048import com.nimbusds.jose.JWEHeader;
049import com.nimbusds.jose.JWSAlgorithm;
050import com.nimbusds.jose.JWSHeader;
051import com.nimbusds.jwt.EncryptedJWT;
052import com.nimbusds.jwt.JWTClaimsSet;
053import com.nimbusds.jwt.SignedJWT;
054
055/**
056 * @author jricher
057 *
058 */
059@Component("userInfoJwtView")
060public class UserInfoJwtView extends UserInfoView {
061
062        private static Logger logger = LoggerFactory.getLogger(UserInfoJwtView.class);
063
064        @Autowired
065        private JwtSigningAndValidationService jwtService;
066
067        @Autowired
068        private ConfigurationPropertiesBean config;
069
070        @Autowired
071        private JWKSetCacheService encrypters;
072
073        @Autowired
074        private SymmetricCacheService symmetricCacheService;
075
076        @Override
077        protected void writeOut(JsonObject json, Map<String, Object> model,
078                        HttpServletRequest request, HttpServletResponse response) {
079
080                try {
081                        ClientDetailsEntity client = (ClientDetailsEntity)model.get("client");
082
083                        // use the parser to import the user claims into the object
084                        StringWriter writer = new StringWriter();
085                        gson.toJson(json, writer);
086
087                        response.setContentType("application/jwt");
088                        
089                        JWTClaimsSet claims = JWTClaimsSet.parse(writer.toString());
090
091                        claims.setAudience(Lists.newArrayList(client.getClientId()));
092
093                        claims.setIssuer(config.getIssuer());
094
095                        claims.setIssueTime(new Date());
096
097                        claims.setJWTID(UUID.randomUUID().toString()); // set a random NONCE in the middle of it
098
099
100                        if (client.getIdTokenEncryptedResponseAlg() != null && !client.getIdTokenEncryptedResponseAlg().equals(Algorithm.NONE)
101                                        && client.getIdTokenEncryptedResponseEnc() != null && !client.getIdTokenEncryptedResponseEnc().equals(Algorithm.NONE)
102                                        && !Strings.isNullOrEmpty(client.getJwksUri())) {
103
104                                // encrypt it to the client's key
105
106                                JwtEncryptionAndDecryptionService encrypter = encrypters.getEncrypter(client.getJwksUri());
107
108                                if (encrypter != null) {
109
110                                        EncryptedJWT encrypted = new EncryptedJWT(new JWEHeader(client.getIdTokenEncryptedResponseAlg(), client.getIdTokenEncryptedResponseEnc()), claims);
111
112                                        encrypter.encryptJwt(encrypted);
113
114
115                                        Writer out = response.getWriter();
116                                        out.write(encrypted.serialize());
117
118                                } else {
119                                        logger.error("Couldn't find encrypter for client: " + client.getClientId());
120                                }
121                        } else {
122
123                                JWSAlgorithm signingAlg = jwtService.getDefaultSigningAlgorithm();
124                                if (client.getUserInfoSignedResponseAlg() != null) {
125                                        signingAlg = client.getUserInfoSignedResponseAlg();
126                                }
127
128                                SignedJWT signed = new SignedJWT(new JWSHeader(signingAlg), claims);
129
130                                if (signingAlg.equals(JWSAlgorithm.HS256)
131                                                || signingAlg.equals(JWSAlgorithm.HS384)
132                                                || signingAlg.equals(JWSAlgorithm.HS512)) {
133
134                                        // sign it with the client's secret
135                                        JwtSigningAndValidationService signer = symmetricCacheService.getSymmetricValidtor(client);
136                                        signer.signJwt(signed);
137
138                                } else {
139                                        // sign it with the server's key
140                                        jwtService.signJwt(signed);
141                                }
142
143                                Writer out = response.getWriter();
144                                out.write(signed.serialize());
145                        }
146                } catch (IOException e) {
147                        logger.error("IO Exception in UserInfoJwtView", e);
148                } catch (ParseException e) {
149                        // TODO Auto-generated catch block
150                        e.printStackTrace();
151                }
152
153        }
154}
