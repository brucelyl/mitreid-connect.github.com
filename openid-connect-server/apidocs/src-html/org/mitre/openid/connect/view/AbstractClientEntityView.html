006import java.io.IOException;

014import org.mitre.jose.JWEAlgorithmEmbed;
015import org.mitre.jose.JWEEncryptionMethodEmbed;
016import org.mitre.jose.JWSAlgorithmEmbed;
017import org.slf4j.Logger;
018import org.slf4j.LoggerFactory;

020import org.springframework.validation.BeanPropertyBindingResult;

023import com.google.gson.ExclusionStrategy;
024import com.google.gson.FieldAttributes;
025import com.google.gson.Gson;
026import com.google.gson.GsonBuilder;
027import com.google.gson.JsonElement;
028import com.google.gson.JsonPrimitive;
029import com.google.gson.JsonSerializationContext;
030import com.google.gson.JsonSerializer;
031
032/**
033 * 
034 * Abstract superclass for client entity view, used with the ClientApi.
035 * 
036 * @see ClientEntityViewForUsers
037 * @see ClientEntityViewForAdmins
038 * 
039 * @author jricher
040 *
041 */
042public abstract class AbstractClientEntityView extends AbstractView {
043        private static Logger logger = LoggerFactory.getLogger(ClientEntityViewForAdmins.class);
044
045        private Gson gson = new GsonBuilder()
046            .setExclusionStrategies(getExclusionStrategy())
047            .registerTypeAdapter(JWSAlgorithmEmbed.class, new JsonSerializer<JWSAlgorithmEmbed>() {
048                        @Override
049            public JsonElement serialize(JWSAlgorithmEmbed src, Type typeOfSrc, JsonSerializationContext context) {
050                                if (src != null) {
051                                        return new JsonPrimitive(src.getAlgorithmName());
052                                } else {
053                                        return null;
054                                }
055            }
056            })
057            .registerTypeAdapter(JWEAlgorithmEmbed.class, new JsonSerializer<JWEAlgorithmEmbed>() {
058                        @Override
059            public JsonElement serialize(JWEAlgorithmEmbed src, Type typeOfSrc, JsonSerializationContext context) {
060                                if (src != null) {
061                                        return new JsonPrimitive(src.getAlgorithmName());
062                                } else {
063                                        return null;
064                                }
065            }
066            })
067            .registerTypeAdapter(JWEEncryptionMethodEmbed.class, new JsonSerializer<JWEEncryptionMethodEmbed>() {
068                        @Override
069            public JsonElement serialize(JWEEncryptionMethodEmbed src, Type typeOfSrc, JsonSerializationContext context) {
070                                if (src != null) {
071                                        return new JsonPrimitive(src.getAlgorithmName());
072                                } else {
073                                        return null;
074                                }
075            }
076            })
077            .serializeNulls()
078            .setDateFormat("yyyy-MM-dd'T'HH:mm:ssZ")
079            .create();
080
081
082        /**
083         * @return
084         */
085    protected abstract ExclusionStrategy getExclusionStrategy();
086 
087        
088    protected void renderMergedOutputModel(Map<String, Object> model, HttpServletRequest request, HttpServletResponse response) {
089
090        response.setContentType("application/json");
091
092        
093                HttpStatus code = (HttpStatus) model.get("code");
094                if (code == null) {
095                        code = HttpStatus.OK; // default to 200
096                }
097                
098                response.setStatus(code.value());
099                
100                try {
101                        
102                        Writer out = response.getWriter();
103                        Object obj = model.get("entity");
104                gson.toJson(obj, out);
105                
106                } catch (IOException e) {
107                        
108                        logger.error("IOException in JsonEntityView.java: ", e);
109                        
110                }
111    }
