030import org.mitre.openid.connect.model.UserInfo;

050@Component("userInfoView")
051public class UserInfoView extends AbstractView {
052        
053        private static Logger logger = LoggerFactory.getLogger(UserInfoView.class);
054        
055        /* (non-Javadoc)
056         * @see org.springframework.web.servlet.view.AbstractView#renderMergedOutputModel(java.util.Map, javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse)
057         */
058        protected void renderMergedOutputModel(Map<String, Object> model, HttpServletRequest request, HttpServletResponse response) {
059                
060                UserInfo userInfo = (UserInfo) model.get("userInfo");
061
062                Set<String> scope = (Set<String>) model.get("scope");
063                
064                Gson gson = new GsonBuilder()
065                        .setExclusionStrategies(new ExclusionStrategy() {
066                                
067                                public boolean shouldSkipField(FieldAttributes f) {
068                                        
069                                        return false;
070                                }
071                                
072                                public boolean shouldSkipClass(Class<?> clazz) {
073                                        // skip the JPA binding wrapper
074                                        if (clazz.equals(BeanPropertyBindingResult.class)) {
075                                                return true;
076                                        }
077                                        return false;
078                                }
079                                                                
080                        }).create();
081
082                response.setContentType("application/json");
083                
084                Writer out;
085                
086                try {
087                        
088                        out = response.getWriter();
089                        
090                        if (model.get("requestObject") != null) {
091                                
092                                try {
093                        String jwtString = (String)model.get("requestObject");
094                        JWT requestObject = JWTParser.parse(jwtString);
095                        
096                        // FIXME: move to GSON for easier processing
097                        JsonObject obj = (JsonObject) new JsonParser().parse(requestObject.getJWTClaimsSet().toJSONObject().toJSONString());
098                        
099                        gson.toJson(toJsonFromRequestObj(userInfo, scope, obj), out);
100                } catch (JsonSyntaxException e) {
101                        // TODO Auto-generated catch block
102                        e.printStackTrace();
103                } catch (JsonIOException e) {
104                        // TODO Auto-generated catch block
105                        e.printStackTrace();
106                } catch (ParseException e) {
107                        // TODO Auto-generated catch block
108                        e.printStackTrace();
109                }
110                        
111                        } else {
112                        
113                                gson.toJson(toJson(userInfo, scope), out);
114                        
115                        }
116                        
117                } catch (IOException e) {
118                        
119                        logger.error("IOException in UserInfoView.java: ", e);
120                        
121                }
122
123        }
124        
125        private JsonObject toJson(UserInfo ui, Set<String> scope) {
126                
127                JsonObject obj = new JsonObject();
128                
129                if (scope.contains("openid")) {
130                        obj.addProperty("sub", ui.getSub());
131                }
132                
133                if (scope.contains("profile")) {
134                        obj.addProperty("name", ui.getName());
135                        obj.addProperty("preferred_username", ui.getPreferredUsername());
136                        obj.addProperty("given_name", ui.getGivenName());
137                        obj.addProperty("family_name", ui.getFamilyName());
138                        obj.addProperty("middle_name", ui.getMiddleName());
139                        obj.addProperty("nickname", ui.getNickname());
140                        obj.addProperty("profile", ui.getProfile());
141                        obj.addProperty("picture", ui.getPicture());
142                        obj.addProperty("website", ui.getWebsite());
143                        obj.addProperty("gender", ui.getGender());
144                        obj.addProperty("zone_info", ui.getZoneinfo());
145                        obj.addProperty("locale", ui.getLocale());
146                        obj.addProperty("updated_time", ui.getUpdatedTime());
147                        obj.addProperty("birthdate", ui.getBirthdate());
148                }
149                
150                if (scope.contains("email")) {
151                        obj.addProperty("email", ui.getEmail());
152                        obj.addProperty("email_verified", ui.getEmailVerified());
153                }
154                
155                if (scope.contains("phone")) {
156                        obj.addProperty("phone_number", ui.getPhoneNumber());
157                }
158                
159                if (scope.contains("address") && ui.getAddress() != null) {
160
161                        JsonObject addr = new JsonObject();
162                        addr.addProperty("formatted", ui.getAddress().getFormatted());
163                        addr.addProperty("street_address", ui.getAddress().getStreetAddress());
164                        addr.addProperty("locality", ui.getAddress().getLocality());
165                        addr.addProperty("region", ui.getAddress().getRegion());
166                        addr.addProperty("postal_code", ui.getAddress().getPostalCode());
167                        addr.addProperty("country", ui.getAddress().getCountry());
168                        
169                        obj.add("address", addr);
170                }
171
172                
173                return obj;
174        }
175        
176        /**
177         * Build a JSON response according to the request object recieved. 
178         * 
179         * Claims requested in requestObj.userinfo.claims are added to any 
180         * claims corresponding to requested scopes, if any.
181         * 
182         * @param ui
183         * @param scope
184         * @param requestObj
185         * @return
186         */
187        private JsonObject toJsonFromRequestObj(UserInfo ui, Set<String> scope, JsonObject requestObj) {
188                
189                JsonObject obj = toJson(ui, scope);
190                
191                //Process list of requested claims out of the request object
192                JsonElement userInfo = requestObj.get("userinfo");
193                if (userInfo == null || !userInfo.isJsonObject()) {
194                        return obj;
195                }
196                
197                JsonElement claims = userInfo.getAsJsonObject().get("claims");
198                if (claims == null || !claims.isJsonObject()) {
199                        return obj;
200                }
201                
202                // TODO: this mehod is likely to be fragile if the data model changes at all
203                
204                //For each claim found, add it if not already present
205                for (Entry<String, JsonElement> i : claims.getAsJsonObject().entrySet()) {
206                        String claimName = i.getKey();
207                        if (!obj.has(claimName)) {
208                                String value = "";
209
210                                
211                                //Process claim names to go from "claim_name" to "ClaimName"
212                                String camelClaimName = CaseFormat.LOWER_UNDERSCORE.to(CaseFormat.UPPER_CAMEL, claimName);
213                                //Now we have "getClaimName"
214                                String methodName = "get" + camelClaimName;
215                                Method getter = null;
216                                try {
217                                        getter = ui.getClass().getMethod(methodName);
218                                        value = (String) getter.invoke(ui);
219                                        obj.addProperty(claimName, value);
220                                } catch (SecurityException e) {
221                                        // TODO Auto-generated catch block
222                                        e.printStackTrace();
223                                } catch (NoSuchMethodException e) {
224                                        // TODO Auto-generated catch block
225                                        e.printStackTrace();
226                                } catch (IllegalArgumentException e) {
227                                        // TODO Auto-generated catch block
228                                        e.printStackTrace();
229                                } catch (IllegalAccessException e) {
230                                        // TODO Auto-generated catch block
231                                        e.printStackTrace();
232                                } catch (InvocationTargetException e) {
233                                        // TODO Auto-generated catch block
234                                        e.printStackTrace();
235                                }
236                        }
237                }
238                
239                
240                
241                return obj;
242                
243        }
