017package org.mitre.openid.connect.web;
018
019import java.util.List;
020
021import org.mitre.oauth2.model.ClientDetailsEntity;
022import org.mitre.oauth2.service.ClientDetailsEntityService;
023import org.mitre.oauth2.service.SystemScopeService;
024import org.mitre.openid.connect.model.UserInfo;
025import org.mitre.openid.connect.service.UserInfoService;
026import org.mitre.openid.connect.view.HttpCodeView;
027import org.mitre.openid.connect.view.UserInfoJWTView;
028import org.mitre.openid.connect.view.UserInfoView;
029import org.slf4j.Logger;
030import org.slf4j.LoggerFactory;
031import org.springframework.beans.factory.annotation.Autowired;
032import org.springframework.http.HttpHeaders;
033import org.springframework.http.HttpStatus;
034import org.springframework.http.MediaType;
035import org.springframework.security.access.prepost.PreAuthorize;
036import org.springframework.security.oauth2.provider.OAuth2Authentication;
037import org.springframework.stereotype.Controller;
038import org.springframework.ui.Model;
039import org.springframework.web.bind.annotation.RequestHeader;
040import org.springframework.web.bind.annotation.RequestMapping;
041import org.springframework.web.bind.annotation.RequestMethod;
042import org.springframework.web.bind.annotation.RequestParam;
043
044import com.google.common.base.Strings;
045
046/**
047 * OpenID Connect UserInfo endpoint, as specified in Standard sec 5 and Messages sec 2.4.
048 *
049 * @author AANGANES
050 *
051 */
052@Controller
053@RequestMapping("/" + UserInfoEndpoint.URL)
054public class UserInfoEndpoint {
055
056        public static final String URL = "userinfo";
057
058        @Autowired
059        private UserInfoService userInfoService;
060
061        @Autowired
062        private ClientDetailsEntityService clientService;
063
064        /**
065         * Logger for this class
066         */
067        private static final Logger logger = LoggerFactory.getLogger(UserInfoEndpoint.class);
068
069        /**
070         * Get information about the user as specified in the accessToken included in this request
071         */
072        @PreAuthorize("hasRole('ROLE_USER') and #oauth2.hasScope('" + SystemScopeService.OPENID_SCOPE + "')")
073        @RequestMapping(method= {RequestMethod.GET, RequestMethod.POST}, produces = {MediaType.APPLICATION_JSON_VALUE, UserInfoJWTView.JOSE_MEDIA_TYPE_VALUE})
074        public String getInfo(@RequestParam(value="claims", required=false) String claimsRequestJsonString,
075                        @RequestHeader(value=HttpHeaders.ACCEPT, required=false) String acceptHeader,
076                        OAuth2Authentication auth, Model model) {
077
078                if (auth == null) {
079                        logger.error("getInfo failed; no principal. Requester is not authorized.");
080                        model.addAttribute(HttpCodeView.CODE, HttpStatus.FORBIDDEN);
081                        return HttpCodeView.VIEWNAME;
082                }
083
084                String username = auth.getName();
085                UserInfo userInfo = userInfoService.getByUsernameAndClientId(username, auth.getOAuth2Request().getClientId());
086
087                if (userInfo == null) {
088                        logger.error("getInfo failed; user not found: " + username);
089                        model.addAttribute(HttpCodeView.CODE, HttpStatus.NOT_FOUND);
090                        return HttpCodeView.VIEWNAME;
091                }
092
093                model.addAttribute(UserInfoView.SCOPE, auth.getOAuth2Request().getScope());
094
095                model.addAttribute(UserInfoView.AUTHORIZED_CLAIMS, auth.getOAuth2Request().getExtensions().get("claims"));
096
097                if (!Strings.isNullOrEmpty(claimsRequestJsonString)) {
098                        model.addAttribute(UserInfoView.REQUESTED_CLAIMS, claimsRequestJsonString);
099                }
100
101                model.addAttribute(UserInfoView.USER_INFO, userInfo);
102
103                // content negotiation
104
105                // start off by seeing if the client has registered for a signed/encrypted JWT from here
106                ClientDetailsEntity client = clientService.loadClientByClientId(auth.getOAuth2Request().getClientId());
107                model.addAttribute(UserInfoJWTView.CLIENT, client);
108
109                List<MediaType> mediaTypes = MediaType.parseMediaTypes(acceptHeader);
110                MediaType.sortBySpecificityAndQuality(mediaTypes);
111
112                if (client.getUserInfoSignedResponseAlg() != null
113                                || client.getUserInfoEncryptedResponseAlg() != null
114                                || client.getUserInfoEncryptedResponseEnc() != null) {
115                        // client has a preference, see if they ask for plain JSON specifically on this request
116                        for (MediaType m : mediaTypes) {
117                                if (!m.isWildcardType() && m.isCompatibleWith(UserInfoJWTView.JOSE_MEDIA_TYPE)) {
118                                        return UserInfoJWTView.VIEWNAME;
119                                } else if (!m.isWildcardType() && m.isCompatibleWith(MediaType.APPLICATION_JSON)) {
120                                        return UserInfoView.VIEWNAME;
121                                }
122                        }
123
124                        // otherwise return JWT
125                        return UserInfoJWTView.VIEWNAME;
126                } else {
127                        // client has no preference, see if they asked for JWT specifically on this request
128                        for (MediaType m : mediaTypes) {
129                                if (!m.isWildcardType() && m.isCompatibleWith(MediaType.APPLICATION_JSON)) {
130                                        return UserInfoView.VIEWNAME;
131                                } else if (!m.isWildcardType() && m.isCompatibleWith(UserInfoJWTView.JOSE_MEDIA_TYPE)) {
132                                        return UserInfoJWTView.VIEWNAME;
133                                }
134                        }
135
136                        // otherwise return JSON
137                        return UserInfoView.VIEWNAME;
138                }
139
140        }
141
142}
