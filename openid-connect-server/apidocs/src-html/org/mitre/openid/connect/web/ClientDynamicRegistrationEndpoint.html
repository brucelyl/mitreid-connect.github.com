063        /**
064         * Create a new Client, issue a client ID, and create a registration access token.
065         * @param jsonString
066         * @param m
067         * @param p
068         * @return
069         */
070        @RequestMapping(method = RequestMethod.POST, consumes = "application/json", produces = "application/json")
071        public String registerNewClient(@RequestBody String jsonString, Model m) {
072                
073                ClientDetailsEntity newClient = parse(jsonString);
074                
075                if (newClient != null) {
076                        // it parsed!
077                        
078                        //
079                        // Now do some post-processing consistency checks on it
080                        //
081                        
082                        // clear out any spurious id/secret (clients don't get to pick)
083                        newClient.setClientId(null);
084                        newClient.setClientSecret(null);
085                        
086                        // set of scopes that are OK for clients to dynamically register for
087                        Set<SystemScope> dynScopes = scopeService.getDynReg();
088
089                        // scopes that the client is asking for
090                        Set<SystemScope> requestedScopes = scopeService.fromStrings(newClient.getScope());
091
092                        // if the client didn't ask for any, give them the defaults
093                        if (requestedScopes == null || requestedScopes.isEmpty()) {
094                                requestedScopes = scopeService.getDefaults();
095                        }
096
097                        // the scopes that the client can have must be a subset of the dynamically allowed scopes
098                        Set<SystemScope> allowedScopes = Sets.intersection(dynScopes, requestedScopes);
099
100                        newClient.setScope(scopeService.toStrings(allowedScopes));
101                        
102
103                        // set default grant types if needed
104                        if (newClient.getGrantTypes() == null || newClient.getGrantTypes().isEmpty()) { 
105                                newClient.setGrantTypes(Sets.newHashSet("authorization_code", "refresh_token")); // allow authorization code and refresh token grant types by default
106                        }
107                        
108                        // set default response types if needed
109                        // TODO: these aren't checked by SECOAUTH
110                        // TODO: the consistency between the response_type and grant_type needs to be checked by the client service, most likely
111                        if (newClient.getResponseTypes() == null || newClient.getResponseTypes().isEmpty()) {
112                                newClient.setResponseTypes(Sets.newHashSet("code")); // default to allowing only the auth code flow
113                        }
114                        
115                        if (newClient.getTokenEndpointAuthMethod() == null) {
116                                newClient.setTokenEndpointAuthMethod(AuthMethod.SECRET_BASIC);
117                        }
118                        
119                        if (newClient.getTokenEndpointAuthMethod() == AuthMethod.SECRET_BASIC ||
120                                        newClient.getTokenEndpointAuthMethod() == AuthMethod.SECRET_JWT ||
121                                        newClient.getTokenEndpointAuthMethod() == AuthMethod.SECRET_POST) {
122                                
123                                // we need to generate a secret
124                                newClient = clientService.generateClientSecret(newClient);
125                        }
126                        
127                        // set some defaults for token timeouts
128                        newClient.setAccessTokenValiditySeconds((int)TimeUnit.HOURS.toSeconds(1)); // access tokens good for 1hr
129                        newClient.setIdTokenValiditySeconds((int)TimeUnit.MINUTES.toSeconds(10)); // id tokens good for 10min
130                        newClient.setRefreshTokenValiditySeconds(null); // refresh tokens good until revoked
131
132                        // this client has been dynamically registered (obviously)
133                        newClient.setDynamicallyRegistered(true);
134                        
135                        // now save it
136                        ClientDetailsEntity savedClient = clientService.saveNewClient(newClient);
137                        
138                        // generate the registration access token
139                        OAuth2AccessTokenEntity token = createRegistrationAccessToken(savedClient);
140                        
141                        // send it all out to the view
142                        m.addAttribute("client", savedClient);
143                        m.addAttribute("code", HttpStatus.CREATED); // http 201
144                        m.addAttribute("token", token);
145                        
146                        return "clientInformationResponseView";
147                } else {
148                        // didn't parse, this is a bad request
149                        logger.error("registerNewClient failed; submitted JSON is malformed");
150                        m.addAttribute("code", HttpStatus.BAD_REQUEST); // http 400
151                        
152                        return "httpCodeView";
153                }
154                
155        }

157        /**
158         * Get the meta information for a client.
159         * @param clientId
160         * @param m
161         * @param auth
162         * @return
163         */
164        @PreAuthorize("hasRole('ROLE_CLIENT') and #oauth2.hasScope('" + OAuth2AccessTokenEntity.REGISTRATION_TOKEN_SCOPE + "')")
165        @RequestMapping(value = "/{id}", method = RequestMethod.GET, produces = "application/json")
166        public String readClientConfiguration(@PathVariable("id") String clientId, Model m, OAuth2Authentication auth) {
167                
168                ClientDetailsEntity client = clientService.loadClientByClientId(clientId);
169                
170                if (client != null && client.getClientId().equals(auth.getAuthorizationRequest().getClientId())) {
171
172                        
173                        // we return the token that we got in
174                        OAuth2AuthenticationDetails details = (OAuth2AuthenticationDetails) auth.getDetails();
175                        OAuth2AccessTokenEntity token = tokenService.readAccessToken(details.getTokenValue());
176                        
177                        // send it all out to the view
178                        m.addAttribute("client", client);
179                        m.addAttribute("code", HttpStatus.OK); // http 200
180                        m.addAttribute("token", token);
181                        
182                        return "clientInformationResponseView";
183                } else {
184                        // client mismatch
185                        logger.error("readClientConfiguration failed, client ID mismatch: " 
186                                        + clientId + " and " + auth.getAuthorizationRequest().getClientId() + " do not match.");
187                        m.addAttribute("code", HttpStatus.FORBIDDEN); // http 403
188                        
189                        return "httpCodeView";
190                }
191        }

193        /**
194         * Update the metainformation for a given client.
195         * @param clientId
196         * @param jsonString
197         * @param m
198         * @param auth
199         * @return
200         */
201        @PreAuthorize("hasRole('ROLE_CLIENT') and #oauth2.hasScope('" + OAuth2AccessTokenEntity.REGISTRATION_TOKEN_SCOPE + "')")
202        @RequestMapping(value = "/{id}", method = RequestMethod.PUT, produces = "application/json", consumes = "application/json")
203        public String updateClient(@PathVariable("id") String clientId, @RequestBody String jsonString, Model m, OAuth2Authentication auth) {
204
205                
206                ClientDetailsEntity newClient = parse(jsonString);
207                ClientDetailsEntity oldClient = clientService.loadClientByClientId(clientId);
208                
209                if (newClient != null && oldClient != null  // we have an existing client and the new one parsed
210                                && oldClient.getClientId().equals(auth.getAuthorizationRequest().getClientId()) // the client passed in the URI matches the one in the auth
211                                && oldClient.getClientId().equals(newClient.getClientId()) // the client passed in the body matches the one in the URI
212                                ) {
213
214                        // a client can't ask to update its own client secret to any particular value
215                        newClient.setClientSecret(oldClient.getClientSecret());
216                        
217                        // we need to copy over all of the local and SECOAUTH fields
218                        newClient.setAccessTokenValiditySeconds(oldClient.getAccessTokenValiditySeconds());
219                        newClient.setIdTokenValiditySeconds(oldClient.getIdTokenValiditySeconds());
220                        newClient.setRefreshTokenValiditySeconds(oldClient.getRefreshTokenValiditySeconds());
221                        newClient.setDynamicallyRegistered(true); // it's still dynamically registered
222                        newClient.setAllowIntrospection(oldClient.isAllowIntrospection());
223                        newClient.setAuthorities(oldClient.getAuthorities());
224                        newClient.setClientDescription(oldClient.getClientDescription());
225                        newClient.setCreatedAt(oldClient.getCreatedAt());
226                        newClient.setReuseRefreshToken(oldClient.isReuseRefreshToken());
227                        
228                        // set of scopes that are OK for clients to dynamically register for
229                        Set<SystemScope> dynScopes = scopeService.getDynReg();
230
231                        // scopes that the client is asking for
232                        Set<SystemScope> requestedScopes = scopeService.fromStrings(newClient.getScope());
233
234                        // the scopes that the client can have must be a subset of the dynamically allowed scopes
235                        Set<SystemScope> allowedScopes = Sets.intersection(dynScopes, requestedScopes);
236
237                        // make sure that the client doesn't ask for scopes it can't have
238                        newClient.setScope(scopeService.toStrings(allowedScopes));
239                        
240                        // save the client
241                        ClientDetailsEntity savedClient = clientService.updateClient(oldClient, newClient);
242                        
243                        // we return the token that we got in
244                        // TODO: rotate this after some set amount of time
245                        OAuth2AuthenticationDetails details = (OAuth2AuthenticationDetails) auth.getDetails();
246                        OAuth2AccessTokenEntity token = tokenService.readAccessToken(details.getTokenValue());
247                        
248                        // send it all out to the view
249                        m.addAttribute("client", savedClient);
250                        m.addAttribute("code", HttpStatus.OK); // http 200
251                        m.addAttribute("token", token);
252                        
253                        return "clientInformationResponseView";
254                } else {
255                        // client mismatch
256                        logger.error("readClientConfiguration failed, client ID mismatch: " 
257                                        + clientId + " and " + auth.getAuthorizationRequest().getClientId() + " do not match.");
258                        m.addAttribute("code", HttpStatus.FORBIDDEN); // http 403
259                        
260                        return "httpCodeView";
261                }
262        }

304        /**
305         * 
306         * Create an unbound ClientDetailsEntity from the given JSON string.
307         * 
308         * @param jsonString
309         * @return the entity if successful, null otherwise
310         */
311    private ClientDetailsEntity parse(String jsonString) {
312                JsonElement jsonEl = parser.parse(jsonString);
313                if (jsonEl.isJsonObject()) {
314
315                        JsonObject o = jsonEl.getAsJsonObject();
316                        ClientDetailsEntity c = new ClientDetailsEntity();
317                        
318                        // TODO: make these field names into constants
319
320                        // these two fields should only be sent in the update request, and MUST match existing values
321                        c.setClientId(getAsString(o, "client_id"));
322                        c.setClientSecret(getAsString(o, "client_secret"));
323                        
324                        // OAuth DynReg
325                        c.setRedirectUris(getAsStringSet(o, "redirect_uris"));
326                        c.setClientName(getAsString(o, "client_name"));
327                        c.setClientUri(getAsString(o, "client_uri"));
328                        c.setLogoUri(getAsString(o, "logo_uri"));
329                        c.setContacts(getAsStringSet(o, "contacts"));
330                        c.setTosUri(getAsString(o, "tos_uri"));
331                        
332                        String authMethod = getAsString(o, "token_endpoint_auth_method");
333                        if (authMethod != null) {
334                                c.setTokenEndpointAuthMethod(AuthMethod.getByValue(authMethod));
335                        }
336                        
337                        // scope is a space-separated string
338                        String scope = getAsString(o, "scope");
339                        if (scope != null) {
340                                c.setScope(Sets.newHashSet(Splitter.on(" ").split(scope)));
341                        }
342                        
343                        c.setGrantTypes(getAsStringSet(o, "grant_types"));
344                        c.setPolicyUri(getAsString(o, "policy_uri"));
345                        c.setJwksUri(getAsString(o, "jwks_uri"));
346                        
347                        
348                        // OIDC Additions
349                        String appType = getAsString(o, "application_type");
350                        if (appType != null) {
351                                c.setApplicationType(AppType.getByValue(appType));
352                        }
353                        
354                        c.setSectorIdentifierUri(getAsString(o, "sector_identifier_uri"));
355                        
356                        String subjectType = getAsString(o, "subject_type");
357                        if (subjectType != null) {
358                                c.setSubjectType(SubjectType.getByValue(subjectType));
359                        }
360                        
361                        c.setRequestObjectSigningAlg(getAsJwsAlgorithm(o, "request_object_signing_alg"));
362                        
363                        c.setUserInfoSignedResponseAlg(getAsJwsAlgorithm(o, "userinfo_signed_response_alg"));
364                        c.setUserInfoEncryptedResponseAlg(getAsJweAlgorithm(o, "userinfo_encrypted_response_alg"));
365                        c.setUserInfoEncryptedResponseEnc(getAsJweEncryptionMethod(o, "userinfo_encrypted_response_enc"));
366                        
367                        c.setIdTokenSignedResponseAlg(getAsJwsAlgorithm(o, "id_token_signed_response_alg"));
368                        c.setIdTokenEncryptedResponseAlg(getAsJweAlgorithm(o, "id_token_encrypted_response_alg"));
369                        c.setIdTokenEncryptedResponseEnc(getAsJweEncryptionMethod(o, "id_token_encrypted_response_enc"));
370                        
371                        if (o.has("default_max_age")) {
372                                if (o.get("default_max_age").isJsonPrimitive()) {
373                                        c.setDefaultMaxAge(o.get("default_max_age").getAsInt());
374                                }
375                        }
376                        
377                        if (o.has("require_auth_time")) {
378                                if (o.get("require_auth_time").isJsonPrimitive()) {
379                                        c.setRequireAuthTime(o.get("require_auth_time").getAsBoolean());
380                                }
381                        }
382                        
383                        c.setDefaultACRvalues(getAsStringSet(o, "default_acr_values"));
384                        c.setInitiateLoginUri(getAsString(o, "initiate_login_uri"));
385                        c.setPostLogoutRedirectUri(getAsString(o, "post_logout_redirect_uri"));
386                        c.setRequestUris(getAsStringSet(o, "request_uris"));
387                        
388                        return c;
389                } else {
390                return null;
391                }
392    }

394        /**
395         * Gets the value of the given given member as a set of strings, null if it doesn't exist
396         */
397    private Set<String> getAsStringSet(JsonObject o, String member) throws JsonSyntaxException {
398        if (o.has(member)) {
399                return gson.fromJson(o.get(member), new TypeToken<Set<String>>(){}.getType());
400        } else {
401                return null;
402        }
403    }

405    /**
406     * Gets the value of the given member as a string, null if it doesn't exist
407     */
408    private String getAsString(JsonObject o, String member) {
409        if (o.has(member)) {
410                JsonElement e = o.get(member);
411                if (e != null && e.isJsonPrimitive()) {
412                        return e.getAsString();
413                } else {
414                        return null;
415                }
416        } else {
417                return null;
418        }
419    }

457        /**
458     * @param client
459     * @return
460     * @throws AuthenticationException
461     */
462    private OAuth2AccessTokenEntity createRegistrationAccessToken(ClientDetailsEntity client) throws AuthenticationException {
463            // create a registration access token, treat it like a client credentials flow
464                // I can't use the auth request interface here because it has no setters and bad constructors -- THIS IS BAD API DESIGN
465                DefaultAuthorizationRequest authorizationRequest = new DefaultAuthorizationRequest(client.getClientId(), Sets.newHashSet(OAuth2AccessTokenEntity.REGISTRATION_TOKEN_SCOPE));
466                authorizationRequest.setApproved(true);
467                authorizationRequest.setAuthorities(Sets.newHashSet(new SimpleGrantedAuthority("ROLE_CLIENT")));
468                OAuth2Authentication authentication = new OAuth2Authentication(authorizationRequest, null);
469                OAuth2AccessTokenEntity registrationAccessToken = (OAuth2AccessTokenEntity) tokenService.createAccessToken(authentication);
470            return registrationAccessToken;
471    }
