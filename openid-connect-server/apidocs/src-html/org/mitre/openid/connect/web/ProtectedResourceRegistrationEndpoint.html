088        /**
089         * Create a new Client, issue a client ID, and create a registration access token.
090         * @param jsonString
091         * @param m
092         * @param p
093         * @return
094         */
095        @RequestMapping(method = RequestMethod.POST, consumes = MediaType.APPLICATION_JSON_VALUE, produces = MediaType.APPLICATION_JSON_VALUE)
096        public String registerNewProtectedResource(@RequestBody String jsonString, Model m) {
097
098                ClientDetailsEntity newClient = null;
099                try {
100                        newClient = ClientDetailsEntityJsonProcessor.parse(jsonString);
101                } catch (JsonSyntaxException e) {
102                        // bad parse
103                        // didn't parse, this is a bad request
104                        logger.error("registerNewProtectedResource failed; submitted JSON is malformed");
105                        m.addAttribute(HttpCodeView.CODE, HttpStatus.BAD_REQUEST); // http 400
106                        return HttpCodeView.VIEWNAME;
107                }
108
109                if (newClient != null) {
110                        // it parsed!
111
112                        //
113                        // Now do some post-processing consistency checks on it
114                        //
115
116                        // clear out any spurious id/secret (clients don't get to pick)
117                        newClient.setClientId(null);
118                        newClient.setClientSecret(null);
119
120                        // do validation on the fields
121                        try {
122                                newClient = validateScopes(newClient);
123                                newClient = validateAuth(newClient);
124                        } catch (ValidationException ve) {
125                                // validation failed, return an error
126                                m.addAttribute(JsonErrorView.ERROR, ve.getError());
127                                m.addAttribute(JsonErrorView.ERROR_MESSAGE, ve.getErrorDescription());
128                                m.addAttribute(HttpCodeView.CODE, ve.getStatus());
129                                return JsonErrorView.VIEWNAME;
130                        }
131
132
133                        // no grant types are allowed
134                        newClient.setGrantTypes(new HashSet<String>());
135                        newClient.setResponseTypes(new HashSet<String>());
136                        newClient.setRedirectUris(new HashSet<String>());
137
138                        // don't issue tokens to this client
139                        newClient.setAccessTokenValiditySeconds(0);
140                        newClient.setIdTokenValiditySeconds(0);
141                        newClient.setRefreshTokenValiditySeconds(0);
142
143                        // clear out unused fields
144                        newClient.setDefaultACRvalues(new HashSet<String>());
145                        newClient.setDefaultMaxAge(null);
146                        newClient.setIdTokenEncryptedResponseAlg(null);
147                        newClient.setIdTokenEncryptedResponseEnc(null);
148                        newClient.setIdTokenSignedResponseAlg(null);
149                        newClient.setInitiateLoginUri(null);
150                        newClient.setPostLogoutRedirectUris(null);
151                        newClient.setRequestObjectSigningAlg(null);
152                        newClient.setRequireAuthTime(null);
153                        newClient.setReuseRefreshToken(false);
154                        newClient.setSectorIdentifierUri(null);
155                        newClient.setSubjectType(null);
156                        newClient.setUserInfoEncryptedResponseAlg(null);
157                        newClient.setUserInfoEncryptedResponseEnc(null);
158                        newClient.setUserInfoSignedResponseAlg(null);
159
160                        // this client has been dynamically registered (obviously)
161                        newClient.setDynamicallyRegistered(true);
162
163                        // this client has access to the introspection endpoint
164                        newClient.setAllowIntrospection(true);
165
166                        // now save it
167                        try {
168                                ClientDetailsEntity savedClient = clientService.saveNewClient(newClient);
169
170                                // generate the registration access token
171                                OAuth2AccessTokenEntity token = connectTokenService.createResourceAccessToken(savedClient);
172                                tokenService.saveAccessToken(token);
173
174                                // send it all out to the view
175
176                                RegisteredClient registered = new RegisteredClient(savedClient, token.getValue(), config.getIssuer() + "resource/" + UriUtils.encodePathSegment(savedClient.getClientId(), "UTF-8"));
177                                m.addAttribute("client", registered);
178                                m.addAttribute(HttpCodeView.CODE, HttpStatus.CREATED); // http 201
179
180                                return ClientInformationResponseView.VIEWNAME;
181                        } catch (UnsupportedEncodingException e) {
182                                logger.error("Unsupported encoding", e);
183                                m.addAttribute(HttpCodeView.CODE, HttpStatus.INTERNAL_SERVER_ERROR);
184                                return HttpCodeView.VIEWNAME;
185                        } catch (IllegalArgumentException e) {
186                                logger.error("Couldn't save client", e);
187
188                                m.addAttribute(JsonErrorView.ERROR, "invalid_client_metadata");
189                                m.addAttribute(JsonErrorView.ERROR_MESSAGE, "Unable to save client due to invalid or inconsistent metadata.");
190                                m.addAttribute(HttpCodeView.CODE, HttpStatus.BAD_REQUEST); // http 400
191
192                                return JsonErrorView.VIEWNAME;
193                        }
194                } else {
195                        // didn't parse, this is a bad request
196                        logger.error("registerNewClient failed; submitted JSON is malformed");
197                        m.addAttribute(HttpCodeView.CODE, HttpStatus.BAD_REQUEST); // http 400
198
199                        return HttpCodeView.VIEWNAME;
200                }
201
202        }

204        private ClientDetailsEntity validateScopes(ClientDetailsEntity newClient) throws ValidationException {
205                // scopes that the client is asking for
206                Set<SystemScope> requestedScopes = scopeService.fromStrings(newClient.getScope());
207
208                // the scopes that the client can have must be a subset of the dynamically allowed scopes
209                Set<SystemScope> allowedScopes = scopeService.removeRestrictedAndReservedScopes(requestedScopes);
210
211                // if the client didn't ask for any, give them the defaults
212                if (allowedScopes == null || allowedScopes.isEmpty()) {
213                        allowedScopes = scopeService.getDefaults();
214                }
215
216                newClient.setScope(scopeService.toStrings(allowedScopes));
217
218                return newClient;
219        }

221        /**
222         * Get the meta information for a client.
223         * @param clientId
224         * @param m
225         * @param auth
226         * @return
227         */
228        @PreAuthorize("hasRole('ROLE_CLIENT') and #oauth2.hasScope('" + SystemScopeService.RESOURCE_TOKEN_SCOPE + "')")
229        @RequestMapping(value = "/{id}", method = RequestMethod.GET, produces = MediaType.APPLICATION_JSON_VALUE)
230        public String readResourceConfiguration(@PathVariable("id") String clientId, Model m, OAuth2Authentication auth) {
231
232                ClientDetailsEntity client = clientService.loadClientByClientId(clientId);
233
234                if (client != null && client.getClientId().equals(auth.getOAuth2Request().getClientId())) {
235
236
237
238                        try {
239                                // possibly update the token
240                                OAuth2AccessTokenEntity token = fetchValidRegistrationToken(auth, client);
241
242                                RegisteredClient registered = new RegisteredClient(client, token.getValue(), config.getIssuer() + "resource/" +  UriUtils.encodePathSegment(client.getClientId(), "UTF-8"));
243
244                                // send it all out to the view
245                                m.addAttribute("client", registered);
246                                m.addAttribute(HttpCodeView.CODE, HttpStatus.OK); // http 200
247
248                                return ClientInformationResponseView.VIEWNAME;
249                        } catch (UnsupportedEncodingException e) {
250                                logger.error("Unsupported encoding", e);
251                                m.addAttribute(HttpCodeView.CODE, HttpStatus.INTERNAL_SERVER_ERROR);
252                                return HttpCodeView.VIEWNAME;
253                        }
254                } else {
255                        // client mismatch
256                        logger.error("readResourceConfiguration failed, client ID mismatch: "
257                                        + clientId + " and " + auth.getOAuth2Request().getClientId() + " do not match.");
258                        m.addAttribute(HttpCodeView.CODE, HttpStatus.FORBIDDEN); // http 403
259
260                        return HttpCodeView.VIEWNAME;
261                }
262        }

265         * Update the metainformation for a given client.
266         * @param clientId
267         * @param jsonString
268         * @param m
269         * @param auth
270         * @return
271         */
272        @PreAuthorize("hasRole('ROLE_CLIENT') and #oauth2.hasScope('" + SystemScopeService.RESOURCE_TOKEN_SCOPE + "')")
273        @RequestMapping(value = "/{id}", method = RequestMethod.PUT, produces = MediaType.APPLICATION_JSON_VALUE, consumes = MediaType.APPLICATION_JSON_VALUE)
274        public String updateProtectedResource(@PathVariable("id") String clientId, @RequestBody String jsonString, Model m, OAuth2Authentication auth) {
275
276
277                ClientDetailsEntity newClient = null;
278                try {
279                        newClient = ClientDetailsEntityJsonProcessor.parse(jsonString);
280                } catch (JsonSyntaxException e) {
281                        // bad parse
282                        // didn't parse, this is a bad request
283                        logger.error("updateProtectedResource failed; submitted JSON is malformed");
284                        m.addAttribute(HttpCodeView.CODE, HttpStatus.BAD_REQUEST); // http 400
285                        return HttpCodeView.VIEWNAME;
286                }
287
288                ClientDetailsEntity oldClient = clientService.loadClientByClientId(clientId);
289
290                if (newClient != null && oldClient != null  // we have an existing client and the new one parsed
291                                && oldClient.getClientId().equals(auth.getOAuth2Request().getClientId()) // the client passed in the URI matches the one in the auth
292                                && oldClient.getClientId().equals(newClient.getClientId()) // the client passed in the body matches the one in the URI
293                                ) {
294
295                        // a client can't ask to update its own client secret to any particular value
296                        newClient.setClientSecret(oldClient.getClientSecret());
297
298                        newClient.setCreatedAt(oldClient.getCreatedAt());
299
300                        // no grant types are allowed
301                        newClient.setGrantTypes(new HashSet<String>());
302                        newClient.setResponseTypes(new HashSet<String>());
303                        newClient.setRedirectUris(new HashSet<String>());
304
305                        // don't issue tokens to this client
306                        newClient.setAccessTokenValiditySeconds(0);
307                        newClient.setIdTokenValiditySeconds(0);
308                        newClient.setRefreshTokenValiditySeconds(0);
309
310                        // clear out unused fields
311                        newClient.setDefaultACRvalues(new HashSet<String>());
312                        newClient.setDefaultMaxAge(null);
313                        newClient.setIdTokenEncryptedResponseAlg(null);
314                        newClient.setIdTokenEncryptedResponseEnc(null);
315                        newClient.setIdTokenSignedResponseAlg(null);
316                        newClient.setInitiateLoginUri(null);
317                        newClient.setPostLogoutRedirectUris(null);
318                        newClient.setRequestObjectSigningAlg(null);
319                        newClient.setRequireAuthTime(null);
320                        newClient.setReuseRefreshToken(false);
321                        newClient.setSectorIdentifierUri(null);
322                        newClient.setSubjectType(null);
323                        newClient.setUserInfoEncryptedResponseAlg(null);
324                        newClient.setUserInfoEncryptedResponseEnc(null);
325                        newClient.setUserInfoSignedResponseAlg(null);
326
327                        // this client has been dynamically registered (obviously)
328                        newClient.setDynamicallyRegistered(true);
329
330                        // this client has access to the introspection endpoint
331                        newClient.setAllowIntrospection(true);
332
333                        // do validation on the fields
334                        try {
335                                newClient = validateScopes(newClient);
336                                newClient = validateAuth(newClient);
337                        } catch (ValidationException ve) {
338                                // validation failed, return an error
339                                m.addAttribute(JsonErrorView.ERROR, ve.getError());
340                                m.addAttribute(JsonErrorView.ERROR_MESSAGE, ve.getErrorDescription());
341                                m.addAttribute(HttpCodeView.CODE, ve.getStatus());
342                                return JsonErrorView.VIEWNAME;
343                        }
344
345
346                        try {
347                                // save the client
348                                ClientDetailsEntity savedClient = clientService.updateClient(oldClient, newClient);
349
350                                // possibly update the token
351                                OAuth2AccessTokenEntity token = fetchValidRegistrationToken(auth, savedClient);
352
353                                RegisteredClient registered = new RegisteredClient(savedClient, token.getValue(), config.getIssuer() + "resource/" + UriUtils.encodePathSegment(savedClient.getClientId(), "UTF-8"));
354
355                                // send it all out to the view
356                                m.addAttribute("client", registered);
357                                m.addAttribute(HttpCodeView.CODE, HttpStatus.OK); // http 200
358
359                                return ClientInformationResponseView.VIEWNAME;
360                        } catch (UnsupportedEncodingException e) {
361                                logger.error("Unsupported encoding", e);
362                                m.addAttribute(HttpCodeView.CODE, HttpStatus.INTERNAL_SERVER_ERROR);
363                                return HttpCodeView.VIEWNAME;
364                        } catch (IllegalArgumentException e) {
365                                logger.error("Couldn't save client", e);
366
367                                m.addAttribute(JsonErrorView.ERROR, "invalid_client_metadata");
368                                m.addAttribute(JsonErrorView.ERROR_MESSAGE, "Unable to save client due to invalid or inconsistent metadata.");
369                                m.addAttribute(HttpCodeView.CODE, HttpStatus.BAD_REQUEST); // http 400
370
371                                return JsonErrorView.VIEWNAME;
372                        }
373                } else {
374                        // client mismatch
375                        logger.error("updateProtectedResource" +
376                                        " failed, client ID mismatch: "
377                                        + clientId + " and " + auth.getOAuth2Request().getClientId() + " do not match.");
378                        m.addAttribute(HttpCodeView.CODE, HttpStatus.FORBIDDEN); // http 403
379
380                        return HttpCodeView.VIEWNAME;
381                }
382        }

384        /**
385         * Delete the indicated client from the system.
386         * @param clientId
387         * @param m
388         * @param auth
389         * @return
390         */
391        @PreAuthorize("hasRole('ROLE_CLIENT') and #oauth2.hasScope('" + SystemScopeService.RESOURCE_TOKEN_SCOPE + "')")
392        @RequestMapping(value = "/{id}", method = RequestMethod.DELETE, produces = MediaType.APPLICATION_JSON_VALUE)
393        public String deleteResource(@PathVariable("id") String clientId, Model m, OAuth2Authentication auth) {
394
395                ClientDetailsEntity client = clientService.loadClientByClientId(clientId);
396
397                if (client != null && client.getClientId().equals(auth.getOAuth2Request().getClientId())) {
398
399                        clientService.deleteClient(client);
400
401                        m.addAttribute(HttpCodeView.CODE, HttpStatus.NO_CONTENT); // http 204
402
403                        return HttpCodeView.VIEWNAME;
404                } else {
405                        // client mismatch
406                        logger.error("readClientConfiguration failed, client ID mismatch: "
407                                        + clientId + " and " + auth.getOAuth2Request().getClientId() + " do not match.");
408                        m.addAttribute(HttpCodeView.CODE, HttpStatus.FORBIDDEN); // http 403
409
410                        return HttpCodeView.VIEWNAME;
411                }
412        }

414        private ClientDetailsEntity validateAuth(ClientDetailsEntity newClient) throws ValidationException {
415                if (newClient.getTokenEndpointAuthMethod() == null) {
416                        newClient.setTokenEndpointAuthMethod(AuthMethod.SECRET_BASIC);
417                }
418
419                if (newClient.getTokenEndpointAuthMethod() == AuthMethod.SECRET_BASIC ||
420                                newClient.getTokenEndpointAuthMethod() == AuthMethod.SECRET_JWT ||
421                                newClient.getTokenEndpointAuthMethod() == AuthMethod.SECRET_POST) {
422
423                        if (Strings.isNullOrEmpty(newClient.getClientSecret())) {
424                                // no secret yet, we need to generate a secret
425                                newClient = clientService.generateClientSecret(newClient);
426                        }
427                } else if (newClient.getTokenEndpointAuthMethod() == AuthMethod.PRIVATE_KEY) {
428                        if (Strings.isNullOrEmpty(newClient.getJwksUri()) && newClient.getJwks() == null) {
429                                throw new ValidationException("invalid_client_metadata", "JWK Set URI required when using private key authentication", HttpStatus.BAD_REQUEST);
430                        }
431
432                        newClient.setClientSecret(null);
433                } else if (newClient.getTokenEndpointAuthMethod() == AuthMethod.NONE) {
434                        newClient.setClientSecret(null);
435                } else {
436                        throw new ValidationException("invalid_client_metadata", "Unknown authentication method", HttpStatus.BAD_REQUEST);
437                }
438                return newClient;
439        }

441        private OAuth2AccessTokenEntity fetchValidRegistrationToken(OAuth2Authentication auth, ClientDetailsEntity client) {
442
443                OAuth2AuthenticationDetails details = (OAuth2AuthenticationDetails) auth.getDetails();
444                OAuth2AccessTokenEntity token = tokenService.readAccessToken(details.getTokenValue());
445
446                if (config.getRegTokenLifeTime() != null) {
447
448                        try {
449                                // Re-issue the token if it has been issued before [currentTime - validity]
450                                Date validToDate = new Date(System.currentTimeMillis() - config.getRegTokenLifeTime() * 1000);
451                                if(token.getJwt().getJWTClaimsSet().getIssueTime().before(validToDate)) {
452                                        logger.info("Rotating the registration access token for " + client.getClientId());
453                                        tokenService.revokeAccessToken(token);
454                                        OAuth2AccessTokenEntity newToken = connectTokenService.createResourceAccessToken(client);
455                                        tokenService.saveAccessToken(newToken);
456                                        return newToken;
457                                } else {
458                                        // it's not expired, keep going
459                                        return token;
460                                }
461                        } catch (ParseException e) {
462                                logger.error("Couldn't parse a known-valid token?", e);
463                                return token;
464                        }
465                } else {
466                        // tokens don't expire, just return it
467                        return token;
468                }
469        }
