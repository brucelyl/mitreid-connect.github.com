001package org.mitre.openid.connect.web;
002
003import java.net.URI;
004import java.net.URISyntaxException;
005import java.text.ParseException;
006
007import javax.servlet.http.HttpServletRequest;
008
009import org.apache.http.client.utils.URIBuilder;
010import org.slf4j.Logger;
011import org.slf4j.LoggerFactory;
012import org.springframework.http.HttpStatus;
013import org.springframework.stereotype.Controller;
014import org.springframework.web.bind.annotation.RequestMapping;
015import org.springframework.web.bind.annotation.RequestParam;
016import org.springframework.web.servlet.ModelAndView;
017
018import com.google.common.base.Strings;
019import com.nimbusds.jwt.JWT;
020import com.nimbusds.jwt.JWTParser;
021
022
023/**
024 * This @Controller is a hack to get around SECOAUTH's AuthorizationEndpoint requiring the response_type parameter to be passed in.
025 * 
026 * @author jricher
027 *
028 */
029@Controller("requestObjectAuthorzationEndpoint")
030//@Component
031public class RequestObjectAuthorizationEndpoint {
032
033        private static Logger logger = LoggerFactory.getLogger(RequestObjectAuthorizationEndpoint.class);
034
035        @RequestMapping(value = "/authorize", params = "request")
036        public String authorizeRequestObject(@RequestParam("request") String jwtString, @RequestParam(value = "response_type", required = false) String responseType, HttpServletRequest request, ModelAndView mav) {
037
038                String query = request.getQueryString();
039                
040                if (responseType == null) {
041                        try {
042                        JWT requestObject = JWTParser.parse(jwtString);
043                        responseType = (String)requestObject.getJWTClaimsSet().getClaim("response_type");
044                        
045                        URI uri = new URIBuilder(Strings.nullToEmpty(request.getServletPath()) + Strings.nullToEmpty(request.getPathInfo()) + "?" + query)
046                                .addParameter("response_type", responseType)
047                                .build();
048                        
049                        query = uri.getRawQuery();//uri.toString();
050                        
051                } catch (ParseException e) {
052                        logger.error("ParseException while attempting to authorize request object: " + e.getStackTrace().toString());
053                        mav.addObject("code", HttpStatus.BAD_REQUEST);
054                        return "httpCodeView";
055                        
056                } catch (URISyntaxException e) {
057                        logger.error("URISyntaxError while attempting to authorize request object: " + e.getStackTrace().toString());
058                        mav.addObject("code", HttpStatus.BAD_REQUEST);
059                        return "httpCodeView";
060            }
061                }
062                
063                return "forward:/oauth/authorize?" + query;
064                
065        }
066        
067}
