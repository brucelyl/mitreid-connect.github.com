020package org.mitre.openid.connect.web;
021
022import java.security.Principal;
023import java.util.Collection;
024
025import org.mitre.openid.connect.model.WhitelistedSite;
026import org.mitre.openid.connect.service.WhitelistedSiteService;
027import org.slf4j.Logger;
028import org.slf4j.LoggerFactory;
029import org.springframework.beans.factory.annotation.Autowired;
030import org.springframework.http.HttpStatus;
031import org.springframework.security.access.prepost.PreAuthorize;
032import org.springframework.stereotype.Controller;
033import org.springframework.ui.ModelMap;
034import org.springframework.web.bind.annotation.PathVariable;
035import org.springframework.web.bind.annotation.RequestBody;
036import org.springframework.web.bind.annotation.RequestMapping;
037import org.springframework.web.bind.annotation.RequestMethod;
038
039import com.google.gson.Gson;
040import com.google.gson.JsonObject;
041import com.google.gson.JsonParseException;
042import com.google.gson.JsonParser;
043
044/**
045 * @author jricher
046 *
047 */
048@Controller
049@RequestMapping("/api/whitelist")
050@PreAuthorize("hasRole('ROLE_USER')")
051public class WhitelistAPI {
052
053        @Autowired
054        private WhitelistedSiteService whitelistService;
055
056        private static Logger logger = LoggerFactory.getLogger(WhitelistAPI.class);
057
058        private Gson gson = new Gson();
059        private JsonParser parser = new JsonParser();
060
061        /**
062         * Get a list of all whitelisted sites
063         * @param m
064         * @return
065         */
066        @RequestMapping(method = RequestMethod.GET, produces = "application/json")
067        public String getAllWhitelistedSites(ModelMap m) {
068
069                Collection<WhitelistedSite> all = whitelistService.getAll();
070
071                m.put("entity", all);
072
073                return "jsonEntityView";
074        }
075
076        /**
077         * Create a new whitelisted site
078         * @param jsonString
079         * @param m
080         * @param p
081         * @return
082         */
083        @PreAuthorize("hasRole('ROLE_ADMIN')")
084        @RequestMapping(method = RequestMethod.POST, consumes = "application/json", produces = "application/json")
085        public String addNewWhitelistedSite(@RequestBody String jsonString, ModelMap m, Principal p) {
086
087                JsonObject json;
088
089                WhitelistedSite whitelist = null;
090                try {
091                        json = parser.parse(jsonString).getAsJsonObject();
092                        whitelist = gson.fromJson(json, WhitelistedSite.class);
093
094                } catch (JsonParseException e) {
095                        logger.error("addNewWhitelistedSite failed due to JsonParseException", e);
096                        m.addAttribute("code", HttpStatus.BAD_REQUEST);
097                        m.addAttribute("errorMessage", "Could not save new whitelisted site. The server encountered a JSON syntax exception. Contact a system administrator for assistance.");
098                        return "jsonErrorView";
099                } catch (IllegalStateException e) {
100                        logger.error("addNewWhitelistedSite failed due to IllegalStateException", e);
101                        m.addAttribute("code", HttpStatus.BAD_REQUEST);
102                        m.addAttribute("errorMessage", "Could not save new whitelisted site. The server encountered an IllegalStateException. Refresh and try again - if the problem persists, contact a system administrator for assistance.");
103                        return "jsonErrorView";
104                }
105
106                // save the id of the person who created this
107                whitelist.setCreatorUserId(p.getName());
108
109                WhitelistedSite newWhitelist = whitelistService.saveNew(whitelist);
110
111                m.put("entity", newWhitelist);
112
113                return "jsonEntityView";
114
115        }
116
117        /**
118         * Update an existing whitelisted site
119         */
120        @PreAuthorize("hasRole('ROLE_ADMIN')")
121        @RequestMapping(value="/{id}", method = RequestMethod.PUT, consumes = "application/json", produces = "application/json")
122        public String updateWhitelistedSite(@PathVariable("id") Long id, @RequestBody String jsonString, ModelMap m, Principal p) {
123
124                JsonObject json;
125
126                WhitelistedSite whitelist = null;
127                try {
128                        json = parser.parse(jsonString).getAsJsonObject();
129                        whitelist = gson.fromJson(json, WhitelistedSite.class);
130
131                } catch (JsonParseException e) {
132                        logger.error("updateWhitelistedSite failed due to JsonParseException", e);
133                        m.put("code", HttpStatus.BAD_REQUEST);
134                        m.put("errorMessage", "Could not update whitelisted site. The server encountered a JSON syntax exception. Contact a system administrator for assistance.");
135                        return "jsonErrorView";
136                } catch (IllegalStateException e) {
137                        logger.error("updateWhitelistedSite failed due to IllegalStateException", e);
138                        m.put("code", HttpStatus.BAD_REQUEST);
139                        m.put("errorMessage", "Could not update whitelisted site. The server encountered an IllegalStateException. Refresh and try again - if the problem persists, contact a system administrator for assistance.");
140                        return "jsonErrorView";
141                }
142
143                WhitelistedSite oldWhitelist = whitelistService.getById(id);
144
145                if (oldWhitelist == null) {
146                        logger.error("updateWhitelistedSite failed; whitelist with id " + id + " could not be found.");
147                        m.put("code", HttpStatus.NOT_FOUND);
148                        m.put("errorMessage", "Could not update whitelisted site. The requested whitelisted site with id " + id + "could not be found.");
149                        return "jsonErrorView";
150                } else {
151
152                        WhitelistedSite newWhitelist = whitelistService.update(oldWhitelist, whitelist);
153
154                        m.put("entity", newWhitelist);
155
156                        return "jsonEntityView";
157                }
158        }
159
160        /**
161         * Delete a whitelisted site
162         * 
163         */
164        @PreAuthorize("hasRole('ROLE_ADMIN')")
165        @RequestMapping(value="/{id}", method = RequestMethod.DELETE)
166        public String deleteWhitelistedSite(@PathVariable("id") Long id, ModelMap m) {
167                WhitelistedSite whitelist = whitelistService.getById(id);
168
169                if (whitelist == null) {
170                        logger.error("deleteWhitelistedSite failed; whitelist with id " + id + " could not be found.");
171                        m.put("code", HttpStatus.NOT_FOUND);
172                        m.put("errorMessage", "Could not delete whitelisted site. The requested whitelisted site with id " + id + "could not be found.");
173                        return "jsonErrorView";
174                } else {
175                        m.put("code", HttpStatus.OK);
176                        whitelistService.remove(whitelist);
177                }
178
179                return "httpCodeView";
180        }
181
182        /**
183         * Get a single whitelisted site
184         */
185        @RequestMapping(value="/{id}", method = RequestMethod.GET, produces = "application/json")
186        public String getWhitelistedSite(@PathVariable("id") Long id, ModelMap m) {
187                WhitelistedSite whitelist = whitelistService.getById(id);
188                if (whitelist == null) {
189                        logger.error("getWhitelistedSite failed; whitelist with id " + id + " could not be found.");
190                        m.put("code", HttpStatus.NOT_FOUND);
191                        m.put("errorMessage", "The requested whitelisted site with id " + id + "could not be found.");
192                        return "jsonErrorView";
193                } else {
194
195                        m.put("entity", whitelist);
196
197                        return "jsonEntityView";
198                }
199
200        }
201}
