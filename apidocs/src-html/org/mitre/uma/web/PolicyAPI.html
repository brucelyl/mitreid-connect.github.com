071        /**
072         * List all resource sets for the current user
073         * @param m
074         * @param auth
075         * @return
076         */
077        @RequestMapping(value = "", method = RequestMethod.GET, produces = MimeTypeUtils.APPLICATION_JSON_VALUE)
078        public String getResourceSetsForCurrentUser(Model m, Authentication auth) {
079
080                Collection<ResourceSet> resourceSets = resourceSetService.getAllForOwner(auth.getName());
081
082                m.addAttribute(JsonEntityView.ENTITY, resourceSets);
083
084                return JsonEntityView.VIEWNAME;
085        }

087        /**
088         * Get the indicated resource set
089         * @param rsid
090         * @param m
091         * @param auth
092         * @return
093         */
094        @RequestMapping(value = "/{rsid}", method = RequestMethod.GET, produces = MimeTypeUtils.APPLICATION_JSON_VALUE)
095        public String getResourceSet(@PathVariable (value = "rsid") Long rsid, Model m, Authentication auth) {
096
097                ResourceSet rs = resourceSetService.getById(rsid);
098
099                if (rs == null) {
100                        m.addAttribute(HttpCodeView.CODE, HttpStatus.NOT_FOUND);
101                        return HttpCodeView.VIEWNAME;
102                }
103
104                if (!rs.getOwner().equals(auth.getName())) {
105                        logger.warn("Unauthorized resource set request from bad user; expected " + rs.getOwner() + " got " + auth.getName());
106
107                        // authenticated user didn't match the owner of the resource set
108                        m.addAttribute(HttpCodeView.CODE, HttpStatus.FORBIDDEN);
109                        return HttpCodeView.VIEWNAME;
110                }
111
112                m.addAttribute(JsonEntityView.ENTITY, rs);
113
114                return JsonEntityView.VIEWNAME;
115        }

178        /**
179         * Create a new policy on the given resource set
180         * @param rsid
181         * @param m
182         * @param auth
183         * @return
184         */
185        @RequestMapping(value = "/{rsid}" + POLICYURL, method = RequestMethod.POST, produces = MimeTypeUtils.APPLICATION_JSON_VALUE)
186        public String createNewPolicyForResourceSet(@PathVariable (value = "rsid") Long rsid, @RequestBody String jsonString, Model m, Authentication auth) {
187                ResourceSet rs = resourceSetService.getById(rsid);
188
189                if (rs == null) {
190                        m.addAttribute(HttpCodeView.CODE, HttpStatus.NOT_FOUND);
191                        return HttpCodeView.VIEWNAME;
192                }
193
194                if (!rs.getOwner().equals(auth.getName())) {
195                        logger.warn("Unauthorized resource set request from bad user; expected " + rs.getOwner() + " got " + auth.getName());
196
197                        // authenticated user didn't match the owner of the resource set
198                        m.addAttribute(HttpCodeView.CODE, HttpStatus.FORBIDDEN);
199                        return HttpCodeView.VIEWNAME;
200                }
201
202                Policy p = gson.fromJson(jsonString, Policy.class);
203
204                if (p.getId() != null) {
205                        logger.warn("Tried to add a policy with a non-null ID: " + p.getId());
206                        m.addAttribute(HttpCodeView.CODE, HttpStatus.BAD_REQUEST);
207                        return HttpCodeView.VIEWNAME;
208                }
209
210                for (Claim claim : p.getClaimsRequired()) {
211                        if (claim.getId() != null) {
212                                logger.warn("Tried to add a policy with a non-null claim ID: " + claim.getId());
213                                m.addAttribute(HttpCodeView.CODE, HttpStatus.BAD_REQUEST);
214                                return HttpCodeView.VIEWNAME;
215                        }
216                }
217
218                rs.getPolicies().add(p);
219                ResourceSet saved = resourceSetService.update(rs, rs);
220
221                // find the new policy object
222                Collection<Policy> newPolicies = Sets.difference(new HashSet<>(saved.getPolicies()), new HashSet<>(rs.getPolicies()));
223
224                if (newPolicies.size() == 1) {
225                        Policy newPolicy = newPolicies.iterator().next();
226                        m.addAttribute(JsonEntityView.ENTITY, newPolicy);
227                        return JsonEntityView.VIEWNAME;
228                } else {
229                        logger.warn("Unexpected result trying to add a new policy object: " + newPolicies);
230                        m.addAttribute(HttpCodeView.CODE, HttpStatus.INTERNAL_SERVER_ERROR);
231                        return HttpCodeView.VIEWNAME;
232                }
233
234        }

236        /**
237         * Get a specific policy
238         * @param rsid
239         * @param pid
240         * @param m
241         * @param auth
242         * @return
243         */
244        @RequestMapping(value = "/{rsid}" + POLICYURL + "/{pid}", method = RequestMethod.GET, produces = MimeTypeUtils.APPLICATION_JSON_VALUE)
245        public String getPolicy(@PathVariable (value = "rsid") Long rsid, @PathVariable (value = "pid") Long pid, Model m, Authentication auth) {
246
247                ResourceSet rs = resourceSetService.getById(rsid);
248
249                if (rs == null) {
250                        m.addAttribute(HttpCodeView.CODE, HttpStatus.NOT_FOUND);
251                        return HttpCodeView.VIEWNAME;
252                }
253
254                if (!rs.getOwner().equals(auth.getName())) {
255                        logger.warn("Unauthorized resource set request from bad user; expected " + rs.getOwner() + " got " + auth.getName());
256
257                        // authenticated user didn't match the owner of the resource set
258                        m.addAttribute(HttpCodeView.CODE, HttpStatus.FORBIDDEN);
259                        return HttpCodeView.VIEWNAME;
260                }
261
262                for (Policy policy : rs.getPolicies()) {
263                        if (policy.getId().equals(pid)) {
264                                // found it!
265                                m.addAttribute(JsonEntityView.ENTITY, policy);
266                                return JsonEntityView.VIEWNAME;
267                        }
268                }
269
270                // if we made it this far, we haven't found it
271                m.addAttribute(HttpCodeView.CODE, HttpStatus.NOT_FOUND);
272                return HttpCodeView.VIEWNAME;
273        }

275        /**
276         * Update a specific policy
277         * @param rsid
278         * @param pid
279         * @param jsonString
280         * @param m
281         * @param auth
282         * @return
283         */
284        @RequestMapping(value = "/{rsid}" + POLICYURL + "/{pid}", method = RequestMethod.PUT, consumes = MimeTypeUtils.APPLICATION_JSON_VALUE, produces = MimeTypeUtils.APPLICATION_JSON_VALUE)
285        public String setClaimsForResourceSet(@PathVariable (value = "rsid") Long rsid, @PathVariable (value = "pid") Long pid, @RequestBody String jsonString, Model m, Authentication auth) {
286
287                ResourceSet rs = resourceSetService.getById(rsid);
288
289                if (rs == null) {
290                        m.addAttribute(HttpCodeView.CODE, HttpStatus.NOT_FOUND);
291                        return HttpCodeView.VIEWNAME;
292                }
293
294                if (!rs.getOwner().equals(auth.getName())) {
295                        logger.warn("Unauthorized resource set request from bad user; expected " + rs.getOwner() + " got " + auth.getName());
296
297                        // authenticated user didn't match the owner of the resource set
298                        m.addAttribute(HttpCodeView.CODE, HttpStatus.FORBIDDEN);
299                        return HttpCodeView.VIEWNAME;
300                }
301
302                Policy p = gson.fromJson(jsonString, Policy.class);
303
304                if (!pid.equals(p.getId())) {
305                        logger.warn("Policy ID mismatch, expected " + pid + " got " + p.getId());
306
307                        m.addAttribute(HttpCodeView.CODE, HttpStatus.BAD_REQUEST);
308                        return HttpCodeView.VIEWNAME;
309                }
310
311                for (Policy policy : rs.getPolicies()) {
312                        if (policy.getId().equals(pid)) {
313                                // found it!
314
315                                // find the existing claim IDs, make sure we're not overwriting anything from another policy
316                                Set<Long> claimIds = new HashSet<>();
317                                for (Claim claim : policy.getClaimsRequired()) {
318                                        claimIds.add(claim.getId());
319                                }
320
321                                for (Claim claim : p.getClaimsRequired()) {
322                                        if (claim.getId() != null && !claimIds.contains(claim.getId())) {
323                                                logger.warn("Tried to add a policy with a an unmatched claim ID: got " + claim.getId() + " expected " + claimIds);
324                                                m.addAttribute(HttpCodeView.CODE, HttpStatus.BAD_REQUEST);
325                                                return HttpCodeView.VIEWNAME;
326                                        }
327                                }
328
329                                // update the existing object with the new values
330                                policy.setClaimsRequired(p.getClaimsRequired());
331                                policy.setName(p.getName());
332                                policy.setScopes(p.getScopes());
333
334                                resourceSetService.update(rs, rs);
335
336                                m.addAttribute(JsonEntityView.ENTITY, policy);
337                                return JsonEntityView.VIEWNAME;
338                        }
339                }
340
341                // if we made it this far, we haven't found it
342                m.addAttribute(HttpCodeView.CODE, HttpStatus.NOT_FOUND);
343                return HttpCodeView.VIEWNAME;
344        }
345
346        /**
347         * Delete a specific policy
348         * @param rsid
349         * @param pid
350         * @param m
351         * @param auth
352         * @return
353         */
354        @RequestMapping(value = "/{rsid}" + POLICYURL + "/{pid}", method = RequestMethod.DELETE, produces = MimeTypeUtils.APPLICATION_JSON_VALUE)
355        public String deleteResourceSet(@PathVariable ("rsid") Long rsid, @PathVariable (value = "pid") Long pid, Model m, Authentication auth) {
356
357                ResourceSet rs = resourceSetService.getById(rsid);
358
359                if (rs == null) {
360                        m.addAttribute(HttpCodeView.CODE, HttpStatus.NOT_FOUND);
361                        m.addAttribute(JsonErrorView.ERROR, "not_found");
362                        return JsonErrorView.VIEWNAME;
363                }
364
365                if (!auth.getName().equals(rs.getOwner())) {
366
367                        logger.warn("Unauthorized resource set request from bad user; expected " + rs.getOwner() + " got " + auth.getName());
368
369                        // it wasn't issued to this user
370                        m.addAttribute(HttpCodeView.CODE, HttpStatus.FORBIDDEN);
371                        return JsonErrorView.VIEWNAME;
372                }
373
374
375                for (Policy policy : rs.getPolicies()) {
376                        if (policy.getId().equals(pid)) {
377                                // found it!
378                                rs.getPolicies().remove(policy);
379                                resourceSetService.update(rs, rs);
380
381                                m.addAttribute(HttpCodeView.CODE, HttpStatus.NO_CONTENT);
382                                return HttpCodeView.VIEWNAME;
383                        }
384                }
385
386                // if we made it this far, we haven't found it
387                m.addAttribute(HttpCodeView.CODE, HttpStatus.NOT_FOUND);
388                return HttpCodeView.VIEWNAME;
389
390        }
