143        /**
144         * @param writer
145         * @throws IOException
146         */
147        private void writeTokenPermissions(JsonWriter writer) throws IOException {
148                for (OAuth2AccessTokenEntity token : tokenRepository.getAllAccessTokens()) {
149                        if (!token.getPermissions().isEmpty()) { // skip tokens that don't have the permissions structure attached
150                                writer.beginObject();
151                                writer.name(TOKEN_ID).value(token.getId());
152                                writer.name(PERMISSIONS);
153                                writer.beginArray();
154                                for (Permission p : token.getPermissions()) {
155                                        writer.beginObject();
156                                        writer.name(RESOURCE_SET).value(p.getResourceSet().getId());
157                                        writer.name(SCOPES);
158                                        writer.beginArray();
159                                        for (String s : p.getScopes()) {
160                                                writer.value(s);
161                                        }
162                                        writer.endArray();
163                                        writer.endObject();
164                                }
165                                writer.endArray();
166
167                                writer.endObject();
168                        }
169                }
170        }

172        /**
173         * @param writer
174         * @throws IOException
175         */
176        private void writePermissionTickets(JsonWriter writer) throws IOException {
177                for (PermissionTicket ticket : permissionRepository.getAll()) {
178                        writer.beginObject();
179
180                        writer.name(CLAIMS_SUPPLIED);
181                        writer.beginArray();
182                        for (Claim claim : ticket.getClaimsSupplied()) {
183                                writer.beginObject();
184
185                                writer.name(ISSUER);
186                                writer.beginArray();
187                                for (String issuer : claim.getIssuer()) {
188                                        writer.value(issuer);
189                                }
190                                writer.endArray();
191                                writer.name(CLAIM_TOKEN_FORMAT);
192                                writer.beginArray();
193                                for (String format : claim.getClaimTokenFormat()) {
194                                        writer.value(format);
195                                }
196                                writer.endArray();
197                                writer.name(CLAIM_TYPE).value(claim.getClaimType());
198                                writer.name(FRIENDLY_NAME).value(claim.getFriendlyName());
199                                writer.name(NAME).value(claim.getName());
200                                writer.name(VALUE).value(claim.getValue().toString());
201                                writer.endObject();
202                        }
203                        writer.endArray();
204
205                        writer.name(EXPIRATION).value(toUTCString(ticket.getExpiration()));
206
207                        writer.name(PERMISSION);
208                        writer.beginObject();
209                        Permission p = ticket.getPermission();
210                        writer.name(RESOURCE_SET).value(p.getResourceSet().getId());
211                        writer.name(SCOPES);
212                        writer.beginArray();
213                        for (String s : p.getScopes()) {
214                                writer.value(s);
215                        }
216                        writer.endArray();
217                        writer.endObject();
218
219                        writer.name(TICKET).value(ticket.getTicket());
220
221                        writer.endObject();
222                }

244                                writer.beginObject();

249                                        writer.value(scope);
250                                }
251                                writer.endArray();

253                                writer.beginArray();
254                                for (Claim claim : policy.getClaimsRequired()) {
255                                        writer.beginObject();
256
257                                        writer.name(ISSUER);
258                                        writer.beginArray();
259                                        for (String issuer : claim.getIssuer()) {
260                                                writer.value(issuer);
261                                        }
262                                        writer.endArray();
263                                        writer.name(CLAIM_TOKEN_FORMAT);
264                                        writer.beginArray();
265                                        for (String format : claim.getClaimTokenFormat()) {
266                                                writer.value(format);
267                                        }
268                                        writer.endArray();
269                                        writer.name(CLAIM_TYPE).value(claim.getClaimType());
270                                        writer.name(FRIENDLY_NAME).value(claim.getFriendlyName());
271                                        writer.name(NAME).value(claim.getName());
272                                        writer.name(VALUE).value(claim.getValue().toString());
273                                        writer.endObject();
274                                }
275                                writer.endArray();
276                                writer.endObject();
277                        }
278                        writer.endArray();
279                        writer.name(SCOPES);
280                        writer.beginArray();
281                        for (String scope : rs.getScopes()) {
282                                writer.value(scope);
283                        }
284                        writer.endArray();
285                        writer.endObject();
286                        logger.debug("Finished writing resource set {}", rs.getId());
287                }

306         * @see org.mitre.openid.connect.service.MITREidDataServiceExtension#importExtensionData(com.google.gson.stream.JsonReader)

308        @Override
309        public boolean importExtensionData(String name, JsonReader reader) throws IOException {
310                if (name.equals(SAVED_REGISTERED_CLIENTS)) {
311                        readSavedRegisteredClients(reader);
312                        return true;
313                } else if (name.equals(RESOURCE_SETS)) {
314                        readResourceSets(reader);
315                        return true;
316                } else if (name.equals(PERMISSION_TICKETS)) {
317                        readPermissionTickets(reader);
318                        return true;
319                } else if (name.equals(TOKEN_PERMISSIONS)) {
320                        readTokenPermissions(reader);
321                        return true;
322                } else {
323                        return false;
324                }
325        }

327        /**
328         * @param reader
329         */
330        private void readTokenPermissions(JsonReader reader) throws IOException {
331                reader.beginArray();
332                while(reader.hasNext()) {
333                        reader.beginObject();
334                        Long tokenId = null;
335                        Set<Long> permissions = new HashSet<>();
336                        while (reader.hasNext()) {
337                                switch(reader.peek()) {
338                                        case END_OBJECT:
339                                                continue;
340                                        case NAME:
341                                                String name = reader.nextName();
342                                                if (name.equals(TOKEN_ID)) {
343                                                        tokenId = reader.nextLong();
344                                                } else if (name.equals(PERMISSIONS)) {
345                                                        reader.beginArray();
346                                                        while (reader.hasNext()) {
347                                                                Permission p = new Permission();
348                                                                Long rsid = null;
349                                                                Set<String> scope = new HashSet<>();
350                                                                reader.beginObject();
351                                                                while (reader.hasNext()) {
352                                                                        switch (reader.peek()) {
353                                                                                case END_OBJECT:
354                                                                                        continue;
355                                                                                case NAME:
356                                                                                        String pname = reader.nextName();
357                                                                                        if (reader.peek() == JsonToken.NULL) {
358                                                                                                reader.skipValue();
359                                                                                        } else if (pname.equals(RESOURCE_SET)) {
360                                                                                                rsid = reader.nextLong();
361                                                                                        } else if (pname.equals(SCOPES)) {
362                                                                                                scope = readSet(reader);
363                                                                                        } else {
364                                                                                                logger.debug("Found unexpected entry");
365                                                                                                reader.skipValue();
366                                                                                        }
367                                                                                        break;
368                                                                                default:
369                                                                                        logger.debug("Found unexpected entry");
370                                                                                        reader.skipValue();
371                                                                                        continue;
372                                                                        }
373                                                                }
374                                                                reader.endObject();
375                                                                p.setScopes(scope);
376                                                                Permission saved = permissionRepository.saveRawPermission(p);
377                                                                permissionToResourceRefs.put(saved.getId(), rsid);
378                                                                permissions.add(saved.getId());
379                                                        }
380                                                        reader.endArray();
381                                                }
382                                                break;
383                                        default:
384                                                logger.debug("Found unexpected entry");
385                                                reader.skipValue();
386                                                continue;
387                                }
388                        }
389                        reader.endObject();
390                        tokenToPermissionRefs.put(tokenId, permissions);
391                }
392                reader.endArray();
393
394        }

456                                                        reader.endArray();
457                                                        ticket.setClaimsSupplied(claimsSupplied);

459                                                        ticket.setExpiration(utcToDate(reader.nextString()));

477                                                                                        logger.debug("Found unexpected entry");

479                                                                                }
480                                                                                break;
481                                                                        default:
482                                                                                logger.debug("Found unexpected entry");

658                                                if (reader.peek() == JsonToken.NULL) {
659                                                        reader.skipValue();
660                                                } else if (name.equals(ISSUER)) {
661                                                        issuer = reader.nextString();
662                                                } else if (name.equals(REGISTERED_CLIENT)) {
663                                                        clientString = reader.nextString();
664                                                } else {
665                                                        logger.debug("Found unexpected entry");
666                                                        reader.skipValue();
667                                                }
668                                                break;
669                                        default:
670                                                logger.debug("Found unexpected entry");
671                                                reader.skipValue();
672                                                continue;
673                                }
674                        }
675                        reader.endObject();
676                        RegisteredClient client = ClientDetailsEntityJsonProcessor.parseRegistered(clientString);
677                        registeredClientService.save(issuer, client);
678                        logger.debug("Saved registered client");
679                }
680                reader.endArray();
681                logger.info("Done reading saved registered clients");
682        }

684        /* (non-Javadoc)
685         * @see org.mitre.openid.connect.service.MITREidDataServiceExtension#fixExtensionObjectReferences()
686         */
687        @Override
688        public void fixExtensionObjectReferences(MITREidDataServiceMaps maps) {
689                for (Long permissionId : permissionToResourceRefs.keySet()) {
690                        Long oldResourceId = permissionToResourceRefs.get(permissionId);
691                        Long newResourceId = resourceSetOldToNewIdMap.get(oldResourceId);
692                        Permission p = permissionRepository.getById(permissionId);
693                        ResourceSet rs = resourceSetRepository.getById(newResourceId);
694                        p.setResourceSet(rs);
695                        permissionRepository.saveRawPermission(p);
696                        logger.debug("Mapping rsid " + oldResourceId + " to " + newResourceId + " for permission " + permissionId);
697                }
698                for (Long tokenId : tokenToPermissionRefs.keySet()) {
699                        Long newTokenId = maps.getAccessTokenOldToNewIdMap().get(tokenId);
700                        OAuth2AccessTokenEntity token = tokenRepository.getAccessTokenById(newTokenId);
701
702                        Set<Permission> permissions = new HashSet<>();
703                        for (Long permissionId : tokenToPermissionRefs.get(tokenId)) {
704                                Permission p = permissionRepository.getById(permissionId);
705                                permissions.add(p);
706                        }
707
708                        token.setPermissions(permissions);
709                        tokenRepository.saveAccessToken(token);
710                }
711                permissionToResourceRefs.clear();
712                resourceSetOldToNewIdMap.clear();
713                tokenToPermissionRefs.clear();
714        }
