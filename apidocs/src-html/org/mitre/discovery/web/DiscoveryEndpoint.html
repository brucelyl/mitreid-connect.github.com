070        // used to map JWA algorithms objects to strings
071        private Function<Algorithm, String> toAlgorithmName = new Function<Algorithm, String>() {
072                @Override
073                public String apply(Algorithm alg) {
074                        if (alg == null) {
075                                return null;
076                        } else {
077                                return alg.getName();
078                        }
079                }

082        @RequestMapping(value={"/.well-known/webfinger"},
083                        params={"resource", "rel=http://openid.net/specs/connect/1.0/issuer"}, produces = "application/json")
084        public String webfinger(@RequestParam("resource") String resource, Model model) {
085
086                if (!resource.equals(config.getIssuer())) {
087                        // it's not the issuer directly, need to check other methods
088
089                        try {
090                                URI resourceUri = new URI(resource);
091                                if (resourceUri != null
092                                                && resourceUri.getScheme() != null
093                                                && resourceUri.getScheme().equals("acct")) {
094                                        // acct: URI
095
096                                        // split out the user and host parts
097                                        List<String> parts = Lists.newArrayList(Splitter.on("@").split(resourceUri.getSchemeSpecificPart()));
098
099                                        UserInfo user = null;
100                                        if (parts.size() > 0) {
101                                                user = userService.getByUsername(parts.get(0)); // first part is the username
102                                        }
103
104                                        if (user == null) {
105                                                logger.info("User not found: " + resource);
106                                                model.addAttribute("code", HttpStatus.NOT_FOUND);
107                                                return "httpCodeView";
108                                        }
109                                        // TODO: check the "host" part against our issuer
110
111                                } else {
112                                        logger.info("Unknown URI format: " + resource);
113                                        model.addAttribute("code", HttpStatus.NOT_FOUND);
114                                        return "httpCodeView";
115                                }
116                        } catch (URISyntaxException e) {
117                                logger.info("URI parsing exception: " + resource, e);
118                                model.addAttribute("code", HttpStatus.NOT_FOUND);
119                                return "httpCodeView";
120                        }
121                }
122
123                // if we got here, then we're good
124                model.addAttribute("resource", resource);
125                model.addAttribute("issuer", config.getIssuer());
126
127                return "webfingerView";
128        }
