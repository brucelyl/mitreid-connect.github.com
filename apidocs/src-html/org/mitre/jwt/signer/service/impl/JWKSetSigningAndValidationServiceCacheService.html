020package org.mitre.jwt.signer.service.impl;
021
022import java.util.concurrent.ExecutionException;
023import java.util.concurrent.TimeUnit;
024
025import org.apache.http.client.HttpClient;
026import org.apache.http.impl.client.DefaultHttpClient;
027import org.mitre.jose.keystore.JWKSetKeyStore;
028import org.mitre.jwt.signer.service.JwtSigningAndValidationService;
029import org.slf4j.Logger;
030import org.slf4j.LoggerFactory;
031import org.springframework.http.client.HttpComponentsClientHttpRequestFactory;
032import org.springframework.stereotype.Service;
033import org.springframework.web.client.RestTemplate;
034
035import com.google.common.cache.CacheBuilder;
036import com.google.common.cache.CacheLoader;
037import com.google.common.cache.LoadingCache;
038import com.nimbusds.jose.jwk.JWKSet;
039
040/**
041 * 
042 * Creates a caching map of JOSE signers and validators keyed on the JWK Set URI.
043 * Dynamically loads JWK Sets to create the signing and validation services.
044 * 
045 * @author jricher
046 *
047 */
048@Service
049public class JWKSetSigningAndValidationServiceCacheService {
050
051        private static Logger logger = LoggerFactory.getLogger(JWKSetSigningAndValidationServiceCacheService.class);
052
053        // map of jwk set uri -> signing/validation service built on the keys found in that jwk set
054        private LoadingCache<String, JwtSigningAndValidationService> cache;
055
056        public JWKSetSigningAndValidationServiceCacheService() {
057                this.cache = CacheBuilder.newBuilder()
058                                .expireAfterWrite(1, TimeUnit.HOURS) // expires 1 hour after fetch
059                                .maximumSize(100)
060                                .build(new JWKSetVerifierFetcher());
061        }
062
063        /**
064         * @param jwksUri
065         * @return
066         * @throws ExecutionException
067         * @see com.google.common.cache.Cache#get(java.lang.Object)
068         */
069        public JwtSigningAndValidationService get(String jwksUri) {
070                try {
071                        return cache.get(jwksUri);
072                } catch (ExecutionException e) {
073                        logger.warn("Couldn't load JWK Set from " + jwksUri, e);
074                        return null;
075                }
076        }
077
078        /**
079         * @author jricher
080         *
081         */
082        private class JWKSetVerifierFetcher extends CacheLoader<String, JwtSigningAndValidationService> {
083                private HttpClient httpClient = new DefaultHttpClient();
084                private HttpComponentsClientHttpRequestFactory httpFactory = new HttpComponentsClientHttpRequestFactory(httpClient);
085                private RestTemplate restTemplate = new RestTemplate(httpFactory);
086
087                /**
088                 * Load the JWK Set and build the appropriate signing service.
089                 */
090                @Override
091                public JwtSigningAndValidationService load(String key) throws Exception {
092
093                        String jsonString = restTemplate.getForObject(key, String.class);
094                        JWKSet jwkSet = JWKSet.parse(jsonString);
095
096                        JWKSetKeyStore keyStore = new JWKSetKeyStore(jwkSet);
097
098                        JwtSigningAndValidationService service = new DefaultJwtSigningAndValidationService(keyStore);
099
100                        return service;
101
102                }
103
104        }
105
106}
