004package org.mitre.oauth2.service.impl;
005
006import org.mitre.openid.connect.service.BlacklistedSiteService;
007import org.springframework.beans.factory.annotation.Autowired;
008import org.springframework.security.oauth2.common.exceptions.InvalidRequestException;
009import org.springframework.security.oauth2.common.exceptions.OAuth2Exception;
010import org.springframework.security.oauth2.provider.ClientDetails;
011import org.springframework.security.oauth2.provider.endpoint.DefaultRedirectResolver;
012import org.springframework.stereotype.Component;
013
014/**
015 * @author jricher
016 *
017 */
018@Component("blacklistAwareRedirectResolver")
019public class BlacklistAwareRedirectResolver extends DefaultRedirectResolver {
020
021        @Autowired
022        private BlacklistedSiteService blacklistService;
023        
024        /* (non-Javadoc)
025         * @see org.springframework.security.oauth2.provider.endpoint.RedirectResolver#resolveRedirect(java.lang.String, org.springframework.security.oauth2.provider.ClientDetails)
026         */
027        @Override
028        public String resolveRedirect(String requestedRedirect, ClientDetails client) throws OAuth2Exception {
029                String redirect = super.resolveRedirect(requestedRedirect, client);
030                if (blacklistService.isBlacklisted(redirect)) {
031                        // don't let it go through
032                        throw new InvalidRequestException("The supplied redirect_uri is not allowed on this server.");
033                } else {
034                        // not blacklisted, passed the parent test, we're fine
035                        return redirect;
036                }
037        }
038
039}
