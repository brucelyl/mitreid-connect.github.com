059@Controller
060@SessionAttributes("authorizationRequest")
061public class OAuthConfirmationController {
062
063        @Autowired
064        private ClientDetailsEntityService clientService;
065
066        @Autowired
067        private SystemScopeService scopeService;
068
069        @Autowired
070        private ScopeClaimTranslationService scopeClaimTranslationService;
071        
072        @Autowired
073        private UserInfoService userInfoService;
074        
075        private static Logger logger = LoggerFactory.getLogger(OAuthConfirmationController.class);
076
077        public OAuthConfirmationController() {
078
079        }
080
081        public OAuthConfirmationController(ClientDetailsEntityService clientService) {
082                this.clientService = clientService;
083        }
084
085        @PreAuthorize("hasRole('ROLE_USER')")
086        @RequestMapping("/oauth/confirm_access")
087        public String confimAccess(Map<String, Object> model, @ModelAttribute("authorizationRequest") AuthorizationRequest clientAuth,
088                        Principal p) {
089
090                // Check the "prompt" parameter to see if we need to do special processing
091
092                String prompt = (String)clientAuth.getExtensions().get("prompt");
093                if ("none".equals(prompt)) {
094                        // we're not supposed to prompt, so "return an error"
095                        logger.info("Client requested no prompt, returning 403 from confirmation endpoint");
096                        model.put("code", HttpStatus.FORBIDDEN);
097                        return "httpCodeView";
098                }
099
100                //AuthorizationRequest clientAuth = (AuthorizationRequest) model.remove("authorizationRequest");
101
102                ClientDetails client = null;
103
104                try {
105                        client = clientService.loadClientByClientId(clientAuth.getClientId());
106                } catch (OAuth2Exception e) {
107                        logger.error("confirmAccess: OAuth2Exception was thrown when attempting to load client", e);
108                        model.put("code", HttpStatus.BAD_REQUEST);
109                        return "httpCodeView";
110                } catch (IllegalArgumentException e) {
111                        logger.error("confirmAccess: IllegalArgumentException was thrown when attempting to load client", e);
112                        model.put("code", HttpStatus.BAD_REQUEST);
113                        return "httpCodeView";
114                }
115
116                if (client == null) {
117                        logger.error("confirmAccess: could not find client " + clientAuth.getClientId());
118                        model.put("code", HttpStatus.NOT_FOUND);
119                        return "httpCodeView";
120                }
121
122                model.put("auth_request", clientAuth);
123                model.put("client", client);
124
125                String redirect_uri = clientAuth.getRedirectUri();
126
127                model.put("redirect_uri", redirect_uri);
128
129                Set<SystemScope> scopes = scopeService.fromStrings(clientAuth.getScope());
130
131                Set<SystemScope> sortedScopes = new LinkedHashSet<SystemScope>(scopes.size());
132                Set<SystemScope> systemScopes = scopeService.getAll();
133
134                // sort scopes for display based on the inherent order of system scopes
135                for (SystemScope s : systemScopes) {
136                        if (scopes.contains(s)) {
137                                sortedScopes.add(s);
138                        }
139                }
140
141                // add in any scopes that aren't system scopes to the end of the list
142                sortedScopes.addAll(Sets.difference(scopes, systemScopes));
143
144                model.put("scopes", sortedScopes);
145
146                // get the userinfo claims for each scope
147                UserInfo user = userInfoService.getByUsername(p.getName());
148                JsonObject userJson = user.toJson();
149                Map<String, Map<String, String>> claimsForScopes = new HashMap<String, Map<String, String>>();
150
151                for (SystemScope systemScope : sortedScopes) {
152                        Map<String, String> claimValues = new HashMap<String, String>();
153                        
154                        Set<String> claims = scopeClaimTranslationService.getClaimsForScope(systemScope.getValue());
155                        for (String claim : claims) {
156                                if (userJson.has(claim) && userJson.get(claim).isJsonPrimitive()) {
157                                        // TODO: this skips the address claim
158                                        claimValues.put(claim, userJson.get(claim).getAsString());
159                                }
160                        }
161                        
162                        claimsForScopes.put(systemScope.getValue(), claimValues);
163                }
164                
165                model.put("claims", claimsForScopes);
166
167                return "approve";
168        }
169

171         * @return the clientService
172         */

174                return clientService;
175        }
