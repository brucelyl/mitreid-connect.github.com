069        private IntrospectionAuthorityGranter introspectionAuthorityGranter = new SimpleIntrospectionAuthorityGranter();

071        private int defaultExpireTime = 300000; // 5 minutes in milliseconds
072        private boolean forceCacheExpireTime = false; // force removal of cached tokens based on default expire time
073        private boolean cacheNonExpiringTokens = false;
074        private boolean cacheTokens = true;
075
076        private HttpClient httpClient = HttpClientBuilder.create()
077                        .useSystemProperties()
078                        .build();
079        private HttpComponentsClientHttpRequestFactory factory = new HttpComponentsClientHttpRequestFactory(httpClient);
080
081        // Inner class to store in the hash map
082        private class TokenCacheObject {
083                OAuth2AccessToken token;
084                OAuth2Authentication auth;
085                Date cacheExpire;
086
087                private TokenCacheObject(OAuth2AccessToken token, OAuth2Authentication auth) {
088                        this.token = token;
089                        this.auth = auth;
090
091                        // we don't need to check the cacheTokens values, because this won't actually be added to the cache if cacheTokens is false
092                        // if the token isn't null we use the token expire time
093                        // if forceCacheExpireTime is also true, we also make sure that the token expire time is shorter than the default expire time
094                        if ((this.token.getExpiration() != null) && (!forceCacheExpireTime || (forceCacheExpireTime && (this.token.getExpiration().getTime() - System.currentTimeMillis() <= defaultExpireTime)))) {
095                                this.cacheExpire = this.token.getExpiration();
096                        } else { // if the token doesn't have an expire time, or if the using forceCacheExpireTime the token expire time is longer than the default, then use the default expire time
097                                Calendar cal = Calendar.getInstance();
098                                cal.add(Calendar.MILLISECOND, defaultExpireTime);
099                                this.cacheExpire = cal.getTime();
100                        }
101                }
102        }

202        /**
203         * Check to see if the introspection end point response for a token has been cached locally
204         * This call will return the token if it has been cached and is still valid according to
205         * the cache expire time on the TokenCacheObject. If a cached value has been found but is
206         * expired, either by default expire times or the token's own expire time, then the token is
207         * removed from the cache and null is returned.
208         * @param key is the token to check
209         * @return the cached TokenCacheObject or null
210         */
211        private TokenCacheObject checkCache(String key) {
212                if (cacheTokens && authCache.containsKey(key)) {
213                        TokenCacheObject tco = authCache.get(key);
214
215                        if (tco != null && tco.cacheExpire != null && tco.cacheExpire.after(new Date())) {
216                                return tco;
217                        } else {
218                                // if the token is expired, don't keep things around.
219                                authCache.remove(key);
220                        }
221                }
222                return null;
223        }
224
225        private OAuth2Request createStoredRequest(final JsonObject token) {
226                String clientId = token.get("client_id").getAsString();
227                Set<String> scopes = new HashSet<>();
228                if (token.has("scope")) {
229                        scopes.addAll(OAuth2Utils.parseParameterList(token.get("scope").getAsString()));
230                }
231                Map<String, String> parameters = new HashMap<>();
232                parameters.put("client_id", clientId);
233                parameters.put("scope", OAuth2Utils.formatParameterList(scopes));
234                OAuth2Request storedRequest = new OAuth2Request(parameters, clientId, null, true, scopes, null, null, null, null);
235                return storedRequest;
236        }
237
238        private Authentication createAuthentication(JsonObject token) {
239                return new PreAuthenticatedAuthenticationToken(token.get("sub").getAsString(), token, introspectionAuthorityGranter.getAuthorities(token));
240        }

242        private OAuth2AccessToken createAccessToken(final JsonObject token, final String tokenString) {
243                OAuth2AccessToken accessToken = new OAuth2AccessTokenImpl(token, tokenString);
244                return accessToken;

247        /**
248         * Validate a token string against the introspection endpoint,
249         * then parse it and store it in the local cache if caching is enabled.
250         *
251         * @param accessToken Token to pass to the introspection endpoint
252         * @return TokenCacheObject containing authentication and token if the token was valid, otherwise null
253         */
254        private TokenCacheObject parseToken(String accessToken) {
255
256                // find out which URL to ask
257                String introspectionUrl;
258                RegisteredClient client;
259                try {
260                        introspectionUrl = introspectionConfigurationService.getIntrospectionUrl(accessToken);
261                        client = introspectionConfigurationService.getClientConfiguration(accessToken);
262                } catch (IllegalArgumentException e) {
263                        logger.error("Unable to load introspection URL or client configuration", e);
264                        return null;
265                }
266                // Use the SpringFramework RestTemplate to send the request to the
267                // endpoint
268                String validatedToken = null;
269
270                RestTemplate restTemplate;
271                MultiValueMap<String, String> form = new LinkedMultiValueMap<>();
272
273                final String clientId = client.getClientId();
274                final String clientSecret = client.getClientSecret();
275
276                if (SECRET_BASIC.equals(client.getTokenEndpointAuthMethod())){
277                        // use BASIC auth if configured to do so
278                        restTemplate = new RestTemplate(factory) {
279
280                                @Override
281                                protected ClientHttpRequest createRequest(URI url, HttpMethod method) throws IOException {
282                                        ClientHttpRequest httpRequest = super.createRequest(url, method);
283                                        httpRequest.getHeaders().add("Authorization",
284                                                        String.format("Basic %s", Base64.encode(String.format("%s:%s", clientId, clientSecret)) ));
285                                        return httpRequest;
286                                }
287                        };
288                } else {  //Alternatively use form based auth
289                        restTemplate = new RestTemplate(factory);
290
291                        form.add("client_id", clientId);
292                        form.add("client_secret", clientSecret);
293                }
294
295                form.add("token", accessToken);
296
297                try {
298                        validatedToken = restTemplate.postForObject(introspectionUrl, form, String.class);
299                } catch (RestClientException rce) {
300                        logger.error("validateToken", rce);
301                }
302                if (validatedToken != null) {
303                        // parse the json
304                        JsonElement jsonRoot = new JsonParser().parse(validatedToken);
305                        if (!jsonRoot.isJsonObject()) {
306                                return null; // didn't get a proper JSON object
307                        }
308
309                        JsonObject tokenResponse = jsonRoot.getAsJsonObject();
310
311                        if (tokenResponse.get("error") != null) {
312                                // report an error?
313                                logger.error("Got an error back: " + tokenResponse.get("error") + ", " + tokenResponse.get("error_description"));
314                                return null;
315                        }
316
317                        if (!tokenResponse.get("active").getAsBoolean()) {
318                                // non-valid token
319                                logger.info("Server returned non-active token");
320                                return null;
321                        }
322                        // create an OAuth2Authentication
323                        OAuth2Authentication auth = new OAuth2Authentication(createStoredRequest(tokenResponse), createAuthentication(tokenResponse));
324                        // create an OAuth2AccessToken
325                        OAuth2AccessToken token = createAccessToken(tokenResponse, accessToken);
326
327                        if (token.getExpiration() == null || token.getExpiration().after(new Date())) {
328                                // Store them in the cache
329                                TokenCacheObject tco = new TokenCacheObject(token, auth);
330                                if (cacheTokens && (cacheNonExpiringTokens || token.getExpiration() != null)) {
331                                        authCache.put(accessToken, tco);
332                                }
333                                return tco;
334                        }
335                }
336
337                // when the token is invalid for whatever reason
338                return null;
339        }
340
341        @Override
342        public OAuth2Authentication loadAuthentication(String accessToken) throws AuthenticationException {
343                // First check if the in memory cache has an Authentication object, and
344                // that it is still valid
345                // If Valid, return it
346                TokenCacheObject cacheAuth = checkCache(accessToken);
347                if (cacheAuth != null) {
348                        return cacheAuth.auth;
349                } else {
350                        cacheAuth = parseToken(accessToken);
351                        if (cacheAuth != null) {
352                                return cacheAuth.auth;
353                        } else {
354                                return null;
355                        }
356                }
357        }
358
359        @Override
360        public OAuth2AccessToken readAccessToken(String accessToken) {
361                // First check if the in memory cache has a Token object, and that it is
362                // still valid
363                // If Valid, return it
364                TokenCacheObject cacheAuth = checkCache(accessToken);
365                if (cacheAuth != null) {
366                        return cacheAuth.token;
367                } else {
368                        cacheAuth = parseToken(accessToken);
369                        if (cacheAuth != null) {
370                                return cacheAuth.token;
371                        } else {
372                                return null;
373                        }
374                }
375        }
