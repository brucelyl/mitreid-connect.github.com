058/**
059 * This ResourceServerTokenServices implementation introspects incoming tokens at a
060 * server's introspection endpoint URL and passes an Authentication object along
061 * based on the response from the introspection endpoint.
062 * @author jricher
063 *
064 */
065public class IntrospectingTokenService implements ResourceServerTokenServices {
066
067        private IntrospectionConfigurationService introspectionConfigurationService;
068        private IntrospectionAuthorityGranter introspectionAuthorityGranter = new SimpleIntrospectionAuthorityGranter();
069
070        private HttpClient httpClient = new SystemDefaultHttpClient();
071        private HttpComponentsClientHttpRequestFactory factory = new HttpComponentsClientHttpRequestFactory(httpClient);
072
073        // Inner class to store in the hash map
074        private class TokenCacheObject {
075                OAuth2AccessToken token;
076                OAuth2Authentication auth;
077
078                private TokenCacheObject(OAuth2AccessToken token, OAuth2Authentication auth) {
079                        this.token = token;
080                        this.auth = auth;
081                }
082        }
083
084        private Map<String, TokenCacheObject> authCache = new HashMap<String, TokenCacheObject>();
085        private static Logger logger = LoggerFactory.getLogger(IntrospectingTokenService.class);
086
087        /**
088         * @return the introspectionConfigurationService
089         */
090        public IntrospectionConfigurationService getIntrospectionConfigurationService() {
091                return introspectionConfigurationService;
092        }
093
094        /**
095         * @param introspectionConfigurationService the introspectionConfigurationService to set
096         */
097        public void setIntrospectionConfigurationService(IntrospectionConfigurationService introspectionUrlProvider) {
098                this.introspectionConfigurationService = introspectionUrlProvider;
099        }

102         * @param introspectionAuthorityGranter the introspectionAuthorityGranter to set

115        // Check if there is a token and authentication in the cache
116        // and check if it is not expired.
117        private TokenCacheObject checkCache(String key) {
118                if (authCache.containsKey(key)) {
119                        TokenCacheObject tco = authCache.get(key);
120                        if (tco.token.getExpiration().after(new Date())) {
121                                return tco;
122                        } else {
123                                // if the token is expired, don't keep things around.
124                                authCache.remove(key);
125                        }
126                }
127                return null;
128        }
129
130        private OAuth2Request createStoredRequest(final JsonObject token) {
131                String clientId = token.get("client_id").getAsString();
132                Set<String> scopes = new HashSet<String>();
133                if (token.has("scope")) {
134                        scopes.addAll(OAuth2Utils.parseParameterList(token.get("scope").getAsString()));
135                }
136                Map<String, String> parameters = new HashMap<String, String>();
137                parameters.put("client_id", clientId);
138                parameters.put("scope", OAuth2Utils.formatParameterList(scopes));
139                OAuth2Request storedRequest = new OAuth2Request(parameters, clientId, null, true, scopes, null, null, null, null);
140                return storedRequest;
141        }
142
143        private Authentication createAuthentication(JsonObject token) {
144                return new PreAuthenticatedAuthenticationToken(token.get("sub").getAsString(), token, introspectionAuthorityGranter.getAuthorities(token));
145        }
146
147        private OAuth2AccessToken createAccessToken(final JsonObject token, final String tokenString) {
148                OAuth2AccessToken accessToken = new OAuth2AccessTokenImpl(token, tokenString);
149                return accessToken;
150        }
151
152        // Validate a token string against the introspection endpoint,
153        // then parse it and store it in the local cache. Return true on
154        // sucess, false otherwise.
155        private boolean parseToken(String accessToken) {
156
157                // find out which URL to ask
158                String introspectionUrl;
159                RegisteredClient client;
160                try {
161                        introspectionUrl = introspectionConfigurationService.getIntrospectionUrl(accessToken);
162                        client = introspectionConfigurationService.getClientConfiguration(accessToken);
163                } catch (IllegalArgumentException e) {
164                        logger.error("Unable to load introspection URL or client configuration", e);
165                        return false;
166                }
167                // Use the SpringFramework RestTemplate to send the request to the
168                // endpoint
169                String validatedToken = null;
170
171                RestTemplate restTemplate;
172                MultiValueMap<String, String> form = new LinkedMultiValueMap<String, String>();
173
174                final String clientId = client.getClientId();
175                final String clientSecret = client.getClientSecret();
176
177                if (SECRET_BASIC.equals(client.getTokenEndpointAuthMethod())){
178                        // use BASIC auth if configured to do so
179                        restTemplate = new RestTemplate(factory) {
180
181                                @Override
182                                protected ClientHttpRequest createRequest(URI url, HttpMethod method) throws IOException {
183                                        ClientHttpRequest httpRequest = super.createRequest(url, method);
184                                        httpRequest.getHeaders().add("Authorization",
185                                                        String.format("Basic %s", Base64.encode(String.format("%s:%s", clientId, clientSecret)) ));
186                                        return httpRequest;
187                                }
188                        };
189                } else {  //Alternatively use form based auth
190                        restTemplate = new RestTemplate(factory);
191
192                        form.add("client_id", clientId);
193                        form.add("client_secret", clientSecret);
194                }
195
196                form.add("token", accessToken);
197
198                try {
199                        validatedToken = restTemplate.postForObject(introspectionUrl, form, String.class);
200                } catch (RestClientException rce) {
201                        logger.error("validateToken", rce);
202                }
203                if (validatedToken != null) {
204                        // parse the json
205                        JsonElement jsonRoot = new JsonParser().parse(validatedToken);
206                        if (!jsonRoot.isJsonObject()) {
207                                return false; // didn't get a proper JSON object
208                        }
209
210                        JsonObject tokenResponse = jsonRoot.getAsJsonObject();
211
212                        if (tokenResponse.get("error") != null) {
213                                // report an error?
214                                logger.error("Got an error back: " + tokenResponse.get("error") + ", " + tokenResponse.get("error_description"));
215                                return false;
216                        }
217
218                        if (!tokenResponse.get("active").getAsBoolean()) {
219                                // non-valid token
220                                logger.info("Server returned non-active token");
221                                return false;
222                        }
223                        // create an OAuth2Authentication
224                        OAuth2Authentication auth = new OAuth2Authentication(createStoredRequest(tokenResponse), createAuthentication(tokenResponse));
225                        // create an OAuth2AccessToken
226                        OAuth2AccessToken token = createAccessToken(tokenResponse, accessToken);
227
228                        if (token.getExpiration().after(new Date())) {
229                                // Store them in the cache
230                                authCache.put(accessToken, new TokenCacheObject(token, auth));
231
232                                return true;
233                        }
234                }
235
236                // If we never put a token and an authentication in the cache...
237                return false;
238        }
239
240        @Override
241        public OAuth2Authentication loadAuthentication(String accessToken) throws AuthenticationException {
242                // First check if the in memory cache has an Authentication object, and
243                // that it is still valid
244                // If Valid, return it
245                TokenCacheObject cacheAuth = checkCache(accessToken);
246                if (cacheAuth != null) {
247                        return cacheAuth.auth;
248                } else {
249                        if (parseToken(accessToken)) {
250                                cacheAuth = authCache.get(accessToken);
251                                if (cacheAuth != null && (cacheAuth.token.getExpiration().after(new Date()))) {
252                                        return cacheAuth.auth;
253                                } else {
254                                        return null;
255                                }
256                        } else {
257                                return null;
258                        }
259                }
260        }
261
262        @Override
263        public OAuth2AccessToken readAccessToken(String accessToken) {
264                // First check if the in memory cache has a Token object, and that it is
265                // still valid
266                // If Valid, return it
267                TokenCacheObject cacheAuth = checkCache(accessToken);
268                if (cacheAuth != null) {
269                        return cacheAuth.token;
270                } else {
271                        if (parseToken(accessToken)) {
272                                cacheAuth = authCache.get(accessToken);
273                                if (cacheAuth != null && (cacheAuth.token.getExpiration().after(new Date()))) {
274                                        return cacheAuth.token;
275                                } else {
276                                        return null;
277                                }
278                        } else {
279                                return null;
280                        }
281                }
282        }
