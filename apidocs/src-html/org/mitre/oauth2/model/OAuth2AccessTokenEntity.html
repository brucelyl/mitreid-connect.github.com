020package org.mitre.oauth2.model;
021
022import java.text.ParseException;
023import java.util.Date;
024import java.util.HashMap;
025import java.util.Map;
026import java.util.Set;
027
028import javax.persistence.Basic;
029import javax.persistence.CascadeType;
030import javax.persistence.CollectionTable;
031import javax.persistence.Column;
032import javax.persistence.ElementCollection;
033import javax.persistence.Entity;
034import javax.persistence.FetchType;
035import javax.persistence.GeneratedValue;
036import javax.persistence.GenerationType;
037import javax.persistence.Id;
038import javax.persistence.JoinColumn;
039import javax.persistence.ManyToOne;
040import javax.persistence.NamedQueries;
041import javax.persistence.NamedQuery;
042import javax.persistence.OneToOne;
043import javax.persistence.Table;
044import javax.persistence.Temporal;
045import javax.persistence.Transient;
046
047import org.springframework.security.oauth2.common.OAuth2AccessToken;
048import org.springframework.security.oauth2.common.OAuth2RefreshToken;
049
050import com.nimbusds.jwt.JWT;
051import com.nimbusds.jwt.JWTParser;
052
053/**
054 * @author jricher
055 *
056 */
057@Entity
058@Table(name="access_token")
059@NamedQueries({
060        @NamedQuery(name = "OAuth2AccessTokenEntity.getByRefreshToken", query = "select a from OAuth2AccessTokenEntity a where a.refreshToken = :refreshToken"),
061        @NamedQuery(name = "OAuth2AccessTokenEntity.getByClient", query = "select a from OAuth2AccessTokenEntity a where a.client = :client"),
062        @NamedQuery(name = "OAuth2AccessTokenEntity.getExpired", query = "select a from OAuth2AccessTokenEntity a where a.expiration is not null and a.expiration < current_timestamp"),
063        @NamedQuery(name = "OAuth2AccessTokenEntity.getByAuthentication", query = "select a from OAuth2AccessTokenEntity a where a.authenticationHolder.authentication = :authentication"),
064        @NamedQuery(name = "OAuth2AccessTokenEntity.getByIdToken", query = "select a from OAuth2AccessTokenEntity a where a.idToken = :idToken"),
065        @NamedQuery(name = "OAuth2AccessTokenEntity.getByTokenValue", query = "select a from OAuth2AccessTokenEntity a where a.value = :tokenValue")
066})
067//@JsonSerialize(using = OAuth2AccessTokenSerializer.class)
068//@JsonDeserialize(using = OAuth2AccessTokenDeserializer.class)
069public class OAuth2AccessTokenEntity implements OAuth2AccessToken {
070
071        public static final String ID_TOKEN_SCOPE = "id-token";
072        public static final String REGISTRATION_TOKEN_SCOPE = "registration-token";
073
074        public static String ID_TOKEN = "id_token";
075
076        private Long id;
077
078        private ClientDetailsEntity client;
079
080        private AuthenticationHolderEntity authenticationHolder; // the authentication that made this access
081
082        private JWT jwtValue; // JWT-encoded access token value
083
084        private OAuth2AccessTokenEntity idToken; // JWT-encoded OpenID Connect IdToken
085
086        private Date expiration;
087
088        private String tokenType = OAuth2AccessToken.BEARER_TYPE;
089
090        private OAuth2RefreshTokenEntity refreshToken;
091
092        private Set<String> scope;
093
094        /**
095         * Create a new, blank access token
096         */
097        public OAuth2AccessTokenEntity() {
098
099        }
100
101        /**
102         * @return the id
103         */
104        @Id
105        @GeneratedValue(strategy = GenerationType.IDENTITY)
106        public Long getId() {
107                return id;
108        }
109
110        /**
111         * @param id the id to set
112         */
113        public void setId(Long id) {
114                this.id = id;
115        }
116
117        /**
118         * Get all additional information to be sent to the serializer. Inserts a copy of the IdToken (in JWT String form).
119         */
120        @Override
121        @Transient
122        public Map<String, Object> getAdditionalInformation() {
123                Map<String, Object> map = new HashMap<String, Object>(); //super.getAdditionalInformation();
124                if (getIdToken() != null) {
125                        map.put(ID_TOKEN, getIdTokenString());
126                }
127                return map;
128        }
129
130        /**
131         * The authentication in place when this token was created.
132         * @return the authentication
133         */
134        @ManyToOne
135        @JoinColumn(name = "auth_holder_id")
136        public AuthenticationHolderEntity getAuthenticationHolder() {
137                return authenticationHolder;
138        }
139
140        /**
141         * @param authentication the authentication to set
142         */
143        public void setAuthenticationHolder(AuthenticationHolderEntity authenticationHolder) {
144                this.authenticationHolder = authenticationHolder;
145        }
146
147        /**
148         * @return the client
149         */
150        @ManyToOne
151        @JoinColumn(name = "client_id")
152        public ClientDetailsEntity getClient() {
153                return client;
154        }
155
156        /**
157         * @param client the client to set
158         */
159        public void setClient(ClientDetailsEntity client) {
160                this.client = client;
161        }
162
163        /**
164         * Get the string-encoded value of this access token.
165         */
166        @Override
167        @Basic
168        @Column(name="token_value")
169        public String getValue() {
170                return jwtValue.serialize();
171        }
172
173        /**
174         * Set the "value" of this Access Token
175         * 
176         * @param value the JWT string
177         * @throws ParseException if "value" is not a properly formatted JWT string
178         */
179        public void setValue(String value) throws ParseException {
180                setJwt(JWTParser.parse(value));
181        }
182
183        @Override
184        @Basic
185        @Temporal(javax.persistence.TemporalType.TIMESTAMP)
186        public Date getExpiration() {
187                return expiration;
188        }
189
190        public void setExpiration(Date expiration) {
191                this.expiration = expiration;
192        }
193
194        @Override
195        @Basic
196        @Column(name="token_type")
197        public String getTokenType() {
198                return tokenType;
199        }
200
201        public void setTokenType(String tokenType) {
202                this.tokenType = tokenType;
203        }
204
205        @Override
206        @ManyToOne
207        @JoinColumn(name="refresh_token_id")
208        public OAuth2RefreshTokenEntity getRefreshToken() {
209                return refreshToken;
210        }
211
212        public void setRefreshToken(OAuth2RefreshTokenEntity refreshToken) {
213                this.refreshToken = refreshToken;
214        }
215
216        public void setRefreshToken(OAuth2RefreshToken refreshToken) {
217                if (!(refreshToken instanceof OAuth2RefreshTokenEntity)) {
218                        // TODO: make a copy constructor instead....
219                        throw new IllegalArgumentException("Not a storable refresh token entity!");
220                }
221                // force a pass through to the entity version
222                setRefreshToken((OAuth2RefreshTokenEntity)refreshToken);
223        }
224
225        @Override
226        @ElementCollection(fetch=FetchType.EAGER)
227        @CollectionTable(
228                        joinColumns=@JoinColumn(name="owner_id"),
229                        name="token_scope"
230                        )
231        public Set<String> getScope() {
232                return scope;
233        }
234
235        public void setScope(Set<String> scope) {
236                this.scope = scope;
237        }
238
239        @Override
240        @Transient
241        public boolean isExpired() {
242                return getExpiration() == null ? false : System.currentTimeMillis() > getExpiration().getTime();
243        }
244
245        /**
246         * @return the idToken
247         */
248        @OneToOne(cascade=CascadeType.ALL) // one-to-one mapping for now
249        @JoinColumn(name = "id_token_id")
250        public OAuth2AccessTokenEntity getIdToken() {
251                return idToken;
252        }
253
254        /**
255         * @param idToken the idToken to set
256         */
257        public void setIdToken(OAuth2AccessTokenEntity idToken) {
258                this.idToken = idToken;
259        }
260
261        /**
262         * @return the idTokenString
263         */
264        @Transient
265        public String getIdTokenString() {
266                if (idToken != null) {
267                        return idToken.getValue(); // get the JWT string value of the id token entity
268                } else {
269                        return null;
270                }
271        }
272
273        /**
274         * @return the jwtValue
275         */
276        @Transient
277        public JWT getJwt() {
278                return jwtValue;
279        }
280
281        /**
282         * @param jwtValue the jwtValue to set
283         */
284        public void setJwt(JWT jwt) {
285                this.jwtValue = jwt;
286        }
287
288        @Override
289        public int getExpiresIn() {
290                // TODO Auto-generated method stub
291                return 0;
292        }
293}
