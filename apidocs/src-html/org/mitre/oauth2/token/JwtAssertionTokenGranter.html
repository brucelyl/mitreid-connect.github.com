060        // read and load up the existing token
061            String incomingTokenValue = authorizationRequest.getAuthorizationParameters().get("assertion");
062            OAuth2AccessTokenEntity incomingToken = tokenServices.readAccessToken(incomingTokenValue);
063            
064            ClientDetailsEntity client = incomingToken.getClient();
065
066            
067            if (incomingToken.getScope().contains(OAuth2AccessTokenEntity.ID_TOKEN_SCOPE)) {
068                    
069                if (!client.getClientId().equals(authorizationRequest.getClientId())) {
070                        throw new InvalidClientException("Not the right client for this token");
071                    }
072
073                    // it's an ID token, process it accordingly
074                
075                try {
076                
077                        // TODO: make this use a more specific idtoken class
078                        JWT idToken = JWTParser.parse(incomingTokenValue);
079                        
080                        OAuth2AccessTokenEntity accessToken = tokenServices.getAccessTokenForIdToken(incomingToken);
081                        
082                        if (accessToken != null) {
083                        
084                                //OAuth2AccessTokenEntity newIdToken = tokenServices.get
085                                
086                                OAuth2AccessTokenEntity newIdTokenEntity = new OAuth2AccessTokenEntity();
087                                
088                                // copy over all existing claims
089                                JWTClaimsSet claims = new JWTClaimsSet(idToken.getJWTClaimsSet());
090        
091                                // update expiration and issued-at claims
092                                        if (client.getIdTokenValiditySeconds() != null) {
093                                                Date expiration = new Date(System.currentTimeMillis() + (client.getIdTokenValiditySeconds() * 1000L));
094                                                claims.setExpirationTime(expiration);
095                                                newIdTokenEntity.setExpiration(expiration);
096                                        }
097                                        claims.setIssueTime(new Date());
098        
099                                        
100                                        SignedJWT newIdToken = new SignedJWT((JWSHeader) idToken.getHeader(), claims);
101                        jwtService.signJwt(newIdToken);
102                                        
103                                        newIdTokenEntity.setJwt(newIdToken);
104                                        newIdTokenEntity.setAuthenticationHolder(incomingToken.getAuthenticationHolder());
105                                        newIdTokenEntity.setScope(incomingToken.getScope());
106                                        newIdTokenEntity.setClient(incomingToken.getClient());
107                                        
108                                        newIdTokenEntity = tokenServices.saveAccessToken(newIdTokenEntity);
109        
110                                        // attach the ID token to the access token entity
111                                        accessToken.setIdToken(newIdTokenEntity);
112                                        accessToken = tokenServices.saveAccessToken(accessToken);
113                                        
114                                        // delete the old ID token
115                                        tokenServices.revokeAccessToken(incomingToken);
116                                        
117                                        return newIdTokenEntity;
118                                
119                        }
120                } catch (ParseException e) {
121                        logger.warn("Couldn't parse id token", e);
122                }
123                
124            }

126            // if we got down here, we didn't actually create any tokens, so return null

128            return null;
