004package org.mitre.openid.connect.repository.impl;
005
006import javax.persistence.EntityManager;
007import javax.persistence.PersistenceContext;
008import javax.persistence.TypedQuery;
009
010import org.mitre.openid.connect.model.PairwiseIdentifier;
011import org.mitre.openid.connect.repository.PairwiseIdentifierRepository;
012import org.springframework.stereotype.Repository;
013import org.springframework.transaction.annotation.Transactional;
014
015import static org.mitre.util.jpa.JpaUtil.getSingleResult;
016import static org.mitre.util.jpa.JpaUtil.saveOrUpdate;
017
018/**
019 * @author jricher
020 *
021 */
022@Repository
023public class JpaPairwiseIdentifierRepository implements PairwiseIdentifierRepository {
024
025        @PersistenceContext
026        private EntityManager manager;
027
028        /* (non-Javadoc)
029         * @see org.mitre.openid.connect.repository.PairwiseIdentifierRepository#getBySectorIdentifier(java.lang.String, java.lang.String)
030         */
031        @Override
032        public PairwiseIdentifier getBySectorIdentifier(String sub, String sectorIdentifierUri) {
033                TypedQuery<PairwiseIdentifier> query = manager.createNamedQuery("PairwiseIdentifier.getBySectorIdentifier", PairwiseIdentifier.class);
034                query.setParameter("sub", sub);
035                query.setParameter("sectorIdentifier", sectorIdentifierUri);
036
037                return getSingleResult(query.getResultList());
038        }
039
040        /* (non-Javadoc)
041         * @see org.mitre.openid.connect.repository.PairwiseIdentifierRepository#save(org.mitre.openid.connect.model.PairwiseIdentifier)
042         */
043        @Override
044        @Transactional
045        public void save(PairwiseIdentifier pairwise) {
046                saveOrUpdate(pairwise.getId(), manager, pairwise);
047        }
048
049}
