017package org.mitre.openid.connect.service;
018
019import java.io.IOException;
020
021import com.google.gson.stream.JsonReader;
022import com.google.gson.stream.JsonWriter;
023
024/**
025 * @author jricher
026 * @author arielak
027 */
028public interface MITREidDataService {
029
030        /**
031         * Data member for 1.X configurations
032         */
033        public static final String MITREID_CONNECT_1_0 = "mitreid-connect-1.0";
034        public static final String MITREID_CONNECT_1_1 = "mitreid-connect-1.1";
035        public static final String MITREID_CONNECT_1_2 = "mitreid-connect-1.2";
036
037        // member names
038        public static final String REFRESHTOKENS = "refreshTokens";
039        public static final String ACCESSTOKENS = "accessTokens";
040        public static final String WHITELISTEDSITES = "whitelistedSites";
041        public static final String BLACKLISTEDSITES = "blacklistedSites";
042        public static final String AUTHENTICATIONHOLDERS = "authenticationHolders";
043        public static final String GRANTS = "grants";
044        public static final String CLIENTS = "clients";
045        public static final String SYSTEMSCOPES = "systemScopes";
046
047        /**
048         * Write out the current server state to the given JSON writer as a JSON object
049         * 
050         * @param writer
051         * @throws IOException
052         */
053        void exportData(JsonWriter writer) throws IOException;
054
055        /**
056         * Read in the current server state from the given JSON reader as a JSON object
057         * 
058         * @param reader
059         */
060        void importData(JsonReader reader) throws IOException;
061
062}
