017package org.mitre.openid.connect.service.impl;
018
019import java.util.HashSet;
020import java.util.Map;
021import java.util.Set;
022
023import org.mitre.openid.connect.service.ScopeClaimTranslationService;
024import org.springframework.stereotype.Service;
025
026import com.google.common.collect.HashMultimap;
027import com.google.common.collect.Maps;
028import com.google.common.collect.SetMultimap;
029
030/**
031 * Service to map scopes to claims, and claims to Java field names
032 * 
033 * @author Amanda Anganes
034 *
035 */
036@Service("scopeClaimTranslator")
037public class DefaultScopeClaimTranslationService implements ScopeClaimTranslationService {
038
039        private SetMultimap<String, String> scopesToClaims = HashMultimap.create();
040        private Map<String, String> claimsToFields = Maps.newHashMap();
041
042        /**
043         * Default constructor; initializes scopesToClaims map
044         */
045        public DefaultScopeClaimTranslationService() {
046
047                scopesToClaims.put("openid", "sub");
048
049                scopesToClaims.put("profile", "name");
050                scopesToClaims.put("profile", "preferred_username");
051                scopesToClaims.put("profile", "given_name");
052                scopesToClaims.put("profile", "family_name");
053                scopesToClaims.put("profile", "middle_name");
054                scopesToClaims.put("profile", "nickname");
055                scopesToClaims.put("profile", "profile");
056                scopesToClaims.put("profile", "picture");
057                scopesToClaims.put("profile", "website");
058                scopesToClaims.put("profile", "gender");
059                scopesToClaims.put("profile", "zone_info");
060                scopesToClaims.put("profile", "locale");
061                scopesToClaims.put("profile", "updated_at");
062                scopesToClaims.put("profile", "birthdate");
063
064                scopesToClaims.put("email", "email");
065                scopesToClaims.put("email", "email_verified");
066
067                scopesToClaims.put("phone", "phone_number");
068                scopesToClaims.put("phone", "phone_number_verified");
069
070                scopesToClaims.put("address", "address");
071
072                claimsToFields.put("sub", "sub");
073
074                claimsToFields.put("name", "name");
075                claimsToFields.put("preferred_username", "preferredUsername");
076                claimsToFields.put("given_name", "givenName");
077                claimsToFields.put("family_name", "familyName");
078                claimsToFields.put("middle_name", "middleName");
079                claimsToFields.put("nickname", "nickname");
080                claimsToFields.put("profile", "profile");
081                claimsToFields.put("picture", "picture");
082                claimsToFields.put("website", "website");
083                claimsToFields.put("gender", "gender");
084                claimsToFields.put("zone_info", "zoneinfo");
085                claimsToFields.put("locale", "locale");
086                claimsToFields.put("updated_at", "updatedTime");
087                claimsToFields.put("birthdate", "birthdate");
088
089                claimsToFields.put("email", "email");
090                claimsToFields.put("email_verified", "emailVerified");
091
092                claimsToFields.put("phone_number", "phoneNumber");
093                claimsToFields.put("phone_number_verified", "phoneNumberVerified");
094
095                claimsToFields.put("address", "address");
096
097        }
098
099        /* (non-Javadoc)
100         * @see org.mitre.openid.connect.service.ScopeClaimTranslationService#getClaimsForScope(java.lang.String)
101         */
102        @Override
103        public Set<String> getClaimsForScope(String scope) {
104                if (scopesToClaims.containsKey(scope)) {
105                        return scopesToClaims.get(scope);
106                } else {
107                        return new HashSet<>();
108                }
109        }
110
111        /* (non-Javadoc)
112         * @see org.mitre.openid.connect.service.ScopeClaimTranslationService#getClaimsForScopeSet(java.util.Set)
113         */
114        @Override
115        public Set<String> getClaimsForScopeSet(Set<String> scopes) {
116                Set<String> result = new HashSet<>();
117                for (String scope : scopes) {
118                        result.addAll(getClaimsForScope(scope));
119                }
120                return result;
121        }
122
123        /* (non-Javadoc)
124         * @see org.mitre.openid.connect.service.ScopeClaimTranslationService#getFieldNameForClaim(java.lang.String)
125         */
126        @Override
127        public String getFieldNameForClaim(String claim) {
128                return claimsToFields.get(claim);
129        }
130
131}
