271        /**
272         * @param writer
273         */
274        private void writeAuthenticationHolders(JsonWriter writer) throws IOException {
275                for (AuthenticationHolderEntity holder : authHolderRepository.getAll()) {
276                        writer.beginObject();
277                        writer.name(ID).value(holder.getId());
278
279                        writer.name(REQUEST_PARAMETERS);
280                        writer.beginObject();
281                        for (Entry<String, String> entry : holder.getRequestParameters().entrySet()) {
282                                writer.name(entry.getKey()).value(entry.getValue());
283                        }
284                        writer.endObject();
285                        writer.name(CLIENT_ID).value(holder.getClientId());
286                        Set<String> scope = holder.getScope();
287                        writer.name(SCOPE);
288                        writer.beginArray();
289                        for (String s : scope) {
290                                writer.value(s);
291                        }
292                        writer.endArray();
293                        writer.name(RESOURCE_IDS);
294                        writer.beginArray();
295                        if (holder.getResourceIds() != null) {
296                                for (String s : holder.getResourceIds()) {
297                                        writer.value(s);
298                                }
299                        }
300                        writer.endArray();
301                        writer.name(AUTHORITIES);
302                        writer.beginArray();
303                        for (GrantedAuthority authority : holder.getAuthorities()) {
304                                writer.value(authority.getAuthority());
305                        }
306                        writer.endArray();
307                        writer.name(APPROVED).value(holder.isApproved());
308                        writer.name(REDIRECT_URI).value(holder.getRedirectUri());
309                        writer.name(RESPONSE_TYPES);
310                        writer.beginArray();
311                        for (String s : holder.getResponseTypes()) {
312                                writer.value(s);
313                        }
314                        writer.endArray();
315                        writer.name(EXTENSIONS);
316                        writer.beginObject();
317                        for (Entry<String, Serializable> entry : holder.getExtensions().entrySet()) {
318                                // while the extension map itself is Serializable, we enforce storage of Strings
319                                if (entry.getValue() instanceof String) {
320                                        writer.name(entry.getKey()).value((String) entry.getValue());
321                                } else {
322                                        logger.warn("Skipping non-string extension: " + entry);
323                                }
324                        }
325                        writer.endObject();
326
327                        writer.name(SAVED_USER_AUTHENTICATION);
328                        if (holder.getUserAuth() != null) {
329                                writer.beginObject();
330                                writer.name(NAME).value(holder.getUserAuth().getName());
331                                writer.name(SOURCE_CLASS).value(holder.getUserAuth().getSourceClass());
332                                writer.name(AUTHENTICATED).value(holder.getUserAuth().isAuthenticated());
333                                writer.name(AUTHORITIES);
334                                writer.beginArray();
335                                for (GrantedAuthority authority : holder.getUserAuth().getAuthorities()) {
336                                        writer.value(authority.getAuthority());
337                                }
338                                writer.endArray();
339
340                                writer.endObject();
341                        } else {
342                                writer.nullValue();
343                        }
344
345
346                        writer.endObject();
347                        logger.debug("Wrote authentication holder {}", holder.getId());
348                }
349                logger.info("Done writing authentication holders");
350        }

536        /* (non-Javadoc)
537         * @see org.mitre.openid.connect.service.MITREidDataService#importData(com.google.gson.stream.JsonReader)
538         */
539        @Override
540        public void importData(JsonReader reader) throws IOException {
541
542                logger.info("Reading configuration for 1.2");
543
544                // this *HAS* to start as an object
545                reader.beginObject();
546
547                while (reader.hasNext()) {
548                        JsonToken tok = reader.peek();
549                        switch (tok) {
550                        case NAME:
551                                String name = reader.nextName();
552                                // find out which member it is
553                                if (name.equals(CLIENTS)) {
554                                        readClients(reader);

561                                } else if (name.equals(AUTHENTICATIONHOLDERS)) {

564                                        readAccessTokens(reader);
565                                } else if (name.equals(REFRESHTOKENS)) {
566                                        readRefreshTokens(reader);
567                                } else if (name.equals(SYSTEMSCOPES)) {
568                                        readSystemScopes(reader);
569                                } else {
570                                        // unknown token, skip it
571                                        reader.skipValue();
572                                }
573                                break;
574                        case END_OBJECT:
575                                // the object ended, we're done here
576                                reader.endObject();
577                                continue;
578                        default:
579                                logger.debug("Found unexpected entry");
580                                reader.skipValue();
581                                continue;
582                        }
583                }
584                fixObjectReferences();

1148        /**
1149         * Read the list of system scopes from the reader and insert them into the
1150         * scope repository.
1151         *
1152         * @param reader
1153         * @throws IOException
1154         */
1155        private void readSystemScopes(JsonReader reader) throws IOException {
1156                reader.beginArray();
1157                while (reader.hasNext()) {
1158                        SystemScope scope = new SystemScope();
1159                        reader.beginObject();
1160                        while (reader.hasNext()) {
1161                                switch (reader.peek()) {
1162                                case END_OBJECT:
1163                                        continue;
1164                                case NAME:
1165                                        String name = reader.nextName();
1166                                        if (reader.peek() == JsonToken.NULL) {
1167                                                reader.skipValue();
1168                                        } else if (name.equals(VALUE)) {
1169                                                scope.setValue(reader.nextString());
1170                                        } else if (name.equals(DESCRIPTION)) {
1171                                                scope.setDescription(reader.nextString());
1172                                        } else if (name.equals(RESTRICTED)) {
1173                                                scope.setRestricted(reader.nextBoolean());
1174                                        } else if (name.equals(DEFAULT_SCOPE)) {
1175                                                scope.setDefaultScope(reader.nextBoolean());
1176                                        } else if (name.equals(ICON)) {
1177                                                scope.setIcon(reader.nextString());
1178                                        } else if (name.equals(STRUCTURED)) {
1179                                                scope.setStructured(reader.nextBoolean());
1180                                        } else if (name.equals(STRUCTURED_PARAMETER)) {
1181                                                scope.setStructuredParamDescription(reader.nextString());
1182                                        } else {
1183                                                logger.debug("found unexpected entry");
1184                                                reader.skipValue();
1185                                        }
1186                                        break;
1187                                default:
1188                                        logger.debug("Found unexpected entry");
1189                                        reader.skipValue();
1190                                        continue;
1191                                }
1192                        }
1193                        reader.endObject();
1194                        sysScopeRepository.save(scope);
1195                }
1196                reader.endArray();
1197                logger.info("Done reading system scopes");
1198        }
