019package org.mitre.openid.connect.service.impl;
020
021import java.io.ByteArrayInputStream;
022import java.io.ByteArrayOutputStream;
023import java.io.IOException;
024import java.io.ObjectInputStream;
025import java.io.ObjectOutputStream;
026import java.io.Serializable;
027import java.util.HashMap;
028import java.util.HashSet;
029import java.util.Map;
030import java.util.Set;
031
032import org.mitre.openid.connect.service.MITREidDataService;
033import org.slf4j.Logger;
034import org.slf4j.LoggerFactory;
035
036import com.google.common.io.BaseEncoding;
037import com.google.gson.stream.JsonReader;
038import com.google.gson.stream.JsonWriter;
039
040/**
041 *
042 * @author arielak
043 */
044public abstract class MITREidDataService_1_X implements MITREidDataService {
045        private static Logger logger = LoggerFactory.getLogger(MITREidDataService_1_X.class);
046    
047    protected static <T> T base64UrlDecodeObject(String encoded, Class<T> type) {
048        if (encoded == null) {
049                return null;
050        } else {
051                T deserialized = null;
052                try {
053                    byte[] decoded = BaseEncoding.base64Url().decode(encoded);
054                    ByteArrayInputStream bais = new ByteArrayInputStream(decoded);
055                    ObjectInputStream ois = new ObjectInputStream(bais);
056                    deserialized = type.cast(ois.readObject());
057                    ois.close();
058                    bais.close();
059                } catch (Exception ex) {
060                    logger.error("Unable to decode object", ex);
061                }
062                return deserialized;
063        }
064    }
065    
066    protected static String base64UrlEncodeObject(Serializable obj) {
067        if (obj == null) {
068                return null;
069        } else {
070                String encoded = null;
071                try {
072                    ByteArrayOutputStream baos = new ByteArrayOutputStream();
073                    ObjectOutputStream oos = new ObjectOutputStream(baos);
074                    oos.writeObject(obj);
075                    encoded = BaseEncoding.base64Url().encode(baos.toByteArray());
076                    oos.close();
077                    baos.close();
078                } catch (IOException ex) {
079                    logger.error("Unable to encode object", ex);
080                }
081                return encoded;
082        }
083    }
084    protected static Set readSet(JsonReader reader) throws IOException {
085        Set arraySet = null;
086        reader.beginArray();
087        switch (reader.peek()) {
088            case STRING:
089                arraySet = new HashSet<String>();
090                while (reader.hasNext()) {
091                    arraySet.add(reader.nextString());
092                }
093                break;
094            case NUMBER:
095                arraySet = new HashSet<Long>();
096                while (reader.hasNext()) {
097                    arraySet.add(reader.nextLong());
098                }
099                break;
100            default:
101                arraySet = new HashSet();
102                break;
103        }
104        reader.endArray();
105        return arraySet;
106    }
107    
108    protected static Map readMap(JsonReader reader) throws IOException {
109        Map map = new HashMap<String, Object>();
110        reader.beginObject();
111        while(reader.hasNext()) {
112            String name = reader.nextName();
113            Object value = null;
114            switch(reader.peek()) {
115                case STRING:
116                    value = reader.nextString();
117                    break;
118                case BOOLEAN:
119                    value = reader.nextBoolean();
120                    break;
121                case NUMBER:
122                    value = reader.nextLong();
123                    break;
124            }
125            map.put(name, value);
126        }
127        reader.endObject();
128        return map;
129    }
130    
131        protected void writeNullSafeArray(JsonWriter writer, Set<String> items)
132                        throws IOException {
133                if (items != null) {
134                        writer.beginArray();
135                    for (String s : items) {
136                        writer.value(s);
137                    }
138                    writer.endArray();
139                } else {
140                        writer.nullValue();
141                }
142        }
143}
