004package org.mitre.openid.connect.service.impl;
005
006import java.util.Set;
007import java.util.UUID;
008
009import org.mitre.oauth2.model.ClientDetailsEntity;
010import org.mitre.openid.connect.model.PairwiseIdentifier;
011import org.mitre.openid.connect.model.UserInfo;
012import org.mitre.openid.connect.repository.PairwiseIdentifierRepository;
013import org.mitre.openid.connect.service.PairwiseIdentiferService;
014import org.slf4j.Logger;
015import org.slf4j.LoggerFactory;
016import org.springframework.beans.factory.annotation.Autowired;
017import org.springframework.stereotype.Service;
018import org.springframework.web.util.UriComponents;
019import org.springframework.web.util.UriComponentsBuilder;
020
021import com.google.common.base.Strings;
022import com.google.common.collect.Iterables;
023
024/**
025 * @author jricher
026 *
027 */
028@Service("uuidPairwiseIdentiferService")
029public class UUIDPairwiseIdentiferService implements PairwiseIdentiferService {
030
031        private static Logger logger = LoggerFactory.getLogger(UUIDPairwiseIdentiferService.class);
032
033        @Autowired
034        private PairwiseIdentifierRepository pairwiseIdentifierRepository;
035
036        @Override
037        public String getIdentifier(UserInfo userInfo, ClientDetailsEntity client) {
038
039                String sectorIdentifier = null;
040
041                if (!Strings.isNullOrEmpty(client.getSectorIdentifierUri())) {
042                        UriComponents uri = UriComponentsBuilder.fromUriString(client.getSectorIdentifierUri()).build();
043                        sectorIdentifier = uri.getHost(); // calculate based on the host component only
044                } else {
045                        Set<String> redirectUris = client.getRedirectUris();
046                        UriComponents uri = UriComponentsBuilder.fromUriString(Iterables.getOnlyElement(redirectUris)).build();
047                        sectorIdentifier = uri.getHost(); // calculate based on the host of the only redirect URI
048                }
049
050                if (sectorIdentifier != null) {
051                        // if there's a sector identifier, use that for the lookup
052                        PairwiseIdentifier pairwise = pairwiseIdentifierRepository.getBySectorIdentifier(userInfo.getSub(), sectorIdentifier);
053
054                        if (pairwise == null) {
055                                // we don't have an identifier, need to make and save one
056
057                                pairwise = new PairwiseIdentifier();
058                                pairwise.setIdentifier(UUID.randomUUID().toString());
059                                pairwise.setUserSub(userInfo.getSub());
060                                pairwise.setSectorIdentifier(sectorIdentifier);
061
062                                pairwiseIdentifierRepository.save(pairwise);
063                        }
064
065                        return pairwise.getIdentifier();
066                } else {
067
068                        return null;
069                }
070        }
071
072}
