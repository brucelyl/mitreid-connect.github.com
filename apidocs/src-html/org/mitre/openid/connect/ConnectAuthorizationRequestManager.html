037@Component("authorizationRequestManager")
038public class ConnectAuthorizationRequestManager implements AuthorizationRequestManager {
039
040        private static Logger logger = LoggerFactory.getLogger(ConnectAuthorizationRequestManager.class);
041        
042        @Autowired
043        private NonceService nonceService;
044        
045        @Autowired
046        private ClientDetailsEntityService clientDetailsService;
047
048        @Autowired
049        private JWKSetSigningAndValidationServiceCacheService validators;
050
051        /**
052         * Constructor with arguments
053         * 
054         * @param clientDetailsService
055         * @param nonceService
056         */
057        public ConnectAuthorizationRequestManager(ClientDetailsEntityService clientDetailsService, NonceService nonceService) {
058                this.clientDetailsService = clientDetailsService;
059                this.nonceService = nonceService;
060        }
061        
062        /**
063         * Default empty constructor
064         */
065        public ConnectAuthorizationRequestManager() {
066                
067        }
068
069        @Override
070        public AuthorizationRequest createAuthorizationRequest(Map<String, String> inputParams) {
071
072                Map<String, String> parameters = processRequestObject(inputParams);
073                
074                String clientId = parameters.get("client_id");
075                if (clientId == null) {
076                        throw new InvalidClientException("A client id must be provided");
077                }
078                ClientDetails client = clientDetailsService.loadClientByClientId(clientId);
079                
080                String requestNonce = parameters.get("nonce");
081                
082                //Only process if the user is authenticated. If the user is not authenticated yet, this 
083                //code will be called a second time once the user is redirected from the login page back 
084                //to the auth endpoint.
085                Object principal = SecurityContextHolder.getContext().getAuthentication().getPrincipal();
086                
087                if (requestNonce != null && principal != null && principal instanceof User) {
088
089                        if (!nonceService.alreadyUsed(clientId, requestNonce)) {
090                                Nonce nonce = nonceService.create(clientId, requestNonce);
091                                nonceService.save(nonce);
092                        }
093                        else {
094                                throw new NonceReuseException(client.getClientId(), requestNonce);
095                        }
096        
097                }
098                
099                Set<String> scopes = OAuth2Utils.parseParameterList(parameters.get("scope"));
100                if ((scopes == null || scopes.isEmpty())) {
101                        //TODO: do we want to allow default scoping at all?
102                        Set<String> clientScopes = client.getScope();
103                        scopes = clientScopes;
104                }
105                
106                
107                // note that we have to inject the processed parameters in at this point so that SECOAUTH can find them later (and this object will get copy-constructored away anyway)
108                DefaultAuthorizationRequest request = new DefaultAuthorizationRequest(parameters, Collections.<String, String> emptyMap(), clientId, scopes);
109                request.addClientDetails(client);
110                return request;
111
112        }
113
114        /**
115         * @param inputParams
116         * @return
117         */
118    private Map<String, String> processRequestObject(Map<String, String> inputParams) {
119        
120        String jwtString = inputParams.get("request");
121        
122        // if there's no request object, bail early
123        if (Strings.isNullOrEmpty(jwtString)) {
124                return inputParams;
125        }
126
127        // start by copying over what's already in there
128        Map<String, String> parameters = new HashMap<String, String>(inputParams);
129
130        // parse the request object
131        try {
132                SignedJWT jwsObject = SignedJWT.parse(jwtString);
133                        JSONObject claims = jwsObject.getPayload().toJSONObject(); 
134                        
135                        // TODO: check parameter consistency, move keys to constants
136                        
137                        String clientId = JSONObjectUtils.getString(claims, "client_id");
138                        if (clientId != null) {
139                                parameters.put("client_id", clientId);
140                        }
141                        
142                        ClientDetailsEntity client = clientDetailsService.loadClientByClientId(clientId);
143
144                        if (client.getJwksUri() == null) {
145                                throw new InvalidClientException("Client must have a JWK URI registered to use request objects.");
146                        }
147                        
148                        // check JWT signature
149                        JwtSigningAndValidationService validator = validators.get(client.getJwksUri());
150                        if (validator == null) {
151                                throw new InvalidClientException("Client must have a JWK URI registered to use request objects.");
152                        }
153                        
154                        if (!validator.validateSignature(jwsObject)) {
155                                throw new AuthenticationServiceException("Signature did not validate for presented JWT request object.");
156                        }
157                        
158                        /*
159                         * if (in Claims):
160                         *              if (in params):
161                         *                      if (equal):
162                         *                              all set
163                         *                      else (not equal):
164                         *                              error
165                         *              else (not in params):
166                         *                      add to params
167                         * else (not in claims):
168                         *              we don't care
169                         */
170                        
171                        String responseTypes = JSONObjectUtils.getString(claims, "response_type");
172                        if (responseTypes != null) {
173                                parameters.put("response_type", responseTypes);
174                        }
175                        
176                        if (claims.get("redirect_uri") != null) {
177                                if (inputParams.containsKey("redirect_uri") == false) {
178                                        parameters.put("redirect_uri", JSONObjectUtils.getString(claims, "redirect_uri"));
179                                }
180                        }
181                        
182                        String state = JSONObjectUtils.getString(claims, "state");
183                        if(state != null) {
184                                if (inputParams.containsKey("state") == false) {
185                                        parameters.put("state", state);
186                                }
187                        }
188                        
189                        String nonce = JSONObjectUtils.getString(claims, "nonce");
190                        if(nonce != null) {
191                                if (inputParams.containsKey("nonce") == false) {
192                                        parameters.put("nonce", nonce);
193                                }
194                        }
195                        
196                        String display = JSONObjectUtils.getString(claims, "display");
197                        if (display != null) {
198                                if (inputParams.containsKey("display") == false) {
199                                        parameters.put("display", display);
200                                }
201                        }
202                        
203                        String prompt = JSONObjectUtils.getString(claims, "prompt");
204                        if (prompt != null) {
205                                if (inputParams.containsKey("prompt") == false) {
206                                        parameters.put("prompt", prompt);
207                                }
208                        }
209                        
210                        String scope = JSONObjectUtils.getString(claims, "scope");
211                        if (scope != null) {
212                                if (inputParams.containsKey("scope") == false) {
213                                        parameters.put("scope", scope);
214                                }
215                        }
216                
217        } catch (ParseException e) {
218                // TODO Auto-generated catch block
219                e.printStackTrace();
220        }
221                return parameters;
222    }
223
224        @Override
225        public void validateParameters(Map<String, String> parameters, ClientDetails clientDetails) {
226                if (parameters.containsKey("scope")) {
227                        if (clientDetails.isScoped()) {
228                                Set<String> validScope = clientDetails.getScope();
229                                for (String scope : OAuth2Utils.parseParameterList(parameters.get("scope"))) {
230                                        if (!validScope.contains(scope)) {
231                                                throw new InvalidScopeException("Invalid scope: " + scope, validScope);
232                                        }
233                                }
234                        }
235                }
236        }
