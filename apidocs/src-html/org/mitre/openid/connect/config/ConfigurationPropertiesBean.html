017package org.mitre.openid.connect.config;
018
019import java.util.List;
020import java.util.Locale;
021
022import javax.annotation.PostConstruct;
023
024import org.slf4j.Logger;
025import org.slf4j.LoggerFactory;
026import org.springframework.beans.factory.BeanCreationException;
027import org.springframework.util.StringUtils;
028
029import com.google.common.collect.Lists;
030import com.google.gson.Gson;
031
032
033
034/**
035 * Bean to hold configuration information that must be injected into various parts
036 * of our application. Set all of the properties here, and autowire a reference
037 * to this bean if you need access to any configuration properties.
038 * 
039 * @author AANGANES
040 *
041 */
042public class ConfigurationPropertiesBean {
043
044        /**
045         * Logger for this class
046         */
047        private static final Logger logger = LoggerFactory.getLogger(ConfigurationPropertiesBean.class);
048
049        private String issuer;
050
051        private String topbarTitle;
052        
053        private String shortTopbarTitle;
054
055        private String logoImageUrl;
056
057        private Long regTokenLifeTime;
058
059        private Long rqpTokenLifeTime;
060
061        private boolean forceHttps = false; // by default we just log a warning for HTTPS deployment
062
063        private Locale locale = Locale.ENGLISH; // we default to the english translation
064        
065        private List<String> languageNamespaces = Lists.newArrayList("messages");
066
067    private boolean dualClient = false;
068    
069    private boolean heartMode = false;
070
071    public ConfigurationPropertiesBean() {
072
073        }
074
075        /**
076         * Endpoints protected by TLS must have https scheme in the URI.
077         * @throws HttpsUrlRequiredException
078         */
079        @PostConstruct
080        public void checkConfigConsistency() {
081                if (!StringUtils.startsWithIgnoreCase(issuer, "https")) {
082                        if (this.forceHttps) {
083                                logger.error("Configured issuer url is not using https scheme. Server will be shut down!");
084                                throw new BeanCreationException("Issuer is not using https scheme as required: " + issuer);
085                        }
086                        else {
087                                logger.warn("\n\n**\n** WARNING: Configured issuer url is not using https scheme.\n**\n\n");
088                        }
089                }
090                
091                if (languageNamespaces == null || languageNamespaces.isEmpty()) {
092                        logger.error("No configured language namespaces! Text rendering will fail!");
093                }
094        }
095
096        /**
097         * @return the issuer baseUrl
098         */
099        public String getIssuer() {
100                return issuer;
101        }
102
103        /**
104         * @param iss the issuer to set
105         */
106        public void setIssuer(String iss) {
107                issuer = iss;
108        }
109
110        /**
111         * @return the topbarTitle
112         */
113        public String getTopbarTitle() {
114                return topbarTitle;
115        }
116
117        /**
118         * @param topbarTitle the topbarTitle to set
119         */
120        public void setTopbarTitle(String topbarTitle) {
121                this.topbarTitle = topbarTitle;
122        }
123        
124        /**
125         * @return If shortTopbarTitle is undefined, returns topbarTitle. 
126         */
127        public String getShortTopbarTitle() {
128                return shortTopbarTitle == null ? topbarTitle : shortTopbarTitle;
129        }
130        
131        public void setShortTopbarTitle(String shortTopbarTitle) {
132                this.shortTopbarTitle = shortTopbarTitle;
133        }
134
135        /**
136         * @return the logoImageUrl
137         */
138        public String getLogoImageUrl() {
139                return logoImageUrl;
140        }
141
142        /**
143         * @param logoImageUrl the logoImageUrl to set
144         */
145        public void setLogoImageUrl(String logoImageUrl) {
146                this.logoImageUrl = logoImageUrl;
147        }
148
149        /**
150         * @return the regTokenLifeTime
151         */
152        public Long getRegTokenLifeTime() {
153                return regTokenLifeTime;
154        }
155
156        /**
157         * @param regTokenLifeTime the registration token lifetime to set in seconds
158         */
159        public void setRegTokenLifeTime(Long regTokenLifeTime) {
160                this.regTokenLifeTime = regTokenLifeTime;
161        }
162
163        /**
164         * @return the rqpTokenLifeTime
165         */
166        public Long getRqpTokenLifeTime() {
167                return rqpTokenLifeTime;
168        }
169
170        /**
171         * @param rqpTokenLifeTime the rqpTokenLifeTime to set
172         */
173        public void setRqpTokenLifeTime(Long rqpTokenLifeTime) {
174                this.rqpTokenLifeTime = rqpTokenLifeTime;
175        }
176
177        public boolean isForceHttps() {
178                return forceHttps;
179        }
180
181        public void setForceHttps(boolean forceHttps) {
182                this.forceHttps = forceHttps;
183        }
184
185        /**
186         * @return the locale
187         */
188        public Locale getLocale() {
189                return locale;
190        }
191
192        /**
193         * @param locale the locale to set
194         */
195        public void setLocale(Locale locale) {
196                this.locale = locale;
197        }
198
199        /**
200         * @return the languageNamespaces
201         */
202        public List<String> getLanguageNamespaces() {
203                return languageNamespaces;
204        }
205
206        /**
207         * @param languageNamespaces the languageNamespaces to set
208         */
209        public void setLanguageNamespaces(List<String> languageNamespaces) {
210                this.languageNamespaces = languageNamespaces;
211        }
212
213        /**
214     * @return true if dual client is configured, otherwise false
215     */
216    public boolean isDualClient() {
217        if (isHeartMode()) {
218                return false; // HEART mode is incompatible with dual client mode
219        } else {
220                return dualClient;
221        }
222    }
223
224    /**
225     * @param dualClient the dual client configuration
226     */
227    public void setDualClient(boolean dualClient) {
228        this.dualClient = dualClient;
229    }
230    
231    /**
232     * Get the list of namespaces as a JSON string, for injection into the JavaScript UI
233     * @return
234     */
235    public String getLanguageNamespacesString() {
236        return new Gson().toJson(getLanguageNamespaces());
237    }
238    
239    /**
240     * Get the default namespace (first in the nonempty list)
241     */
242    public String getDefaultLanguageNamespace() {
243        return getLanguageNamespaces().get(0);
244    }
245
246        /**
247         * @return the heartMode
248         */
249        public boolean isHeartMode() {
250                return heartMode;
251        }
252
253        /**
254         * @param heartMode the heartMode to set
255         */
256        public void setHeartMode(boolean heartMode) {
257                this.heartMode = heartMode;
258        }
259}
