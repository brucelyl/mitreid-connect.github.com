017package org.mitre.openid.connect.config;
018
019
020
021/**
022 * 
023 * Container class for a client's view of a server's configuration
024 * 
025 * @author nemonik, jricher
026 * 
027 */
028public class ServerConfiguration {
029
030        private String authorizationEndpointUri;
031
032        private String tokenEndpointUri;
033
034        private String registrationEndpointUri;
035
036        private String issuer;
037
038        private String jwksUri;
039
040        private String userInfoUri;
041
042        /**
043         * @return the authorizationEndpointUri
044         */
045        public String getAuthorizationEndpointUri() {
046                return authorizationEndpointUri;
047        }
048
049        /**
050         * @param authorizationEndpointUri the authorizationEndpointUri to set
051         */
052        public void setAuthorizationEndpointUri(String authorizationEndpointUri) {
053                this.authorizationEndpointUri = authorizationEndpointUri;
054        }
055
056        /**
057         * @return the tokenEndpointUri
058         */
059        public String getTokenEndpointUri() {
060                return tokenEndpointUri;
061        }
062
063        /**
064         * @param tokenEndpointUri the tokenEndpointUri to set
065         */
066        public void setTokenEndpointUri(String tokenEndpointUri) {
067                this.tokenEndpointUri = tokenEndpointUri;
068        }
069
070        /**
071         * @return the issuer
072         */
073        public String getIssuer() {
074                return issuer;
075        }
076
077        /**
078         * @param issuer the issuer to set
079         */
080        public void setIssuer(String issuer) {
081                this.issuer = issuer;
082        }
083
084        /**
085         * @return the jwksUri
086         */
087        public String getJwksUri() {
088                return jwksUri;
089        }
090
091        /**
092         * @param jwksUri the jwksUri to set
093         */
094        public void setJwksUri(String jwksUri) {
095                this.jwksUri = jwksUri;
096        }
097
098        /**
099         * @return the userInfoUri
100         */
101        public String getUserInfoUri() {
102                return userInfoUri;
103        }
104
105        /**
106         * @param userInfoUri the userInfoUri to set
107         */
108        public void setUserInfoUri(String userInfoUri) {
109                this.userInfoUri = userInfoUri;
110        }
111
112        /**
113         * @return the registrationEndpointUri
114         */
115        public String getRegistrationEndpointUri() {
116                return registrationEndpointUri;
117        }
118
119        /**
120         * @param registrationEndpointUri the registrationEndpointUri to set
121         */
122        public void setRegistrationEndpointUri(String registrationEndpointUri) {
123                this.registrationEndpointUri = registrationEndpointUri;
124        }
125
126        /* (non-Javadoc)
127         * @see java.lang.Object#hashCode()
128         */
129        @Override
130        public int hashCode() {
131                final int prime = 31;
132                int result = 1;
133                result = prime * result + ((authorizationEndpointUri == null) ? 0 : authorizationEndpointUri.hashCode());
134                result = prime * result + ((issuer == null) ? 0 : issuer.hashCode());
135                result = prime * result + ((jwksUri == null) ? 0 : jwksUri.hashCode());
136                result = prime * result + ((registrationEndpointUri == null) ? 0 : registrationEndpointUri.hashCode());
137                result = prime * result + ((tokenEndpointUri == null) ? 0 : tokenEndpointUri.hashCode());
138                result = prime * result + ((userInfoUri == null) ? 0 : userInfoUri.hashCode());
139                return result;
140        }
141
142        /* (non-Javadoc)
143         * @see java.lang.Object#equals(java.lang.Object)
144         */
145        @Override
146        public boolean equals(Object obj) {
147                if (this == obj) {
148                        return true;
149                }
150                if (obj == null) {
151                        return false;
152                }
153                if (getClass() != obj.getClass()) {
154                        return false;
155                }
156                ServerConfiguration other = (ServerConfiguration) obj;
157                if (authorizationEndpointUri == null) {
158                        if (other.authorizationEndpointUri != null) {
159                                return false;
160                        }
161                } else if (!authorizationEndpointUri.equals(other.authorizationEndpointUri)) {
162                        return false;
163                }
164                if (issuer == null) {
165                        if (other.issuer != null) {
166                                return false;
167                        }
168                } else if (!issuer.equals(other.issuer)) {
169                        return false;
170                }
171                if (jwksUri == null) {
172                        if (other.jwksUri != null) {
173                                return false;
174                        }
175                } else if (!jwksUri.equals(other.jwksUri)) {
176                        return false;
177                }
178                if (registrationEndpointUri == null) {
179                        if (other.registrationEndpointUri != null) {
180                                return false;
181                        }
182                } else if (!registrationEndpointUri.equals(other.registrationEndpointUri)) {
183                        return false;
184                }
185                if (tokenEndpointUri == null) {
186                        if (other.tokenEndpointUri != null) {
187                                return false;
188                        }
189                } else if (!tokenEndpointUri.equals(other.tokenEndpointUri)) {
190                        return false;
191                }
192                if (userInfoUri == null) {
193                        if (other.userInfoUri != null) {
194                                return false;
195                        }
196                } else if (!userInfoUri.equals(other.userInfoUri)) {
197                        return false;
198                }
199                return true;
200        }
