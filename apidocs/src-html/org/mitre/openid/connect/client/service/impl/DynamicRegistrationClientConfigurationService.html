020package org.mitre.openid.connect.client.service.impl;
021
022import java.util.concurrent.ExecutionException;
023
024import org.apache.http.client.HttpClient;
025import org.apache.http.impl.client.DefaultHttpClient;
026import org.mitre.oauth2.model.ClientDetailsEntity;
027import org.mitre.openid.connect.ClientDetailsEntityJsonProcessor;
028import org.mitre.openid.connect.client.service.ClientConfigurationService;
029import org.mitre.openid.connect.config.ServerConfiguration;
030import org.slf4j.Logger;
031import org.slf4j.LoggerFactory;
032import org.springframework.http.HttpEntity;
033import org.springframework.http.HttpHeaders;
034import org.springframework.http.MediaType;
035import org.springframework.http.client.HttpComponentsClientHttpRequestFactory;
036import org.springframework.security.oauth2.provider.ClientDetails;
037import org.springframework.web.client.RestTemplate;
038
039import com.google.common.cache.CacheBuilder;
040import com.google.common.cache.CacheLoader;
041import com.google.common.cache.LoadingCache;
042import com.google.common.collect.Lists;
043import com.google.gson.JsonObject;
044import com.google.gson.JsonParser;
045
046/**
047 * @author jricher
048 *
049 */
050public class DynamicRegistrationClientConfigurationService implements ClientConfigurationService {
051
052        private static Logger logger = LoggerFactory.getLogger(DynamicServerConfigurationService.class);
053
054        private LoadingCache<ServerConfiguration, ClientDetailsEntity> clients;
055
056        private ClientDetailsEntity template;
057
058        public DynamicRegistrationClientConfigurationService() {
059                clients = CacheBuilder.newBuilder().build(new DynamicClientRegistrationLoader());
060        }
061
062        @Override
063        public ClientDetails getClientConfiguration(ServerConfiguration issuer) {
064                try {
065                        return clients.get(issuer);
066                } catch (ExecutionException e) {
067                        logger.warn("Unable to get client configuration", e);
068                        return null;
069                }
070        }
071
072        /**
073         * @return the template
074         */
075        public ClientDetailsEntity getTemplate() {
076                return template;
077        }
078
079        /**
080         * @param template the template to set
081         */
082        public void setTemplate(ClientDetailsEntity template) {
083                this.template = template;
084        }
085
086        public class DynamicClientRegistrationLoader extends CacheLoader<ServerConfiguration, ClientDetailsEntity> {
087                private HttpClient httpClient = new DefaultHttpClient();
088                private HttpComponentsClientHttpRequestFactory httpFactory = new HttpComponentsClientHttpRequestFactory(httpClient);
089                private JsonParser parser = new JsonParser();
090
091                @Override
092                public ClientDetailsEntity load(ServerConfiguration serverConfig) throws Exception {
093                        RestTemplate restTemplate = new RestTemplate(httpFactory);
094
095                        // dynamically register this client
096                        JsonObject jsonRequest = ClientDetailsEntityJsonProcessor.serialize(template, null, null);
097
098                        HttpHeaders headers = new HttpHeaders();
099                        headers.setContentType(MediaType.APPLICATION_JSON);
100                        headers.setAccept(Lists.newArrayList(MediaType.APPLICATION_JSON));
101
102                        HttpEntity<String> entity = new HttpEntity<String>(jsonRequest.toString(), headers);
103
104                        String registered = restTemplate.postForObject(serverConfig.getRegistrationEndpointUri(), entity, String.class);
105                        // TODO: handle HTTP errors
106
107                        // TODO: save registration token and other important bits
108                        ClientDetailsEntity client = ClientDetailsEntityJsonProcessor.parse(registered);
109
110                        return client;
111                }
112
113        }
114
115}
