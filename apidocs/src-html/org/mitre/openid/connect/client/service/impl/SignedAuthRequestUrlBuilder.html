004package org.mitre.openid.connect.client.service.impl;
005
006import java.net.URISyntaxException;
007import java.security.NoSuchAlgorithmException;
008
009import javax.servlet.http.HttpServletRequest;
010import javax.servlet.http.HttpServletResponse;
011import javax.servlet.http.HttpSession;
012
013import org.apache.http.client.utils.URIBuilder;
014import org.mitre.jwt.signer.service.JwtSigningAndValidationService;
015import org.mitre.openid.connect.client.service.AuthRequestUrlBuilder;
016import org.mitre.openid.connect.config.ServerConfiguration;
017import org.springframework.security.authentication.AuthenticationServiceException;
018import org.springframework.security.oauth2.provider.ClientDetails;
019
020import com.google.common.base.Joiner;
021import com.nimbusds.jose.JWSHeader;
022import com.nimbusds.jwt.JWTClaimsSet;
023import com.nimbusds.jwt.SignedJWT;
024
025/**
026 * @author jricher
027 *
028 */
029public class SignedAuthRequestUrlBuilder implements AuthRequestUrlBuilder {
030
031        private JwtSigningAndValidationService signingAndValidationService;
032
033        /* (non-Javadoc)
034         * @see org.mitre.openid.connect.client.service.AuthRequestUrlBuilder#buildAuthRequestUrl(org.mitre.openid.connect.config.ServerConfiguration, org.springframework.security.oauth2.provider.ClientDetails, java.lang.String, java.lang.String, java.lang.String)
035         */
036        @Override
037        public String buildAuthRequestUrl(ServerConfiguration serverConfig, ClientDetails clientConfig, String redirectUri, String nonce, String state) {
038
039                // create our signed JWT for the request object
040                JWTClaimsSet claims = new JWTClaimsSet();
041                
042                //set parameters to JwtClaims
043                claims.setCustomClaim("response_type", "code");
044                claims.setCustomClaim("client_id", clientConfig.getClientId());
045                claims.setCustomClaim("scope", Joiner.on(" ").join(clientConfig.getScope()));
046                
047                // build our redirect URI
048                claims.setCustomClaim("redirect_uri", redirectUri);
049                
050                // this comes back in the id token
051                claims.setCustomClaim("nonce", nonce);
052                
053                // this comes back in the auth request return
054                claims.setCustomClaim("state", state);
055                
056                SignedJWT jwt = new SignedJWT(new JWSHeader(signingAndValidationService.getDefaultSigningAlgorithm()), claims);
057                
058                signingAndValidationService.signJwt(jwt);
059                
060                try {
061                URIBuilder uriBuilder = new URIBuilder(serverConfig.getAuthorizationEndpointUri());
062                uriBuilder.addParameter("request", jwt.serialize());
063                
064                // build out the URI
065                return uriBuilder.build().toString();
066        } catch (URISyntaxException e) {
067                throw new AuthenticationServiceException("Malformed Authorization Endpoint Uri", e);
068        }
069        }
070
071        /**
072         * @return the signingAndValidationService
073         */
074        public JwtSigningAndValidationService getSigningAndValidationService() {
075                return signingAndValidationService;
076        }
077
078        /**
079         * @param signingAndValidationService the signingAndValidationService to set
080         */
081        public void setSigningAndValidationService(JwtSigningAndValidationService signingAndValidationService) {
082                this.signingAndValidationService = signingAndValidationService;
083        }
084
085}
