053/**
054 * Use Webfinger to discover the appropriate issuer for a user-given input string.
055 * @author jricher
056 *
057 */
058public class WebfingerIssuerService implements IssuerService {
059
060        private static Logger logger = LoggerFactory.getLogger(WebfingerIssuerService.class);
061
062        // pattern used to parse user input; we can't use the built-in java URI parser
063        private static final Pattern pattern = Pattern.compile("^((https|acct|http|mailto):(//)?)?((([^@]+)@)?(([^:]+)(:(\\d*))?))([^\\?]+)?(\\?([^#]+))?(#(.*))?$");
064
065        // map of user input -> issuer, loaded dynamically from webfinger discover
066        private LoadingCache<UriComponents, String> issuers;
067
068        private Set<String> whitelist = new HashSet<String>();
069        private Set<String> blacklist = new HashSet<String>();
070
071        /**
072         * Name of the incoming parameter to check for discovery purposes.
073         */
074        private String parameterName = "identifier";
075
076        /**
077         * URL of the page to forward to if no identifier is given.
078         */
079        private String loginPageUrl;
080
081        public WebfingerIssuerService() {
082                issuers = CacheBuilder.newBuilder().build(new WebfingerIssuerFetcher());
083        }
084
085        /* (non-Javadoc)
086         * @see org.mitre.openid.connect.client.service.IssuerService#getIssuer(javax.servlet.http.HttpServletRequest)
087         */
088        @Override
089        public IssuerServiceResponse getIssuer(HttpServletRequest request) {
090
091                String identifier = request.getParameter(parameterName);
092                if (!Strings.isNullOrEmpty(identifier)) {
093                        try {
094                                String issuer = issuers.get(normalizeResource(identifier));
095                                if (!whitelist.isEmpty() && !whitelist.contains(issuer)) {
096                                        throw new AuthenticationServiceException("Whitelist was nonempty, issuer was not in whitelist: " + issuer);
097                                }
098                                
099                                if (blacklist.contains(issuer)) {
100                                        throw new AuthenticationServiceException("Issuer was in blacklist: " + issuer);
101                                }
102                                
103                                return new IssuerServiceResponse(issuer, null, null);
104                        } catch (ExecutionException e) {
105                                logger.warn("Issue fetching issuer for user input: " + identifier, e);
106                                return null;
107                        }
108
109                } else {
110                        logger.warn("No user input given, directing to login page: " + loginPageUrl);
111                        return new IssuerServiceResponse(loginPageUrl);
112                }
113        }
114
115        /**
116         * Normalize the resource string as per OIDC Discovery.
117         * @param identifier
118         * @return the normalized string, or null if the string can't be normalized
119         */
120        private UriComponents normalizeResource(String identifier) {
121                // try to parse the URI
122                // NOTE: we can't use the Java built-in URI class because it doesn't split the parts appropriately
123
124                if (Strings.isNullOrEmpty(identifier)) {
125                        logger.warn("Can't normalize null or empty URI: " + identifier);
126                        return null; // nothing we can do
127                } else {
128
129                        //UriComponentsBuilder builder = UriComponentsBuilder.fromUriString(identifier);
130                        UriComponentsBuilder builder = UriComponentsBuilder.newInstance();
131
132                        //Pattern regex = Pattern.compile("^(([^:/?#]+):)?(//(([^@/]*)@)?([^/?#:]*)(:(\\d*))?)?([^?#]*)(\\?([^#]*))?(#(.*))?");
133                        Matcher m = pattern.matcher(identifier);
134                        if (m.matches()) {
135                                builder.scheme(m.group(2));
136                                builder.userInfo(m.group(6));
137                                builder.host(m.group(8));
138                                String port = m.group(10);
139                                if (!Strings.isNullOrEmpty(port)) {
140                                        builder.port(Integer.parseInt(port));
141                                }
142                                builder.path(m.group(11));
143                                builder.query(m.group(13));
144                                builder.fragment(m.group(15)); // we throw away the hash, but this is the group it would be if we kept it
145                        } else {
146                                // doesn't match the pattern, throw it out
147                                logger.warn("Parser couldn't match input: " + identifier);
148                                return null;
149                        }
150                        
151                        UriComponents n = builder.build();
152                        
153                        if (Strings.isNullOrEmpty(n.getScheme())) {
154                                if (!Strings.isNullOrEmpty(n.getUserInfo())
155                                                && Strings.isNullOrEmpty(n.getPath())
156                                                && Strings.isNullOrEmpty(n.getQuery())
157                                                && n.getPort() < 0) {
158                                        
159                                        // scheme empty, userinfo is not empty, path/query/port are empty
160                                        // set to "acct" (rule 2)
161                                        builder.scheme("acct");
162                                        
163                                } else {
164                                        // scheme is empty, but rule 2 doesn't apply
165                                        // set scheme to "https" (rule 3)
166                                        builder.scheme("https");
167                                }
168                        }
169                        
170                        // fragment must be stripped (rule 4)
171                        builder.fragment(null);
172                        
173                        return builder.build();
174                }
175
176
177        }

237        /**
238         * @author jricher
239         *
240         */
241        private class WebfingerIssuerFetcher extends CacheLoader<UriComponents, String> {
242                private HttpClient httpClient = new DefaultHttpClient();
243                private HttpComponentsClientHttpRequestFactory httpFactory = new HttpComponentsClientHttpRequestFactory(httpClient);
244                private JsonParser parser = new JsonParser();
245
246                @Override
247                public String load(UriComponents key) throws Exception {
248
249                        RestTemplate restTemplate = new RestTemplate(httpFactory);
250                        // construct the URL to go to
251                        
252                        // preserving http scheme is strictly for demo system use only.
253                        String scheme = key.getScheme();
254                        if (!Strings.isNullOrEmpty(scheme) && scheme.equals("http")) {
255                                scheme = "http://"; // add on colon and slashes.
256                                logger.warn("Webfinger endpoint MUST use the https URI scheme.");
257                        } else {
258                                scheme = "https://";
259                        }
260
261                        // do a webfinger lookup
262                        URIBuilder builder = new URIBuilder(scheme 
263                                                                                                + key.getHost()
264                                                                                                + (key.getPort() >= 0 ? ":" + key.getPort() : "")
265                                                                                                + Strings.nullToEmpty(key.getPath())
266                                                                                                + "/.well-known/webfinger" 
267                                                                                                + (Strings.isNullOrEmpty(key.getQuery()) ? "" : "?" + key.getQuery())
268                                                                                                );
269                        builder.addParameter("resource", key.toString());
270                        builder.addParameter("rel", "http://openid.net/specs/connect/1.0/issuer");
271
272                        // do the fetch
273                        logger.info("Loading: " + builder.toString());
274                        String webfingerResponse = restTemplate.getForObject(builder.build(), String.class);
275
276                        // TODO: catch and handle HTTP errors
277
278                        JsonElement json = parser.parse(webfingerResponse);
279
280                        // TODO: catch and handle JSON errors
281
282                        if (json != null && json.isJsonObject()) {
283                                // find the issuer
284                                JsonArray links = json.getAsJsonObject().get("links").getAsJsonArray();
285                                for (JsonElement link : links) {
286                                        if (link.isJsonObject()) {
287                                                JsonObject linkObj = link.getAsJsonObject();
288                                                if (linkObj.has("href")
289                                                                && linkObj.has("rel")
290                                                                && linkObj.get("rel").getAsString().equals("http://openid.net/specs/connect/1.0/issuer")) {
291
292                                                        // we found the issuer, return it
293                                                        return linkObj.get("href").getAsString();
294                                                }
295                                        }
296                                }
297                        }
298
299                        // we couldn't find it
300                        
301                        if (key.getScheme().equals("http") || key.getScheme().equals("https")) {
302                                // if it looks like HTTP then punt and return the input
303                                logger.warn("Returning normalized input string as issuer, hoping for the best: " + key.toString());
304                                return key.toString();
305                        } else {
306                                // if it's not HTTP, give up
307                                logger.warn("Couldn't find issuer: " + key.toString());
308                                return null;
309                        }
310                        
311                }
312
313        }
