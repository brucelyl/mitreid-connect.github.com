053/**
054 * Use Webfinger to discover the appropriate issuer for a user-given input string.
055 * @author jricher
056 *
057 */
058public class WebfingerIssuerService implements IssuerService {
059
060        /**
061         * Logger for this class
062         */
063        private static final Logger logger = LoggerFactory.getLogger(WebfingerIssuerService.class);
064
065        // map of user input -> issuer, loaded dynamically from webfinger discover
066        private LoadingCache<String, LoadingResult> issuers;
067
068        // private data shuttle class to get back two bits of info from the cache loader
069        private class LoadingResult {
070                public String loginHint;
071                public String issuer;
072                public LoadingResult(String loginHint, String issuer) {
073                        this.loginHint = loginHint;
074                        this.issuer = issuer;
075                }
076        }
077        
078        private Set<String> whitelist = new HashSet<>();
079        private Set<String> blacklist = new HashSet<>();
080
081        /**
082         * Name of the incoming parameter to check for discovery purposes.
083         */
084        private String parameterName = "identifier";
085
086        /**
087         * URL of the page to forward to if no identifier is given.
088         */
089        private String loginPageUrl;
090
091        /**
092         * Strict enfocement of "https"
093         */
094        private boolean forceHttps = true;
095
096        public WebfingerIssuerService() {
097                issuers = CacheBuilder.newBuilder().build(new WebfingerIssuerFetcher());
098        }
099
100        /* (non-Javadoc)
101         * @see org.mitre.openid.connect.client.service.IssuerService#getIssuer(javax.servlet.http.HttpServletRequest)
102         */
103        @Override
104        public IssuerServiceResponse getIssuer(HttpServletRequest request) {
105
106                String identifier = request.getParameter(parameterName);
107                if (!Strings.isNullOrEmpty(identifier)) {
108                        try {
109                                LoadingResult lr = issuers.get(identifier);
110                                if (!whitelist.isEmpty() && !whitelist.contains(lr.issuer)) {
111                                        throw new AuthenticationServiceException("Whitelist was nonempty, issuer was not in whitelist: " + lr.issuer);
112                                }
113
114                                if (blacklist.contains(lr.issuer)) {
115                                        throw new AuthenticationServiceException("Issuer was in blacklist: " + lr.issuer);
116                                }
117
118                                return new IssuerServiceResponse(lr.issuer, lr.loginHint, null);
119                        } catch (UncheckedExecutionException | ExecutionException e) {
120                                logger.warn("Issue fetching issuer for user input: " + identifier + ": " + e.getMessage());
121                                return null;
122                        }
123
124                } else {
125                        logger.warn("No user input given, directing to login page: " + loginPageUrl);
126                        return new IssuerServiceResponse(loginPageUrl);
127                }
128        }

201        /**
202         * @author jricher
203         *
204         */
205        private class WebfingerIssuerFetcher extends CacheLoader<String, LoadingResult> {
206                private HttpClient httpClient = HttpClientBuilder.create()
207                                .useSystemProperties()
208                                .build();
209                private HttpComponentsClientHttpRequestFactory httpFactory = new HttpComponentsClientHttpRequestFactory(httpClient);
210                private JsonParser parser = new JsonParser();
211
212                @Override
213                public LoadingResult load(String identifier) throws Exception {
214
215                        UriComponents key = WebfingerURLNormalizer.normalizeResource(identifier);
216                        
217                        RestTemplate restTemplate = new RestTemplate(httpFactory);
218                        // construct the URL to go to
219
220                        String scheme = key.getScheme();
221
222                        // preserving http scheme is strictly for demo system use only.
223                        if (!Strings.isNullOrEmpty(scheme) &&scheme.equals("http")) {
224                                if (forceHttps) {
225                                        throw new IllegalArgumentException("Scheme must not be 'http'");
226                                } else {
227                                        logger.warn("Webfinger endpoint MUST use the https URI scheme, overriding by configuration");
228                                        scheme = "http://"; // add on colon and slashes.
229                                }
230                        } else {
231                                // otherwise we don't know the scheme, assume HTTPS
232                                scheme = "https://";
233                        }
234
235                        // do a webfinger lookup
236                        URIBuilder builder = new URIBuilder(scheme
237                                        + key.getHost()
238                                        + (key.getPort() >= 0 ? ":" + key.getPort() : "")
239                                        + Strings.nullToEmpty(key.getPath())
240                                        + "/.well-known/webfinger"
241                                        + (Strings.isNullOrEmpty(key.getQuery()) ? "" : "?" + key.getQuery())
242                                        );
243                        builder.addParameter("resource", identifier);
244                        builder.addParameter("rel", "http://openid.net/specs/connect/1.0/issuer");
245
246                        try {
247
248                                // do the fetch
249                                logger.info("Loading: " + builder.toString());
250                                String webfingerResponse = restTemplate.getForObject(builder.build(), String.class);
251
252                                JsonElement json = parser.parse(webfingerResponse);
253
254                                if (json != null && json.isJsonObject()) {
255                                        // find the issuer
256                                        JsonArray links = json.getAsJsonObject().get("links").getAsJsonArray();
257                                        for (JsonElement link : links) {
258                                                if (link.isJsonObject()) {
259                                                        JsonObject linkObj = link.getAsJsonObject();
260                                                        if (linkObj.has("href")
261                                                                        && linkObj.has("rel")
262                                                                        && linkObj.get("rel").getAsString().equals("http://openid.net/specs/connect/1.0/issuer")) {
263
264                                                                // we found the issuer, return it
265                                                                String href = linkObj.get("href").getAsString();
266                                                                
267                                                                if (identifier.equals(href)
268                                                                                || identifier.startsWith("http")) {
269                                                                        // try to avoid sending a URL as the login hint
270                                                                        return new LoadingResult(null, href);
271                                                                } else {
272                                                                        // otherwise pass back whatever the user typed as a login hint
273                                                                        return new LoadingResult(identifier, href);
274                                                                }
275                                                        }
276                                                }
277                                        }
278                                }
279                        } catch (JsonParseException | RestClientException e) {
280                                logger.warn("Failure in fetching webfinger input", e.getMessage());
281                        }
282
283                        // we couldn't find it!
284
285                        if (key.getScheme().equals("http") || key.getScheme().equals("https")) {
286                                // if it looks like HTTP then punt: return the input, hope for the best
287                                logger.warn("Returning normalized input string as issuer, hoping for the best: " + identifier);
288                                return new LoadingResult(null, identifier);
289                        } else {
290                                // if it's not HTTP, give up
291                                logger.warn("Couldn't find issuer: " + identifier);
292                                throw new IllegalArgumentException();
293                        }
294
295                }
296
297        }
