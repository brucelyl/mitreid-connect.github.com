031/**
032 * Use Webfinger to discover the appropriate issuer for a user-given input string.
033 * @author jricher
034 *
035 */
036public class WebfingerIssuerService implements IssuerService {
037
038        private static Logger logger = LoggerFactory.getLogger(WebfingerIssuerService.class);
039        
040        // pattern used to parse user input; we can't use the built-in java URI parser
041        private static final Pattern pattern = Pattern.compile("(https://|acct:|http://|mailto:)?(([^@]+)@)?([^\\?]+)(\\?([^#]+))?(#(.*))?");
042
043        // map of user input -> issuer, loaded dynamically from webfinger discover
044        private LoadingCache<NormalizedURI, String> issuers;
045
046        /**
047         * Name of the incoming parameter to check for discovery purposes.
048         */
049        private String parameterName = "identifier";
050        
051        /**
052         * URL of the page to forward to if no identifier is given.
053         */
054        private String loginPageUrl;
055
056        public WebfingerIssuerService() {
057                issuers = CacheBuilder.newBuilder().build(new WebfingerIssuerFetcher());
058        }
059        
060        /* (non-Javadoc)
061         * @see org.mitre.openid.connect.client.service.IssuerService#getIssuer(javax.servlet.http.HttpServletRequest)
062         */
063        @Override
064        public IssuerServiceResponse getIssuer(HttpServletRequest request) {
065                
066                String identifier = request.getParameter(parameterName);
067                if (!Strings.isNullOrEmpty(identifier)) {
068            try {
069                String issuer = issuers.get(normalizeResource(identifier));
070                return new IssuerServiceResponse(issuer, null, null);
071            } catch (ExecutionException e) {
072                logger.warn("Issue fetching issuer for user input: " + identifier, e);
073                    return null;
074            }
075            
076                } else {
077                        logger.warn("No user input given, directing to login page: " + loginPageUrl);
078                        return new IssuerServiceResponse(loginPageUrl);
079                }
080        }
081        
082        /**
083         * Normalize the resource string as per OIDC Discovery.
084         * @param identifier
085         * @return the normalized string, or null if the string can't be normalized
086         */
087        private NormalizedURI normalizeResource(String identifier) {
088                // try to parse the URI         
089                // NOTE: we can't use the Java built-in URI class because it doesn't split the parts appropriately
090                
091                if (Strings.isNullOrEmpty(identifier)) {
092                        logger.warn("Can't normalize null or empty URI: " + identifier);
093                        return null; // nothing we can do
094                } else {
095                                
096                NormalizedURI n = new NormalizedURI();                  
097                Matcher m = pattern.matcher(identifier);
098                
099                        if (m.matches()) {
100                                n.scheme = m.group(1); // includes colon and maybe initial slashes
101                                n.user = m.group(2); // includes at sign
102                                n.hostportpath = m.group(4);
103                                n.query = m.group(5); // includes question mark
104                                n.hash = m.group(7); // includes hash mark
105                                
106                                // normalize scheme portion
107                                if (Strings.isNullOrEmpty(n.scheme)) {
108                                        if (!Strings.isNullOrEmpty(n.user)) {
109                                                // no scheme, but have a user, assume acct:
110                                                n.scheme = "acct:";
111                                        } else {
112                                                // no scheme, no user, assume https://
113                                                n.scheme = "https://";
114                                        }
115                                }
116                                
117                                n.source = Strings.nullToEmpty(n.scheme) +
118                                                Strings.nullToEmpty(n.user) +
119                                                Strings.nullToEmpty(n.hostportpath) + 
120                                                Strings.nullToEmpty(n.query); // note: leave fragment off
121                                
122                                return n;
123                        } else {
124                                logger.warn("Parser couldn't match input: " + identifier);
125                                return null;
126                        }
127                        
128                }
129                
130                
131        }

164        /**
165     * @author jricher
166     *
167     */
168    private class WebfingerIssuerFetcher extends CacheLoader<NormalizedURI, String> {
169        private HttpClient httpClient = new DefaultHttpClient();
170        private HttpComponentsClientHttpRequestFactory httpFactory = new HttpComponentsClientHttpRequestFactory(httpClient);
171                private JsonParser parser = new JsonParser();
172
173                @Override
174        public String load(NormalizedURI key) throws Exception {
175
176                RestTemplate restTemplate = new RestTemplate(httpFactory);
177                // construct the URL to go to
178                
179                //String url = "https://" + key.hostportpath + "/.well-known/webfinger?resource="
180                String scheme = key.scheme;
181                if (!Strings.isNullOrEmpty(scheme) && !scheme.startsWith("http")) {
182                        // do discovery on http or https URLs
183                        scheme = "https://";
184                }
185                URIBuilder builder = new URIBuilder(scheme + key.hostportpath + "/.well-known/webfinger" + Strings.nullToEmpty(key.query));
186                builder.addParameter("resource", key.source);
187                builder.addParameter("rel", "http://openid.net/specs/connect/1.0/issuer");
188                
189                // do the fetch
190                logger.info("Loading: " + builder.toString());
191                String webfingerResponse = restTemplate.getForObject(builder.build(), String.class);
192
193                // TODO: catch and handle HTTP errors
194                
195                JsonElement json = parser.parse(webfingerResponse);
196                
197                // TODO: catch and handle JSON errors
198                
199                if (json != null && json.isJsonObject()) {
200                        // find the issuer
201                        JsonArray links = json.getAsJsonObject().get("links").getAsJsonArray();
202                        for (JsonElement link : links) {
203                        if (link.isJsonObject()) {
204                                JsonObject linkObj = link.getAsJsonObject();
205                                if (linkObj.has("href") 
206                                                && linkObj.has("rel") 
207                                                && linkObj.get("rel").getAsString().equals("http://openid.net/specs/connect/1.0/issuer")) {
208                                        
209                                        // we found the issuer, return it
210                                        return linkObj.get("href").getAsString();
211                                }
212                        }
213                }
214                }
215                
216                // we couldn't find it
217                logger.warn("Couldn't find issuer");
218                return null;
219        }
220
221    }
