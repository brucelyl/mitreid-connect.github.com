107        /*
108         * This is the main entry point for the filter.
109         * 
110         * (non-Javadoc)
111         * 
112         * @see org.springframework.security.web.authentication.
113         * AbstractAuthenticationProcessingFilter
114         * #attemptAuthentication(javax.servlet.http.HttpServletRequest,
115         * javax.servlet.http.HttpServletResponse)
116         */
117        @Override
118        public Authentication attemptAuthentication(HttpServletRequest request, HttpServletResponse response) throws AuthenticationException, IOException, ServletException {
119
120                if (!Strings.isNullOrEmpty(request.getParameter("error"))) {
121
122                        // there's an error coming back from the server, need to handle this
123                        handleError(request, response);
124                        return null; // no auth, response is sent to display page or something
125
126                } else if (!Strings.isNullOrEmpty(request.getParameter("code"))) {
127
128                        // we got back the code, need to process this to get our tokens
129                        Authentication auth = handleAuthorizationCodeResponse(request, response);
130                        return auth;
131
132                } else {
133
134                        // not an error, not a code, must be an initial login of some type
135                        handleAuthorizationRequest(request, response);
136
137                        return null; // no auth, response redirected to the server's Auth Endpoint (or possibly to the account chooser)
138                }
139
140        }

142        /**
143         * Initiate an Authorization request
144         * 
145         * @param request
146         *            The request from which to extract parameters and perform the
147         *            authentication
148         * @param response
149         * @throws IOException
150         *             If an input or output exception occurs
151         */
152        protected void handleAuthorizationRequest(HttpServletRequest request, HttpServletResponse response) throws IOException {
153
154                HttpSession session = request.getSession();
155
156                IssuerServiceResponse issResp = issuerService.getIssuer(request);
157
158                if (issResp == null) {
159                        logger.error("Null issuer response returned from service.");
160                        throw new AuthenticationServiceException("No issuer found.");
161                }
162
163                if (issResp.shouldRedirect()) {
164                        response.sendRedirect(issResp.getRedirectUrl());
165                } else {
166                        String issuer = issResp.getIssuer();
167
168                        if (Strings.isNullOrEmpty(issuer)) {
169                                logger.error("No issuer found: " + issuer);
170                                throw new AuthenticationServiceException("No issuer found: " + issuer);
171                        }
172
173                        session.setAttribute(ISSUER_SESSION_VARIABLE, issuer);
174
175                        ServerConfiguration serverConfig = servers.getServerConfiguration(issuer);
176                        if (serverConfig == null) {
177                                logger.error("No server configuration found for issuer: " + issuer);
178                                throw new AuthenticationServiceException("No server configuration found for issuer: " + issuer);
179                        }
180
181
182                        RegisteredClient clientConfig = clients.getClientConfiguration(serverConfig);
183                        if (clientConfig == null) {
184                                logger.error("No client configuration found for issuer: " + issuer);
185                                throw new AuthenticationServiceException("No client configuration found for issuer: " + issuer);
186                        }
187
188                        String redirectUri = null; 
189                        if (clientConfig.getRegisteredRedirectUri() != null && clientConfig.getRegisteredRedirectUri().size() == 1) {
190                                // if there's a redirect uri configured (and only one), use that
191                                redirectUri = clientConfig.getRegisteredRedirectUri().toArray(new String[] {})[0];
192                        } else {
193                                // otherwise our redirect URI is this current URL, with no query parameters
194                                redirectUri = request.getRequestURL().toString();
195                        }
196                        session.setAttribute(REDIRECT_URI_SESION_VARIABLE, redirectUri);
197
198                        // this value comes back in the id token and is checked there
199                        String nonce = createNonce(session);
200
201                        // this value comes back in the auth code response
202                        String state = createState(session);
203
204                        String authRequest = authRequestBuilder.buildAuthRequestUrl(serverConfig, clientConfig, redirectUri, nonce, state);
205
206                        logger.debug("Auth Request:  " + authRequest);
207
208                        response.sendRedirect(authRequest);
209                }
210        }

212        /**
213         * @param request
214         *            The request from which to extract parameters and perform the
215         *            authentication
216         * @return The authenticated user token, or null if authentication is
217         *         incomplete.
218         */
219        protected Authentication handleAuthorizationCodeResponse(HttpServletRequest request, HttpServletResponse response) {
220
221                String authorizationCode = request.getParameter("code");
222
223                HttpSession session = request.getSession();
224
225                // check for state, if it doesn't match we bail early
226                String storedState = getStoredState(session);
227                if (!Strings.isNullOrEmpty(storedState)) {
228                        String state = request.getParameter("state");
229                        if (!storedState.equals(state)) {
230                                throw new AuthenticationServiceException("State parameter mismatch on return. Expected " + storedState + " got " + state);
231                        }
232                }
233
234                // look up the issuer that we set out to talk to
235                String issuer = getStoredSessionString(session, ISSUER_SESSION_VARIABLE);
236
237                // pull the configurations based on that issuer
238                ServerConfiguration serverConfig = servers.getServerConfiguration(issuer);
239                final RegisteredClient clientConfig = clients.getClientConfiguration(serverConfig);
240
241                MultiValueMap<String, String> form = new LinkedMultiValueMap<String, String>();
242                form.add("grant_type", "authorization_code");
243                form.add("code", authorizationCode);
244
245                String redirectUri = getStoredSessionString(session, REDIRECT_URI_SESION_VARIABLE);
246                if (redirectUri != null) {
247                        form.add("redirect_uri", redirectUri);
248                }
249
250                // Handle Token Endpoint interaction
251                DefaultHttpClient httpClient = new DefaultHttpClient();
252
253                httpClient.getParams().setParameter("http.socket.timeout", new Integer(httpSocketTimeout));
254
255                HttpComponentsClientHttpRequestFactory factory = new HttpComponentsClientHttpRequestFactory(httpClient);
256
257                RestTemplate restTemplate;
258
259                if (SECRET_BASIC.equals(clientConfig.getTokenEndpointAuthMethod())){
260                        // use BASIC auth if configured to do so
261                        restTemplate = new RestTemplate(factory) {
262
263                                @Override
264                                protected ClientHttpRequest createRequest(URI url, HttpMethod method) throws IOException {
265                                        ClientHttpRequest httpRequest = super.createRequest(url, method);
266                                        httpRequest.getHeaders().add("Authorization",
267                                                        String.format("Basic %s", Base64.encode(String.format("%s:%s", clientConfig.getClientId(), clientConfig.getClientSecret())) ));
268
269
270
271                                        return httpRequest;
272                                }
273                        };
274                } else {  //Alternatively use form based auth
275                        restTemplate = new RestTemplate(factory);
276
277                        form.add("client_id", clientConfig.getClientId());
278                        form.add("client_secret", clientConfig.getClientSecret());
279                }
280
281                logger.debug("tokenEndpointURI = " + serverConfig.getTokenEndpointUri());
282                logger.debug("form = " + form);
283
284                String jsonString = null;
285
286                try {
287                        jsonString = restTemplate.postForObject(serverConfig.getTokenEndpointUri(), form, String.class);
288                } catch (HttpClientErrorException httpClientErrorException) {
289
290                        // Handle error
291
292                        logger.error("Token Endpoint error response:  "
293                                        + httpClientErrorException.getStatusText() + " : "
294                                        + httpClientErrorException.getMessage());
295
296                        throw new AuthenticationServiceException("Unable to obtain Access Token: " + httpClientErrorException.getMessage());
297                }
298
299                logger.debug("from TokenEndpoint jsonString = " + jsonString);
300
301                JsonElement jsonRoot = new JsonParser().parse(jsonString);
302                if (!jsonRoot.isJsonObject()) {
303                        throw new AuthenticationServiceException("Token Endpoint did not return a JSON object: " + jsonRoot);
304                }
305
306                JsonObject tokenResponse = jsonRoot.getAsJsonObject();
307
308                if (tokenResponse.get("error") != null) {
309
310                        // Handle error
311
312                        String error = tokenResponse.get("error").getAsString();
313
314                        logger.error("Token Endpoint returned: " + error);
315
316                        throw new AuthenticationServiceException("Unable to obtain Access Token.  Token Endpoint returned: " + error);
317
318                } else {
319
320                        // Extract the id_token to insert into the
321                        // OIDCAuthenticationToken
322
323                        // get out all the token strings
324                        String accessTokenValue = null;
325                        String idTokenValue = null;
326                        String refreshTokenValue = null;
327
328                        if (tokenResponse.has("access_token")) {
329                                accessTokenValue = tokenResponse.get("access_token").getAsString();
330                        } else {
331                                throw new AuthenticationServiceException("Token Endpoint did not return an access_token: " + jsonString);
332                        }
333
334                        if (tokenResponse.has("id_token")) {
335                                idTokenValue = tokenResponse.get("id_token").getAsString();
336                        } else {
337                                logger.error("Token Endpoint did not return an id_token");
338                                throw new AuthenticationServiceException("Token Endpoint did not return an id_token");
339                        }
340
341                        if (tokenResponse.has("refresh_token")) {
342                                refreshTokenValue = tokenResponse.get("refresh_token").getAsString();
343                        }
344
345                        try {
346                                SignedJWT idToken = SignedJWT.parse(idTokenValue);
347
348                                // validate our ID Token over a number of tests
349                                ReadOnlyJWTClaimsSet idClaims = idToken.getJWTClaimsSet();
350
351                                // check the signature
352                                JwtSigningAndValidationService jwtValidator = validationServices.get(serverConfig.getJwksUri());
353                                if (jwtValidator != null) {
354                                        if(!jwtValidator.validateSignature(idToken)) {
355                                                throw new AuthenticationServiceException("Signature validation failed");
356                                        }
357                                } else {
358                                        logger.info("No validation service found. Skipping signature validation");
359                                }
360
361                                // check the issuer
362                                if (idClaims.getIssuer() == null) {
363                                        throw new AuthenticationServiceException("Id Token Issuer is null");
364                                } else if (!idClaims.getIssuer().equals(serverConfig.getIssuer())){
365                                        throw new AuthenticationServiceException("Issuers do not match, expected " + serverConfig.getIssuer() + " got " + idClaims.getIssuer());
366                                }
367
368                                // check expiration
369                                if (idClaims.getExpirationTime() == null) {
370                                        throw new AuthenticationServiceException("Id Token does not have required expiration claim");
371                                } else {
372                                        // it's not null, see if it's expired
373                                        Date now = new Date(System.currentTimeMillis() - (timeSkewAllowance * 1000));
374                                        if (now.after(idClaims.getExpirationTime())) {
375                                                throw new AuthenticationServiceException("Id Token is expired: " + idClaims.getExpirationTime());
376                                        }
377                                }
378
379                                // check not before
380                                if (idClaims.getNotBeforeTime() != null) {
381                                        Date now = new Date(System.currentTimeMillis() + (timeSkewAllowance * 1000));
382                                        if (now.before(idClaims.getNotBeforeTime())){
383                                                throw new AuthenticationServiceException("Id Token not valid untill: " + idClaims.getNotBeforeTime());
384                                        }
385                                }
386
387                                // check issued at
388                                if (idClaims.getIssueTime() == null) {
389                                        throw new AuthenticationServiceException("Id Token does not have required issued-at claim");
390                                } else {
391                                        // since it's not null, see if it was issued in the future
392                                        Date now = new Date(System.currentTimeMillis() + (timeSkewAllowance * 1000));
393                                        if (now.before(idClaims.getIssueTime())) {
394                                                throw new AuthenticationServiceException("Id Token was issued in the future: " + idClaims.getIssueTime());
395                                        }
396                                }
397
398                                // check audience
399                                if (idClaims.getAudience() == null) {
400                                        throw new AuthenticationServiceException("Id token audience is null");
401                                } else if (!idClaims.getAudience().contains(clientConfig.getClientId())) {
402                                        throw new AuthenticationServiceException("Audience does not match, expected " + clientConfig.getClientId() + " got " + idClaims.getAudience());
403                                }
404
405                                // compare the nonce to our stored claim
406                                // TODO: Nimbus claims as strings?
407                                String nonce = (String) idClaims.getCustomClaim("nonce");
408                                if (Strings.isNullOrEmpty(nonce)) {
409
410                                        logger.error("ID token did not contain a nonce claim.");
411
412                                        throw new AuthenticationServiceException("ID token did not contain a nonce claim.");
413                                }
414
415                                String storedNonce = getStoredNonce(session);
416                                if (!nonce.equals(storedNonce)) {
417                                        logger.error("Possible replay attack detected! The comparison of the nonce in the returned "
418                                                        + "ID Token to the session " + NONCE_SESSION_VARIABLE + " failed. Expected " + storedNonce + " got " + nonce + ".");
419
420                                        throw new AuthenticationServiceException(
421                                                        "Possible replay attack detected! The comparison of the nonce in the returned "
422                                                                        + "ID Token to the session " + NONCE_SESSION_VARIABLE + " failed. Expected " + storedNonce + " got " + nonce + ".");
423                                }
424
425                                // pull the subject (user id) out as a claim on the id_token
426
427                                String userId = idClaims.getSubject();
428
429                                // construct an OIDCAuthenticationToken and return a Authentication object w/the userId and the idToken
430
431                                OIDCAuthenticationToken token = new OIDCAuthenticationToken(userId, idClaims.getIssuer(), serverConfig, idTokenValue, accessTokenValue, refreshTokenValue);
432
433                                Authentication authentication = this.getAuthenticationManager().authenticate(token);
434
435                                return authentication;
436                        } catch (ParseException e) {
437                                throw new AuthenticationServiceException("Couldn't parse idToken: ", e);
438                        }
439
440
441
442                }
443        }

445        /**
446         * Handle Authorization Endpoint error
447         * 
448         * @param request
449         *            The request from which to extract parameters and handle the
450         *            error
451         * @param response
452         *            The response, needed to do a redirect to display the error
453         * @throws IOException
454         *             If an input or output exception occurs
455         */
456        protected void handleError(HttpServletRequest request, HttpServletResponse response) throws IOException {
457
458                String error = request.getParameter("error");
459                String errorDescription = request.getParameter("error_description");
460                String errorURI = request.getParameter("error_uri");
461
462                throw new AuthenticationServiceException("Error from Authorization Endpoint: " + error + " " + errorDescription + " " + errorURI);
463        }

465        /**
466         * Get the named stored session variable as a string. Return null if not found or not a string.
467         * @param session
468         * @param key
469         * @return
470         */
471        private static String getStoredSessionString(HttpSession session, String key) {
472                Object o = session.getAttribute(key);
473                if (o != null && o instanceof String) {
474                        return o.toString();
475                } else {
476                        return null;
477                }
478        }

480        /**
481         * Create a cryptographically random nonce and store it in the session
482         * @param session
483         * @return
484         */
485        protected static String createNonce(HttpSession session) {
486                String nonce = new BigInteger(50, new SecureRandom()).toString(16);
487                session.setAttribute(NONCE_SESSION_VARIABLE, nonce);
488
489                return nonce;
490        }
491
492        /**
493         * Get the nonce we stored in the session
494         * @param session
495         * @return
496         */
497        protected static String getStoredNonce(HttpSession session) {
498                return getStoredSessionString(session, NONCE_SESSION_VARIABLE);
499        }
500
501        /**
502         * Create a cryptographically random state and store it in the session
503         * @param session
504         * @return
505         */
506        protected static String createState(HttpSession session) {
507                String state = new BigInteger(50, new SecureRandom()).toString(16);
508                session.setAttribute(STATE_SESSION_VARIABLE, state);
509
510                return state;
511        }
512
513        /**
514         * Get the state we stored in the session
515         * @param session
516         * @return
517         */
518        protected static String getStoredState(HttpSession session) {
519                return getStoredSessionString(session, STATE_SESSION_VARIABLE);
520        }
