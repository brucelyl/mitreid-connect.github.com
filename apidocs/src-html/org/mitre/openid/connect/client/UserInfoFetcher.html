001package org.mitre.openid.connect.client;
002
003import org.apache.http.client.HttpClient;
004import org.apache.http.impl.client.DefaultHttpClient;
005import org.mitre.openid.connect.model.DefaultUserInfo;
006import org.mitre.openid.connect.model.UserInfo;
007import org.springframework.http.client.HttpComponentsClientHttpRequestFactory;
008import org.springframework.util.LinkedMultiValueMap;
009import org.springframework.util.MultiValueMap;
010import org.springframework.web.client.RestTemplate;
011
012import com.google.gson.JsonObject;
013import com.google.gson.JsonParser;
014
015public class UserInfoFetcher {
016
017        public UserInfo loadUserInfo(OIDCAuthenticationToken token) {
018                
019                HttpClient httpClient = new DefaultHttpClient();
020
021                HttpComponentsClientHttpRequestFactory factory = new HttpComponentsClientHttpRequestFactory(httpClient);
022
023                RestTemplate restTemplate = new RestTemplate(factory);
024
025                MultiValueMap<String, String> form = new LinkedMultiValueMap<String, String>();
026                form.add("access_token", token.getAccessTokenValue());
027                form.add("schema", "openid");
028                
029                String userInfoString = restTemplate.postForObject(token.getServerConfiguration().getUserInfoUri(), form, String.class);
030                
031                JsonObject userInfoJson = new JsonParser().parse(userInfoString).getAsJsonObject();
032                
033                UserInfo userInfo = DefaultUserInfo.fromJson(userInfoJson);
034                
035                return userInfo;
036                
037    }
038
039}
