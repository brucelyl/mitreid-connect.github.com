016package org.mitre.openid.connect.view;
017
018import java.io.IOException;
019import java.io.Writer;
020import java.util.Map;
021import java.util.Set;
022
023import javax.servlet.http.HttpServletRequest;
024import javax.servlet.http.HttpServletResponse;
025
026import org.mitre.openid.connect.model.UserInfo;
027import org.slf4j.Logger;
028import org.slf4j.LoggerFactory;
029import org.springframework.stereotype.Component;
030import org.springframework.validation.BeanPropertyBindingResult;
031import org.springframework.web.servlet.view.AbstractView;
032
033import com.google.gson.ExclusionStrategy;
034import com.google.gson.FieldAttributes;
035import com.google.gson.Gson;
036import com.google.gson.GsonBuilder;
037import com.google.gson.JsonArray;
038import com.google.gson.JsonObject;
039
040@Component("pocoUserInfoView")
041public class POCOUserInfoView extends AbstractView {
042        
043        private static Logger logger = LoggerFactory.getLogger(POCOUserInfoView.class);
044        
045        /* (non-Javadoc)
046         * @see org.springframework.web.servlet.view.AbstractView#renderMergedOutputModel(java.util.Map, javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse)
047         */
048        protected void renderMergedOutputModel(Map<String, Object> model, HttpServletRequest request, HttpServletResponse response) {
049                
050                UserInfo userInfo = (UserInfo) model.get("userInfo");
051
052                Set<String> scope = (Set<String>) model.get("scope");
053                
054                Gson gson = new GsonBuilder()
055                        .setExclusionStrategies(new ExclusionStrategy() {
056                                
057                                public boolean shouldSkipField(FieldAttributes f) {
058                                        
059                                        return false;
060                                }
061                                
062                                public boolean shouldSkipClass(Class<?> clazz) {
063                                        // skip the JPA binding wrapper
064                                        if (clazz.equals(BeanPropertyBindingResult.class)) {
065                                                return true;
066                                        }
067                                        return false;
068                                }
069                                                                
070                        }).create();
071
072                response.setContentType("application/json");
073                
074                Writer out;
075                
076                try {
077                        
078                        out = response.getWriter();
079                        gson.toJson(toPoco(userInfo, scope), out);
080                        
081                } catch (IOException e) {
082                        
083                        logger.error("IOException in POCOUserInfoView.java: ", e);
084                        
085                }
086                
087        }
088        
089        private JsonObject toPoco(UserInfo ui, Set<String> scope) {
090                JsonObject poco = new JsonObject();
091                
092                // Envelope Info
093                poco.addProperty("startIndex", 0);
094                poco.addProperty("itemsPerPage", 1);
095                poco.addProperty("totalResults", 1);
096                
097                // Build the entry for this userInfo, then add it to entries, then add it to poco
098                JsonObject entry = new JsonObject();
099                
100                if (scope.contains("openid")) {
101                        entry.addProperty("id", ui.getSub());
102                }
103                
104                if (scope.contains("profile")) {
105                        entry.addProperty("displayName", ui.getNickname());
106                        
107                        if (ui.getFamilyName() != null 
108                                        || ui.getGivenName() != null 
109                                        || ui.getMiddleName() != null
110                                        || ui.getName() != null) {
111                                JsonObject name = new JsonObject();
112                                name.addProperty("familyName", ui.getFamilyName());
113                                name.addProperty("givenName", ui.getGivenName());
114                                name.addProperty("middleName", ui.getMiddleName());
115                                name.addProperty("formatted", ui.getName());
116                                entry.add("name", name);
117                        }
118                
119                        entry.addProperty("gender", ui.getGender());
120                        entry.addProperty("preferredUsername", ui.getPreferredUsername());
121                        if (ui.getPicture() != null){
122                                JsonObject photo = new JsonObject();
123                                photo.addProperty("value", ui.getPicture());
124                                
125                                JsonArray photoArray = new JsonArray();
126                                photoArray.add(photo);
127                                entry.add("photos", photoArray);
128                        }
129                        
130                        if (ui.getWebsite() != null) {
131                                JsonObject website = new JsonObject();
132                                website.addProperty("value", ui.getWebsite());
133                                
134                                JsonArray websiteArray = new JsonArray();
135                                websiteArray.add(website);
136                                entry.add("urls", websiteArray);
137                        }
138
139                        entry.addProperty("updated", ui.getUpdatedTime());
140                        
141                }
142                
143                if (scope.contains("email")) {
144                        if (ui.getEmail() != null) {
145                                JsonObject email = new JsonObject();
146                                email.addProperty("value", ui.getEmail());
147                                
148                                JsonArray emailArray = new JsonArray();
149                                emailArray.add(email);
150                                entry.add("emails", emailArray);
151                        }
152                }
153                
154                if (scope.contains("phone")) {
155                        if (ui.getPhoneNumber() != null){
156                                JsonObject phone = new JsonObject();
157                                phone.addProperty("value", ui.getPhoneNumber());
158                                
159                                JsonArray phoneArray = new JsonArray();
160                                phoneArray.add(phone);
161                                entry.add("phoneNumbers", phoneArray);
162                        }
163                
164                }
165                
166                if (scope.contains("address")) {
167                        if (ui.getAddress() != null) {
168                                JsonObject addr = new JsonObject();
169                                addr.addProperty("formatted", ui.getAddress().getFormatted());
170                                addr.addProperty("streetAddress", ui.getAddress().getStreetAddress());
171                                addr.addProperty("locality", ui.getAddress().getLocality());
172                                addr.addProperty("region", ui.getAddress().getRegion());
173                                addr.addProperty("postalCode", ui.getAddress().getPostalCode());
174                                addr.addProperty("country", ui.getAddress().getCountry());
175                                
176                                JsonArray addrArray = new JsonArray();
177                                addrArray.add(addr);
178                                entry.add("addresses", addrArray);
179                        }
180                }
181                
182                JsonArray entryArray = new JsonArray();
183                entryArray.add(entry);
184                poco.add("entry", entryArray);
185                return poco;
186        }
187
188}
