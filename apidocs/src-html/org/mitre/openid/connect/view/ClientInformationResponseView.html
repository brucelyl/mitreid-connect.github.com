004package org.mitre.openid.connect.view;
005
006import java.io.IOException;
007import java.io.Writer;
008import java.util.Map;
009import java.util.Set;
010
011import javax.servlet.http.HttpServletRequest;
012import javax.servlet.http.HttpServletResponse;
013
014import org.mitre.oauth2.model.ClientDetailsEntity;
015import org.mitre.oauth2.model.OAuth2AccessTokenEntity;
016import org.springframework.http.HttpStatus;
017import org.springframework.stereotype.Component;
018import org.springframework.web.servlet.view.AbstractView;
019
020import com.google.common.base.Joiner;
021import com.google.gson.Gson;
022import com.google.gson.JsonElement;
023import com.google.gson.JsonIOException;
024import com.google.gson.JsonObject;
025import com.google.gson.reflect.TypeToken;
026
027/**
028 * 
029 * Provides a minimal representation of a client's registration information, to be shown from the dynamic registration endpoint
030 * on the client_register and rotate_secret operations.
031 * 
032 * @author jricher
033 *
034 */
035@Component("clientInformationResponseView")
036public class ClientInformationResponseView extends AbstractView {
037
038        // note that this won't serialize nulls by default
039        private Gson gson = new Gson();
040        
041        /* (non-Javadoc)
042         * @see org.springframework.web.servlet.view.AbstractView#renderMergedOutputModel(java.util.Map, javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse)
043         */
044        @Override
045        protected void renderMergedOutputModel(Map<String, Object> model, HttpServletRequest request, HttpServletResponse response) {
046
047                response.setContentType("application/json");
048                
049                ClientDetailsEntity c = (ClientDetailsEntity) model.get("client");
050                OAuth2AccessTokenEntity token = (OAuth2AccessTokenEntity) model.get("token");
051                HttpStatus code = (HttpStatus) model.get("code");
052                if (code == null) {
053                        code = HttpStatus.OK;
054                }
055                
056                JsonObject o = new JsonObject();
057                
058                o.addProperty("client_id", c.getClientId());
059                if (c.getClientSecret() != null) {
060                        o.addProperty("client_secret", c.getClientSecret());
061                        o.addProperty("expires_at", 0); // TODO: do we want to let secrets expire?
062                }
063                
064                if (c.getCreatedAt() != null) {
065                        o.addProperty("issued_at", c.getCreatedAt().getTime());
066                }
067
068                o.addProperty("registration_access_token", token.getValue());
069                
070                // TODO: urlencode the client id for safety?
071                String uri = request.getRequestURL() + "/" + c.getClientId();           
072                o.addProperty("registration_client_uri", uri);
073                
074                
075                // add in all other client properties
076                
077                // OAuth DynReg
078                o.add("redirect_uris", getAsArray(c.getRedirectUris()));
079                o.addProperty("client_name", c.getClientName());
080                o.addProperty("client_uri", c.getClientUri());
081                o.addProperty("logo_uri", c.getLogoUri());
082                o.add("contacts", getAsArray(c.getContacts()));
083                o.addProperty("tos_uri", c.getTosUri());
084                o.addProperty("token_endpoint_auth_method", c.getTokenEndpointAuthMethod() != null ? c.getTokenEndpointAuthMethod().getValue() : null);
085                o.addProperty("scope", c.getScope() != null ? Joiner.on(" ").join(c.getScope()) : null);
086                o.add("grant_types", getAsArray(c.getGrantTypes()));
087                o.addProperty("policy_uri", c.getPolicyUri());
088                o.addProperty("jwks_uri", c.getJwksUri());
089                
090                // OIDC Registration
091                o.addProperty("application_type", c.getApplicationType() != null ? c.getApplicationType().getValue() : null);
092                o.addProperty("sector_identifier_uri", c.getSectorIdentifierUri());
093                o.addProperty("subject_type", c.getSubjectType() != null ? c.getSubjectType().getValue() : null);
094                o.addProperty("request_object_signing_alg", c.getRequestObjectSigningAlg() != null ? c.getRequestObjectSigningAlg().getAlgorithmName() : null);
095                o.addProperty("userinfo_signed_response_alg", c.getUserInfoSignedResponseAlg() != null ? c.getUserInfoSignedResponseAlg().getAlgorithmName() : null);
096                o.addProperty("userinfo_encrypted_response_alg", c.getUserInfoEncryptedResponseAlg() != null ? c.getUserInfoEncryptedResponseAlg().getAlgorithmName() : null);
097                o.addProperty("userinfo_encrypted_response_enc", c.getUserInfoEncryptedResponseEnc() != null ? c.getUserInfoEncryptedResponseEnc().getAlgorithmName() : null);
098                o.addProperty("id_token_signed_response_alg", c.getIdTokenSignedResponseAlg() != null ? c.getIdTokenSignedResponseAlg().getAlgorithmName() : null);
099                o.addProperty("id_token_encrypted_response_alg", c.getIdTokenEncryptedResponseAlg() != null ? c.getIdTokenEncryptedResponseAlg().getAlgorithmName() : null);
100                o.addProperty("id_token_encrypted_response_enc", c.getIdTokenEncryptedResponseEnc() != null ? c.getIdTokenEncryptedResponseEnc().getAlgorithmName() : null);
101                o.addProperty("default_max_age", c.getDefaultMaxAge());
102                o.addProperty("require_auth_time", c.getRequireAuthTime());
103                o.add("default_acr_values", getAsArray(c.getDefaultACRvalues()));
104                o.addProperty("initiate_login_uri", c.getInitiateLoginUri());
105                o.addProperty("post_logout_redirect_uri", c.getPostLogoutRedirectUri());
106                o.add("request_uris", getAsArray(c.getRequestUris()));
107                
108                try {
109                Writer out = response.getWriter();
110                gson.toJson(o, out);
111        } catch (JsonIOException e) {
112                // TODO Auto-generated catch block
113                e.printStackTrace();
114        } catch (IOException e) {
115                // TODO Auto-generated catch block
116                e.printStackTrace();
117        }               
118                        
119        }
120        
121        private JsonElement getAsArray(Set<String> value) {
122                return gson.toJsonTree(value, new TypeToken<Set<String>>(){}.getType());
123        }
124
125}
