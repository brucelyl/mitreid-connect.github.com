017package org.mitre.openid.connect.web;

019import java.net.URI;
020import java.net.URISyntaxException;
021import java.text.ParseException;
022
023import javax.servlet.http.HttpServletRequest;
024
025import org.apache.http.client.utils.URIBuilder;
026import org.slf4j.Logger;
027import org.slf4j.LoggerFactory;
028import org.springframework.http.HttpStatus;
029import org.springframework.stereotype.Controller;
030import org.springframework.web.bind.annotation.RequestMapping;
031import org.springframework.web.bind.annotation.RequestParam;
032import org.springframework.web.servlet.ModelAndView;
033
034import com.google.common.base.Strings;
035import com.nimbusds.jwt.JWT;
036import com.nimbusds.jwt.JWTParser;
037
038
039/**
040 * This @Controller is a hack to get around SECOAUTH's AuthorizationEndpoint requiring the response_type parameter to be passed in.
041 * 
042 * @author jricher
043 *
044 */
045@Controller("requestObjectAuthorzationEndpoint")
046//@Component
047public class RequestObjectAuthorizationEndpoint {
048
049        private static Logger logger = LoggerFactory.getLogger(RequestObjectAuthorizationEndpoint.class);
050
051        @RequestMapping(value = "/authorize", params = "request")
052        public String authorizeRequestObject(@RequestParam("request") String jwtString, @RequestParam(value = "response_type", required = false) String responseType, HttpServletRequest request, ModelAndView mav) {
053
054                String query = request.getQueryString();
055
056                if (responseType == null) {
057                        try {
058                                JWT requestObject = JWTParser.parse(jwtString);
059                                responseType = (String)requestObject.getJWTClaimsSet().getClaim("response_type");
060
061                                URI uri = new URIBuilder(Strings.nullToEmpty(request.getServletPath()) + Strings.nullToEmpty(request.getPathInfo()) + "?" + query)
062                                .addParameter("response_type", responseType)
063                                .build();
064
065                                query = uri.getRawQuery();//uri.toString();
066
067                        } catch (ParseException e) {
068                                logger.error("ParseException while attempting to authorize request object: " + e.getStackTrace().toString());
069                                mav.addObject("code", HttpStatus.BAD_REQUEST);
070                                return "httpCodeView";
071
072                        } catch (URISyntaxException e) {
073                                logger.error("URISyntaxError while attempting to authorize request object: " + e.getStackTrace().toString());
074                                mav.addObject("code", HttpStatus.BAD_REQUEST);
075                                return "httpCodeView";
076                        }
077                }
078
079                return "forward:/oauth/authorize?" + query;
080
081        }
082
083}
