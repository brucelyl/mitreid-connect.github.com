052        /**
053         * Create a new Client, issue a client ID, and create a registration access token.
054         * @param jsonString
055         * @param m
056         * @param p
057         * @return
058         */
059        @RequestMapping(method = RequestMethod.POST, consumes = "application/json", produces = "application/json")
060        public String registerNewClient(@RequestBody String jsonString, Model m) {
061                
062                ClientDetailsEntity newClient = ClientDetailsEntityJsonProcessor.parse(jsonString);
063                
064                if (newClient != null) {
065                        // it parsed!
066                        
067                        //
068                        // Now do some post-processing consistency checks on it
069                        //
070                        
071                        // clear out any spurious id/secret (clients don't get to pick)
072                        newClient.setClientId(null);
073                        newClient.setClientSecret(null);
074                        
075                        // set of scopes that are OK for clients to dynamically register for
076                        Set<SystemScope> dynScopes = scopeService.getDynReg();
077
078                        // scopes that the client is asking for
079                        Set<SystemScope> requestedScopes = scopeService.fromStrings(newClient.getScope());
080
081                        // if the client didn't ask for any, give them the defaults
082                        if (requestedScopes == null || requestedScopes.isEmpty()) {
083                                requestedScopes = scopeService.getDefaults();
084                        }
085
086                        // the scopes that the client can have must be a subset of the dynamically allowed scopes
087                        Set<SystemScope> allowedScopes = Sets.intersection(dynScopes, requestedScopes);
088
089                        newClient.setScope(scopeService.toStrings(allowedScopes));
090                        
091
092                        // set default grant types if needed
093                        if (newClient.getGrantTypes() == null || newClient.getGrantTypes().isEmpty()) { 
094                                if (newClient.getScope().contains("offline_access")) { // client asked for offline access
095                                        newClient.setGrantTypes(Sets.newHashSet("authorization_code", "refresh_token")); // allow authorization code and refresh token grant types by default
096                                } else {
097                                        newClient.setGrantTypes(Sets.newHashSet("authorization_code")); // allow authorization code grant type by default
098                                }
099                        }
100                        
101                        // set default response types if needed
102                        // TODO: these aren't checked by SECOAUTH
103                        // TODO: the consistency between the response_type and grant_type needs to be checked by the client service, most likely
104                        if (newClient.getResponseTypes() == null || newClient.getResponseTypes().isEmpty()) {
105                                newClient.setResponseTypes(Sets.newHashSet("code")); // default to allowing only the auth code flow
106                        }
107                        
108                        if (newClient.getTokenEndpointAuthMethod() == null) {
109                                newClient.setTokenEndpointAuthMethod(AuthMethod.SECRET_BASIC);
110                        }
111                        
112                        if (newClient.getTokenEndpointAuthMethod() == AuthMethod.SECRET_BASIC ||
113                                        newClient.getTokenEndpointAuthMethod() == AuthMethod.SECRET_JWT ||
114                                        newClient.getTokenEndpointAuthMethod() == AuthMethod.SECRET_POST) {
115                                
116                                // we need to generate a secret
117                                newClient = clientService.generateClientSecret(newClient);
118                        }
119                        
120                        // set some defaults for token timeouts
121                        newClient.setAccessTokenValiditySeconds((int)TimeUnit.HOURS.toSeconds(1)); // access tokens good for 1hr
122                        newClient.setIdTokenValiditySeconds((int)TimeUnit.MINUTES.toSeconds(10)); // id tokens good for 10min
123                        newClient.setRefreshTokenValiditySeconds(null); // refresh tokens good until revoked
124
125                        // this client has been dynamically registered (obviously)
126                        newClient.setDynamicallyRegistered(true);
127                        
128                        // now save it
129                        ClientDetailsEntity savedClient = clientService.saveNewClient(newClient);
130                        
131                        // generate the registration access token
132                        OAuth2AccessTokenEntity token = createRegistrationAccessToken(savedClient);
133                        
134                        // send it all out to the view
135                        m.addAttribute("client", savedClient);
136                        m.addAttribute("code", HttpStatus.CREATED); // http 201
137                        m.addAttribute("token", token);
138                        
139                        return "clientInformationResponseView";
140                } else {
141                        // didn't parse, this is a bad request
142                        logger.error("registerNewClient failed; submitted JSON is malformed");
143                        m.addAttribute("code", HttpStatus.BAD_REQUEST); // http 400
144                        
145                        return "httpCodeView";
146                }
147                
148        }

150        /**
151         * Get the meta information for a client.
152         * @param clientId
153         * @param m
154         * @param auth
155         * @return
156         */
157        @PreAuthorize("hasRole('ROLE_CLIENT') and #oauth2.hasScope('" + OAuth2AccessTokenEntity.REGISTRATION_TOKEN_SCOPE + "')")
158        @RequestMapping(value = "/{id}", method = RequestMethod.GET, produces = "application/json")
159        public String readClientConfiguration(@PathVariable("id") String clientId, Model m, OAuth2Authentication auth) {
160                
161                ClientDetailsEntity client = clientService.loadClientByClientId(clientId);
162                
163                if (client != null && client.getClientId().equals(auth.getAuthorizationRequest().getClientId())) {
164
165                        
166                        // we return the token that we got in
167                        OAuth2AuthenticationDetails details = (OAuth2AuthenticationDetails) auth.getDetails();
168                        OAuth2AccessTokenEntity token = tokenService.readAccessToken(details.getTokenValue());
169                        
170                        // send it all out to the view
171                        m.addAttribute("client", client);
172                        m.addAttribute("code", HttpStatus.OK); // http 200
173                        m.addAttribute("token", token);
174                        
175                        return "clientInformationResponseView";
176                } else {
177                        // client mismatch
178                        logger.error("readClientConfiguration failed, client ID mismatch: " 
179                                        + clientId + " and " + auth.getAuthorizationRequest().getClientId() + " do not match.");
180                        m.addAttribute("code", HttpStatus.FORBIDDEN); // http 403
181                        
182                        return "httpCodeView";
183                }
184        }

186        /**
187         * Update the metainformation for a given client.
188         * @param clientId
189         * @param jsonString
190         * @param m
191         * @param auth
192         * @return
193         */
194        @PreAuthorize("hasRole('ROLE_CLIENT') and #oauth2.hasScope('" + OAuth2AccessTokenEntity.REGISTRATION_TOKEN_SCOPE + "')")
195        @RequestMapping(value = "/{id}", method = RequestMethod.PUT, produces = "application/json", consumes = "application/json")
196        public String updateClient(@PathVariable("id") String clientId, @RequestBody String jsonString, Model m, OAuth2Authentication auth) {
197
198                
199                ClientDetailsEntity newClient = ClientDetailsEntityJsonProcessor.parse(jsonString);
200                ClientDetailsEntity oldClient = clientService.loadClientByClientId(clientId);
201                
202                if (newClient != null && oldClient != null  // we have an existing client and the new one parsed
203                                && oldClient.getClientId().equals(auth.getAuthorizationRequest().getClientId()) // the client passed in the URI matches the one in the auth
204                                && oldClient.getClientId().equals(newClient.getClientId()) // the client passed in the body matches the one in the URI
205                                ) {
206
207                        // a client can't ask to update its own client secret to any particular value
208                        newClient.setClientSecret(oldClient.getClientSecret());
209                        
210                        // we need to copy over all of the local and SECOAUTH fields
211                        newClient.setAccessTokenValiditySeconds(oldClient.getAccessTokenValiditySeconds());
212                        newClient.setIdTokenValiditySeconds(oldClient.getIdTokenValiditySeconds());
213                        newClient.setRefreshTokenValiditySeconds(oldClient.getRefreshTokenValiditySeconds());
214                        newClient.setDynamicallyRegistered(true); // it's still dynamically registered
215                        newClient.setAllowIntrospection(oldClient.isAllowIntrospection());
216                        newClient.setAuthorities(oldClient.getAuthorities());
217                        newClient.setClientDescription(oldClient.getClientDescription());
218                        newClient.setCreatedAt(oldClient.getCreatedAt());
219                        newClient.setReuseRefreshToken(oldClient.isReuseRefreshToken());
220                        
221                        // set of scopes that are OK for clients to dynamically register for
222                        Set<SystemScope> dynScopes = scopeService.getDynReg();
223
224                        // scopes that the client is asking for
225                        Set<SystemScope> requestedScopes = scopeService.fromStrings(newClient.getScope());
226
227                        // the scopes that the client can have must be a subset of the dynamically allowed scopes
228                        Set<SystemScope> allowedScopes = Sets.intersection(dynScopes, requestedScopes);
229
230                        // make sure that the client doesn't ask for scopes it can't have
231                        newClient.setScope(scopeService.toStrings(allowedScopes));
232                        
233                        // save the client
234                        ClientDetailsEntity savedClient = clientService.updateClient(oldClient, newClient);
235                        
236                        // we return the token that we got in
237                        // TODO: rotate this after some set amount of time
238                        OAuth2AuthenticationDetails details = (OAuth2AuthenticationDetails) auth.getDetails();
239                        OAuth2AccessTokenEntity token = tokenService.readAccessToken(details.getTokenValue());
240                        
241                        // send it all out to the view
242                        m.addAttribute("client", savedClient);
243                        m.addAttribute("code", HttpStatus.OK); // http 200
244                        m.addAttribute("token", token);
245                        
246                        return "clientInformationResponseView";
247                } else {
248                        // client mismatch
249                        logger.error("readClientConfiguration failed, client ID mismatch: " 
250                                        + clientId + " and " + auth.getAuthorizationRequest().getClientId() + " do not match.");
251                        m.addAttribute("code", HttpStatus.FORBIDDEN); // http 403
252                        
253                        return "httpCodeView";
254                }
255        }

257        /**
258         * Delete the indicated client from the system.
259         * @param clientId

261         * @param auth

264        @PreAuthorize("hasRole('ROLE_CLIENT') and #oauth2.hasScope('" + OAuth2AccessTokenEntity.REGISTRATION_TOKEN_SCOPE + "')")
265        @RequestMapping(value = "/{id}", method = RequestMethod.DELETE, produces = "application/json")
266        public String deleteClient(@PathVariable("id") String clientId, Model m, OAuth2Authentication auth) {
267                
268                ClientDetailsEntity client = clientService.loadClientByClientId(clientId);

270                if (client != null && client.getClientId().equals(auth.getAuthorizationRequest().getClientId())) {
271
272                        clientService.deleteClient(client);
273                        
274                        // we return the token that we got in

277                        
278                        // send it all out to the view
279                        m.addAttribute("client", client);
280                        m.addAttribute("code", HttpStatus.OK); // http 200

284                } else {
285                        // client mismatch
286                        logger.error("readClientConfiguration failed, client ID mismatch: " 
287                                        + clientId + " and " + auth.getAuthorizationRequest().getClientId() + " do not match.");

297        /**
298     * @param client
299     * @return
300     * @throws AuthenticationException
301     */
302    private OAuth2AccessTokenEntity createRegistrationAccessToken(ClientDetailsEntity client) throws AuthenticationException {
303            // create a registration access token, treat it like a client credentials flow
304                // I can't use the auth request interface here because it has no setters and bad constructors -- THIS IS BAD API DESIGN
305                DefaultAuthorizationRequest authorizationRequest = new DefaultAuthorizationRequest(client.getClientId(), Sets.newHashSet(OAuth2AccessTokenEntity.REGISTRATION_TOKEN_SCOPE));
306                authorizationRequest.setApproved(true);
307                authorizationRequest.setAuthorities(Sets.newHashSet(new SimpleGrantedAuthority("ROLE_CLIENT")));
308                OAuth2Authentication authentication = new OAuth2Authentication(authorizationRequest, null);
309                OAuth2AccessTokenEntity registrationAccessToken = (OAuth2AccessTokenEntity) tokenService.createAccessToken(authentication);
310            return registrationAccessToken;
311    }
