087        /**
088         * Create a new Client, issue a client ID, and create a registration access token.
089         * @param jsonString
090         * @param m
091         * @param p
092         * @return
093         */
094        @RequestMapping(method = RequestMethod.POST, consumes = "application/json", produces = "application/json")
095        public String registerNewClient(@RequestBody String jsonString, Model m) {
096
097                ClientDetailsEntity newClient = null;
098                try {
099                        newClient = ClientDetailsEntityJsonProcessor.parse(jsonString);
100                } catch (JsonSyntaxException e) {
101                        // bad parse
102                        // didn't parse, this is a bad request
103                        logger.error("registerNewClient failed; submitted JSON is malformed");
104                        m.addAttribute("code", HttpStatus.BAD_REQUEST); // http 400
105                        return "httpCodeView";
106                }
107
108                if (newClient != null) {
109                        // it parsed!
110
111                        //
112                        // Now do some post-processing consistency checks on it
113                        //
114
115                        // clear out any spurious id/secret (clients don't get to pick)
116                        newClient.setClientId(null);
117                        newClient.setClientSecret(null);
118
119                        // do validation on the fields
120                        try {
121                                newClient = validateScopes(newClient);
122                                newClient = validateResponseTypes(newClient);
123                                newClient = validateGrantTypes(newClient);
124                                newClient = validateRedirectUris(newClient);
125                                newClient = validateAuth(newClient);
126                        } catch (ValidationException ve) {
127                                // validation failed, return an error
128                                m.addAttribute("error", ve.getError());
129                                m.addAttribute("errorMessage", ve.getErrorDescription());
130                                m.addAttribute("code", ve.getStatus());
131                                return "jsonErrorView";
132                        }
133                        
134                        if (newClient.getTokenEndpointAuthMethod() == null) {
135                                newClient.setTokenEndpointAuthMethod(AuthMethod.SECRET_BASIC);
136                        }
137
138                        if (newClient.getTokenEndpointAuthMethod() == AuthMethod.SECRET_BASIC ||
139                                        newClient.getTokenEndpointAuthMethod() == AuthMethod.SECRET_JWT ||
140                                        newClient.getTokenEndpointAuthMethod() == AuthMethod.SECRET_POST) {
141
142                                // we need to generate a secret
143                                newClient = clientService.generateClientSecret(newClient);
144                        }
145
146                        // set some defaults for token timeouts
147                        newClient.setAccessTokenValiditySeconds((int)TimeUnit.HOURS.toSeconds(1)); // access tokens good for 1hr
148                        newClient.setIdTokenValiditySeconds((int)TimeUnit.MINUTES.toSeconds(10)); // id tokens good for 10min
149                        newClient.setRefreshTokenValiditySeconds(null); // refresh tokens good until revoked
150
151                        // this client has been dynamically registered (obviously)
152                        newClient.setDynamicallyRegistered(true);
153
154                        // this client can't do token introspection
155                        newClient.setAllowIntrospection(false);
156
157                        // now save it
158                        try {
159                                ClientDetailsEntity savedClient = clientService.saveNewClient(newClient);
160
161                                // generate the registration access token
162                                OAuth2AccessTokenEntity token = connectTokenService.createRegistrationAccessToken(savedClient);
163                                tokenService.saveAccessToken(token);
164
165                                // send it all out to the view
166
167                                RegisteredClient registered = new RegisteredClient(savedClient, token.getValue(), config.getIssuer() + "register/" + UriUtils.encodePathSegment(savedClient.getClientId(), "UTF-8"));
168                                m.addAttribute("client", registered);
169                                m.addAttribute("code", HttpStatus.CREATED); // http 201
170
171                                return "clientInformationResponseView";
172                        } catch (UnsupportedEncodingException e) {
173                                logger.error("Unsupported encoding", e);
174                                m.addAttribute("code", HttpStatus.INTERNAL_SERVER_ERROR);
175                                return "httpCodeView";
176                        } catch (IllegalArgumentException e) {
177                                logger.error("Couldn't save client", e);
178                                
179                                m.addAttribute("error", "invalid_client_metadata");
180                                m.addAttribute("errorMessage", "Unable to save client due to invalid or inconsistent metadata.");
181                                m.addAttribute("code", HttpStatus.BAD_REQUEST); // http 400
182                                
183                                return "jsonErrorView";
184                        }
185                } else {
186                        // didn't parse, this is a bad request
187                        logger.error("registerNewClient failed; submitted JSON is malformed");
188                        m.addAttribute("code", HttpStatus.BAD_REQUEST); // http 400
189
190                        return "httpCodeView";
191                }
192
193        }

195        /**
196         * Get the meta information for a client.
197         * @param clientId
198         * @param m
199         * @param auth
200         * @return
201         */
202        @PreAuthorize("hasRole('ROLE_CLIENT') and #oauth2.hasScope('" + SystemScopeService.REGISTRATION_TOKEN_SCOPE + "')")
203        @RequestMapping(value = "/{id}", method = RequestMethod.GET, produces = "application/json")
204        public String readClientConfiguration(@PathVariable("id") String clientId, Model m, OAuth2Authentication auth) {
205
206                ClientDetailsEntity client = clientService.loadClientByClientId(clientId);
207
208                if (client != null && client.getClientId().equals(auth.getOAuth2Request().getClientId())) {
209
210                        try {
211                                OAuth2AccessTokenEntity token = fetchValidRegistrationToken(auth, client);
212                                RegisteredClient registered = new RegisteredClient(client, token.getValue(), config.getIssuer() + "register/" +  UriUtils.encodePathSegment(client.getClientId(), "UTF-8"));
213
214                                // send it all out to the view
215                                m.addAttribute("client", registered);
216                                m.addAttribute("code", HttpStatus.OK); // http 200
217
218                                return "clientInformationResponseView";
219                        } catch (UnsupportedEncodingException e) {
220                                logger.error("Unsupported encoding", e);
221                                m.addAttribute("code", HttpStatus.INTERNAL_SERVER_ERROR);
222                                return "httpCodeView";
223                        }
224                        
225                } else {
226                        // client mismatch
227                        logger.error("readClientConfiguration failed, client ID mismatch: "
228                                        + clientId + " and " + auth.getOAuth2Request().getClientId() + " do not match.");
229                        m.addAttribute("code", HttpStatus.FORBIDDEN); // http 403
230
231                        return "httpCodeView";
232                }
233        }

235        /**
236         * Update the metainformation for a given client.
237         * @param clientId
238         * @param jsonString
239         * @param m
240         * @param auth
241         * @return
242         */
243        @PreAuthorize("hasRole('ROLE_CLIENT') and #oauth2.hasScope('" + SystemScopeService.REGISTRATION_TOKEN_SCOPE + "')")
244        @RequestMapping(value = "/{id}", method = RequestMethod.PUT, produces = "application/json", consumes = "application/json")
245        public String updateClient(@PathVariable("id") String clientId, @RequestBody String jsonString, Model m, OAuth2Authentication auth) {
246
247
248                ClientDetailsEntity newClient = null;
249                try {
250                        newClient = ClientDetailsEntityJsonProcessor.parse(jsonString);
251                } catch (JsonSyntaxException e) {
252                        // bad parse
253                        // didn't parse, this is a bad request
254                        logger.error("updateClient failed; submitted JSON is malformed");
255                        m.addAttribute("code", HttpStatus.BAD_REQUEST); // http 400
256                        return "httpCodeView";
257                }
258                ClientDetailsEntity oldClient = clientService.loadClientByClientId(clientId);
259
260                if (newClient != null && oldClient != null  // we have an existing client and the new one parsed
261                                && oldClient.getClientId().equals(auth.getOAuth2Request().getClientId()) // the client passed in the URI matches the one in the auth
262                                && oldClient.getClientId().equals(newClient.getClientId()) // the client passed in the body matches the one in the URI
263                                ) {
264
265                        // a client can't ask to update its own client secret to any particular value
266                        newClient.setClientSecret(oldClient.getClientSecret());
267
268                        // we need to copy over all of the local and SECOAUTH fields
269                        newClient.setAccessTokenValiditySeconds(oldClient.getAccessTokenValiditySeconds());
270                        newClient.setIdTokenValiditySeconds(oldClient.getIdTokenValiditySeconds());
271                        newClient.setRefreshTokenValiditySeconds(oldClient.getRefreshTokenValiditySeconds());
272                        newClient.setDynamicallyRegistered(true); // it's still dynamically registered
273                        newClient.setAllowIntrospection(false); // dynamically registered clients can't do introspection -- use the resource registration instead
274                        newClient.setAuthorities(oldClient.getAuthorities());
275                        newClient.setClientDescription(oldClient.getClientDescription());
276                        newClient.setCreatedAt(oldClient.getCreatedAt());
277                        newClient.setReuseRefreshToken(oldClient.isReuseRefreshToken());
278
279                        // do validation on the fields
280                        try {
281                                newClient = validateScopes(newClient);
282                                newClient = validateResponseTypes(newClient);
283                                newClient = validateGrantTypes(newClient);
284                                newClient = validateRedirectUris(newClient);
285                                newClient = validateAuth(newClient);
286                        } catch (ValidationException ve) {
287                                // validation failed, return an error
288                                m.addAttribute("error", ve.getError());
289                                m.addAttribute("errorMessage", ve.getErrorDescription());
290                                m.addAttribute("code", ve.getStatus());
291                                return "jsonErrorView";
292                        }
293                        
294                        try {
295                                // save the client
296                                ClientDetailsEntity savedClient = clientService.updateClient(oldClient, newClient);
297
298                                OAuth2AccessTokenEntity token = fetchValidRegistrationToken(auth, savedClient);
299
300                                RegisteredClient registered = new RegisteredClient(savedClient, token.getValue(), config.getIssuer() + "register/" + UriUtils.encodePathSegment(savedClient.getClientId(), "UTF-8"));
301
302                                // send it all out to the view
303                                m.addAttribute("client", registered);
304                                m.addAttribute("code", HttpStatus.OK); // http 200
305
306                                return "clientInformationResponseView";
307                        } catch (UnsupportedEncodingException e) {
308                                logger.error("Unsupported encoding", e);
309                                m.addAttribute("code", HttpStatus.INTERNAL_SERVER_ERROR);
310                                return "httpCodeView";
311                        } catch (IllegalArgumentException e) {
312                                logger.error("Couldn't save client", e);
313                                
314                                m.addAttribute("error", "invalid_client_metadata");
315                                m.addAttribute("errorMessage", "Unable to save client due to invalid or inconsistent metadata.");
316                                m.addAttribute("code", HttpStatus.BAD_REQUEST); // http 400
317                                
318                                return "jsonErrorView";
319                        }
320                } else {
321                        // client mismatch
322                        logger.error("updateClient failed, client ID mismatch: "
323                                        + clientId + " and " + auth.getOAuth2Request().getClientId() + " do not match.");
324                        m.addAttribute("code", HttpStatus.FORBIDDEN); // http 403
325
326                        return "httpCodeView";
327                }
328        }

360        private ClientDetailsEntity validateScopes(ClientDetailsEntity newClient) throws ValidationException {
361                // set of scopes that are OK for clients to dynamically register for
362                Set<SystemScope> dynScopes = scopeService.getDynReg();
363
364                // scopes that the client is asking for
365                Set<SystemScope> requestedScopes = scopeService.fromStrings(newClient.getScope());
366
367                // the scopes that the client can have must be a subset of the dynamically allowed scopes
368                Set<SystemScope> allowedScopes = Sets.intersection(dynScopes, requestedScopes);
369
370                // if the client didn't ask for any, give them the defaults
371                if (allowedScopes == null || allowedScopes.isEmpty()) {
372                        allowedScopes = scopeService.getDefaults();
373                }
374
375                newClient.setScope(scopeService.toStrings(allowedScopes));
376                
377                return newClient;
378        }

380        private ClientDetailsEntity validateResponseTypes(ClientDetailsEntity newClient) throws ValidationException {
381                if (newClient.getResponseTypes() == null) {
382                        newClient.setResponseTypes(new HashSet<String>());
383                }
384                return newClient;
385        }

387        private ClientDetailsEntity validateGrantTypes(ClientDetailsEntity newClient) throws ValidationException {
388                // set default grant types if needed
389                if (newClient.getGrantTypes() == null || newClient.getGrantTypes().isEmpty()) {
390                        if (newClient.getScope().contains("offline_access")) { // client asked for offline access
391                                newClient.setGrantTypes(Sets.newHashSet("authorization_code", "refresh_token")); // allow authorization code and refresh token grant types by default
392                        } else {
393                                newClient.setGrantTypes(Sets.newHashSet("authorization_code")); // allow authorization code grant type by default
394                        }
395                }
396                
397                // filter out unknown grant types
398                // TODO: make this a pluggable service
399                Set<String> requestedGrantTypes = new HashSet<String>(newClient.getGrantTypes());
400                requestedGrantTypes.retainAll(
401                                        ImmutableSet.of("authorization_code", "implicit", 
402                                                        "password", "client_credentials", "refresh_token",
403                                                        "urn:ietf:params:oauth:grant_type:redelegate"));
404                
405                // don't allow "password" grant type for dynamic registration
406                if (newClient.getGrantTypes().contains("password")) {
407                        // return an error, you can't dynamically register for the password grant
408                        throw new ValidationException("invalid_client_metadata", "The password grant type is not allowed in dynamic registration on this server.", HttpStatus.BAD_REQUEST);
409                }
410
411                // don't allow clients to have multiple incompatible grant types and scopes
412                if (newClient.getGrantTypes().contains("authorization_code")) {
413
414                        // check for incompatible grants
415                        if (newClient.getGrantTypes().contains("implicit") ||
416                                        newClient.getGrantTypes().contains("client_credentials")) {
417                                // return an error, you can't have these grant types together
418                                throw new ValidationException("invalid_client_metadata", "Incompatible grant types requested: " + newClient.getGrantTypes(), HttpStatus.BAD_REQUEST);
419                        }
420
421                        if (newClient.getResponseTypes().contains("token")) {
422                                // return an error, you can't have this grant type and response type together
423                                throw new ValidationException("invalid_client_metadata", "Incompatible response types requested: " + newClient.getGrantTypes() + " / " + newClient.getResponseTypes(), HttpStatus.BAD_REQUEST);
424                        }
425                        
426                        newClient.getResponseTypes().add("code");
427                        
428                
429                }
430                
431                if (newClient.getGrantTypes().contains("implicit")) {
432
433                        // check for incompatible grants
434                        if (newClient.getGrantTypes().contains("authorization_code") ||
435                                        newClient.getGrantTypes().contains("client_credentials")) {
436                                // return an error, you can't have these grant types together
437                                throw new ValidationException("invalid_client_metadata", "Incompatible grant types requested: " + newClient.getGrantTypes(), HttpStatus.BAD_REQUEST);
438                        }
439                        
440                        if (newClient.getResponseTypes().contains("code")) {
441                                // return an error, you can't have this grant type and response type together
442                                throw new ValidationException("invalid_client_metadata", "Incompatible response types requested: " + newClient.getGrantTypes() + " / " + newClient.getResponseTypes(), HttpStatus.BAD_REQUEST);
443                        }
444                        
445                        newClient.getResponseTypes().add("token");
446                        
447                        // don't allow refresh tokens in implicit clients
448                        newClient.getGrantTypes().remove("refresh_token");
449                        newClient.getScope().remove("offline_access");
450                }
451                
452                if (newClient.getGrantTypes().contains("client_credentials")) {
453
454                        // check for incompatible grants
455                        if (newClient.getGrantTypes().contains("authorization_code") ||
456                                        newClient.getGrantTypes().contains("implicit")) {
457                                // return an error, you can't have these grant types together
458                                throw new ValidationException("invalid_client_metadata", "Incompatible grant types requested: " + newClient.getGrantTypes(), HttpStatus.BAD_REQUEST);
459                        }
460                        
461                        if (!newClient.getResponseTypes().isEmpty()) {
462                                // return an error, you can't have this grant type and response type together
463                                throw new ValidationException("invalid_client_metadata", "Incompatible response types requested: " + newClient.getGrantTypes() + " / " + newClient.getResponseTypes(), HttpStatus.BAD_REQUEST);
464                        }
465                        
466                        // don't allow refresh tokens or id tokens in client_credentials clients
467                        newClient.getGrantTypes().remove("refresh_token");
468                        newClient.getScope().remove("offline_access");
469                        newClient.getScope().remove("openid");
470                }
471                
472                if (newClient.getGrantTypes().isEmpty()) {
473                        // return an error, you need at least one grant type selected
474                        throw new ValidationException("invalid_client_metadata", "Clients must register at least one grant type.", HttpStatus.BAD_REQUEST);
475                }               
476                return newClient;
477        }

479        private ClientDetailsEntity validateRedirectUris(ClientDetailsEntity newClient) throws ValidationException {
480                // check to make sure this client registered a redirect URI if using a redirect flow
481                if (newClient.getGrantTypes().contains("authorization_code") || newClient.getGrantTypes().contains("implicit")) {
482                        if (newClient.getRedirectUris() == null || newClient.getRedirectUris().isEmpty()) {
483                                // return an error
484                                throw new ValidationException("invalid_redirect_uri", "Clients using a redirect-based grant type must register at least one redirect URI.", HttpStatus.BAD_REQUEST);
485                        }
486
487                        for (String uri : newClient.getRedirectUris()) {
488                                if (blacklistService.isBlacklisted(uri)) {
489                                        // return an error
490                                        throw new ValidationException("invalid_redirect_uri", "Redirect URI is not allowed: " + uri, HttpStatus.BAD_REQUEST);
491                                }
492                                
493                                if (uri.contains("#")) { 
494                                        // if it contains the hash symbol then it has a fragment, which isn't allowed
495                                        throw new ValidationException("invalid_redirect_uri", "Redirect URI can not have a fragment", HttpStatus.BAD_REQUEST);
496                                }
497                        }
498                }
499                
500                return newClient;
501        }

503        private ClientDetailsEntity validateAuth(ClientDetailsEntity newClient) throws ValidationException {
504                if (newClient.getTokenEndpointAuthMethod() == null) {
505                        newClient.setTokenEndpointAuthMethod(AuthMethod.SECRET_BASIC);
506                }
507
508                if (newClient.getTokenEndpointAuthMethod() == AuthMethod.SECRET_BASIC ||
509                                newClient.getTokenEndpointAuthMethod() == AuthMethod.SECRET_JWT ||
510                                newClient.getTokenEndpointAuthMethod() == AuthMethod.SECRET_POST) {
511
512                        if (Strings.isNullOrEmpty(newClient.getClientSecret())) {
513                                // no secret yet, we need to generate a secret
514                                newClient = clientService.generateClientSecret(newClient);
515                        }
516                } else if (newClient.getTokenEndpointAuthMethod() == AuthMethod.PRIVATE_KEY) {
517                        if (Strings.isNullOrEmpty(newClient.getJwksUri())) {
518                                throw new ValidationException("invalid_client_metadata", "JWK Set URI required when using private key authentication", HttpStatus.BAD_REQUEST);
519                        }
520                        
521                        newClient.setClientSecret(null);
522                } else if (newClient.getTokenEndpointAuthMethod() == AuthMethod.NONE) {
523                        newClient.setClientSecret(null);
524                } else {
525                        throw new ValidationException("invalid_client_metadata", "Unknown authentication method", HttpStatus.BAD_REQUEST);
526                }
527                return newClient;
528        }

530        private OAuth2AccessTokenEntity fetchValidRegistrationToken(OAuth2Authentication auth, ClientDetailsEntity client) {
531                
532                OAuth2AuthenticationDetails details = (OAuth2AuthenticationDetails) auth.getDetails();
533                OAuth2AccessTokenEntity token = tokenService.readAccessToken(details.getTokenValue());
534                
535                if (config.getRegTokenLifeTime() != null) {
536                
537                        try {
538                                // Re-issue the token if it has been issued before [currentTime - validity]
539                                Date validToDate = new Date(System.currentTimeMillis() - config.getRegTokenLifeTime() * 1000);
540                                if(token.getJwt().getJWTClaimsSet().getIssueTime().before(validToDate)) {
541                                        logger.info("Rotating the registration access token for " + client.getClientId());
542                                        tokenService.revokeAccessToken(token);
543                                        OAuth2AccessTokenEntity newToken = connectTokenService.createRegistrationAccessToken(client);
544                                        tokenService.saveAccessToken(newToken);
545                                        return newToken;
546                                } else {
547                                        // it's not expired, keep going
548                                        return token;
549                                }
550                        } catch (ParseException e) {
551                                logger.error("Couldn't parse a known-valid token?", e);
552                                return token;
553                        }
554                } else {
555                        // tokens don't expire, just return it
556                        return token;
557                }
558        }
