004package org.mitre.openid.connect.web;
005
006import java.security.Principal;
007
008import javax.servlet.http.HttpServletRequest;
009import javax.servlet.http.HttpServletResponse;
010
011import org.mitre.openid.connect.model.UserInfo;
012import org.mitre.openid.connect.repository.UserInfoRepository;
013import org.mitre.openid.connect.service.UserInfoService;
014import org.springframework.beans.factory.annotation.Autowired;
015import org.springframework.web.servlet.ModelAndView;
016import org.springframework.web.servlet.handler.HandlerInterceptorAdapter;
017
018/**
019 * Injects the UserInfo object for the current user into the current model's context, if both exist. Allows JSPs and the like to call "userInfo.name" and other fields.
020 * 
021 * @author jricher
022 *
023 */
024public class UserInfoInterceptor extends HandlerInterceptorAdapter {
025
026        @Autowired
027        private UserInfoService userInfoService;
028        
029    @Override
030    public void postHandle(HttpServletRequest request, HttpServletResponse response, Object handler, ModelAndView modelAndView) throws Exception {
031
032        if (modelAndView != null) { // skip checking at all if we have no model and view to hand the user to
033                // get our principal from the security context
034                Principal p = request.getUserPrincipal();
035                
036                if (p != null && p.getName() != null) { // don't bother checking if we don't have a principal
037                
038                        // try to look up a user based on it
039                        UserInfo user = userInfoService.getBySubject(p.getName());
040                        
041                        // if we have one, inject it so views can use it
042                        if (user != null) {
043                                modelAndView.addObject("userInfo", user);
044                        }
045                }
046        }
047        
048    }
049        
050        
051        
052
053}
