004package org.mitre.openid.connect.web;
005
006import javax.servlet.http.HttpServletRequest;
007import javax.servlet.http.HttpServletResponse;
008
009import org.mitre.openid.connect.config.ConfigurationPropertiesBean;
010import org.springframework.beans.factory.annotation.Autowired;
011import org.springframework.web.servlet.ModelAndView;
012import org.springframework.web.servlet.handler.HandlerInterceptorAdapter;
013
014/**
015 * 
016 * Injects the server configuration bean into the Model context, if it exists. Allows JSPs and the like to call "config.logoUrl" among others. 
017 * 
018 * @author jricher
019 *
020 */
021public class ServerConfigInterceptor extends HandlerInterceptorAdapter {
022
023        @Autowired
024        private ConfigurationPropertiesBean config;
025        
026    @Override
027    public void postHandle(HttpServletRequest request, HttpServletResponse response, Object handler, ModelAndView modelAndView) throws Exception {
028        if (modelAndView != null) { // skip checking at all if we have no model and view to hand the config to
029                modelAndView.addObject("config", config);
030        }
031    }
032
033}
