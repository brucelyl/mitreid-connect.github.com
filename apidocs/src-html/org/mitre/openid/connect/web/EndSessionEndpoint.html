054/**
055 * Implementation of the End Session Endpoint from OIDC session management
056 * 
057 * @author jricher
058 *
059 */
060@Controller
061public class EndSessionEndpoint {
062
063        public static final String URL = "endsession";
064        
065        private static final String CLIENT_KEY = "client";
066        private static final String STATE_KEY = "state";
067        private static final String REDIRECT_URI_KEY = "redirectUri";
068
069        private static Logger logger = LoggerFactory.getLogger(EndSessionEndpoint.class);
070        
071        @Autowired
072        private SelfAssertionValidator validator;
073        
074        @Autowired
075        private UserInfoService userInfoService;
076        
077        @Autowired
078        private ClientDetailsEntityService clientService;
079        
080        @RequestMapping(value = "/" + URL, method = RequestMethod.GET)
081        public String endSession(@RequestParam (value = "id_token_hint", required = false) String idTokenHint,  
082                    @RequestParam (value = "post_logout_redirect_uri", required = false) String postLogoutRedirectUri,
083                    @RequestParam (value = STATE_KEY, required = false) String state,
084                    HttpServletRequest request,
085                    HttpServletResponse response,
086                    HttpSession session,
087                    Authentication auth, Model m) {
088
089                // conditionally filled variables
090                JWTClaimsSet idTokenClaims = null; // pulled from the parsed and validated ID token
091                ClientDetailsEntity client = null; // pulled from ID token's audience field
092                
093                if (!Strings.isNullOrEmpty(postLogoutRedirectUri)) {
094                        session.setAttribute(REDIRECT_URI_KEY, postLogoutRedirectUri);
095                }
096                if (!Strings.isNullOrEmpty(state)) {
097                        session.setAttribute(STATE_KEY, state);
098                }
099                
100                // parse the ID token hint to see if it's valid
101                if (!Strings.isNullOrEmpty(idTokenHint)) {
102                        try {
103                                JWT idToken = JWTParser.parse(idTokenHint);
104                                
105                                if (validator.isValid(idToken)) {
106                                        // we issued this ID token, figure out who it's for
107                                        idTokenClaims = idToken.getJWTClaimsSet();
108                                        
109                                        String clientId = Iterables.getOnlyElement(idTokenClaims.getAudience());
110                                        
111                                        client = clientService.loadClientByClientId(clientId);
112                                        
113                                        // save a reference in the session for us to pick up later
114                                        //session.setAttribute("endSession_idTokenHint_claims", idTokenClaims);
115                                        session.setAttribute(CLIENT_KEY, client);
116                                }
117                        } catch (ParseException e) {
118                                // it's not a valid ID token, ignore it
119                                logger.debug("Invalid id token hint", e);
120                        } catch (InvalidClientException e) {
121                                // couldn't find the client, ignore it
122                                logger.debug("Invalid client", e);
123                        }
124                }
125                
126                // are we logged in or not?
127                if (auth == null || !request.isUserInRole("ROLE_USER")) {
128                        // we're not logged in anyway, process the final redirect bits if needed
129                        return processLogout(null, request, response, session, auth, m);
130                } else {
131                        // we are logged in, need to prompt the user before we log out
132                
133                        // see who the current user is
134                        UserInfo ui = userInfoService.getByUsername(auth.getName()); 
135                        
136                        if (idTokenClaims != null) {
137                                String subject = idTokenClaims.getSubject();
138                                // see if the current user is the same as the one in the ID token
139                                // TODO: should we do anything different in these cases?
140                                if (!Strings.isNullOrEmpty(subject) && subject.equals(ui.getSub())) {
141                                        // it's the same user
142                                } else { 
143                                        // it's not the same user
144                                }
145                        }
146
147                        m.addAttribute("client", client);
148                        m.addAttribute("idToken", idTokenClaims);
149                        
150                        // display the log out confirmation page
151                        return "logoutConfirmation";
152                }
153        }
154        
155        @RequestMapping(value = "/" + URL, method = RequestMethod.POST)
156        public String processLogout(@RequestParam(value = "approve", required = false) String approved,
157                        HttpServletRequest request,
158                        HttpServletResponse response,
159                    HttpSession session,
160                    Authentication auth, Model m) {
161
162                String redirectUri = (String) session.getAttribute(REDIRECT_URI_KEY);
163                String state = (String) session.getAttribute(STATE_KEY);
164                ClientDetailsEntity client = (ClientDetailsEntity) session.getAttribute(CLIENT_KEY);
165                
166                if (!Strings.isNullOrEmpty(approved)) {
167                        // use approved, perform the logout
168                        if (auth != null){    
169                                new SecurityContextLogoutHandler().logout(request, response, auth);
170                        }
171                        SecurityContextHolder.getContext().setAuthentication(null);
172                        // TODO: hook into other logout post-processing
173                }
174                
175                // if the user didn't approve, don't log out but hit the landing page anyway for redirect as needed
176
177                
178                
179                // if we have a client AND the client has post-logout redirect URIs
180                // registered AND the URI given is in that list, then...
181                if (!Strings.isNullOrEmpty(redirectUri) && 
182                        client != null && client.getPostLogoutRedirectUris() != null) {
183                        
184                        if (client.getPostLogoutRedirectUris().contains(redirectUri)) {
185                                // TODO: future, add the redirect URI to the model for the display page for an interstitial
186                                // m.addAttribute("redirectUri", postLogoutRedirectUri);
187                                
188                                UriComponents uri = UriComponentsBuilder.fromHttpUrl(redirectUri).queryParam("state", state).build();
189                                
190                                return "redirect:" + uri;
191                        }
192                }
193                
194                // otherwise, return to a nice post-logout landing page
195                return "postLogout";
196        }
