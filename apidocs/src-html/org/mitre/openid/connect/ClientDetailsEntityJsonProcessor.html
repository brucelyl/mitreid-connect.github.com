053        /**
054         * 
055         * Create an unbound ClientDetailsEntity from the given JSON string.
056         * 
057         * @param jsonString
058         * @return the entity if successful, null otherwise
059         */
060        public static ClientDetailsEntity parse(String jsonString) {
061                JsonElement jsonEl = parser.parse(jsonString);
062                if (jsonEl.isJsonObject()) {
063
064                        JsonObject o = jsonEl.getAsJsonObject();
065                        ClientDetailsEntity c = new ClientDetailsEntity();
066
067                        // TODO: make these field names into constants
068
069                        // these two fields should only be sent in the update request, and MUST match existing values
070                        c.setClientId(getAsString(o, "client_id"));
071                        c.setClientSecret(getAsString(o, "client_secret"));
072
073                        // OAuth DynReg
074                        c.setRedirectUris(getAsStringSet(o, "redirect_uris"));
075                        c.setClientName(getAsString(o, "client_name"));
076                        c.setClientUri(getAsString(o, "client_uri"));
077                        c.setLogoUri(getAsString(o, "logo_uri"));
078                        c.setContacts(getAsStringSet(o, "contacts"));
079                        c.setTosUri(getAsString(o, "tos_uri"));
080
081                        String authMethod = getAsString(o, "token_endpoint_auth_method");
082                        if (authMethod != null) {
083                                c.setTokenEndpointAuthMethod(AuthMethod.getByValue(authMethod));
084                        }
085
086                        // scope is a space-separated string
087                        String scope = getAsString(o, "scope");
088                        if (scope != null) {
089                                c.setScope(Sets.newHashSet(Splitter.on(" ").split(scope)));
090                        }
091
092                        c.setGrantTypes(getAsStringSet(o, "grant_types"));
093                        c.setResponseTypes(getAsStringSet(o, "response_types"));
094                        c.setPolicyUri(getAsString(o, "policy_uri"));
095                        c.setJwksUri(getAsString(o, "jwks_uri"));
096
097
098                        // OIDC Additions
099                        String appType = getAsString(o, "application_type");
100                        if (appType != null) {
101                                c.setApplicationType(AppType.getByValue(appType));
102                        }
103
104                        c.setSectorIdentifierUri(getAsString(o, "sector_identifier_uri"));
105
106                        String subjectType = getAsString(o, "subject_type");
107                        if (subjectType != null) {
108                                c.setSubjectType(SubjectType.getByValue(subjectType));
109                        }
110
111                        c.setRequestObjectSigningAlg(getAsJwsAlgorithm(o, "request_object_signing_alg"));
112
113                        c.setUserInfoSignedResponseAlg(getAsJwsAlgorithm(o, "userinfo_signed_response_alg"));
114                        c.setUserInfoEncryptedResponseAlg(getAsJweAlgorithm(o, "userinfo_encrypted_response_alg"));
115                        c.setUserInfoEncryptedResponseEnc(getAsJweEncryptionMethod(o, "userinfo_encrypted_response_enc"));
116
117                        c.setIdTokenSignedResponseAlg(getAsJwsAlgorithm(o, "id_token_signed_response_alg"));
118                        c.setIdTokenEncryptedResponseAlg(getAsJweAlgorithm(o, "id_token_encrypted_response_alg"));
119                        c.setIdTokenEncryptedResponseEnc(getAsJweEncryptionMethod(o, "id_token_encrypted_response_enc"));
120
121                        if (o.has("default_max_age")) {
122                                if (o.get("default_max_age").isJsonPrimitive()) {
123                                        c.setDefaultMaxAge(o.get("default_max_age").getAsInt());
124                                }
125                        }
126
127                        if (o.has("require_auth_time")) {
128                                if (o.get("require_auth_time").isJsonPrimitive()) {
129                                        c.setRequireAuthTime(o.get("require_auth_time").getAsBoolean());
130                                }
131                        }
132
133                        c.setDefaultACRvalues(getAsStringSet(o, "default_acr_values"));
134                        c.setInitiateLoginUri(getAsString(o, "initiate_login_uri"));
135                        c.setPostLogoutRedirectUri(getAsString(o, "post_logout_redirect_uri"));
136                        c.setRequestUris(getAsStringSet(o, "request_uris"));
137
138                        return c;
139                } else {
140                        return null;
141                }
142        }

144        /**
145         * Parse the JSON as a RegisteredClient (useful in the dynamic client filter)
146         */
147        public static RegisteredClient parseRegistered(String jsonString) {
148
149
150                JsonElement jsonEl = parser.parse(jsonString);
151                if (jsonEl.isJsonObject()) {
152
153                        JsonObject o = jsonEl.getAsJsonObject();
154                        ClientDetailsEntity c = parse(jsonString);
155
156                        RegisteredClient rc = new RegisteredClient(c);
157                        // get any fields from the registration
158                        rc.setRegistrationAccessToken(getAsString(o, "registration_access_token"));
159                        rc.setRegistrationClientUri(getAsString(o, "registration_client_uri"));
160                        rc.setClientIdIssuedAt(getAsDate(o, "client_id_issued_at"));
161                        rc.setClientSecretExpiresAt(getAsDate(o, "client_secret_expires_at"));
162
163                        return rc;
164                } else {
165                        return null;
166                }
167        }

169        /**
170         * @param c
171         * @param token
172         * @param registrationUri
173         * @return
174         */
175        public static JsonObject serialize(RegisteredClient c) {
176                JsonObject o = new JsonObject();
177
178                o.addProperty("client_id", c.getClientId());
179                if (c.getClientSecret() != null) {
180                        o.addProperty("client_secret", c.getClientSecret());
181
182                        if (c.getClientSecretExpiresAt() == null) {
183                                o.addProperty("client_secret_expires_at", 0); // TODO: do we want to let secrets expire?
184                        } else {
185                                o.addProperty("client_secret_expires_at", c.getClientSecretExpiresAt().getTime() / 1000L);
186                        }
187                }
188
189                if (c.getClientIdIssuedAt() != null) {
190                        o.addProperty("client_id_issued_at", c.getClientIdIssuedAt().getTime() / 1000L);
191                } else if (c.getCreatedAt() != null) {
192                        o.addProperty("client_id_issued_at", c.getCreatedAt().getTime() / 1000L);
193                }
194                if (c.getRegistrationAccessToken() != null) {
195                        o.addProperty("registration_access_token", c.getRegistrationAccessToken());
196                }
197
198                if (c.getRegistrationClientUri() != null) {
199                        o.addProperty("registration_client_uri", c.getRegistrationClientUri());
200                }
201
202
203                // add in all other client properties
204
205                // OAuth DynReg
206                o.add("redirect_uris", getAsArray(c.getRedirectUris()));
207                o.addProperty("client_name", c.getClientName());
208                o.addProperty("client_uri", c.getClientUri());
209                o.addProperty("logo_uri", c.getLogoUri());
210                o.add("contacts", getAsArray(c.getContacts()));
211                o.addProperty("tos_uri", c.getTosUri());
212                o.addProperty("token_endpoint_auth_method", c.getTokenEndpointAuthMethod() != null ? c.getTokenEndpointAuthMethod().getValue() : null);
213                o.addProperty("scope", c.getScope() != null ? Joiner.on(" ").join(c.getScope()) : null);
214                o.add("grant_types", getAsArray(c.getGrantTypes()));
215                o.add("response_types", getAsArray(c.getResponseTypes()));
216                o.addProperty("policy_uri", c.getPolicyUri());
217                o.addProperty("jwks_uri", c.getJwksUri());
218
219                // OIDC Registration
220                o.addProperty("application_type", c.getApplicationType() != null ? c.getApplicationType().getValue() : null);
221                o.addProperty("sector_identifier_uri", c.getSectorIdentifierUri());
222                o.addProperty("subject_type", c.getSubjectType() != null ? c.getSubjectType().getValue() : null);
223                o.addProperty("request_object_signing_alg", c.getRequestObjectSigningAlg() != null ? c.getRequestObjectSigningAlg().getAlgorithmName() : null);
224                o.addProperty("userinfo_signed_response_alg", c.getUserInfoSignedResponseAlg() != null ? c.getUserInfoSignedResponseAlg().getAlgorithmName() : null);
225                o.addProperty("userinfo_encrypted_response_alg", c.getUserInfoEncryptedResponseAlg() != null ? c.getUserInfoEncryptedResponseAlg().getAlgorithmName() : null);
226                o.addProperty("userinfo_encrypted_response_enc", c.getUserInfoEncryptedResponseEnc() != null ? c.getUserInfoEncryptedResponseEnc().getAlgorithmName() : null);
227                o.addProperty("id_token_signed_response_alg", c.getIdTokenSignedResponseAlg() != null ? c.getIdTokenSignedResponseAlg().getAlgorithmName() : null);
228                o.addProperty("id_token_encrypted_response_alg", c.getIdTokenEncryptedResponseAlg() != null ? c.getIdTokenEncryptedResponseAlg().getAlgorithmName() : null);
229                o.addProperty("id_token_encrypted_response_enc", c.getIdTokenEncryptedResponseEnc() != null ? c.getIdTokenEncryptedResponseEnc().getAlgorithmName() : null);
230                o.addProperty("default_max_age", c.getDefaultMaxAge());
231                o.addProperty("require_auth_time", c.getRequireAuthTime());
232                o.add("default_acr_values", getAsArray(c.getDefaultACRvalues()));
233                o.addProperty("initiate_login_uri", c.getInitiateLoginUri());
234                o.addProperty("post_logout_redirect_uri", c.getPostLogoutRedirectUri());
235                o.add("request_uris", getAsArray(c.getRequestUris()));
236                return o;
237        }

239        /**
240         * Gets the value of the given member as a JWE Algorithm, null if it doesn't exist
241         */
242        private static JWEAlgorithmEmbed getAsJweAlgorithm(JsonObject o, String member) {
243                String s = getAsString(o, member);
244                if (s != null) {
245                        return JWEAlgorithmEmbed.getForAlgorithmName(s);
246                } else {
247                        return null;
248                }
249        }

275        /**
276         * Gets the value of the given member as a string, null if it doesn't exist
277         */
278        private static String getAsString(JsonObject o, String member) {
279                if (o.has(member)) {
280                        JsonElement e = o.get(member);
281                        if (e != null && e.isJsonPrimitive()) {
282                                return e.getAsString();
283                        } else {
284                                return null;
285                        }
286                } else {
287                        return null;
288                }
289        }
290
291        /**
292         * Gets the value of the given member (expressed as integer seconds since epoch) as a Date
293         */
294        private static Date getAsDate(JsonObject o, String member) {
295                if (o.has(member)) {
296                        JsonElement e = o.get(member);
297                        if (e != null && e.isJsonPrimitive()) {
298                                return new Date(e.getAsInt() * 1000L);
299                        } else {
300                                return null;
301                        }
302                } else {
303                        return null;
304                }
305        }
306
307        /**
308         * Gets the value of the given given member as a set of strings, null if it doesn't exist
309         */
310        private static Set<String> getAsStringSet(JsonObject o, String member) throws JsonSyntaxException {
311                if (o.has(member)) {
312                        return gson.fromJson(o.get(member), new TypeToken<Set<String>>(){}.getType());
313                } else {
314                        return null;
315                }
316        }
317
318
319        /**
320         * Translate a set of strings to a JSON array
321         * @param value
322         * @return
323         */
324        private static JsonElement getAsArray(Set<String> value) {
325                return gson.toJsonTree(value, new TypeToken<Set<String>>(){}.getType());
326        }
