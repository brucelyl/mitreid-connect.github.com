004package org.mitre.openid.connect.assertion;
005
006import java.util.Collection;
007
008import org.springframework.security.authentication.AbstractAuthenticationToken;
009import org.springframework.security.core.GrantedAuthority;
010
011import com.nimbusds.jwt.JWT;
012
013/**
014 * @author jricher
015 *
016 */
017public class JwtBearerAssertionAuthenticationToken extends AbstractAuthenticationToken {
018
019        private String clientId;
020        private JWT jwt;
021        
022        /**
023         * Create an unauthenticated token with the given client ID and jwt
024         * @param clientId
025         * @param jwt
026         */
027        public JwtBearerAssertionAuthenticationToken(String clientId, JWT jwt) {
028            super(null);
029            this.clientId = clientId;
030            this.jwt = jwt;
031            setAuthenticated(false);
032    }
033        
034        /**
035         * Create an authenticated token with the given clientID, jwt, and authorities set
036         * @param clientId
037         * @param jwt
038         * @param authorities
039         */
040        public JwtBearerAssertionAuthenticationToken(String clientId, JWT jwt, Collection<? extends GrantedAuthority> authorities) {
041            super(authorities);
042            this.clientId = clientId;
043            this.jwt = jwt;
044            setAuthenticated(true);
045    }
046
047        /* (non-Javadoc)
048         * @see org.springframework.security.core.Authentication#getCredentials()
049         */
050        @Override
051        public Object getCredentials() {
052                return jwt;
053        }
054
055        /* (non-Javadoc)
056         * @see org.springframework.security.core.Authentication#getPrincipal()
057         */
058        @Override
059        public Object getPrincipal() {
060                return clientId;
061        }
062
063        /**
064     * @return the clientId
065     */
066    public String getClientId() {
067        return clientId;
068    }
069
070        /**
071     * @param clientId the clientId to set
072     */
073    public void setClientId(String clientId) {
074        this.clientId = clientId;
075    }
076
077        /**
078     * @return the jwt
079     */
080    public JWT getJwt() {
081        return jwt;
082    }
083
084        /**
085     * @param jwt the jwt to set
086     */
087    public void setJwt(JWT jwt) {
088        this.jwt = jwt;
089    }
090
091        /**
092         * Clear out the JWT that this token holds.
093     */
094    @Override
095    public void eraseCredentials() {
096            super.eraseCredentials();
097            setJwt(null);
098    }
099        
100        
101
102}
