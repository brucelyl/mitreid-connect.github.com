020package org.mitre.openid.connect.assertion;
021
022import java.text.ParseException;
023import java.util.Date;
024
025import org.mitre.jwt.signer.service.JwtSigningAndValidationService;
026import org.mitre.jwt.signer.service.impl.JWKSetCacheService;
027import org.mitre.oauth2.model.ClientDetailsEntity;
028import org.mitre.oauth2.service.ClientDetailsEntityService;
029import org.mitre.openid.connect.config.ConfigurationPropertiesBean;
030import org.slf4j.Logger;
031import org.slf4j.LoggerFactory;
032import org.springframework.beans.factory.annotation.Autowired;
033import org.springframework.security.authentication.AuthenticationProvider;
034import org.springframework.security.authentication.AuthenticationServiceException;
035import org.springframework.security.core.Authentication;
036import org.springframework.security.core.AuthenticationException;
037import org.springframework.security.core.userdetails.UsernameNotFoundException;
038import org.springframework.security.oauth2.common.exceptions.InvalidClientException;
039
040import com.nimbusds.jwt.JWT;
041import com.nimbusds.jwt.ReadOnlyJWTClaimsSet;
042import com.nimbusds.jwt.SignedJWT;
043
044/**
045 * @author jricher
046 *
047 */
048public class JwtBearerAuthenticationProvider implements AuthenticationProvider {
049
050        private static final Logger logger = LoggerFactory.getLogger(JwtBearerAuthenticationProvider.class);
051
052        // map of verifiers, load keys for clients
053        @Autowired
054        private JWKSetCacheService validators;
055
056        // Allow for time sync issues by having a window of X seconds.
057        private int timeSkewAllowance = 300;
058
059        // to load clients
060        @Autowired
061        private ClientDetailsEntityService clientService;
062
063        // to get our server's issuer url
064        @Autowired
065        private ConfigurationPropertiesBean config;
066
067        /**
068         * Try to validate the client credentials by parsing and validating the JWT.
069         */
070        @Override
071        public Authentication authenticate(Authentication authentication) throws AuthenticationException {
072
073                JwtBearerAssertionAuthenticationToken jwtAuth = (JwtBearerAssertionAuthenticationToken)authentication;
074
075
076                try {
077                        ClientDetailsEntity client = clientService.loadClientByClientId(jwtAuth.getClientId());
078
079                        JWT jwt = jwtAuth.getJwt();
080                        ReadOnlyJWTClaimsSet jwtClaims = jwt.getJWTClaimsSet();
081
082                        // check the signature with nimbus
083                        if (jwt instanceof SignedJWT) {
084                                SignedJWT jws = (SignedJWT)jwt;
085                                JwtSigningAndValidationService validator = validators.getValidator(client.getJwksUri());
086                                if (validator == null || !validator.validateSignature(jws)) {
087                                        throw new AuthenticationServiceException("Invalid signature");
088                                }
089                        }
090
091                        // check the issuer
092                        if (jwtClaims.getIssuer() == null) {
093                                throw new AuthenticationServiceException("Assertion Token Issuer is null");
094                        } else if (!jwtClaims.getIssuer().equals(client.getClientId())){
095                                throw new AuthenticationServiceException("Issuers do not match, expected " + client.getClientId() + " got " + jwtClaims.getIssuer());
096                        }
097
098                        // check expiration
099                        if (jwtClaims.getExpirationTime() == null) {
100                                throw new AuthenticationServiceException("Assertion Token does not have required expiration claim");
101                        } else {
102                                // it's not null, see if it's expired
103                                Date now = new Date(System.currentTimeMillis() - (timeSkewAllowance * 1000));
104                                if (now.after(jwtClaims.getExpirationTime())) {
105                                        throw new AuthenticationServiceException("Assertion Token is expired: " + jwtClaims.getExpirationTime());
106                                }
107                        }
108
109                        // check not before
110                        if (jwtClaims.getNotBeforeTime() != null) {
111                                Date now = new Date(System.currentTimeMillis() + (timeSkewAllowance * 1000));
112                                if (now.before(jwtClaims.getNotBeforeTime())){
113                                        throw new AuthenticationServiceException("Assertion Token not valid untill: " + jwtClaims.getNotBeforeTime());
114                                }
115                        }
116
117                        // check issued at
118                        if (jwtClaims.getIssueTime() != null) {
119                                // since it's not null, see if it was issued in the future
120                                Date now = new Date(System.currentTimeMillis() + (timeSkewAllowance * 1000));
121                                if (now.before(jwtClaims.getIssueTime())) {
122                                        throw new AuthenticationServiceException("Assertion Token was issued in the future: " + jwtClaims.getIssueTime());
123                                }
124                        }
125
126                        // check audience
127                        if (jwtClaims.getAudience() == null) {
128                                throw new AuthenticationServiceException("Assertion token audience is null");
129                        } else if (!jwtClaims.getAudience().contains(config.getIssuer())) {
130                                throw new AuthenticationServiceException("Audience does not match, expected " + config.getIssuer() + " got " + jwtClaims.getAudience());
131                        }
132
133                        // IFF we managed to get all the way down here, the token is valid
134                        return new JwtBearerAssertionAuthenticationToken(client.getClientId(), jwt, client.getAuthorities());
135
136                } catch (InvalidClientException e) {
137                        throw new UsernameNotFoundException("Could not find client: " + jwtAuth.getClientId());
138                } catch (ParseException e) {
139
140                        logger.error("Failure during authentication, error was: ", e);
141
142                        throw new AuthenticationServiceException("Invalid JWT format");
143                }
144        }
145
146        /**
147         * We support {@link JwtBearerAssertionAuthenticationToken}s only.
148         */
149        @Override
150        public boolean supports(Class<?> authentication) {
151                return (JwtBearerAssertionAuthenticationToken.class.isAssignableFrom(authentication));
152        }
153
154
155}
