004package org.mitre.openid.connect.assertion;
005
006import java.text.ParseException;
007import java.util.Date;
008
009import org.mitre.jwt.signer.service.JwtSigningAndValidationService;
010import org.mitre.jwt.signer.service.impl.JWKSetSigningAndValidationServiceCacheService;
011import org.mitre.oauth2.model.ClientDetailsEntity;
012import org.mitre.oauth2.service.ClientDetailsEntityService;
013import org.mitre.openid.connect.config.ConfigurationPropertiesBean;
014import org.slf4j.Logger;
015import org.slf4j.LoggerFactory;
016import org.springframework.beans.factory.annotation.Autowired;
017import org.springframework.security.authentication.AuthenticationProvider;
018import org.springframework.security.authentication.AuthenticationServiceException;
019import org.springframework.security.core.Authentication;
020import org.springframework.security.core.AuthenticationException;
021import org.springframework.security.core.userdetails.UsernameNotFoundException;
022import org.springframework.security.oauth2.common.exceptions.InvalidClientException;
023
024import com.nimbusds.jwt.JWT;
025import com.nimbusds.jwt.ReadOnlyJWTClaimsSet;
026import com.nimbusds.jwt.SignedJWT;
027
028/**
029 * @author jricher
030 *
031 */
032public class JwtBearerAuthenticationProvider implements AuthenticationProvider {
033
034        private static final Logger logger = LoggerFactory.getLogger(JwtBearerAuthenticationProvider.class);
035
036        // map of verifiers, load keys for clients
037        @Autowired
038        private JWKSetSigningAndValidationServiceCacheService validators;
039        
040        // Allow for time sync issues by having a window of X seconds.
041        private int timeSkewAllowance = 300;
042
043        // to load clients
044        @Autowired
045        private ClientDetailsEntityService clientService;
046        
047        // to get our server's issuer url
048        @Autowired
049        private ConfigurationPropertiesBean config;
050        
051        /**
052         * Try to validate the client credentials by parsing and validating the JWT.
053     */
054    @Override
055    public Authentication authenticate(Authentication authentication) throws AuthenticationException {
056        
057        JwtBearerAssertionAuthenticationToken jwtAuth = (JwtBearerAssertionAuthenticationToken)authentication;
058        
059        
060        try {
061                ClientDetailsEntity client = clientService.loadClientByClientId(jwtAuth.getClientId());
062
063                JWT jwt = jwtAuth.getJwt();
064                ReadOnlyJWTClaimsSet jwtClaims = jwt.getJWTClaimsSet();
065
066                // check the signature with nimbus
067                if (jwt instanceof SignedJWT) {
068                        SignedJWT jws = (SignedJWT)jwt;
069                        JwtSigningAndValidationService validator = validators.get(client.getJwksUri());
070                        if (validator == null || !validator.validateSignature(jws)) {
071                                throw new AuthenticationServiceException("Invalid signature");
072                        }
073                }
074                
075                        // check the issuer
076                        if (jwtClaims.getIssuer() == null) {
077                                throw new AuthenticationServiceException("Assertion Token Issuer is null");
078                        } else if (!jwtClaims.getIssuer().equals(client.getClientId())){
079                                throw new AuthenticationServiceException("Issuers do not match, expected " + client.getClientId() + " got " + jwtClaims.getIssuer());
080                        }
081                        
082                        // check expiration
083                        if (jwtClaims.getExpirationTime() == null) {
084                                throw new AuthenticationServiceException("Assertion Token does not have required expiration claim");
085                        } else {
086                                // it's not null, see if it's expired
087                                Date now = new Date(System.currentTimeMillis() - (timeSkewAllowance * 1000));
088                                if (now.after(jwtClaims.getExpirationTime())) {
089                                        throw new AuthenticationServiceException("Assertion Token is expired: " + jwtClaims.getExpirationTime());
090                                }
091                        }
092                        
093                        // check not before
094                        if (jwtClaims.getNotBeforeTime() != null) {
095                                Date now = new Date(System.currentTimeMillis() + (timeSkewAllowance * 1000));
096                                if (now.before(jwtClaims.getNotBeforeTime())){
097                                        throw new AuthenticationServiceException("Assertion Token not valid untill: " + jwtClaims.getNotBeforeTime());
098                                }
099                        }
100                        
101                        // check issued at
102                        if (jwtClaims.getIssueTime() != null) {
103                                // since it's not null, see if it was issued in the future
104                                Date now = new Date(System.currentTimeMillis() + (timeSkewAllowance * 1000));
105                                if (now.before(jwtClaims.getIssueTime())) {
106                                        throw new AuthenticationServiceException("Assertion Token was issued in the future: " + jwtClaims.getIssueTime());
107                                }
108                        }
109                        
110                        // check audience
111                        if (jwtClaims.getAudience() == null) {
112                                throw new AuthenticationServiceException("Assertion token audience is null"); 
113                        } else if (!jwtClaims.getAudience().contains(config.getIssuer())) {
114                                throw new AuthenticationServiceException("Audience does not match, expected " + config.getIssuer() + " got " + jwtClaims.getAudience());
115                        }
116                        
117                // IFF we managed to get all the way down here, the token is valid
118                        return new JwtBearerAssertionAuthenticationToken(client.getClientId(), jwt, client.getAuthorities());
119                
120        } catch (InvalidClientException e) {
121                throw new UsernameNotFoundException("Could not find client: " + jwtAuth.getClientId());
122        } catch (ParseException e) {
123                // TODO Auto-generated catch block
124                throw new AuthenticationServiceException("Invalid JWT format");
125        }
126    }
127
128        /**
129         * We support {@link JwtBearerAssertionAuthenticationToken}s only.
130     */
131    @Override
132    public boolean supports(Class<?> authentication) {
133            return (JwtBearerAssertionAuthenticationToken.class.isAssignableFrom(authentication));
134    }
135
136        
137}
