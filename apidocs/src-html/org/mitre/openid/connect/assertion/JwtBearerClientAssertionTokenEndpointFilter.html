004package org.mitre.openid.connect.assertion;
005
006import java.io.IOException;
007import java.text.ParseException;
008
009import javax.servlet.ServletException;
010import javax.servlet.http.HttpServletRequest;
011import javax.servlet.http.HttpServletResponse;
012
013import org.springframework.security.authentication.BadCredentialsException;
014import org.springframework.security.core.Authentication;
015import org.springframework.security.core.AuthenticationException;
016import org.springframework.security.oauth2.provider.client.ClientCredentialsTokenEndpointFilter;
017
018import com.google.common.base.Strings;
019import com.nimbusds.jwt.JWT;
020import com.nimbusds.jwt.JWTParser;
021
022/**
023 * Filter to check client authentication via JWT Bearer assertions.
024 * 
025 * @author jricher
026 *
027 */
028public class JwtBearerClientAssertionTokenEndpointFilter extends ClientCredentialsTokenEndpointFilter {
029
030        public JwtBearerClientAssertionTokenEndpointFilter() {
031            super();
032            // TODO Auto-generated constructor stub
033    }
034
035        public JwtBearerClientAssertionTokenEndpointFilter(String path) {
036            super(path);
037            // TODO Auto-generated constructor stub
038    }
039
040        /**
041         * Pull the assertion out of the request and send it up to the auth manager for processing.
042     */
043    @Override
044    public Authentication attemptAuthentication(HttpServletRequest request, HttpServletResponse response) throws AuthenticationException, IOException, ServletException {
045
046        // check for appropriate parameters
047        String assertionType = request.getParameter("client_assertion_type");
048        String assertion = request.getParameter("client_assertion");
049        
050        try {
051                JWT jwt = JWTParser.parse(assertion);
052        
053                String clientId = jwt.getJWTClaimsSet().getSubject();
054                
055                Authentication authRequest = new JwtBearerAssertionAuthenticationToken(clientId, jwt);
056        
057                return this.getAuthenticationManager().authenticate(authRequest);
058        } catch (ParseException e) {
059                throw new BadCredentialsException("Invalid JWT credential: " + assertion);
060        }
061    }
062
063        /**
064         * Check to see if the "client_assertion_type" and "client_assertion" parameters are present and contain the right values.
065     */
066    @Override
067    protected boolean requiresAuthentication(HttpServletRequest request, HttpServletResponse response) {
068        // check for appropriate parameters
069        String assertionType = request.getParameter("client_assertion_type");
070        String assertion = request.getParameter("client_assertion");
071
072        if (Strings.isNullOrEmpty(assertionType) || Strings.isNullOrEmpty(assertion)) {
073                return false;
074        } else if (!assertionType.equals("urn:ietf:params:oauth:client-assertion-type:jwt-bearer")) {
075                return false;
076        }
077        
078        
079        // Can't call to superclass here b/c client creds would break for lack of client_id
080//          return super.requiresAuthentication(request, response);
081        
082        String uri = request.getRequestURI();
083        int pathParamIndex = uri.indexOf(';');
084
085        if (pathParamIndex > 0) {
086            // strip everything after the first semi-colon
087            uri = uri.substring(0, pathParamIndex);
088        }
089
090        if ("".equals(request.getContextPath())) {
091            return uri.endsWith(getFilterProcessesUrl());
092        }
093
094        return uri.endsWith(request.getContextPath() + getFilterProcessesUrl());
095
096    }
