004package org.mitre.openid.connect.model;
005
006import javax.persistence.Basic;
007import javax.persistence.Column;
008import javax.persistence.Entity;
009import javax.persistence.GeneratedValue;
010import javax.persistence.GenerationType;
011import javax.persistence.Id;
012import javax.persistence.NamedQueries;
013import javax.persistence.NamedQuery;
014import javax.persistence.Table;
015
016/**
017 * 
018 * Holds the generated pairwise identifiers for a user. Can be tied to either a client ID or a sector identifier URL.
019 * 
020 * @author jricher
021 *
022 */
023@Entity
024@Table(name = "pairwise_identifier")
025@NamedQueries({
026        @NamedQuery(name="PairwiseIdentifier.getAll", query = "select p from PairwiseIdentifier p"),
027        @NamedQuery(name="PairwiseIdentifier.getBySectorIdentifier", query = "select p from PairwiseIdentifier p WHERE p.userSub = :sub AND p.sectorIdentifier = :sectorIdentifier")
028})
029public class PairwiseIdentifier {
030
031        private Long id;
032        private String identifier;
033        private String userSub;
034        private String sectorIdentifier;
035
036        /**
037         * @return the id
038         */
039        @Id
040        @GeneratedValue(strategy=GenerationType.IDENTITY)
041        @Column(name = "id")
042        public Long getId() {
043                return id;
044        }
045
046        /**
047         * @param id the id to set
048         */
049        public void setId(Long id) {
050                this.id = id;
051        }
052
053        /**
054         * @return the identifier
055         */
056        @Basic
057        @Column(name = "identifier")
058        public String getIdentifier() {
059                return identifier;
060        }
061
062        /**
063         * @param identifier the identifier to set
064         */
065        public void setIdentifier(String identifier) {
066                this.identifier = identifier;
067        }
068
069        /**
070         * @return the userSub
071         */
072        @Basic
073        @Column(name = "sub")
074        public String getUserSub() {
075                return userSub;
076        }
077
078        /**
079         * @param userSub the userSub to set
080         */
081        public void setUserSub(String userSub) {
082                this.userSub = userSub;
083        }
084
085        /**
086         * @return the sectorIdentifier
087         */
088        @Basic
089        @Column(name = "sector_identifier")
090        public String getSectorIdentifier() {
091                return sectorIdentifier;
092        }
093
094        /**
095         * @param sectorIdentifier the sectorIdentifier to set
096         */
097        public void setSectorIdentifier(String sectorIdentifier) {
098                this.sectorIdentifier = sectorIdentifier;
099        }
100}
