044/**
045 * @author jricher
046 *
047 */
048public class ClientDetailsEntityJsonProcessor {
049
050        private static JsonParser parser = new JsonParser();
051
052        /**
053         * 
054         * Create an unbound ClientDetailsEntity from the given JSON string.
055         * 
056         * @param jsonString
057         * @return the entity if successful, null otherwise
058         */
059        public static ClientDetailsEntity parse(String jsonString) {
060                JsonElement jsonEl = parser.parse(jsonString);
061                if (jsonEl.isJsonObject()) {
062
063                        JsonObject o = jsonEl.getAsJsonObject();
064                        ClientDetailsEntity c = new ClientDetailsEntity();
065
066                        // TODO: make these field names into constants
067
068                        // these two fields should only be sent in the update request, and MUST match existing values
069                        c.setClientId(getAsString(o, "client_id"));
070                        c.setClientSecret(getAsString(o, "client_secret"));
071
072                        // OAuth DynReg
073                        c.setRedirectUris(getAsStringSet(o, "redirect_uris"));
074                        c.setClientName(getAsString(o, "client_name"));
075                        c.setClientUri(getAsString(o, "client_uri"));
076                        c.setLogoUri(getAsString(o, "logo_uri"));
077                        c.setContacts(getAsStringSet(o, "contacts"));
078                        c.setTosUri(getAsString(o, "tos_uri"));
079
080                        String authMethod = getAsString(o, "token_endpoint_auth_method");
081                        if (authMethod != null) {
082                                c.setTokenEndpointAuthMethod(AuthMethod.getByValue(authMethod));
083                        }
084
085                        // scope is a space-separated string
086                        String scope = getAsString(o, "scope");
087                        if (scope != null) {
088                                c.setScope(Sets.newHashSet(Splitter.on(" ").split(scope)));
089                        }
090
091                        c.setGrantTypes(getAsStringSet(o, "grant_types"));
092                        c.setResponseTypes(getAsStringSet(o, "response_types"));
093                        c.setPolicyUri(getAsString(o, "policy_uri"));
094                        c.setJwksUri(getAsString(o, "jwks_uri"));
095
096
097                        // OIDC Additions
098                        String appType = getAsString(o, "application_type");
099                        if (appType != null) {
100                                c.setApplicationType(AppType.getByValue(appType));
101                        }
102
103                        c.setSectorIdentifierUri(getAsString(o, "sector_identifier_uri"));
104
105                        String subjectType = getAsString(o, "subject_type");
106                        if (subjectType != null) {
107                                c.setSubjectType(SubjectType.getByValue(subjectType));
108                        }
109
110                        c.setRequestObjectSigningAlg(getAsJwsAlgorithm(o, "request_object_signing_alg"));
111
112                        c.setUserInfoSignedResponseAlg(getAsJwsAlgorithm(o, "userinfo_signed_response_alg"));
113                        c.setUserInfoEncryptedResponseAlg(getAsJweAlgorithm(o, "userinfo_encrypted_response_alg"));
114                        c.setUserInfoEncryptedResponseEnc(getAsJweEncryptionMethod(o, "userinfo_encrypted_response_enc"));
115
116                        c.setIdTokenSignedResponseAlg(getAsJwsAlgorithm(o, "id_token_signed_response_alg"));
117                        c.setIdTokenEncryptedResponseAlg(getAsJweAlgorithm(o, "id_token_encrypted_response_alg"));
118                        c.setIdTokenEncryptedResponseEnc(getAsJweEncryptionMethod(o, "id_token_encrypted_response_enc"));
119
120                        c.setTokenEndpointAuthSigningAlg(getAsJwsAlgorithm(o, "token_endpoint_auth_signing_alg"));
121
122                        if (o.has("default_max_age")) {
123                                if (o.get("default_max_age").isJsonPrimitive()) {
124                                        c.setDefaultMaxAge(o.get("default_max_age").getAsInt());
125                                }
126                        }
127
128                        if (o.has("require_auth_time")) {
129                                if (o.get("require_auth_time").isJsonPrimitive()) {
130                                        c.setRequireAuthTime(o.get("require_auth_time").getAsBoolean());
131                                }
132                        }
133
134                        c.setDefaultACRvalues(getAsStringSet(o, "default_acr_values"));
135                        c.setInitiateLoginUri(getAsString(o, "initiate_login_uri"));
136                        c.setPostLogoutRedirectUri(getAsString(o, "post_logout_redirect_uri"));
137                        c.setRequestUris(getAsStringSet(o, "request_uris"));
138
139                        return c;
140                } else {
141                        return null;
142                }
143        }
144
145        /**
146         * Parse the JSON as a RegisteredClient (useful in the dynamic client filter)
147         */
148        public static RegisteredClient parseRegistered(String jsonString) {
149
150
151                JsonElement jsonEl = parser.parse(jsonString);
152                if (jsonEl.isJsonObject()) {
153
154                        JsonObject o = jsonEl.getAsJsonObject();
155                        ClientDetailsEntity c = parse(jsonString);
156
157                        RegisteredClient rc = new RegisteredClient(c);
158                        // get any fields from the registration
159                        rc.setRegistrationAccessToken(getAsString(o, "registration_access_token"));
160                        rc.setRegistrationClientUri(getAsString(o, "registration_client_uri"));
161                        rc.setClientIdIssuedAt(getAsDate(o, "client_id_issued_at"));
162                        rc.setClientSecretExpiresAt(getAsDate(o, "client_secret_expires_at"));
163
164                        return rc;
165                } else {
166                        return null;
167                }
168        }
169
170        /**
171         * @param c
172         * @param token
173         * @param registrationUri
174         * @return
175         */
176        public static JsonObject serialize(RegisteredClient c) {
177                JsonObject o = new JsonObject();
178
179                o.addProperty("client_id", c.getClientId());
180                if (c.getClientSecret() != null) {
181                        o.addProperty("client_secret", c.getClientSecret());
182
183                        if (c.getClientSecretExpiresAt() == null) {
184                                o.addProperty("client_secret_expires_at", 0); // TODO: do we want to let secrets expire?
185                        } else {
186                                o.addProperty("client_secret_expires_at", c.getClientSecretExpiresAt().getTime() / 1000L);
187                        }
188                }
189
190                if (c.getClientIdIssuedAt() != null) {
191                        o.addProperty("client_id_issued_at", c.getClientIdIssuedAt().getTime() / 1000L);
192                } else if (c.getCreatedAt() != null) {
193                        o.addProperty("client_id_issued_at", c.getCreatedAt().getTime() / 1000L);
194                }
195                if (c.getRegistrationAccessToken() != null) {
196                        o.addProperty("registration_access_token", c.getRegistrationAccessToken());
197                }
198
199                if (c.getRegistrationClientUri() != null) {
200                        o.addProperty("registration_client_uri", c.getRegistrationClientUri());
201                }
202
203
204                // add in all other client properties
205
206                // OAuth DynReg
207                o.add("redirect_uris", getAsArray(c.getRedirectUris()));
208                o.addProperty("client_name", c.getClientName());
209                o.addProperty("client_uri", c.getClientUri());
210                o.addProperty("logo_uri", c.getLogoUri());
211                o.add("contacts", getAsArray(c.getContacts()));
212                o.addProperty("tos_uri", c.getTosUri());
213                o.addProperty("token_endpoint_auth_method", c.getTokenEndpointAuthMethod() != null ? c.getTokenEndpointAuthMethod().getValue() : null);
214                o.addProperty("scope", c.getScope() != null ? Joiner.on(" ").join(c.getScope()) : null);
215                o.add("grant_types", getAsArray(c.getGrantTypes()));
216                o.add("response_types", getAsArray(c.getResponseTypes()));
217                o.addProperty("policy_uri", c.getPolicyUri());
218                o.addProperty("jwks_uri", c.getJwksUri());
219
220                // OIDC Registration
221                o.addProperty("application_type", c.getApplicationType() != null ? c.getApplicationType().getValue() : null);
222                o.addProperty("sector_identifier_uri", c.getSectorIdentifierUri());
223                o.addProperty("subject_type", c.getSubjectType() != null ? c.getSubjectType().getValue() : null);
224                o.addProperty("request_object_signing_alg", c.getRequestObjectSigningAlg() != null ? c.getRequestObjectSigningAlg().getName() : null);
225                o.addProperty("userinfo_signed_response_alg", c.getUserInfoSignedResponseAlg() != null ? c.getUserInfoSignedResponseAlg().getName() : null);
226                o.addProperty("userinfo_encrypted_response_alg", c.getUserInfoEncryptedResponseAlg() != null ? c.getUserInfoEncryptedResponseAlg().getName() : null);
227                o.addProperty("userinfo_encrypted_response_enc", c.getUserInfoEncryptedResponseEnc() != null ? c.getUserInfoEncryptedResponseEnc().getName() : null);
228                o.addProperty("id_token_signed_response_alg", c.getIdTokenSignedResponseAlg() != null ? c.getIdTokenSignedResponseAlg().getName() : null);
229                o.addProperty("id_token_encrypted_response_alg", c.getIdTokenEncryptedResponseAlg() != null ? c.getIdTokenEncryptedResponseAlg().getName() : null);
230                o.addProperty("id_token_encrypted_response_enc", c.getIdTokenEncryptedResponseEnc() != null ? c.getIdTokenEncryptedResponseEnc().getName() : null);
231                o.addProperty("token_endpoint_auth_signing_alg", c.getTokenEndpointAuthSigningAlg() != null ? c.getTokenEndpointAuthSigningAlg().getName() : null);
232                o.addProperty("default_max_age", c.getDefaultMaxAge());
233                o.addProperty("require_auth_time", c.getRequireAuthTime());
234                o.add("default_acr_values", getAsArray(c.getDefaultACRvalues()));
235                o.addProperty("initiate_login_uri", c.getInitiateLoginUri());
236                o.addProperty("post_logout_redirect_uri", c.getPostLogoutRedirectUri());
237                o.add("request_uris", getAsArray(c.getRequestUris()));
238                return o;
239        }
