047public class ClientDetailsEntityJsonProcessor {

049        private static Gson gson = new Gson();
050        private static JsonParser parser = new JsonParser();
051
052        /**
053         * 
054         * Create an unbound ClientDetailsEntity from the given JSON string.
055         * 
056         * @param jsonString
057         * @return the entity if successful, null otherwise
058         */
059        public static ClientDetailsEntity parse(String jsonString) {
060                JsonElement jsonEl = parser.parse(jsonString);
061                if (jsonEl.isJsonObject()) {
062
063                        JsonObject o = jsonEl.getAsJsonObject();
064                        ClientDetailsEntity c = new ClientDetailsEntity();
065
066                        // TODO: make these field names into constants
067
068                        // these two fields should only be sent in the update request, and MUST match existing values
069                        c.setClientId(getAsString(o, "client_id"));
070                        c.setClientSecret(getAsString(o, "client_secret"));
071
072                        // OAuth DynReg
073                        c.setRedirectUris(getAsStringSet(o, "redirect_uris"));
074                        c.setClientName(getAsString(o, "client_name"));
075                        c.setClientUri(getAsString(o, "client_uri"));
076                        c.setLogoUri(getAsString(o, "logo_uri"));
077                        c.setContacts(getAsStringSet(o, "contacts"));
078                        c.setTosUri(getAsString(o, "tos_uri"));
079
080                        String authMethod = getAsString(o, "token_endpoint_auth_method");
081                        if (authMethod != null) {
082                                c.setTokenEndpointAuthMethod(AuthMethod.getByValue(authMethod));
083                        }
084
085                        // scope is a space-separated string
086                        String scope = getAsString(o, "scope");
087                        if (scope != null) {
088                                c.setScope(Sets.newHashSet(Splitter.on(" ").split(scope)));
089                        }
090
091                        c.setGrantTypes(getAsStringSet(o, "grant_types"));
092                        c.setPolicyUri(getAsString(o, "policy_uri"));
093                        c.setJwksUri(getAsString(o, "jwks_uri"));
094
095
096                        // OIDC Additions
097                        String appType = getAsString(o, "application_type");
098                        if (appType != null) {
099                                c.setApplicationType(AppType.getByValue(appType));
100                        }
101
102                        c.setSectorIdentifierUri(getAsString(o, "sector_identifier_uri"));
103
104                        String subjectType = getAsString(o, "subject_type");
105                        if (subjectType != null) {
106                                c.setSubjectType(SubjectType.getByValue(subjectType));
107                        }
108
109                        c.setRequestObjectSigningAlg(getAsJwsAlgorithm(o, "request_object_signing_alg"));
110
111                        c.setUserInfoSignedResponseAlg(getAsJwsAlgorithm(o, "userinfo_signed_response_alg"));
112                        c.setUserInfoEncryptedResponseAlg(getAsJweAlgorithm(o, "userinfo_encrypted_response_alg"));
113                        c.setUserInfoEncryptedResponseEnc(getAsJweEncryptionMethod(o, "userinfo_encrypted_response_enc"));
114
115                        c.setIdTokenSignedResponseAlg(getAsJwsAlgorithm(o, "id_token_signed_response_alg"));
116                        c.setIdTokenEncryptedResponseAlg(getAsJweAlgorithm(o, "id_token_encrypted_response_alg"));
117                        c.setIdTokenEncryptedResponseEnc(getAsJweEncryptionMethod(o, "id_token_encrypted_response_enc"));
118
119                        if (o.has("default_max_age")) {
120                                if (o.get("default_max_age").isJsonPrimitive()) {
121                                        c.setDefaultMaxAge(o.get("default_max_age").getAsInt());
122                                }
123                        }
124
125                        if (o.has("require_auth_time")) {
126                                if (o.get("require_auth_time").isJsonPrimitive()) {
127                                        c.setRequireAuthTime(o.get("require_auth_time").getAsBoolean());
128                                }
129                        }
130
131                        c.setDefaultACRvalues(getAsStringSet(o, "default_acr_values"));
132                        c.setInitiateLoginUri(getAsString(o, "initiate_login_uri"));
133                        c.setPostLogoutRedirectUri(getAsString(o, "post_logout_redirect_uri"));
134                        c.setRequestUris(getAsStringSet(o, "request_uris"));
135
136                        return c;
137                } else {
138                        return null;
139                }
140        }
141
142        public static JsonElement getAsArray(Set<String> value) {
143                return gson.toJsonTree(value, new TypeToken<Set<String>>(){}.getType());
144        }
145
146        /**
147         * @param c
148         * @param token
149         * @param registrationUri
150         * @return
151         */
152        public static JsonObject serialize(ClientDetailsEntity c, OAuth2AccessTokenEntity token, String registrationUri) {
153                JsonObject o = new JsonObject();
154
155                o.addProperty("client_id", c.getClientId());
156                if (c.getClientSecret() != null) {
157                        o.addProperty("client_secret", c.getClientSecret());
158                        o.addProperty("expires_at", 0); // TODO: do we want to let secrets expire?
159                }
160
161                if (c.getCreatedAt() != null) {
162                        o.addProperty("issued_at", c.getCreatedAt().getTime());
163                }
164                if (token != null) {
165                        o.addProperty("registration_access_token", token.getValue());
166                }
167
168                if (registrationUri != null) {
169                        o.addProperty("registration_client_uri", registrationUri);
170                }
171
172
173                // add in all other client properties
174
175                // OAuth DynReg
176                o.add("redirect_uris", getAsArray(c.getRedirectUris()));
177                o.addProperty("client_name", c.getClientName());
178                o.addProperty("client_uri", c.getClientUri());
179                o.addProperty("logo_uri", c.getLogoUri());
180                o.add("contacts", getAsArray(c.getContacts()));
181                o.addProperty("tos_uri", c.getTosUri());
182                o.addProperty("token_endpoint_auth_method", c.getTokenEndpointAuthMethod() != null ? c.getTokenEndpointAuthMethod().getValue() : null);
183                o.addProperty("scope", c.getScope() != null ? Joiner.on(" ").join(c.getScope()) : null);
184                o.add("grant_types", getAsArray(c.getGrantTypes()));
185                o.addProperty("policy_uri", c.getPolicyUri());
186                o.addProperty("jwks_uri", c.getJwksUri());
187
188                // OIDC Registration
189                o.addProperty("application_type", c.getApplicationType() != null ? c.getApplicationType().getValue() : null);
190                o.addProperty("sector_identifier_uri", c.getSectorIdentifierUri());
191                o.addProperty("subject_type", c.getSubjectType() != null ? c.getSubjectType().getValue() : null);
192                o.addProperty("request_object_signing_alg", c.getRequestObjectSigningAlg() != null ? c.getRequestObjectSigningAlg().getAlgorithmName() : null);
193                o.addProperty("userinfo_signed_response_alg", c.getUserInfoSignedResponseAlg() != null ? c.getUserInfoSignedResponseAlg().getAlgorithmName() : null);
194                o.addProperty("userinfo_encrypted_response_alg", c.getUserInfoEncryptedResponseAlg() != null ? c.getUserInfoEncryptedResponseAlg().getAlgorithmName() : null);
195                o.addProperty("userinfo_encrypted_response_enc", c.getUserInfoEncryptedResponseEnc() != null ? c.getUserInfoEncryptedResponseEnc().getAlgorithmName() : null);
196                o.addProperty("id_token_signed_response_alg", c.getIdTokenSignedResponseAlg() != null ? c.getIdTokenSignedResponseAlg().getAlgorithmName() : null);
197                o.addProperty("id_token_encrypted_response_alg", c.getIdTokenEncryptedResponseAlg() != null ? c.getIdTokenEncryptedResponseAlg().getAlgorithmName() : null);
198                o.addProperty("id_token_encrypted_response_enc", c.getIdTokenEncryptedResponseEnc() != null ? c.getIdTokenEncryptedResponseEnc().getAlgorithmName() : null);
199                o.addProperty("default_max_age", c.getDefaultMaxAge());
200                o.addProperty("require_auth_time", c.getRequireAuthTime());
201                o.add("default_acr_values", getAsArray(c.getDefaultACRvalues()));
202                o.addProperty("initiate_login_uri", c.getInitiateLoginUri());
203                o.addProperty("post_logout_redirect_uri", c.getPostLogoutRedirectUri());
204                o.add("request_uris", getAsArray(c.getRequestUris()));
205                return o;
206        }

208        /**
209         * Gets the value of the given member as a JWE Algorithm, null if it doesn't exist
210         */
211        public static JWEAlgorithmEmbed getAsJweAlgorithm(JsonObject o, String member) {
212                String s = getAsString(o, member);
213                if (s != null) {
214                        return JWEAlgorithmEmbed.getForAlgorithmName(s);
215                } else {
216                        return null;
217                }
218        }
219
220        /**
221         * Gets the value of the given member as a JWE Encryption Method, null if it doesn't exist
222         */
223        public static JWEEncryptionMethodEmbed getAsJweEncryptionMethod(JsonObject o, String member) {
224                String s = getAsString(o, member);
225                if (s != null) {
226                        return JWEEncryptionMethodEmbed.getForAlgorithmName(s);
227                } else {
228                        return null;
229                }
230        }
231
232        /**
233         * Gets the value of the given member as a JWS Algorithm, null if it doesn't exist
234         */
235        public static JWSAlgorithmEmbed getAsJwsAlgorithm(JsonObject o, String member) {
236                String s = getAsString(o, member);
237                if (s != null) {
238                        return JWSAlgorithmEmbed.getForAlgorithmName(s);
239                } else {
240                        return null;
241                }
242        }
243
244        /**
245         * Gets the value of the given member as a string, null if it doesn't exist
246         */
247        public static String getAsString(JsonObject o, String member) {
248                if (o.has(member)) {
249                        JsonElement e = o.get(member);
250                        if (e != null && e.isJsonPrimitive()) {
251                                return e.getAsString();
252                        } else {
253                                return null;
254                        }
255                } else {
256                        return null;
257                }
258        }
259
260        /**
261         * Gets the value of the given given member as a set of strings, null if it doesn't exist
262         */
263        public static Set<String> getAsStringSet(JsonObject o, String member) throws JsonSyntaxException {
264                if (o.has(member)) {
265                        return gson.fromJson(o.get(member), new TypeToken<Set<String>>(){}.getType());
266                } else {
267                        return null;
268                }
269        }
