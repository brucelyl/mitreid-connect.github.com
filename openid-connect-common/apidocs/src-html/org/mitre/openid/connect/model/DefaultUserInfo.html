373    /**
374         * Parse a JsonObject into a UserInfo.
375         * @param o
376         * @return
377         */
378        public static UserInfo fromJson(JsonObject obj) {
379                DefaultUserInfo ui = new DefaultUserInfo();
380
381                ui.setSub(obj.has("sub") ? obj.get("sub").getAsString() : null);
382                
383                ui.setName(obj.has("name") ? obj.get("name").getAsString() : null);
384                ui.setPreferredUsername(obj.has("preferred_username") ? obj.get("preferred_username").getAsString() : null);
385                ui.setGivenName(obj.has("given_name") ? obj.get("given_name").getAsString() : null);
386                ui.setFamilyName(obj.has("family_name") ? obj.get("family_name").getAsString() : null);
387                ui.setMiddleName(obj.has("middle_name") ? obj.get("middle_name").getAsString() : null);
388                ui.setNickname(obj.has("nickname") ? obj.get("nickname").getAsString() : null);
389                ui.setProfile(obj.has("profile") ? obj.get("profile").getAsString() : null);
390                ui.setPicture(obj.has("picture") ? obj.get("picture").getAsString() : null);
391                ui.setWebsite(obj.has("website") ? obj.get("website").getAsString() : null);
392                ui.setGender(obj.has("gender") ? obj.get("gender").getAsString() : null);
393                ui.setZoneinfo(obj.has("zone_info") ? obj.get("zone_info").getAsString() : null);
394                ui.setLocale(obj.has("locale") ? obj.get("locale").getAsString() : null);
395                ui.setUpdatedTime(obj.has("updated_time") ? obj.get("updated_time").getAsString() : null);
396                ui.setBirthdate(obj.has("birthdate") ? obj.get("birthdate").getAsString() : null);
397                
398                ui.setEmail(obj.has("email") ? obj.get("email").getAsString() : null);
399                ui.setEmailVerified(obj.has("email_verified") ? obj.get("email_verified").getAsBoolean() : null);
400                
401                ui.setPhoneNumber(obj.has("phone_number") ? obj.get("phone_number").getAsString() : null);
402
403                
404                if (obj.has("address") && obj.get("address").isJsonObject()) {
405                        JsonObject addr = obj.get("address").getAsJsonObject();
406                        ui.setAddress(new Address());
407                        
408                        ui.getAddress().setFormatted(addr.has("formatted") ? addr.get("formatted").getAsString() : null);
409                        ui.getAddress().setStreetAddress(addr.has("street_address") ? addr.get("street_address").getAsString() : null);
410                        ui.getAddress().setLocality(addr.has("locality") ? addr.get("locality").getAsString() : null);
411                        ui.getAddress().setRegion(addr.has("region") ? addr.get("region").getAsString() : null);
412                        ui.getAddress().setPostalCode(addr.has("postal_code") ? addr.get("postal_code").getAsString() : null);
413                        ui.getAddress().setCountry(addr.has("country") ? addr.get("country").getAsString() : null);
414                        
415                }
416
417                
418                return ui;
419                
420        }
