385        /**
386         * Parse a JsonObject into a UserInfo.
387         * @param o
388         * @return
389         */
390        public static UserInfo fromJson(JsonObject obj) {
391                DefaultUserInfo ui = new DefaultUserInfo();
392
393                ui.setSub(obj.has("sub") ? obj.get("sub").getAsString() : null);
394
395                ui.setName(obj.has("name") ? obj.get("name").getAsString() : null);
396                ui.setPreferredUsername(obj.has("preferred_username") ? obj.get("preferred_username").getAsString() : null);
397                ui.setGivenName(obj.has("given_name") ? obj.get("given_name").getAsString() : null);
398                ui.setFamilyName(obj.has("family_name") ? obj.get("family_name").getAsString() : null);
399                ui.setMiddleName(obj.has("middle_name") ? obj.get("middle_name").getAsString() : null);
400                ui.setNickname(obj.has("nickname") ? obj.get("nickname").getAsString() : null);
401                ui.setProfile(obj.has("profile") ? obj.get("profile").getAsString() : null);
402                ui.setPicture(obj.has("picture") ? obj.get("picture").getAsString() : null);
403                ui.setWebsite(obj.has("website") ? obj.get("website").getAsString() : null);
404                ui.setGender(obj.has("gender") ? obj.get("gender").getAsString() : null);
405                ui.setZoneinfo(obj.has("zone_info") ? obj.get("zone_info").getAsString() : null);
406                ui.setLocale(obj.has("locale") ? obj.get("locale").getAsString() : null);
407                ui.setUpdatedTime(obj.has("updated_time") ? obj.get("updated_time").getAsString() : null);
408                ui.setBirthdate(obj.has("birthdate") ? obj.get("birthdate").getAsString() : null);
409
410                ui.setEmail(obj.has("email") ? obj.get("email").getAsString() : null);
411                ui.setEmailVerified(obj.has("email_verified") ? obj.get("email_verified").getAsBoolean() : null);
412
413                ui.setPhoneNumber(obj.has("phone_number") ? obj.get("phone_number").getAsString() : null);
414
415
416                if (obj.has("address") && obj.get("address").isJsonObject()) {
417                        JsonObject addr = obj.get("address").getAsJsonObject();
418                        ui.setAddress(new Address());
419
420                        ui.getAddress().setFormatted(addr.has("formatted") ? addr.get("formatted").getAsString() : null);
421                        ui.getAddress().setStreetAddress(addr.has("street_address") ? addr.get("street_address").getAsString() : null);
422                        ui.getAddress().setLocality(addr.has("locality") ? addr.get("locality").getAsString() : null);
423                        ui.getAddress().setRegion(addr.has("region") ? addr.get("region").getAsString() : null);
424                        ui.getAddress().setPostalCode(addr.has("postal_code") ? addr.get("postal_code").getAsString() : null);
425                        ui.getAddress().setCountry(addr.has("country") ? addr.get("country").getAsString() : null);
426
427                }
428
429
430                return ui;
431
432        }
