001package org.mitre.oauth2.model;
002
003import javax.persistence.Basic;
004import javax.persistence.Column;
005import javax.persistence.Entity;
006import javax.persistence.FetchType;
007import javax.persistence.GeneratedValue;
008import javax.persistence.GenerationType;
009import javax.persistence.Id;
010import javax.persistence.Lob;
011import javax.persistence.NamedQueries;
012import javax.persistence.NamedQuery;
013import javax.persistence.Table;
014
015import org.springframework.security.oauth2.provider.code.AuthorizationRequestHolder;
016
017/**
018 * Entity class for authorization codes
019 * 
020 * @author aanganes
021 *
022 */
023@Entity
024@Table(name="authorization_code")
025@NamedQueries({
026        @NamedQuery(name = "AuthorizationCodeEntity.getByValue", query = "select a from AuthorizationCodeEntity a where a.code = :code")
027})
028public class AuthorizationCodeEntity {
029
030        private Long id;
031        
032        private String code;
033        
034        private AuthorizationRequestHolder authorizationRequestHolder;
035
036        /**
037         * Default constructor.
038         */
039        public AuthorizationCodeEntity() {
040                
041        }
042        
043        /**
044         * Create a new AuthorizationCodeEntity with the given code and AuthorizationRequestHolder.
045         * 
046         * @param code                  the authorization code
047         * @param authRequest   the AuthoriztionRequestHolder associated with the original code request
048         */
049        public AuthorizationCodeEntity(String code, AuthorizationRequestHolder authRequest) {
050                this.code = code;
051                this.authorizationRequestHolder = authRequest;
052        }
053        
054        /**
055         * @return the id
056         */
057        @Id
058    @GeneratedValue(strategy = GenerationType.IDENTITY)
059        public Long getId() {
060                return id;
061        }
062
063        /**
064         * @param id the id to set
065         */
066        public void setId(Long id) {
067                this.id = id;
068        }
069
070        /**
071         * @return the code
072         */
073        @Basic
074        public String getCode() {
075                return code;
076        }
077
078        /**
079         * @param code the code to set
080         */
081        public void setCode(String code) {
082                this.code = code;
083        }
084
085        /**
086         * @return the authorizationRequestHolder
087         */
088        @Lob
089        @Basic(fetch=FetchType.EAGER)
090        @Column(name="authorization_request_holder")
091        public AuthorizationRequestHolder getAuthorizationRequestHolder() {
092                return authorizationRequestHolder;
093        }
094
095        /**
096         * @param authorizationRequestHolder the authorizationRequestHolder to set
097         */
098        public void setAuthorizationRequestHolder(AuthorizationRequestHolder authorizationRequestHolder) {
099                this.authorizationRequestHolder = authorizationRequestHolder;
100        }
101        
102}
