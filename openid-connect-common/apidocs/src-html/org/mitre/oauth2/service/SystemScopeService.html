020package org.mitre.oauth2.service;
021
022import java.util.Set;
023
024import org.mitre.oauth2.model.SystemScope;
025
026import com.google.common.collect.Sets;
027
028/**
029 * @author jricher
030 *
031 */
032public interface SystemScopeService {
033
034        public static final String OFFLINE_ACCESS = "offline_access";
035        public static final String OPENID_SCOPE = "openid";
036        public static final String ID_TOKEN_SCOPE = "id-token"; // ID tokens are generated using this scope
037        public static final String REGISTRATION_TOKEN_SCOPE = "registration-token"; // this scope manages dynamic client registrations
038        public static final String RESOURCE_TOKEN_SCOPE = "resource-token"; // this scope manages client-style protected resources
039        public static final String UMA_PROTECTION_SCOPE = "uma_protection";
040        public static final String UMA_AUTHORIZATION_SCOPE = "uma_authorization";
041        
042        public static final Set<SystemScope> reservedScopes = 
043                Sets.newHashSet(
044                        new SystemScope(ID_TOKEN_SCOPE),
045                        new SystemScope(REGISTRATION_TOKEN_SCOPE),
046                        new SystemScope(RESOURCE_TOKEN_SCOPE)
047                );
048
049        public Set<SystemScope> getAll();
050
051        /**
052         * Get all scopes that are defaulted to new clients on this system
053         * @return
054         */
055        public Set<SystemScope> getDefaults();
056
057        /**
058         * Get all the reserved system scopes. These can't be used
059         * by clients directly, but are instead tied to special system
060         * tokens like id tokens and registration access tokens.
061         * 
062         * @return
063         */
064        public Set<SystemScope> getReserved();
065
066        /**
067         * Get all the registered scopes that are restricted.
068         * @return
069         */
070        public Set<SystemScope> getRestricted();
071
072        /**
073         * Get all the registered scopes that aren't restricted.
074         * @return
075         */
076        public Set<SystemScope> getUnrestricted();
077
078        public SystemScope getById(Long id);
079
080        public SystemScope getByValue(String value);
081
082        public void remove(SystemScope scope);
083
084        public SystemScope save(SystemScope scope);
085
086        /**
087         * Translate the set of scope strings into a set of SystemScope objects.
088         * @param scope
089         * @return
090         */
091        public Set<SystemScope> fromStrings(Set<String> scope);
092
093        /**
094         * Pluck the scope values from the set of SystemScope objects and return a list of strings
095         * @param scope
096         * @return
097         */
098        public Set<String> toStrings(Set<SystemScope> scope);
099
100        /**
101         * Test whether the scopes in both sets are compatible, with special
102         * processing for structured scopes. All scopes in "actual" must exist in
103         * "expected". If a scope in "expected" is structured and has a value, it
104         * must be matched exactly by its corresponding scope in "actual". If a
105         * scope in "expected" is structured but has no value, it may be matched by
106         * a scope with or without a value in "actual".
107         */
108        public boolean scopesMatch(Set<String> expected, Set<String> actual);
109
110        /**
111         * Remove any system-reserved or registered restricted scopes from the
112         * set and return the result.
113         * @param scopes
114         * @return
115         */
116        public Set<SystemScope> removeRestrictedAndReservedScopes(Set<SystemScope> scopes);
117
118        /**
119         * Remove any system-reserved scopes from the set and return the result.
120         * @param scopes
121         * @return
122         */
123        public Set<SystemScope> removeReservedScopes(Set<SystemScope> scopes);
124
125}
