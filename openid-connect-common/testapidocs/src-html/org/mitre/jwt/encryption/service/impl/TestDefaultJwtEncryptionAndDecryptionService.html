096        private DefaultJwtEncryptionAndDecryptionService service;

098        @Before
099        public void prepare() throws NoSuchAlgorithmException, InvalidKeySpecException, JOSEException {
100                
101                service = new DefaultJwtEncryptionAndDecryptionService(keys);
102                
103                claimsSet.setIssuer(issuer);
104                claimsSet.setSubject(subject);
105        }
106        
107        @Test
108        public void decrypt_RSA() throws ParseException {
109                
110                service.setDefaultDecryptionKeyId(RSAkid);
111                service.setDefaultEncryptionKeyId(RSAkid);
112                
113                JWEObject jwt = JWEObject.parse(compactSerializedJwe);
114                
115                assertThat(jwt.getPayload(), nullValue()); // observe..nothing is there
116                
117                service.decryptJwt(jwt);
118                String result = jwt.getPayload().toString(); // and voila! decrypto-magic
119                
120                assertEquals(plainText, result);
121        }
122        
123        @Test
124        public void encryptThenDecrypt_RSA() throws ParseException {
125                
126                service.setDefaultDecryptionKeyId(RSAkid);
127                service.setDefaultEncryptionKeyId(RSAkid);
128                
129                JWEHeader header = new JWEHeader(JWEAlgorithm.RSA_OAEP, EncryptionMethod.A256GCM);
130                
131                EncryptedJWT jwt = new EncryptedJWT(header, claimsSet);
132                
133                service.encryptJwt(jwt);
134                String serialized = jwt.serialize();
135                
136                EncryptedJWT encryptedJwt = EncryptedJWT.parse(serialized);
137                assertThat(encryptedJwt.getJWTClaimsSet(), nullValue());
138                service.decryptJwt(encryptedJwt);
139                
140                ReadOnlyJWTClaimsSet resultClaims = encryptedJwt.getJWTClaimsSet();
141                
142                assertEquals(claimsSet.getIssuer(), resultClaims.getIssuer());
143                assertEquals(claimsSet.getSubject(), resultClaims.getSubject());
144        }
