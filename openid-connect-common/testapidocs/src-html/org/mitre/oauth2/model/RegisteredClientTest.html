040        /**
041         * Test method for {@link org.mitre.oauth2.model.RegisteredClient#RegisteredClient()}.
042         */
043        @Test
044        public void testRegisteredClient() {
045
046                // make sure all the pass-through getters and setters work
047
048                RegisteredClient c = new RegisteredClient();
049
050                c.setClientId("s6BhdRkqt3");
051                c.setClientSecret("ZJYCqe3GGRvdrudKyZS0XhGv_Z45DuKhCUk0gBR1vZk");
052                c.setClientSecretExpiresAt(new Date(1577858400L * 1000L));
053                c.setRegistrationAccessToken("this.is.an.access.token.value.ffx83");
054                c.setRegistrationClientUri("https://server.example.com/connect/register?client_id=s6BhdRkqt3");
055                c.setApplicationType(ClientDetailsEntity.AppType.WEB);
056                c.setRedirectUris(ImmutableSet.of("https://client.example.org/callback", "https://client.example.org/callback2"));
057                c.setClientName("My Example");
058                c.setLogoUri("https://client.example.org/logo.png");
059                c.setSubjectType(ClientDetailsEntity.SubjectType.PAIRWISE);
060                c.setSectorIdentifierUri("https://other.example.net/file_of_redirect_uris.json");
061                c.setTokenEndpointAuthMethod(ClientDetailsEntity.AuthMethod.SECRET_BASIC);
062                c.setJwksUri("https://client.example.org/my_public_keys.jwks");
063                c.setUserInfoEncryptedResponseAlg(new JWEAlgorithmEmbed(JWEAlgorithm.RSA1_5));
064                c.setUserInfoEncryptedResponseEnc(new JWEEncryptionMethodEmbed(EncryptionMethod.A128CBC_HS256));
065                c.setContacts(ImmutableSet.of("ve7jtb@example.org", "mary@example.org"));
066                c.setRequestUris(ImmutableSet.of("https://client.example.org/rf.txt#qpXaRLh_n93TTR9F252ValdatUQvQiJi5BDub2BeznA"));
067
068                assertEquals("s6BhdRkqt3", c.getClientId());
069                assertEquals("ZJYCqe3GGRvdrudKyZS0XhGv_Z45DuKhCUk0gBR1vZk", c.getClientSecret());
070                assertEquals(new Date(1577858400L * 1000L), c.getClientSecretExpiresAt());
071                assertEquals("this.is.an.access.token.value.ffx83", c.getRegistrationAccessToken());
072                assertEquals("https://server.example.com/connect/register?client_id=s6BhdRkqt3", c.getRegistrationClientUri());
073                assertEquals(ClientDetailsEntity.AppType.WEB, c.getApplicationType());
074                assertEquals(ImmutableSet.of("https://client.example.org/callback", "https://client.example.org/callback2"), c.getRedirectUris());
075                assertEquals("My Example", c.getClientName());
076                assertEquals("https://client.example.org/logo.png", c.getLogoUri());
077                assertEquals(ClientDetailsEntity.SubjectType.PAIRWISE, c.getSubjectType());
078                assertEquals("https://other.example.net/file_of_redirect_uris.json", c.getSectorIdentifierUri());
079                assertEquals(ClientDetailsEntity.AuthMethod.SECRET_BASIC, c.getTokenEndpointAuthMethod());
080                assertEquals("https://client.example.org/my_public_keys.jwks", c.getJwksUri());
081                assertEquals(JWEAlgorithm.RSA1_5, c.getUserInfoEncryptedResponseAlg().getAlgorithm());
082                assertEquals(EncryptionMethod.A128CBC_HS256, c.getUserInfoEncryptedResponseEnc().getAlgorithm());
083                assertEquals(ImmutableSet.of("ve7jtb@example.org", "mary@example.org"), c.getContacts());
084                assertEquals(ImmutableSet.of("https://client.example.org/rf.txt#qpXaRLh_n93TTR9F252ValdatUQvQiJi5BDub2BeznA"), c.getRequestUris());
085        }
